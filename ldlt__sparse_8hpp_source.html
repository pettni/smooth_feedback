46 template<typename Scalar, int Layout>
47 inline auto ldlt_nnz(const Eigen::SparseMatrix<Scalar, Layout> & A)
48 {
49  Eigen::Index n = A.cols();
50 
51  // nnz(k) = # nonzeros in row k of L (excluding ones on diagonal)
52  Eigen::Matrix<int, -1, 1> nnz = Eigen::Matrix<int, -1, 1>::Zero(n);
53 
54  // elimination tree: tree(i) = min { j : j > i, l_{ji} != 0 }
55  // elimination tree is s.t. a non-zero at A_ki causes fill-in in row k for tree successors of i
56  Eigen::Matrix<int, -1, 1> tree(n);
57  Eigen::Matrix<int, -1, 1> visited(n);
58 
59  // traverse each row of L
60  for (Eigen::Index row = 0; row != n; ++row) {
61  tree(row) = -1; // no parent yet
62  visited(row) = row; // mark
63 
64  // traverse non-zeros in corresponding row/column of symmetric A (up to diagonal)
65  for (It<Scalar, Layout> it(A, row); it && it.index() < row; ++it) {
66  // traverse elimination tree to determine fill-in from non-zero A_{col, row}
67  for (int col = it.index(); visited(col) != row; col = tree(col)) {
68  // found new non-zero
69  if (tree(col) == -1) { tree(col) = row; } // first non-zero in this column
70  ++nnz(col); // increment count of column non-zeros
71  visited(col) = row;
72  }
73  }
74  }
75 
76  return std::make_tuple(nnz, tree);
77 }

82 template<typename Scalar>
83 class LDLTSparse
84 {
85 public:
96  inline LDLTSparse(const Eigen::SparseMatrix<Scalar> & A)
97  {
98  auto n = A.cols();
99 
100  // working memory
101  Eigen::Matrix<int, -1, 1> visited(n);
102  Eigen::Matrix<int, -1, 1> Yidx(n), bfr(n); // nonzero indices of Y and temporary buffer
103  Eigen::Matrix<Scalar, -1, 1> Yval(n); // values of Y stored at nonzero indices
104 
105  const auto [nnz_col, tree] = ldlt_nnz(A);
106  L_.resize(n, n);
107  L_.reserve(nnz_col);
108  d_inv_.resize(n);
109 
110  Scalar d_new = A.coeff(0, 0);
111  if (d_new == 0) {
112  info_ = 1;
113  return;
114  }
115  d_inv_(0) = Scalar(1.) / d_new;
116 
117  // Fill in L row-wise (implicit ones on diagonal)
118  for (Eigen::Index row = 1; row != n; ++row) {
119  // solve the triangular sparse system L[:k, :k] y = A[:k, k] w.r.t. y
120 
121  // find Yidx -- the non-zero indices of y
122  visited[row] = row;
123  int Ynnz = 0;
124  Yval(row) = 0.;
125  for (It<Scalar> it(A, row); it && it.index() < row; ++it) {
126  int branch_nnz = 0;
127  for (int node = it.index(); visited[node] != row; node = tree(node)) {
128  visited[node] = row;
129  bfr(branch_nnz++) = node;
130  }
131  while (branch_nnz > 0) {
132  Yidx[Ynnz++] = bfr[--branch_nnz]; // store branch in reverse order
133  }
134  }
135 
136  for (It<Scalar> it(A, row); it && it.index() < row; ++it) {
137  Yval(it.index()) = it.value(); // set y[i] = A(row, i)
138  }
139 
140  // pass two: iterate over columns k of L and perform subtractions y_k -= l_{k, i} y_i
141  for (Eigen::Index i = 0; i != Ynnz; ++i) {
142  const int col = Yidx[Ynnz - i - 1]; // traverse reverse branch-wise
143  for (It<Scalar> it_l(L_, col); it_l && it_l.index(); ++it_l) {
144  Yval(it_l.index()) -= Yval(col) * it_l.value();
145  }
146  }
147 
148  // Now y defined by Ynnz, Yidx and Yval solves system above
149 
150  // set L[row, :] = Dinv[:row, :row] * y
151  // and calculate d_new = A[row, row] - y' Dinv y
152  Scalar d_new = A.coeff(row, row);
153  for (Eigen::Index i = 0; i != Ynnz; ++i) {
154  const int yi = Yidx[Ynnz - i - 1];
155  const Scalar yval_i = Yval[yi];
156  Yval[yi] = 0; // reset for next iteration
157  L_.insert(row, yi) = yval_i * d_inv_(yi); // columns of L are filled in order
158  d_new -= yval_i * yval_i * d_inv_(yi);
159  }
160 
161  if (d_new == 0) {
162  info_ = row + 1;
163  return;
164  }
165  d_inv_(row) = Scalar(1) / d_new;
166  }
167 
168  L_.makeCompressed();
169 
170  info_ = 0;
171  }

197  inline void solve_inplace(Eigen::Matrix<Scalar, -1, 1> & b) const
198  {
199  const Eigen::Index N = L_.cols();
200 
201  // solve L x = b in-place
202  for (Eigen::Index col = 0; col != N; ++col) {
203  for (It<Scalar> it(L_, col); it; ++it) { b(it.index()) -= b(col) * it.value(); }
204  }
205 
206  // solve D x = b in-place
207  for (int col = 0; col != N; ++col) { b(col) *= d_inv_(col); }
208 
209  // solve L' x = b in-place
210  for (Eigen::Index row = N - 1; row >= 0; --row) {
211  for (It<Scalar> it(L_, row); it; ++it) { b(row) -= b(it.index()) * it.value(); }
212  }
213  }

221  inline Eigen::Matrix<Scalar, -1, 1> solve(const Eigen::Matrix<Scalar, -1, 1> & b) const
222  {
223  Eigen::Matrix<Scalar, -1, 1> x(b);
224  solve_inplace(x);
225  return x;
226  }
