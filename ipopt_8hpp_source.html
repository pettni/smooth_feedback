75 inline bool get_nlp_info(
76 Ipopt::Index & n,
77 Ipopt::Index & m,
78 Ipopt::Index & nnz_jac_g,
79 Ipopt::Index & nnz_h_lag,
80 IndexStyleEnum & index_style) override
81 {
82 n = nlp_.n();
83 m = nlp_.m();
84
85 const auto J = nlp_.dg_dx(Eigen::VectorXd::Zero(n));
86 nnz_jac_g = J.nonZeros();
87
88 nnz_h_lag = 0; // default
89 if constexpr (HessianNLP<Problem>) {
90 if (use_hessian_) {
91 H_ = nlp_.d2f_dx2(Eigen::VectorXd::Zero(n));
92 H_ += nlp_.d2g_dx2(Eigen::VectorXd::Zero(n), Eigen::VectorXd::Zero(m));
93 H_.makeCompressed();
94 nnz_h_lag = H_.nonZeros();
95 }
96 } else {
97 assert(use_hessian_ == false);
98 }
99
100 index_style = TNLP::C_STYLE; // zero-based
101
102 return true;
103 }

194 inline bool eval_jac_g(
195 Ipopt::Index n,
196 const Ipopt::Number * x,
197 [[maybe_unused]] bool new_x,
198 [[maybe_unused]] Ipopt::Index m,
199 [[maybe_unused]] Ipopt::Index nele_jac,
200 Ipopt::Index * iRow,
201 Ipopt::Index * jCol,
202 Ipopt::Number * values) override
203 {
204 if (values == NULL) {
205 const auto & J = nlp_.dg_dx(Eigen::VectorXd::Zero(n));
206 assert(nele_jac == J.nonZeros());
207
208 for (auto cntr = 0u, od = 0u; od < J.outerSize(); ++od) {
209 for (Eigen::InnerIterator it(J, od); it; ++it) {
210 iRow[cntr] = it.row();
211 jCol[cntr++] = it.col();
212 }
213 }
214 } else {
215 const auto & J = nlp_.dg_dx(Eigen::Map<const Eigen::VectorXd>(x, n));
216 assert(nele_jac == J.nonZeros());
217
218 for (auto cntr = 0u, od = 0u; od < J.outerSize(); ++od) {
219 for (Eigen::InnerIterator it(J, od); it; ++it) { values[cntr++] = it.value(); }
220 }
221 }
222 return true;
223 }

228 inline bool eval_h(
229 Ipopt::Index n,
230 const Ipopt::Number * x,
231 [[maybe_unused]] bool new_x,
232 Ipopt::Number sigma,
233 Ipopt::Index m,
234 const Ipopt::Number * lambda,
235 [[maybe_unused]] bool new_lambda,
236 [[maybe_unused]] Ipopt::Index nele_hess,
237 Ipopt::Index * iRow,
238 Ipopt::Index * jCol,
239 Ipopt::Number * values) override
240 {
241 assert(use_hessian_);
242 assert(HessianNLP<Problem>);
243 assert(H_.isCompressed());
244 assert(H_.nonZeros() == nele_hess);
245
246 H_.coeffs().setZero();
247
248 if (values == NULL) {
249 for (auto cntr = 0u, od = 0u; od < H_.outerSize(); ++od) {
250 for (Eigen::InnerIterator it(H_, od); it; ++it) {
251 // transpose: H upper triangular but ipopt expects lower-triangular
252 iRow[cntr] = it.col();
253 jCol[cntr++] = it.row();
254 }
255 }
256 } else {
257 Eigen::Map<const Eigen::VectorXd> xvar(x, n);
258 Eigen::Map<const Eigen::VectorXd> lvar(lambda, m);
259
260 H_ += nlp_.d2f_dx2(xvar);
261 H_ *= sigma;
262 H_ += nlp_.d2g_dx2(xvar, lvar);
263
264 for (auto cntr = 0u, od = 0u; od < H_.outerSize(); ++od) {
265 for (Eigen::InnerIterator it(H_, od); it; ++it) { values[cntr++] = it.value(); }
266 }
267 }
268
269 return true;
270 }

275 inline void finalize_solution(
276 Ipopt::SolverReturn status,
277 Ipopt::Index n,
278 const Ipopt::Number * x,
279 const Ipopt::Number * z_L,
280 const Ipopt::Number * z_U,
281 Ipopt::Index m,
282 [[maybe_unused]] const Ipopt::Number * g,
283 const Ipopt::Number * lambda,
284 Ipopt::Number obj_value,
285 const Ipopt::IpoptData * ip_data,
286 [[maybe_unused]] Ipopt::IpoptCalculatedQuantities * ip_cq) override
287 {
288 switch (status) {
289 case Ipopt::SolverReturn::SUCCESS:
290 sol_.status = NLPSolution::Status::Optimal;
291 break;
292 case Ipopt::SolverReturn::STOP_AT_ACCEPTABLE_POINT:
293 sol_.status = NLPSolution::Status::Optimal;
294 break;
295 case Ipopt::SolverReturn::MAXITER_EXCEEDED:
296 sol_.status = NLPSolution::Status::MaxIterations;
297 break;
298 case Ipopt::SolverReturn::CPUTIME_EXCEEDED:
299 sol_.status = NLPSolution::Status::MaxTime;
300 break;
301 case Ipopt::SolverReturn::LOCAL_INFEASIBILITY:
302 sol_.status = NLPSolution::Status::PrimalInfeasible;
303 break;
304 case Ipopt::SolverReturn::DIVERGING_ITERATES:
305 sol_.status = NLPSolution::Status::DualInfeasible;
306 break;
307 default:
308 sol_.status = NLPSolution::Status::Unknown;
309 break;
310 }
311
312 sol_.iter = ip_data->iter_count();
313
314 sol_.x = Eigen::Map<const Eigen::VectorXd>(x, n);
315 sol_.zl = Eigen::Map<const Eigen::VectorXd>(z_L, n);
316 sol_.zu = Eigen::Map<const Eigen::VectorXd>(z_U, n);
317 sol_.lambda = Eigen::Map<const Eigen::VectorXd>(lambda, m);
318
319 sol_.objective = obj_value;
320 }

322private:
323 Problem nlp_;
324 bool use_hessian_;
325 NLPSolution sol_;
326 Eigen::SparseMatrix<double> H_;

340inline NLPSolution solve_nlp_ipopt(
341 NLP auto && nlp,
342 std::optional<NLPSolution> warmstart = {},
343 std::vector<std::pair<std::string, int>> opts_integer = {},
344 std::vector<std::pair<std::string, std::string>> opts_string = {},
345 std::vector<std::pair<std::string, double>> opts_numeric = {})
346{
347 using nlp_t = std::decay_t<decltype(nlp)>;
348 bool use_hessian = false;
349 const auto n = nlp.n();
350
351 auto it = std::find_if(opts_string.begin(), opts_string.end(), [](const auto & p) {
352 return p.first == "hessian_approximation";
353 });
354 if (it != opts_string.end() && it->second == "exact") { use_hessian = true; }
355
356 Ipopt::SmartPtr<IpoptNLP<nlp_t>> ipopt_nlp =
357 new IpoptNLP<nlp_t>(std::forward<decltype(nlp)>(nlp), use_hessian);
358 Ipopt::SmartPtr<Ipopt::IpoptApplication> app = new Ipopt::IpoptApplication();
359
360 // silence welcome message
361 app->Options()->SetStringValue("sb", "yes");
362
363 if (warmstart.has_value()) {
364 // initial guess is given
365 app->Options()->SetStringValue("warm_start_init_point", "yes");
366 ipopt_nlp->sol() = warmstart.value();
367 } else {
368 // initial guess not given, set to zero
369 app->Options()->SetStringValue("warm_start_init_point", "no");
370 ipopt_nlp->sol().x.setZero(n);
371 }
372
373 // override with user-provided options
374 for (auto [opt, val] : opts_integer) { app->Options()->SetIntegerValue(opt, val); }
375 for (auto [opt, val] : opts_string) { app->Options()->SetStringValue(opt, val); }
376 for (auto [opt, val] : opts_numeric) { app->Options()->SetNumericValue(opt, val); }
377
378 app->Initialize();
379 app->OptimizeTNLP(ipopt_nlp);
380
381 return ipopt_nlp->sol();
382}
