168  inline bool eval_jac_g(
169  Ipopt::Index n,
170  const Ipopt::Number * x,
171  [[maybe_unused]] bool new_x,
172  [[maybe_unused]] Ipopt::Index m,
173  [[maybe_unused]] Ipopt::Index nele_jac,
174  Ipopt::Index * iRow,
175  Ipopt::Index * jCol,
176  Ipopt::Number * values) override
177  {
178  if (values == NULL) {
179  const auto J = nlp_.dg_dx(Eigen::VectorXd::Zero(n));
180  assert(nele_jac == J.nonZeros());
181 
182  for (auto cntr = 0u, col = 0u; col < J.cols(); ++col) {
183  for (typename decltype(J)::InnerIterator it(J, col); it; ++it) {
184  iRow[cntr] = it.index();
185  jCol[cntr++] = col;
186  }
187  }
188  } else {
189  const auto J = nlp_.dg_dx(Eigen::Map<const Eigen::VectorXd>(x, n));
190  assert(nele_jac == J.nonZeros());
191 
192  for (auto cntr = 0u, col = 0u; col < J.cols(); ++col) {
193  for (typename decltype(J)::InnerIterator it(J, col); it; ++it) {
194  values[cntr++] = it.value();
195  }
196  }
197  }
198  return true;
199  }

204  inline void finalize_solution(
205  Ipopt::SolverReturn status,
206  Ipopt::Index n,
207  const Ipopt::Number * x,
208  const Ipopt::Number * z_L,
209  const Ipopt::Number * z_U,
210  Ipopt::Index m,
211  [[maybe_unused]] const Ipopt::Number * g,
212  const Ipopt::Number * lambda,
213  Ipopt::Number obj_value,
214  const Ipopt::IpoptData * ip_data,
215  [[maybe_unused]] Ipopt::IpoptCalculatedQuantities * ip_cq) override
216  {
217  switch (status) {
218  case Ipopt::SolverReturn::SUCCESS:
219  sol_.status = NLPSolution::Status::Optimal;
220  break;
221  case Ipopt::SolverReturn::STOP_AT_ACCEPTABLE_POINT:
222  sol_.status = NLPSolution::Status::Optimal;
223  break;
224  case Ipopt::SolverReturn::MAXITER_EXCEEDED:
225  sol_.status = NLPSolution::Status::MaxIterations;
226  break;
227  case Ipopt::SolverReturn::CPUTIME_EXCEEDED:
228  sol_.status = NLPSolution::Status::MaxTime;
229  break;
230  case Ipopt::SolverReturn::LOCAL_INFEASIBILITY:
231  sol_.status = NLPSolution::Status::PrimalInfeasible;
232  break;
233  case Ipopt::SolverReturn::DIVERGING_ITERATES:
234  sol_.status = NLPSolution::Status::DualInfeasible;
235  break;
236  default:
237  sol_.status = NLPSolution::Status::Unknown;
238  break;
239  }
240 
241  sol_.iter = ip_data->iter_count();
242 
243  sol_.x = Eigen::Map<const Eigen::VectorXd>(x, n);
244  sol_.zl = Eigen::Map<const Eigen::VectorXd>(z_L, n);
245  sol_.zu = Eigen::Map<const Eigen::VectorXd>(z_U, n);
246  sol_.lambda = Eigen::Map<const Eigen::VectorXd>(lambda, m);
247 
248  sol_.objective = obj_value;
249  }

253  NLPSolution sol_;

267 inline NLPSolution solve_nlp_ipopt(
268  const NLP & nlp,
269  std::optional<NLPSolution> warmstart = {},
270  std::vector<std::pair<std::string, int>> opts_integer = {},
271  std::vector<std::pair<std::string, std::string>> opts_string = {},
272  std::vector<std::pair<std::string, double>> opts_numeric = {})
273 {
274  Ipopt::SmartPtr<IpoptNLP> ipopt_nlp = new IpoptNLP(nlp);
275  Ipopt::SmartPtr<Ipopt::IpoptApplication> app = new Ipopt::IpoptApplication();
276 
277  for (auto [opt, val] : opts_integer) { app->Options()->SetIntegerValue(opt, val); }
278  for (auto [opt, val] : opts_string) { app->Options()->SetStringValue(opt, val); }
279  for (auto [opt, val] : opts_numeric) { app->Options()->SetNumericValue(opt, val); }
280 
281  // silence welcome message
282  app->Options()->SetStringValue("sb", "yes");
283 
284  if (warmstart.has_value()) {
285  // initial guess is given
286  app->Options()->SetStringValue("warm_start_init_point", "yes");
287  ipopt_nlp->sol() = warmstart.value();
288  } else {
289  // initial guess not given, set to zero
290  app->Options()->SetStringValue("warm_start_init_point", "no");
291  ipopt_nlp->sol().x.setZero(nlp.n);
292  }
293 
294  app->Initialize();
295  app->OptimizeTNLP(ipopt_nlp);
296 
297  return ipopt_nlp->sol();
298 }
