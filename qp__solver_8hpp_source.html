92template<typename Pbm, typename D1, typename D2>
93bool polish_qp(
94 const Pbm & pbm,
95 qp_solution_t<Pbm> & sol,
96 const QPSolverParams & prm,
97 const typename decltype(Pbm::A)::Scalar c,
98 const Eigen::MatrixBase<D1> & sx,
99 const Eigen::MatrixBase<D2> & sy)
100{
101 using AmatT = decltype(Pbm::A);
102 using Scalar = typename AmatT::Scalar;
103 static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
104
105 static constexpr Scalar inf = std::numeric_limits<Scalar>::infinity();
106 static constexpr Scalar eps = std::numeric_limits<Scalar>::epsilon();
107
108 static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
109 const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows();
110
111 // FIND ACTIVE CONSTRAINT SETS
112
113 Eigen::Index nl = 0, nu = 0;
114 for (auto idx = 0; idx < m; ++idx) {
115 if (sol.dual[idx] < -100 * eps && pbm.l[idx] != -inf) { nl++; }
116 if (sol.dual[idx] > 100 * eps && pbm.u[idx] != inf) { nu++; }
117 }
118
119 Eigen::VectorXi LU_idx(nl + nu);
120 for (auto idx = 0, lcntr = 0, ucntr = 0; idx < m; ++idx) {
121 if (sol.dual[idx] < -100 * eps && pbm.l[idx] != -inf) { LU_idx(lcntr++) = idx; }
122 if (sol.dual[idx] > 100 * eps && pbm.u[idx] != inf) { LU_idx(nl + ucntr++) = idx; }
123 }
124
125 // FORM REDUCED SYSTEMS (27) AND (30)
126
127 // square symmetric system matrix
128 using HT = std::conditional_t<sparse, Eigen::SparseMatrix<Scalar>, Eigen::Matrix<Scalar, -1, -1>>;
129 HT H(n + nl + nu, n + nl + nu), Hp(n + nl + nu, n + nl + nu);
130
131 // fill up H
132 if constexpr (sparse) {
133 // preallocate nonzeros
134 Eigen::VectorXi nnz(n + nl + nu);
135 for (auto i = 0u; i != n; ++i) { nnz(i) = pbm.P.outerIndexPtr()[i + 1] - pbm.P.outerIndexPtr()[i]; }
136 for (auto i = 0u; i != nl + nu; ++i) {
137 nnz(n + i) = pbm.A.outerIndexPtr()[LU_idx(i) + 1] - pbm.A.outerIndexPtr()[LU_idx(i)];
138 }
139 H.reserve(nnz);
140 Hp.reserve(nnz + Eigen::VectorXi::Ones(n + nl + nu));
141
142 // fill P in top left block
143 for (auto k = 0u; k < n; ++k) {
144 for (Eigen::InnerIterator it(pbm.P, k); it; ++it) {
145 const Scalar pij = c * sx(it.col()) * sx(it.row()) * it.value();
146 H.insert(it.row(), it.col()) = pij;
147 Hp.insert(it.row(), it.col()) = pij;
148 }
149 }
150
151 // fill selected rows of A in top right block
152 for (auto a_row = 0u; a_row != nl + nu; ++a_row) {
153 for (Eigen::InnerIterator it(pbm.A, LU_idx(a_row)); it; ++it) {
154 const Scalar Aij = sy(it.row()) * sx(it.col()) * it.value();
155 H.insert(it.col(), n + a_row) = Aij;
156 Hp.insert(it.col(), n + a_row) = Aij;
157 }
158 }
159 } else {
160 H.setZero();
161 H.topLeftCorner(n, n) = c * sx.asDiagonal() * pbm.P * sx.asDiagonal();
162 for (auto i = 0u; i != nl + nu; ++i) {
163 H.col(n + i).template head<N>(n) = sy(LU_idx(i)) * pbm.A.row(LU_idx(i)) * sx.asDiagonal();
164 }
165 Hp = H;
166 }
167
168 // add perturbing diagonal elements to Hp
169 if constexpr (sparse) {
170 for (auto i = 0u; i != n; ++i) { Hp.coeffRef(i, i) += prm.delta; }
171 for (auto i = 0u; i != nl + nu; ++i) { Hp.coeffRef(n + i, n + i) -= prm.delta; }
172 H.makeCompressed();
173 Hp.makeCompressed();
174 } else {
175 Hp.topLeftCorner(n, n) += Eigen::VectorX<Scalar>::Constant(n, prm.delta).asDiagonal();
176 Hp.bottomRightCorner(nl + nu, nl + nu) -= Eigen::VectorX<Scalar>::Constant(nl + nu, prm.delta).asDiagonal();
177 }
178
179 Eigen::VectorX<Scalar> h(n + nl + nu);
180 h.head(n) = -c * sx.cwiseProduct(pbm.q);
181 for (auto i = 0u; i != nl; ++i) { h(n + i) = sy(LU_idx(i)) * pbm.l(LU_idx(i)); }
182 for (auto i = 0u; i != nu; ++i) { h(n + nl + i) = sy(LU_idx(nl + i)) * pbm.u(LU_idx(nl + i)); }
183
184 // ITERATIVE REFINEMENT
185
186 // factorize Hp
187 std::conditional_t<sparse, Eigen::SimplicialLDLT<decltype(H), Eigen::Upper>, Eigen::LDLT<decltype(H), Eigen::Upper>>
188 ldlt(Hp);
189
190 if (ldlt.info()) { return false; }
191
192 Eigen::VectorX<Scalar> t_hat = Eigen::VectorX<Scalar>::Zero(n + nl + nu);
193 for (auto i = 0u; i != prm.polish_iter; ++i) {
194 t_hat += ldlt.solve(h - H.template selfadjointView<Eigen::Upper>() * t_hat);
195 }
196
197 // UPDATE SOLUTION
198
199 sol.primal = t_hat.template head<N>(n);
200 for (auto i = 0u; i < nl; ++i) { sol.dual(LU_idx(i)) = t_hat(n + i); }
201 for (auto i = 0u; i < nu; ++i) { sol.dual(LU_idx(nl + i)) = t_hat(n + nl + i); }
202
203 return true;
204}

240template<typename Pbm>
241class QPSolver
242{
243 using AmatT = decltype(Pbm::A);
244 using Scalar = typename AmatT::Scalar;
245 static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
246
247 // static sizes
248 static constexpr Eigen::Index M = AmatT::RowsAtCompileTime;
249 static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
250 static constexpr Eigen::Index K = (N == -1 || M == -1) ? Eigen::Index(-1) : N + M;
251
252 // typedefs
253 using Rn = Eigen::Vector<Scalar, N>;
254 using Rm = Eigen::Vector<Scalar, M>;
255 using Rk = Eigen::Vector<Scalar, K>;
256 using Ht = std::conditional_t<sparse, Eigen::SparseMatrix<Scalar>, Eigen::Matrix<Scalar, K, K>>;
257 using LDLTt = std::conditional_t<sparse, Eigen::SimplicialLDLT<Ht, Eigen::Upper>, Eigen::LDLT<Ht, Eigen::Upper>>;
258
259 static inline const Scalar inf = std::numeric_limits<Scalar>::infinity();
260
261public:
265 QPSolver(const QPSolverParams & prm = {}) : prm_(prm) {}
266
274 QPSolver(const Pbm & pbm, const QPSolverParams & prm = {}) : prm_(prm) { analyze(pbm); }
275
277 QPSolver(const QPSolver &) = default;
279 QPSolver(QPSolver &&) = default;
281 QPSolver & operator=(const QPSolver &) = default;
283 QPSolver & operator=(QPSolver &&) = default;

288 const QPSolution<M, N, Scalar> & sol() const { return sol_; }

293 void analyze(const Pbm & pbm)
294 {
295 const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows(), k = n + m;
296
297 // solution
298 sol_.primal.setZero(n);
299 sol_.dual.setZero(m);
300
301 // scaling variables and working memory
302 c_ = 1;
303 sx_.setOnes(n);
304 sy_.setOnes(m);
305 sx_inc_.setZero(n);
306 sy_inc_.setZero(m);
307
308 // solve working memory
309 z_.resize(m);
310 z_next_.resize(m);
311 rho_.resize(m);
312 p_.resize(k);
313 pt_.resize(k);
314
315 // stopping working memory
316 x_us_.resize(n);
317 dx_us_.resize(n);
318 y_us_.resize(m);
319 dy_us_.resize(m);
320 z_us_.resize(m);
321 Px_.resize(n);
322 Aty_.resize(n);
323 Ax_.resize(m);
324
325 // preallocate nonzeros in H
326 H_.setZero();
327 H_.resize(k, k);
328 if constexpr (sparse) {
329 Eigen::VectorXi nnz(k);
330 for (auto i = 0u; i < n; ++i) { nnz(i) = pbm.P.outerIndexPtr()[i + 1] - pbm.P.outerIndexPtr()[i] + 1; }
331 for (auto i = 0u; i < m; ++i) { nnz(n + i) = pbm.A.outerIndexPtr()[i + 1] - pbm.A.outerIndexPtr()[i] + 1; }
332 H_.reserve(nnz);
333 }
334 }
335
339 const QPSolution<M, N, Scalar> &
340 solve(const Pbm & pbm, std::optional<std::reference_wrapper<const QPSolution<M, N, Scalar>>> warmstart = {})
341 {
342 // update problem scaling
343 if (prm_.scaling) { scale(pbm); }
344
345 // dynamic sizes
346 const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows();
347
348 // cast parameters to scalar type
349 const Scalar rho_bar = static_cast<Scalar>(prm_.rho);
350 const Scalar alpha = static_cast<Scalar>(prm_.alpha);
351 const Scalar alpha_comp = Scalar(1) - alpha;
352 const Scalar sigma = static_cast<Scalar>(prm_.sigma);
353
354 // return code: when set algorithm is finished
355 std::optional<QPSolutionStatus> ret_code = std::nullopt;
356
357 for (auto i = 0u; i != m; ++i) {
358 if (pbm.l(i) == inf || pbm.u(i) == -inf || pbm.u(i) - pbm.l(i) < Scalar(0.)) {
359 ret_code = QPSolutionStatus::PrimalInfeasible; // feasible set trivially empty
360 }
361
362 // set rho depending on constraint type
363 if (pbm.l(i) == -inf && pbm.u(i) == inf) {
364 rho_(i) = Scalar(1e-6); // unbounded
365 } else if (sy_(i) * abs(pbm.l(i) - pbm.u(i)) < 1e-5) {
366 rho_(i) = Scalar(1e3) * rho_bar; // equality
367 } else {
368 rho_(i) = rho_bar; // inequality
369 }
370 }
371
372 const auto t0 = std::chrono::high_resolution_clock::now();
373
374 // fill square symmetric system matrix H = [P A'; A 0]
375 if constexpr (sparse) {
376 if (H_.isCompressed()) { H_.coeffs().setZero(); }
377
378 for (auto i = 0u; i < pbm.P.outerSize(); ++i) {
379 for (Eigen::InnerIterator it(pbm.P, i); it; ++it) {
380 if (it.col() >= it.row()) {
381 H_.coeffRef(it.row(), it.col()) = c_ * sx_(it.row()) * sx_(it.col()) * it.value();
382 }
383 }
384 }
385 block_add_identity(H_, 0, 0, n, sigma);
386 for (auto i = 0u; i < pbm.A.outerSize(); ++i) {
387 for (Eigen::InnerIterator it(pbm.A, i); it; ++it) {
388 H_.coeffRef(it.col(), n + it.row()) = sy_(it.row()) * sx_(it.col()) * it.value();
389 }
390 }
391 for (auto row = 0u; row < m; ++row) { H_.coeffRef(n + row, n + row) = Scalar(-1) / rho_(row); }
392
393 if (!H_.isCompressed()) { H_.makeCompressed(); }
394 } else {
395 H_.setZero();
396
397 H_.template topLeftCorner<N, N>(n, n) = c_ * sx_.asDiagonal() * pbm.P * sx_.asDiagonal();
398 H_.template topLeftCorner<N, N>(n, n) += Eigen::Vector<Scalar, N>::Constant(n, sigma).asDiagonal();
399 H_.template topRightCorner<N, M>(n, m) = (sy_.asDiagonal() * pbm.A * sx_.asDiagonal()).transpose();
400 H_.template bottomRightCorner<M, M>(m, m) = (-rho_).cwiseInverse().asDiagonal();
401 }
402
403 const auto t_fill = std::chrono::high_resolution_clock::now();
404
405 if (prm_.verbose) {
406 using std::cout, std::left, std::setw, std::right;
407 // clang-format off
408 cout << "========================= QP Solver =========================" << '\n';
409 cout << "Solving " << (sparse ? "sparse" : "dense") << " QP with n=" << n << ", m=" << m << '\n';
410 cout << setw(8) << right << "ITER"
411 << setw(14) << right << "OBJ"
412 << setw(14) << right << "PRI_RES"
413 << setw(14) << right << "DUA_RES"
414 << setw(10) << right << "TIME" << '\n';
415 // clang-format on
416 }
417
418 // factorize H
419 if constexpr (sparse) {
420 if (ldlt_.first) { ldlt_.ldlt.analyzePattern(H_); }
421 ldlt_.ldlt.factorize(H_); // makes copy for permuted matrix..
422 ldlt_.first = false;
423 } else {
424 ldlt_.ldlt.compute(H_);
425 }
426
427 const auto t_factor = std::chrono::high_resolution_clock::now();
428
429 if (ldlt_.ldlt.info()) { ret_code = QPSolutionStatus::Unknown; }
430
431 // initialize solver variables
432 if (warmstart.has_value()) {
433 // warmstart variables must be scaled
434 sol_.primal = sx_.cwiseInverse().cwiseProduct(warmstart.value().get().primal);
435 sol_.dual = c_ * sy_.cwiseInverse().cwiseProduct(warmstart.value().get().dual);
436 z_.noalias() = sy_.asDiagonal() * pbm.A * warmstart.value().get().primal;
437 } else {
438 sol_.primal.setZero();
439 sol_.dual.setZero();
440 z_.setZero();
441 }
442
443 // main optimization loop
444 auto iter = 0u;
445 for (; (!prm_.max_iter || iter != prm_.max_iter.value()) && !ret_code; ++iter) {
446 p_.template segment<N>(0, n) = sigma * sol_.primal - c_ * sx_.asDiagonal() * pbm.q;
447 p_.template segment<M>(n, m) = z_ - rho_.cwiseInverse().cwiseProduct(sol_.dual);
448 if constexpr (sparse) {
449 // p_ = ldlt_.solve(p_);
450 // manual solve because ldlt_.solve() uses temporaries...
451 // Pinv L D L' P x = b <==> x = Pinv * (L' \ Dinv(L \ (P * b)))
452 pt_.noalias() = ldlt_.ldlt.permutationP() * p_;
453 ldlt_.ldlt.matrixL().solveInPlace(pt_);
454 pt_.applyOnTheLeft(ldlt_.ldlt.vectorD().cwiseInverse().asDiagonal());
455 ldlt_.ldlt.matrixU().solveInPlace(pt_);
456 p_.noalias() = ldlt_.ldlt.permutationPinv() * pt_;
457 } else {
458 ldlt_.ldlt.solveInPlace(p_);
459 }
460
461 if (iter % prm_.stop_check_iter == 1) {
462 // termination checking requires difference, store old scaled values
463 dx_us_ = sol_.primal, dy_us_ = sol_.dual;
464 }
465
466 sol_.primal = alpha * p_.template segment<N>(0, n) + alpha_comp * sol_.primal;
467 z_next_ = (alpha * rho_.cwiseInverse().cwiseProduct(p_.template segment<M>(n, m))
468 + alpha_comp * rho_.cwiseInverse().cwiseProduct(sol_.dual) + z_)
469 .cwiseMax(sy_.cwiseProduct(pbm.l))
470 .cwiseMin(sy_.cwiseProduct(pbm.u));
471 sol_.dual = alpha_comp * sol_.dual + alpha * p_.template segment<M>(n, m) + rho_.cwiseProduct(z_)
472 - rho_.cwiseProduct(z_next_);
473 std::swap(z_, z_next_);
474
475 if (iter % prm_.stop_check_iter == 1) {
476 // unscale solution
477 x_us_ = sx_.cwiseProduct(sol_.primal);
478 y_us_ = sy_.cwiseProduct(sol_.dual) / c_;
479 z_us_ = sy_.cwiseInverse().cwiseProduct(z_);
480 dx_us_ = sx_.cwiseProduct(sol_.primal - dx_us_);
481 dy_us_ = sy_.cwiseProduct(sol_.dual - dy_us_) / c_;
482
483 // check stopping criteria for unscaled problem and unscaled variables
484 ret_code = check_stopping(pbm);
485
486 if (prm_.verbose) {
487 using std::cout, std::setw, std::right, std::chrono::microseconds;
488 // clang-format off
489 cout << setw(7) << right << iter << ":"
490 << std::scientific
491 << setw(14) << right << (0.5 * pbm.P * x_us_ + pbm.q).dot(x_us_)
492 << setw(14) << right << (pbm.A * x_us_ - z_us_).template lpNorm<Eigen::Infinity>()
493 << setw(14) << right << (pbm.P * x_us_ + pbm.q + pbm.A.transpose() * y_us_).template lpNorm<Eigen::Infinity>()
494 << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count()
495 << '\n';
496 // clang-format on
497 }
498
499 // check for timeout
500 if (!ret_code) {
501 if (prm_.max_time && std::chrono::high_resolution_clock::now() > t0 + prm_.max_time.value()) {
502 ret_code = QPSolutionStatus::MaxTime;
503 }
504 }
505 }
506 }
507
508 const auto t_iter = std::chrono::high_resolution_clock::now();
509
510 // polish solution if optimal
511 if (ret_code.has_value() && ret_code.value() == QPSolutionStatus::Optimal && prm_.polish) {
512 if (detail::polish_qp(pbm, sol_, prm_, c_, sx_, sy_)) {
513 if (prm_.verbose) {
514 // unscale solution
515 x_us_ = sx_.cwiseProduct(sol_.primal);
516 y_us_ = sy_.cwiseProduct(sol_.dual) / c_;
517 z_us_ = sy_.cwiseInverse().cwiseProduct(z_);
518
519 using std::cout, std::setw, std::right, std::chrono::microseconds;
520 // clang-format off
521 cout << setw(8) << right << "polish:"
522 << std::scientific
523 << setw(14) << right << (0.5 * pbm.P * x_us_ + pbm.q).dot(x_us_)
524 << setw(14) << right << (pbm.A * x_us_ - z_us_).template lpNorm<Eigen::Infinity>()
525 << setw(14) << right << (pbm.P * x_us_ + pbm.q + pbm.A.transpose() * y_us_).template lpNorm<Eigen::Infinity>()
526 << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count()
527 << '\n';
528 // clang-format on
529 }
530
531 } else {
532 if (prm_.verbose) { std::cout << "Polish failed" << '\n'; }
533 sol_.code = QPSolutionStatus::PolishFailed;
534 }
535 }
536
537 const auto t_polish = std::chrono::high_resolution_clock::now();
538
539 // unscale solution
540 sol_.code = ret_code.value_or(QPSolutionStatus::MaxIterations);
541 sol_.primal = sx_.cwiseProduct(sol_.primal);
542 sol_.dual = sy_.cwiseProduct(sol_.dual) / c_;
543 sol_.objective = sol_.primal.dot(0.5 * pbm.P * sol_.primal + pbm.q);
544 sol_.iter = iter;
545
546 if (prm_.verbose) {
547 using std::cout, std::left, std::right, std::setw, std::chrono::microseconds;
548
549 // clang-format off
550 cout << "QP solver summary:" << '\n';
551 cout << "Result " << static_cast<int>(sol_.code) << '\n';
552
553 cout << setw(25) << left << "Iterations" << setw(10) << right << iter - 1 << '\n';
554 cout << setw(26) << left << "Total time (s)" << setw(10) << right << duration_cast<microseconds>(t_polish - t0).count() << '\n';
555 cout << setw(25) << left << " Matrix filling" << setw(10) << right << duration_cast<microseconds>(t_fill - t0).count() << '\n';
556 cout << setw(25) << left << " Factorization" << setw(10) << right << duration_cast<microseconds>(t_factor - t_fill).count() << '\n';
557 cout << setw(25) << left << " Iteration" << setw(10) << right << duration_cast<microseconds>(t_iter - t_factor).count() << '\n';
558 cout << setw(25) << left << " Polish" << setw(10) << right << duration_cast<microseconds>(t_polish - t_iter).count() << '\n';
559 cout << "=============================================================" << '\n';
560 // clang-format on
561 }
562
563 return sol_;
564 }
565
566protected:
570 std::optional<QPSolutionStatus> check_stopping(const Pbm & pbm)
571 {
572 const Eigen::Index m = pbm.A.rows();
573
574 // norm function
575 static const auto norm = [](auto && t) -> Scalar { return t.template lpNorm<Eigen::Infinity>(); };
576
577 // OPTIMALITY
578
579 // check primal
580 Ax_.noalias() = pbm.A * x_us_;
581 const Scalar Ax_norm = norm(Ax_);
582 Ax_ -= z_us_;
583 if (norm(Ax_) <= prm_.eps_abs + prm_.eps_rel * std::max<Scalar>(Ax_norm, norm(z_us_))) {
584 // primal succeeded, check dual
585 Px_.noalias() = pbm.P * x_us_;
586 Aty_.noalias() = pbm.A.transpose() * y_us_;
587 const Scalar dual_scale = std::max<Scalar>({norm(Px_), norm(pbm.q), norm(Aty_)});
588 Px_ += pbm.q + Aty_;
589 if (norm(Px_) <= prm_.eps_abs + prm_.eps_rel * dual_scale) { return QPSolutionStatus::Optimal; }
590 }
591
592 // PRIMAL INFEASIBILITY
593
594 Aty_.noalias() = pbm.A.transpose() * dy_us_; // NOTE new value A' * dy
595 const Scalar Edy_norm = norm(dy_us_);
596
597 Scalar u_dyp_plus_l_dyn = Scalar(0);
598 for (auto i = 0u; i != m; ++i) {
599 if (pbm.u(i) != inf) {
600 u_dyp_plus_l_dyn += pbm.u(i) * std::max<Scalar>(Scalar(0), dy_us_(i));
601 } else if (dy_us_(i) > prm_.eps_primal_inf * Edy_norm) {
602 // contributes +inf to sum --> no certificate
603 u_dyp_plus_l_dyn = inf;
604 break;
605 }
606 if (pbm.l(i) != -inf) {
607 u_dyp_plus_l_dyn += pbm.l(i) * std::min<Scalar>(Scalar(0), dy_us_(i));
608 } else if (dy_us_(i) < -prm_.eps_primal_inf * Edy_norm) {
609 // contributes +inf to sum --> no certificate
610 u_dyp_plus_l_dyn = inf;
611 break;
612 }
613 }
614
615 if (std::max<Scalar>(norm(Aty_), u_dyp_plus_l_dyn) < prm_.eps_primal_inf * Edy_norm) {
616 return QPSolutionStatus::PrimalInfeasible;
617 }
618
619 // DUAL INFEASIBILITY
620
621 Ax_.noalias() = pbm.A * dx_us_; // note new value A * dx
622 const Scalar dx_norm = norm(dx_us_);
623 Px_.noalias() = pbm.P * dx_us_;
624
625 bool dual_infeasible =
626 (norm(Px_) <= prm_.eps_dual_inf * dx_norm) && (pbm.q.dot(dx_us_) <= prm_.eps_dual_inf * dx_norm);
627 for (auto i = 0u; i != m && dual_infeasible; ++i) {
628 if (pbm.u(i) == inf) {
629 dual_infeasible &= (Ax_(i) >= -prm_.eps_dual_inf * dx_norm);
630 } else if (pbm.l(i) == -inf) {
631 dual_infeasible &= (Ax_(i) <= prm_.eps_dual_inf * dx_norm);
632 } else {
633 dual_infeasible &= std::abs(Ax_(i)) < prm_.eps_dual_inf * dx_norm;
634 }
635 }
636
637 if (dual_infeasible) { return QPSolutionStatus::DualInfeasible; }
638
639 return std::nullopt;
640 }
641
669 void scale(const Pbm & pbm)
670 {
671 sx_.setOnes();
672 sy_.setOnes();
673
674 sx_inc_.setZero();
675
676 // find "norm" of cost function
677 for (auto i = 0u; i < pbm.P.outerSize(); ++i) {
678 for (Eigen::InnerIterator it(pbm.P, i); it; ++it) {
679 sx_inc_(it.col()) = std::max(sx_inc_(it.col()), std::abs(it.value()));
680 }
681 }
682
683 // if there are "zero cols"
684 for (auto i = 0u; i != sx_.size(); ++i) {
685 if (sx_inc_(i) == 0) { sx_inc_(i) = 1; }
686 }
687
688 // scale cost function
689 c_ = Scalar(1) / std::max({1e-6, sx_inc_.mean(), pbm.q.template lpNorm<Eigen::Infinity>()});
690
691 int iter = 0;
692
693 // calculate inf-norm for every column of [Ps As' ; As 0]
694 do {
695 sx_inc_.setZero();
696 sy_inc_.setZero();
697 for (auto k = 0u; k < pbm.P.outerSize(); ++k) {
698 for (Eigen::InnerIterator it(pbm.P, k); it; ++it) {
699 // upper left block of H
700 sx_inc_(it.col()) = std::max({
701 sx_inc_(it.col()),
702 std::abs(c_ * sx_(it.row()) * sx_(it.col()) * it.value()),
703 });
704 }
705 }
706 for (auto k = 0u; k < pbm.A.outerSize(); ++k) {
707 for (Eigen::InnerIterator it(pbm.A, k); it; ++it) {
708 const Scalar Aij = std::abs(sy_(it.row()) * sx_(it.col()) * it.value());
709 sx_inc_(it.col()) = std::max(sx_inc_(it.col()), Aij); // bottom left block of H
710 sy_inc_(it.row()) = std::max(sy_inc_(it.row()), Aij); // upper right block of H
711 }
712 }
713
714 // if there are "zero cols" we don't scale
715 for (auto i = 0u; i < sx_.size(); ++i) {
716 if (sx_inc_(i) == 0) { sx_inc_(i) = 1; }
717 }
718 for (auto i = 0u; i < sy_.size(); ++i) {
719 if (sy_inc_(i) == 0) { sy_inc_(i) = 1; }
720 }
721
722 sx_.applyOnTheLeft(sx_inc_.cwiseMax(1e-8).cwiseInverse().cwiseSqrt().asDiagonal());
723 sy_.applyOnTheLeft(sy_inc_.cwiseMax(1e-8).cwiseInverse().cwiseSqrt().asDiagonal());
724 } while (iter++ < 10
725 && std::max((sx_inc_.array() - 1).abs().maxCoeff(), (sy_inc_.array() - 1).abs().maxCoeff()) > 0.1);
726 }
