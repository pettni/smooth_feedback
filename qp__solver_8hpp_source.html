116template<typename Pbm, typename D1, typename D2>
117bool polish_qp(
118 const Pbm & pbm,
119 qp_solution_t<Pbm> & sol,
120 const QPSolverParams & prm,
121 const typename decltype(Pbm::A)::Scalar c,
122 const Eigen::MatrixBase<D1> & sx,
123 const Eigen::MatrixBase<D2> & sy)
124{
125 using AmatT = decltype(Pbm::A);
126 using Scalar = typename AmatT::Scalar;
127 static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
128
129 static constexpr Scalar inf = std::numeric_limits<Scalar>::infinity();
130 static constexpr Scalar eps = std::numeric_limits<Scalar>::epsilon();
131
132 static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
133 const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows();
134
135 // FIND ACTIVE CONSTRAINT SETS
136
137 Eigen::Index nl = 0, nu = 0;
138 for (Eigen::Index idx = 0; idx < m; ++idx) {
139 if (sol.dual[idx] < -100 * eps && pbm.l[idx] != -inf) { nl++; }
140 if (sol.dual[idx] > 100 * eps && pbm.u[idx] != inf) { nu++; }
141 }
142
143 Eigen::VectorXi LU_idx(nl + nu);
144 for (Eigen::Index idx = 0, lcntr = 0, ucntr = 0; idx < m; ++idx) {
145 if (sol.dual[idx] < -100 * eps && pbm.l[idx] != -inf) { LU_idx(lcntr++) = idx; }
146 if (sol.dual[idx] > 100 * eps && pbm.u[idx] != inf) { LU_idx(nl + ucntr++) = idx; }
147 }
148
149 // FORM REDUCED SYSTEMS (27) AND (30)
150
151 // square symmetric system matrix
152 using HT = std::conditional_t<sparse, Eigen::SparseMatrix<Scalar>, Eigen::Matrix<Scalar, -1, -1>>;
153 HT H(n + nl + nu, n + nl + nu), Hp(n + nl + nu, n + nl + nu);
154
155 // fill up H
156 if constexpr (sparse) {
157 // preallocate nonzeros
158 Eigen::VectorXi nnz(n + nl + nu);
159 for (auto i = 0u; i != n; ++i) {
160 nnz(i) = pbm.P.outerIndexPtr()[i + 1] - pbm.P.outerIndexPtr()[i];
161 }
162 for (auto i = 0u; i != nl + nu; ++i) {
163 nnz(n + i) = pbm.A.outerIndexPtr()[LU_idx(i) + 1] - pbm.A.outerIndexPtr()[LU_idx(i)];
164 }
165 H.reserve(nnz);
166 Hp.reserve(nnz + Eigen::VectorXi::Ones(n + nl + nu));
167
168 // fill P in top left block
169 for (auto k = 0u; k < n; ++k) {
170 for (Eigen::InnerIterator it(pbm.P, k); it; ++it) {
171 const Scalar pij = c * sx(it.col()) * sx(it.row()) * it.value();
172 H.insert(it.row(), it.col()) = pij;
173 Hp.insert(it.row(), it.col()) = pij;
174 }
175 }
176
177 // fill selected rows of A in top right block
178 for (auto a_row = 0u; a_row != nl + nu; ++a_row) {
179 for (Eigen::InnerIterator it(pbm.A, LU_idx(a_row)); it; ++it) {
180 const Scalar Aij = sy(it.row()) * sx(it.col()) * it.value();
181 H.insert(it.col(), n + a_row) = Aij;
182 Hp.insert(it.col(), n + a_row) = Aij;
183 }
184 }
185 } else {
186 H.setZero();
187 H.topLeftCorner(n, n) = c * sx.asDiagonal() * pbm.P * sx.asDiagonal();
188 for (auto i = 0u; i != nl + nu; ++i) {
189 H.col(n + i).template head<N>(n) = sy(LU_idx(i)) * pbm.A.row(LU_idx(i)) * sx.asDiagonal();
190 }
191 Hp = H;
192 }
193
194 // add perturbing diagonal elements to Hp
195 if constexpr (sparse) {
196 for (auto i = 0u; i != n; ++i) { Hp.coeffRef(i, i) += prm.delta; }
197 for (auto i = 0u; i != nl + nu; ++i) { Hp.coeffRef(n + i, n + i) -= prm.delta; }
198 H.makeCompressed();
199 Hp.makeCompressed();
200 } else {
201 Hp.topLeftCorner(n, n) += Eigen::VectorX<Scalar>::Constant(n, prm.delta).asDiagonal();
202 Hp.bottomRightCorner(nl + nu, nl + nu) -=
203 Eigen::VectorX<Scalar>::Constant(nl + nu, prm.delta).asDiagonal();
204 }
205
206 Eigen::VectorX<Scalar> h(n + nl + nu);
207 h.head(n) = -c * sx.cwiseProduct(pbm.q);
208 for (auto i = 0u; i != nl; ++i) { h(n + i) = sy(LU_idx(i)) * pbm.l(LU_idx(i)); }
209 for (auto i = 0u; i != nu; ++i) { h(n + nl + i) = sy(LU_idx(nl + i)) * pbm.u(LU_idx(nl + i)); }
210
211 // ITERATIVE REFINEMENT
212
213 // factorize Hp
214 std::conditional_t<
215 sparse,
216 Eigen::SimplicialLDLT<decltype(H), Eigen::Upper>,
217 Eigen::LDLT<decltype(H), Eigen::Upper>>
218 ldlt(Hp);
219
220 if (ldlt.info()) { return false; }
221
222 Eigen::VectorX<Scalar> t_hat = Eigen::VectorX<Scalar>::Zero(n + nl + nu);
223 for (auto i = 0u; i != prm.polish_iter; ++i) {
224 t_hat += ldlt.solve(h - H.template selfadjointView<Eigen::Upper>() * t_hat);
225 }
226
227 // UPDATE SOLUTION
228
229 sol.primal = t_hat.template head<N>(n);
230 for (auto i = 0u; i < nl; ++i) { sol.dual(LU_idx(i)) = t_hat(n + i); }
231 for (auto i = 0u; i < nu; ++i) { sol.dual(LU_idx(nl + i)) = t_hat(n + nl + i); }
232
233 return true;
234}

271class QPSolver
272{
273 using AmatT = decltype(Pbm::A);
274 using Scalar = typename AmatT::Scalar;
275 static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
276
277 // static sizes
278 static constexpr Eigen::Index M = AmatT::RowsAtCompileTime;
279 static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
280 static constexpr Eigen::Index K = (N == -1 || M == -1) ? Eigen::Index(-1) : N + M;
281
282 // typedefs
283 using Rn = Eigen::Vector<Scalar, N>;
284 using Rm = Eigen::Vector<Scalar, M>;
285 using Rk = Eigen::Vector<Scalar, K>;
286 using Ht = std::conditional_t<sparse, Eigen::SparseMatrix<Scalar>, Eigen::Matrix<Scalar, K, K>>;
287 using LDLTt = std::
288 conditional_t<sparse, Eigen::SimplicialLDLT<Ht, Eigen::Upper>, Eigen::LDLT<Ht, Eigen::Upper>>;
289
290 static inline const Scalar inf = std::numeric_limits<Scalar>::infinity();
291
292public:
296 QPSolver(const QPSolverParams & prm = {}) : prm_(prm) {}

305 QPSolver(const Pbm & pbm, const QPSolverParams & prm = {}) : prm_(prm) { analyze(pbm); }

308 QPSolver(const QPSolver &) = default;

312 QPSolver & operator=(const QPSolver &) = default;
314 QPSolver & operator=(QPSolver &&) = default;

324 void analyze(const Pbm & pbm)
325 {
326 const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows(), k = n + m;
327
328 // solution
329 sol_.primal.setZero(n);
330 sol_.dual.setZero(m);
331
332 // scaling variables and working memory
333 c_ = 1;
334 sx_.setOnes(n);
335 sy_.setOnes(m);
336 sx_inc_.setZero(n);
337 sy_inc_.setZero(m);
338
339 // solve working memory
340 z_.resize(m);
341 z_next_.resize(m);
342 rho_.resize(m);
343 p_.resize(k);
344 pt_.resize(k);
345
346 // stopping working memory
347 x_us_.resize(n);
348 dx_us_.resize(n);
349 y_us_.resize(m);
350 dy_us_.resize(m);
351 z_us_.resize(m);
352 Px_.resize(n);
353 Aty_.resize(n);
354 Ax_.resize(m);
355
356 // preallocate nonzeros in H
357 H_.setZero();
358 H_.resize(k, k);
359 if constexpr (sparse) {
360 Eigen::VectorXi nnz(k);
361 for (auto i = 0u; i < n; ++i) {
362 nnz(i) = pbm.P.outerIndexPtr()[i + 1] - pbm.P.outerIndexPtr()[i] + 1;
363 }
364 for (auto i = 0u; i < m; ++i) {
365 nnz(n + i) = pbm.A.outerIndexPtr()[i + 1] - pbm.A.outerIndexPtr()[i] + 1;
366 }
367 H_.reserve(nnz);
368 }
369 }

374 const QPSolution<M, N, Scalar> & solve(
375 const Pbm & pbm,
376 std::optional<std::reference_wrapper<const QPSolution<M, N, Scalar>>> warmstart = {})
377 {
378 // update problem scaling
379 if (prm_.scaling) { scale(pbm); }
380
381 // dynamic sizes
382 const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows();
383
384 // cast parameters to scalar type
385 const Scalar rho_bar = static_cast<Scalar>(prm_.rho);
386 const Scalar alpha = static_cast<Scalar>(prm_.alpha);
387 const Scalar alpha_comp = Scalar(1) - alpha;
388 const Scalar sigma = static_cast<Scalar>(prm_.sigma);
389
390 // return code: when set algorithm is finished
391 std::optional<QPSolutionStatus> ret_code = std::nullopt;
392
393 for (auto i = 0u; i != m; ++i) {
394 if (pbm.l(i) == inf || pbm.u(i) == -inf || pbm.u(i) - pbm.l(i) < Scalar(0.)) {
395 ret_code = QPSolutionStatus::PrimalInfeasible; // feasible set trivially empty
396 }
397
398 // set rho depending on constraint type
399 if (pbm.l(i) == -inf && pbm.u(i) == inf) {
400 rho_(i) = Scalar(1e-6); // unbounded
401 } else if (sy_(i) * abs(pbm.l(i) - pbm.u(i)) < 1e-5) {
402 rho_(i) = Scalar(1e3) * rho_bar; // equality
403 } else {
404 rho_(i) = rho_bar; // inequality
405 }
406 }
407
408 const auto t0 = std::chrono::high_resolution_clock::now();
409
410 // fill square symmetric system matrix H = [P A'; A 0]
411 if constexpr (sparse) {
412 if (H_.isCompressed()) { H_.coeffs().setZero(); }
413
414 for (auto i = 0u; i < pbm.P.outerSize(); ++i) {
415 for (Eigen::InnerIterator it(pbm.P, i); it; ++it) {
416 if (it.col() >= it.row()) {
417 H_.coeffRef(it.row(), it.col()) = c_ * sx_(it.row()) * sx_(it.col()) * it.value();
418 }
419 }
420 }
421 block_add_identity(H_, 0, 0, n, sigma);
422 for (auto i = 0u; i < pbm.A.outerSize(); ++i) {
423 for (Eigen::InnerIterator it(pbm.A, i); it; ++it) {
424 H_.coeffRef(it.col(), n + it.row()) = sy_(it.row()) * sx_(it.col()) * it.value();
425 }
426 }
427 for (auto row = 0u; row < m; ++row) {
428 H_.coeffRef(n + row, n + row) = Scalar(-1) / rho_(row);
429 }
430
431 if (!H_.isCompressed()) { H_.makeCompressed(); }
432 } else {
433 H_.setZero();
434
435 H_.template topLeftCorner<N, N>(n, n) = c_ * sx_.asDiagonal() * pbm.P * sx_.asDiagonal();
436 H_.template topLeftCorner<N, N>(n, n) +=
437 Eigen::Vector<Scalar, N>::Constant(n, sigma).asDiagonal();
438 H_.template topRightCorner<N, M>(n, m) =
439 (sy_.asDiagonal() * pbm.A * sx_.asDiagonal()).transpose();
440 H_.template bottomRightCorner<M, M>(m, m) = (-rho_).cwiseInverse().asDiagonal();
441 }
442
443 const auto t_fill = std::chrono::high_resolution_clock::now();
444
445 if (prm_.verbose) {
446 using std::cout, std::left, std::setw, std::right;
447 // clang-format off
448 cout << "========================= QP Solver =========================" << '\n';
449 cout << "Solving " << (sparse ? "sparse" : "dense") << " QP with n=" << n << ", m=" << m << '\n';
450 cout << setw(8) << right << "ITER"
451 << setw(14) << right << "OBJ"
452 << setw(14) << right << "PRI_RES"
453 << setw(14) << right << "DUA_RES"
454 << setw(10) << right << "TIME" << '\n';
455 // clang-format on
456 }
457
458 // factorize H
459 if constexpr (sparse) {
460 if (ldlt_.first) { ldlt_.ldlt.analyzePattern(H_); }
461 ldlt_.ldlt.factorize(H_); // makes copy for permuted matrix..
462 ldlt_.first = false;
463 } else {
464 ldlt_.ldlt.compute(H_);
465 }
466
467 const auto t_factor = std::chrono::high_resolution_clock::now();
468
469 if (ldlt_.ldlt.info()) { ret_code = QPSolutionStatus::Unknown; }
470
471 // initialize solver variables
472 if (warmstart.has_value()) {
473 // warmstart variables must be scaled
474 sol_.primal = sx_.cwiseInverse().cwiseProduct(warmstart.value().get().primal);
475 sol_.dual = c_ * sy_.cwiseInverse().cwiseProduct(warmstart.value().get().dual);
476 z_.noalias() = sy_.asDiagonal() * pbm.A * warmstart.value().get().primal;
477 } else {
478 sol_.primal.setZero();
479 sol_.dual.setZero();
480 z_.setZero();
481 }
482
483 // main optimization loop
484 auto iter = 0u;
485 for (; (!prm_.max_iter || iter != prm_.max_iter.value()) && !ret_code; ++iter) {
486 p_.template segment<N>(0, n) = sigma * sol_.primal - c_ * sx_.asDiagonal() * pbm.q;
487 p_.template segment<M>(n, m) = z_ - rho_.cwiseInverse().cwiseProduct(sol_.dual);
488 if constexpr (sparse) {
489 // p_ = ldlt_.solve(p_);
490 // manual solve because ldlt_.solve() uses temporaries...
491 // Pinv L D L' P x = b <==> x = Pinv * (L' \ Dinv(L \ (P * b)))
492 pt_.noalias() = ldlt_.ldlt.permutationP() * p_;
493 ldlt_.ldlt.matrixL().solveInPlace(pt_);
494 pt_.applyOnTheLeft(ldlt_.ldlt.vectorD().cwiseInverse().asDiagonal());
495 ldlt_.ldlt.matrixU().solveInPlace(pt_);
496 p_.noalias() = ldlt_.ldlt.permutationPinv() * pt_;
497 } else {
498 ldlt_.ldlt.solveInPlace(p_);
499 }
500
501 if (iter % prm_.stop_check_iter == 1) {
502 // termination checking requires difference, store old scaled values
503 dx_us_ = sol_.primal, dy_us_ = sol_.dual;
504 }
505
506 sol_.primal = alpha * p_.template segment<N>(0, n) + alpha_comp * sol_.primal;
507 z_next_ = (alpha * rho_.cwiseInverse().cwiseProduct(p_.template segment<M>(n, m))
508 + alpha_comp * rho_.cwiseInverse().cwiseProduct(sol_.dual) + z_)
509 .cwiseMax(sy_.cwiseProduct(pbm.l))
510 .cwiseMin(sy_.cwiseProduct(pbm.u));
511 sol_.dual = alpha_comp * sol_.dual + alpha * p_.template segment<M>(n, m)
512 + rho_.cwiseProduct(z_) - rho_.cwiseProduct(z_next_);
513 std::swap(z_, z_next_);
514
515 if (iter % prm_.stop_check_iter == 1) {
516 // unscale solution
517 x_us_ = sx_.cwiseProduct(sol_.primal);
518 y_us_ = sy_.cwiseProduct(sol_.dual) / c_;
519 z_us_ = sy_.cwiseInverse().cwiseProduct(z_);
520 dx_us_ = sx_.cwiseProduct(sol_.primal - dx_us_);
521 dy_us_ = sy_.cwiseProduct(sol_.dual - dy_us_) / c_;
522
523 // check stopping criteria for unscaled problem and unscaled variables
524 ret_code = check_stopping(pbm);
525
526 if (prm_.verbose) {
527 using std::cout, std::setw, std::right, std::chrono::microseconds;
528 // clang-format off
529 cout << setw(7) << right << iter << ":"
530 << std::scientific
531 << setw(14) << right << (0.5 * pbm.P * x_us_ + pbm.q).dot(x_us_)
532 << setw(14) << right << (pbm.A * x_us_ - z_us_).template lpNorm<Eigen::Infinity>()
533 << setw(14) << right << (pbm.P * x_us_ + pbm.q + pbm.A.transpose() * y_us_).template lpNorm<Eigen::Infinity>()
534 << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count()
535 << '\n';
536 // clang-format on
537 }
538
539 // check for timeout
540 if (!ret_code) {
541 if (
542 prm_.max_time
543 && std::chrono::high_resolution_clock::now() > t0 + prm_.max_time.value()) {
544 ret_code = QPSolutionStatus::MaxTime;
545 }
546 }
547 }
548 }
549
550 const auto t_iter = std::chrono::high_resolution_clock::now();
551
552 // polish solution if optimal
553 if (ret_code.has_value() && ret_code.value() == QPSolutionStatus::Optimal && prm_.polish) {
554 if (detail::polish_qp(pbm, sol_, prm_, c_, sx_, sy_)) {
555 if (prm_.verbose) {
556 // unscale solution
557 x_us_ = sx_.cwiseProduct(sol_.primal);
558 y_us_ = sy_.cwiseProduct(sol_.dual) / c_;
559 z_us_ = sy_.cwiseInverse().cwiseProduct(z_);
560
561 using std::cout, std::setw, std::right, std::chrono::microseconds;
562 // clang-format off
563 cout << setw(8) << right << "polish:"
564 << std::scientific
565 << setw(14) << right << (0.5 * pbm.P * x_us_ + pbm.q).dot(x_us_)
566 << setw(14) << right << (pbm.A * x_us_ - z_us_).template lpNorm<Eigen::Infinity>()
567 << setw(14) << right << (pbm.P * x_us_ + pbm.q + pbm.A.transpose() * y_us_).template lpNorm<Eigen::Infinity>()
568 << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count()
569 << '\n';
570 // clang-format on
571 }
572
573 } else {
574 if (prm_.verbose) { std::cout << "Polish failed" << '\n'; }
575 sol_.code = QPSolutionStatus::PolishFailed;
576 }
577 }
578
579 const auto t_polish = std::chrono::high_resolution_clock::now();
580
581 // unscale solution
582 sol_.code = ret_code.value_or(QPSolutionStatus::MaxIterations);
583 sol_.primal = sx_.cwiseProduct(sol_.primal);
584 sol_.dual = sy_.cwiseProduct(sol_.dual) / c_;
585 sol_.objective = sol_.primal.dot(0.5 * pbm.P * sol_.primal + pbm.q);
586 sol_.iter = iter;
587
588 if (prm_.verbose) {
589 using std::cout, std::left, std::right, std::setw, std::chrono::microseconds;
590
591 // clang-format off
592 cout << "QP solver summary:" << '\n';
593 cout << "Result " << static_cast<int>(sol_.code) << '\n';
594
595 cout << setw(25) << left << "Iterations" << setw(10) << right << iter - 1 << '\n';
596 cout << setw(26) << left << "Total time (µs)" << setw(10) << right << duration_cast<microseconds>(t_polish - t0).count() << '\n';
597 cout << setw(25) << left << " Matrix filling" << setw(10) << right << duration_cast<microseconds>(t_fill - t0).count() << '\n';
598 cout << setw(25) << left << " Factorization" << setw(10) << right << duration_cast<microseconds>(t_factor - t_fill).count() << '\n';
599 cout << setw(25) << left << " Iteration" << setw(10) << right << duration_cast<microseconds>(t_iter - t_factor).count() << '\n';
600 cout << setw(25) << left << " Polish" << setw(10) << right << duration_cast<microseconds>(t_polish - t_iter).count() << '\n';
601 cout << "=============================================================" << '\n';
602 // clang-format on
603 }
604
605 return sol_;
606 }

608protected:
612 std::optional<QPSolutionStatus> check_stopping(const Pbm & pbm)
613 {
614 const Eigen::Index m = pbm.A.rows();
615
616 // norm function
617 static const auto norm = [](auto && t) -> Scalar {
618 return t.template lpNorm<Eigen::Infinity>();
619 };
620
621 // OPTIMALITY
622
623 // check primal
624 Ax_.noalias() = pbm.A * x_us_;
625 const Scalar Ax_norm = norm(Ax_);
626 Ax_ -= z_us_;
627 if (norm(Ax_) <= prm_.eps_abs + prm_.eps_rel * std::max<Scalar>(Ax_norm, norm(z_us_))) {
628 // primal succeeded, check dual
629 Px_.noalias() = pbm.P * x_us_;
630 Aty_.noalias() = pbm.A.transpose() * y_us_;
631 const Scalar dual_scale = std::max<Scalar>({norm(Px_), norm(pbm.q), norm(Aty_)});
632 Px_ += pbm.q + Aty_;
633 if (norm(Px_) <= prm_.eps_abs + prm_.eps_rel * dual_scale) {
634 return QPSolutionStatus::Optimal;
635 }
636 }
637
638 // PRIMAL INFEASIBILITY
639
640 Aty_.noalias() = pbm.A.transpose() * dy_us_; // NOTE new value A' * dy
641 const Scalar Edy_norm = norm(dy_us_);
642
643 Scalar u_dyp_plus_l_dyn = Scalar(0);
644 for (auto i = 0u; i != m; ++i) {
645 if (pbm.u(i) != inf) {
646 u_dyp_plus_l_dyn += pbm.u(i) * std::max<Scalar>(Scalar(0), dy_us_(i));
647 } else if (dy_us_(i) > prm_.eps_primal_inf * Edy_norm) {
648 // contributes +inf to sum --> no certificate
649 u_dyp_plus_l_dyn = inf;
650 break;
651 }
652 if (pbm.l(i) != -inf) {
653 u_dyp_plus_l_dyn += pbm.l(i) * std::min<Scalar>(Scalar(0), dy_us_(i));
654 } else if (dy_us_(i) < -prm_.eps_primal_inf * Edy_norm) {
655 // contributes +inf to sum --> no certificate
656 u_dyp_plus_l_dyn = inf;
657 break;
658 }
659 }
660
661 if (std::max<Scalar>(norm(Aty_), u_dyp_plus_l_dyn) < prm_.eps_primal_inf * Edy_norm) {
662 return QPSolutionStatus::PrimalInfeasible;
663 }
664
665 // DUAL INFEASIBILITY
666
667 Ax_.noalias() = pbm.A * dx_us_; // note new value A * dx
668 const Scalar dx_norm = norm(dx_us_);
669 Px_.noalias() = pbm.P * dx_us_;
670
671 bool dual_infeasible = (norm(Px_) <= prm_.eps_dual_inf * dx_norm)
672 && (pbm.q.dot(dx_us_) <= prm_.eps_dual_inf * dx_norm);
673 for (auto i = 0u; i != m && dual_infeasible; ++i) {
674 if (pbm.u(i) == inf) {
675 dual_infeasible &= (Ax_(i) >= -prm_.eps_dual_inf * dx_norm);
676 } else if (pbm.l(i) == -inf) {
677 dual_infeasible &= (Ax_(i) <= prm_.eps_dual_inf * dx_norm);
678 } else {
679 dual_infeasible &= std::abs(Ax_(i)) < prm_.eps_dual_inf * dx_norm;
680 }
681 }
682
683 if (dual_infeasible) { return QPSolutionStatus::DualInfeasible; }
684
685 return std::nullopt;
686 }

715 void scale(const Pbm & pbm)
716 {
717 sx_.setOnes();
718 sy_.setOnes();
719
720 sx_inc_.setZero();
721
722 // find "norm" of cost function
723 for (auto i = 0u; i < pbm.P.outerSize(); ++i) {
724 for (Eigen::InnerIterator it(pbm.P, i); it; ++it) {
725 sx_inc_(it.col()) = std::max(sx_inc_(it.col()), std::abs(it.value()));
726 }
727 }
728
729 // if there are "zero cols"
730 for (auto i = 0u; i != sx_.size(); ++i) {
731 if (sx_inc_(i) == 0) { sx_inc_(i) = 1; }
732 }
733
734 // scale cost function
735 c_ = Scalar(1) / std::max({1e-6, sx_inc_.mean(), pbm.q.template lpNorm<Eigen::Infinity>()});
736
737 int iter = 0;
738
739 // calculate inf-norm for every column of [Ps As' ; As 0]
740 do {
741 sx_inc_.setZero();
742 sy_inc_.setZero();
743 for (auto k = 0u; k < pbm.P.outerSize(); ++k) {
744 for (Eigen::InnerIterator it(pbm.P, k); it; ++it) {
745 // upper left block of H
746 sx_inc_(it.col()) = std::max({
747 sx_inc_(it.col()),
748 std::abs(c_ * sx_(it.row()) * sx_(it.col()) * it.value()),
749 });
750 }
751 }
752 for (auto k = 0u; k < pbm.A.outerSize(); ++k) {
753 for (Eigen::InnerIterator it(pbm.A, k); it; ++it) {
754 const Scalar Aij = std::abs(sy_(it.row()) * sx_(it.col()) * it.value());
755 sx_inc_(it.col()) = std::max(sx_inc_(it.col()), Aij); // bottom left block of H
756 sy_inc_(it.row()) = std::max(sy_inc_(it.row()), Aij); // upper right block of H
757 }
758 }
759
760 // if there are "zero cols" we don't scale
761 for (auto i = 0u; i < sx_.size(); ++i) {
762 if (sx_inc_(i) == 0) { sx_inc_(i) = 1; }
763 }
764 for (auto i = 0u; i < sy_.size(); ++i) {
765 if (sy_inc_(i) == 0) { sy_inc_(i) = 1; }
766 }
767
768 sx_.applyOnTheLeft(sx_inc_.cwiseMax(1e-8).cwiseInverse().cwiseSqrt().asDiagonal());
769 sy_.applyOnTheLeft(sy_inc_.cwiseMax(1e-8).cwiseInverse().cwiseSqrt().asDiagonal());
770 } while (
771 iter++ < 10
772 && std::max((sx_inc_.array() - 1).abs().maxCoeff(), (sy_inc_.array() - 1).abs().maxCoeff())
773 > 0.1);
774 }
