151 template<LieGroup G, Manifold U>
152  requires(Dof<G> > 0 && Dof<U> > 0)
153 void ocp_to_qp_allocate(
154  const OptimalControlProblem<G, U> & pbm,
155  std::size_t K,
156  QuadraticProgramSparse<double> & qp,
157  bool lin_con = false)
158 {
159  // problem info
160  static constexpr int Nx = Dof<G>;
161  static constexpr int Nu = Dof<U>;
162 
163  static_assert(Nx > 0, "State space dimension must be static");
164  static_assert(Nu > 0, "Input space dimension must be static");
165 
166  const uint32_t n_eq = K * Nx;
167  const uint32_t NU_iq = K * pbm.ulim.A.rows();
168  const uint32_t NX_iq = K * pbm.glim.A.rows();
169  const uint32_t NXLIN_iq = lin_con ? K * Nu : 0;
170 
171  const uint32_t nvar = K * Nx + K * Nu;
172  const uint32_t ncon = n_eq + NU_iq + NX_iq + NXLIN_iq;
173 
174  // Matrix sizes
175  qp.P.resize(nvar, nvar);
176  qp.q.resize(nvar);
177 
178  qp.A.resize(ncon, nvar);
179  qp.l.resize(ncon);
180  qp.u.resize(ncon);
181 
182  // SPARSITY PATTERN FOR P
183 
184  Eigen::Matrix<int, -1, 1> Pp(nvar);
185  for (auto k = 0u; k != K; ++k) {
186  for (auto i = 0u; i != Nu; ++i) { Pp[k * Nu + i] = 1 + i; }
187  }
188 
189  for (auto k = 0u; k != K; ++k) {
190  for (auto i = 0u; i != Nx; ++i) { Pp[K * Nu + k * Nx + i] = 1 + i; }
191  }
192 
193  // SPARSITY PATTERN FOR A
194 
195  Eigen::Matrix<int, -1, 1> Ap(ncon);
196  int Arow = 0;
197 
198  // state constraint k = 0
199  Ap.segment(Arow, Nx).setConstant(1 + Nu);
200  Arow += Nx;
201 
202  // state constraints k = 1, ... K
203  Ap.segment(Arow, (K - 1) * Nx).setConstant(1 + Nu + Nx);
204  Arow += (K - 1) * Nx;
205 
206  // input constraints
207  if (NU_iq > 0) { Ap.segment(Arow, NU_iq).setConstant(Nu); }
208  Arow += NU_iq;
209 
210  // state constraints
211  if (NX_iq > 0) { Ap.segment(Arow, NX_iq).setConstant(Nx); }
212  Arow += NX_iq;
213 
214  // state linearization constraints
215  if (NXLIN_iq > 0) { Ap.segment(Arow, NXLIN_iq).setConstant(1); }
216 
217  qp.P.reserve(Pp);
218  qp.A.reserve(Ap);
219 }

224 template<LieGroup G, Manifold U, typename Dyn, diff::Type DT = diff::Type::DEFAULT>
225  requires(Dof<G> > 0 && Dof<U> > 0)
226 void ocp_to_qp_fill(
227  const OptimalControlProblem<G, U> & pbm,
228  std::size_t K,
229  const Dyn & f,
230  const LinearizationInfo<G, U> & lin,
231  QuadraticProgramSparse<double> & qp)
232 {
233  using std::placeholders::_1;
234 
235  static constexpr int Nx = Dof<G>;
236  static constexpr int Nu = Dof<U>;
237  const int NU = K * Nu;
238 
239  static_assert(Nx > 0, "State space dimension must be static");
240  static_assert(Nu > 0, "Input space dimension must be static");
241 
242  const bool lin_con = lin.g_domain.minCoeff() < std::numeric_limits<double>::infinity();
243 
244  const uint32_t Nu_iq = pbm.ulim.A.rows();
245  const uint32_t Nx_iq = pbm.glim.A.rows();
246  const uint32_t Nxlin_iq = lin_con ? Nx : 0;
247 
248  const double dt = pbm.T / static_cast<double>(K);
249 
253 
254  assert(static_cast<uint32_t>(qp.A.rows()) == K * (Nx + Nu_iq + Nx_iq + Nxlin_iq));
255  assert(static_cast<uint32_t>(qp.A.cols()) == K * (Nx + Nu));
256  assert(static_cast<uint32_t>(qp.l.rows()) == K * (Nx + Nu_iq + Nx_iq + Nxlin_iq));
257  assert(static_cast<uint32_t>(qp.u.rows()) == K * (Nx + Nu_iq + Nx_iq + Nxlin_iq));
258 
259  int Arow = 0;
260 
261  for (auto k = 0u; k != K; ++k) {
262  using AT = Eigen::Matrix<double, Nx, Nx>;
263  using BT = Eigen::Matrix<double, Nx, Nu>;
264  using ET = Eigen::Matrix<double, Nx, 1>;
265 
266  const double t = k * dt;
267 
268  // LINEARIZATION
269 
270  const auto [xl, dxl] = lin.g(t);
271  const auto ul = lin.u(t);
272 
273  // clang-format off
274  const auto [flin, df_xu] = diff::dr<DT>(
275  [&f, &t]<typename T>(const CastT<T, G> & vx, const CastT<T, U> & vu) -> Tangent<CastT<T, G>> {
276  return f(t, vx, vu);
277  },
278  wrt(xl, ul)
279  );
280  // clang-format on
281 
282  // cltv system \dot x = At x(t) + Bt u(t) + Et
283  const AT At = -0.5 * ad<G>(flin) - 0.5 * ad<G>(dxl) + df_xu.template leftCols<Nx>();
284  const BT Bt = df_xu.template rightCols<Nu>();
285  const ET Et = flin - dxl;
286 
287  // TIME DISCRETIZATION
288 
289  const AT At2 = At * At;
290  const double dt2 = dt * dt;
291  const double dt3 = dt2 * dt;
292 
293  // dltv system x^+ = Ak x + Bk u + Ek by truncated taylor expansion of the matrix exponential
294  const AT Ak = AT::Identity() + At * dt + At2 * dt2 / 2. + At2 * At * dt3 / 6.;
295  const BT Bk = Bt * dt + At * Bt * dt2 / 2. + At2 * Bt * dt3 / 6.;
296  const ET Ek = Et * dt + At * Et * dt2 / 2. + At2 * Et * dt3 / 6.;
297 
298  // DYNAMICS CONSTRANTS
299 
300  if (k == 0) {
301  // x(1) - B u(0) = A x0 + E
302 
303  // identity matrix on x(1)
304  for (auto i = 0u; i != Nx; ++i) {
305  const Eigen::Index Ari = Nx * k + i;
306  assert(qp.A.outerIndexPtr()[Ari + 1] - qp.A.outerIndexPtr()[Ari] == Nu + 1);
307 
308  // B matrix on u0
309  for (auto j = 0u; j != Nu; ++j) { qp.A.coeffRef(Ari, Nu * k + j) = -Bk(i, j); }
310 
311  // Identity matrix on x1
312  qp.A.coeffRef(Ari, NU + Nx * k + i) = 1;
313  }
314 
315  // B matrix on u(0)
316  for (auto i = 0u; i != Nx; ++i) {}
317 
318  qp.u.template segment<Nx>(Nx * k) = Ak * rminus(pbm.x0, xl) + Ek;
319  qp.l.template segment<Nx>(Nx * k) = qp.u.template segment<Nx>(Nx * k);
320  } else {
321  // x(k+1) - A x(k) - B u(k) = E
322  for (auto i = 0u; i != Nx; ++i) {
323  const Eigen::Index Ari = Nx * k + i;
324  assert(qp.A.outerIndexPtr()[Ari + 1] - qp.A.outerIndexPtr()[Ari] == Nu + Nx + 1);
325 
326  // B matrix on u(k)
327  for (auto j = 0u; j != Nu; ++j) { qp.A.coeffRef(Ari, Nu * k + j) = -Bk(i, j); }
328 
329  // A matrix on x(k)
330  for (auto j = 0u; j != Nx; ++j) { qp.A.coeffRef(Ari, NU + Nx * (k - 1) + j) = -Ak(i, j); }
331 
332  // identity matrix on x(k+1)
333  qp.A.coeffRef(Ari, NU + Nx * k + i) = 1;
334  }
335 
336  qp.u.template segment<Nx>(Nx * k) = Ek;
337  qp.l.template segment<Nx>(Nx * k) = Ek;
338  }
339  }
340  Arow += K * Nx;
341 
342  // INPUT CONSTRAINTS
343 
344  if (Nu_iq > 0) {
345  for (auto k = 0u; k < K; ++k) {
346  for (auto i = 0u; i != Nu_iq; ++i) {
347  const Eigen::Index Ari = Arow + k * Nu_iq + i;
348  assert(qp.A.outerIndexPtr()[Ari + 1] - qp.A.outerIndexPtr()[Ari] == Nu);
349 
350  for (auto j = 0u; j != Nu; ++j) { qp.A.coeffRef(Ari, k * Nu + j) = pbm.ulim.A(i, j); }
351  }
352  // clang-format off
353  qp.l.segment(Arow + k * Nu_iq, Nu_iq) = pbm.ulim.l - pbm.ulim.A * rminus(lin.u(k * dt), pbm.ulim.c);
354  qp.u.segment(Arow + k * Nu_iq, Nu_iq) = pbm.ulim.u - pbm.ulim.A * rminus(lin.u(k * dt), pbm.ulim.c);
355  // clang-format on
356  }
357  }
358  Arow += K * Nu_iq;
359 
360  // STATE CONSTRAINTS
361 
362  if (Nx_iq > 0) {
363  for (auto k = 1u; k != K + 1; ++k) {
364  for (auto i = 0u; i != Nx_iq; ++i) {
365  const Eigen::Index Ari = Arow + (k - 1) * Nx_iq + i;
366  assert(qp.A.outerIndexPtr()[Ari + 1] - qp.A.outerIndexPtr()[Ari] == Nx);
367 
368  for (auto j = 0u; j != Nx; ++j) {
369  qp.A.coeffRef(Ari, NU + (k - 1) * Nx + j) = pbm.glim.A(i, j);
370  }
371  }
372  // clang-format off
373  qp.l.segment(Arow + (k - 1) * Nx_iq, Nx_iq) = pbm.glim.l - pbm.glim.A * rminus(lin.g(k * dt).first, pbm.glim.c);
374  qp.u.segment(Arow + (k - 1) * Nx_iq, Nx_iq) = pbm.glim.u - pbm.glim.A * rminus(lin.g(k * dt).first, pbm.glim.c);
375  // clang-format on
376  }
377  }
378  Arow += K * Nx_iq;
379 
380  // STATE LINEARIZATION BOUNDS
381 
382  if (Nxlin_iq > 0) {
383  for (auto k = 1u; k < K + 1; ++k) {
384  for (auto i = 0u; i != Nxlin_iq; ++i) {
385  const Eigen::Index Ari = Arow + (k - 1) * Nxlin_iq + i;
386  assert(qp.A.outerIndexPtr()[Ari + 1] - qp.A.outerIndexPtr()[Ari] == 1);
387 
388  qp.A.coeffRef(Ari, NU + (k - 1) * Nx + i) = 1.;
389  }
390  qp.l.segment(Arow + (k - 1) * Nxlin_iq, Nxlin_iq) = -lin.g_domain;
391  qp.u.segment(Arow + (k - 1) * Nxlin_iq, Nxlin_iq) = lin.g_domain;
392  }
393  }
394  Arow += K * Nxlin_iq;
395 
399 
400  assert(static_cast<uint32_t>(qp.P.cols()) == K * (Nx + Nu));
401  assert(static_cast<uint32_t>(qp.P.cols()) == K * (Nx + Nu));
402  assert(static_cast<uint32_t>(qp.q.rows()) == K * (Nx + Nu));
403 
404  // INPUT COSTS
405 
406  for (auto k = 0u; k < K; ++k) {
407  for (auto i = 0u; i != Nu; ++i) {
408  const Eigen::Index Pci = k * Nu + i;
409  assert(
410  qp.P.outerIndexPtr()[Pci + 1] - qp.P.outerIndexPtr()[Pci]
411  == static_cast<decltype(qp.P)::StorageIndex>(i) + 1);
412 
413  for (auto j = 0u; j != i + 1; ++j) {
414  qp.P.coeffRef(k * Nu + j, Pci) = pbm.weights.R(j, i) * dt;
415  }
416  }
417  qp.q.segment(k * Nu, Nu) = pbm.weights.R * rminus(lin.u(k * dt), pbm.udes(k * dt)) * dt;
418  }
419 
420  // STATE COSTS
421 
422  // intermediate states x(1) ... x(K-1)
423  for (auto k = 1u; k < K; ++k) {
424  for (auto i = 0u; i != Nx; ++i) {
425  const Eigen::Index Pci = NU + (k - 1) * Nx + i;
426  assert(
427  qp.P.outerIndexPtr()[Pci + 1] - qp.P.outerIndexPtr()[Pci]
428  == static_cast<decltype(qp.P)::StorageIndex>(i) + 1);
429 
430  for (auto j = 0u; j != i + 1; ++j) {
431  qp.P.coeffRef(NU + (k - 1) * Nx + j, Pci) = pbm.weights.Q(j, i) * dt;
432  }
433  }
434  qp.q.segment(NU + (k - 1) * Nx, Nx) =
435  pbm.weights.Q * rminus(lin.g(k * dt).first, pbm.gdes(k * dt)) * dt;
436  }
437 
438  // last state x(K) ~ x(T)
439  for (auto i = 0u; i != Nx; ++i) {
440  const Eigen::Index Pci = NU + (K - 1) * Nx + i;
441  assert(
442  qp.P.outerIndexPtr()[Pci + 1] - qp.P.outerIndexPtr()[Pci]
443  == static_cast<decltype(qp.P)::StorageIndex>(i) + 1);
444 
445  for (auto j = 0u; j != i + 1; ++j) {
446  qp.P.coeffRef(NU + (K - 1) * Nx + j, Pci) = pbm.weights.QT(j, i);
447  }
448  }
449  qp.q.segment(NU + (K - 1) * Nx, Nx) =
450  pbm.weights.QT * rminus(lin.g(pbm.T).first, pbm.gdes(pbm.T));
451 
452  qp.A.makeCompressed();
453  qp.P.makeCompressed();
454 }
