123 inline bool check_ocp(const OCPType auto & ocp)
124 {
125  using X = typename std::decay_t<decltype(ocp)>::X;
126  using U = typename std::decay_t<decltype(ocp)>::U;
127 
128  const double t = 0;
129  const X x = Default<X>(ocp.nx);
130  const U u = Default<U>(ocp.nu);
131 
132  if (!(static_cast<std::size_t>(dof(x)) == ocp.nx)) { return false; }
133  if (!(static_cast<std::size_t>(dof(u)) == ocp.nu)) { return false; }
134 
135  const auto dx = ocp.f.template operator()<double>(t, x, u);
136  if (!(static_cast<std::size_t>(dx.size()) == ocp.nx)) { return false; }
137 
138  const auto g = ocp.g.template operator()<double>(t, x, u);
139  if (!(static_cast<std::size_t>(g.size()) == ocp.nq)) { return false; }
140 
141  [[maybe_unused]] const double obj = ocp.theta.template operator()<double>(t, x, x, g);
142 
143  const auto cr = ocp.cr.template operator()<double>(t, x, u);
144  if (!(static_cast<std::size_t>(cr.size()) == ocp.ncr)) { return false; }
145  if (!(static_cast<std::size_t>(ocp.crl.size()) == ocp.ncr)) { return false; }
146  if (!(static_cast<std::size_t>(ocp.cru.size()) == ocp.ncr)) { return false; }
147 
148  const auto ce = ocp.ce.template operator()<double>(t, x, x, g);
149  if (!(static_cast<std::size_t>(ce.size()) == ocp.nce)) { return false; }
150  if (!(static_cast<std::size_t>(ocp.cel.size()) == ocp.nce)) { return false; }
151  if (!(static_cast<std::size_t>(ocp.ceu.size()) == ocp.nce)) { return false; }
152 
153  return true;
154 }

165 inline auto flatten_ocp(const OCPType auto & ocp, auto && xl_fun, auto && ul_fun)
166 {
167  using Eigen::VectorX;
168  using X = typename std::decay_t<decltype(ocp)>::X;
169  using U = typename std::decay_t<decltype(ocp)>::U;
170 
171  assert(Dof<X> == -1 || ocp.nx == Dof<X>);
172  assert(Dof<U> == -1 || ocp.nu == Dof<U>);
173 
174  auto f_new = [f = ocp.f, xl_fun = xl_fun, ul_fun = ul_fun]<typename T>(
175  const T & t, const VectorX<T> & xe, const VectorX<T> & ue) -> VectorX<T> {
176  // can not double-differentiate, so we neglect derivative of linearization w.r.t. t
177  const double tdbl = static_cast<double>(t);
178  const auto [xl, dxl] = diff::dr(xl_fun, wrt(tdbl));
179  const auto ul = ul_fun(tdbl);
180 
181  const X x = rplus(xl, xe);
182  const U u = rplus(ul, ue);
183 
184  return dr_expinv<X>(xe) * f.template operator()<T>(t, x, u) - dl_expinv<X>(xe) * dxl;
185  };
186 
187  auto g_new = [g = ocp.g, xl_fun = xl_fun, ul_fun = ul_fun]<typename T>(
188  const T & t, const VectorX<T> & xe, const VectorX<T> & ue) -> VectorX<T> {
189  return g.template operator()<T>(t, rplus(xl_fun(t), xe), rplus(ul_fun(t), ue));
190  };
191 
192  auto cr_new = [cr = ocp.cr, xl_fun = xl_fun, ul_fun = ul_fun]<typename T>(
193  const T & t, const VectorX<T> & xe, const VectorX<T> & ue) -> VectorX<T> {
194  return cr.template operator()<T>(t, rplus(xl_fun(t), xe), rplus(ul_fun(t), ue));
195  };
196 
197  auto theta_new =
198  [theta = ocp.theta, xl_fun = xl_fun]<typename T>(
199  const T & tf, const VectorX<T> & xe0, const VectorX<T> & xef, const VectorX<T> & q) -> T {
200  return theta.template operator()<T>(tf, rplus(xl_fun(0.), xe0), rplus(xl_fun(tf), xef), q);
201  };
202 
203  auto ce_new = [ce = ocp.ce, xl_fun = xl_fun]<typename T>(
204  const T & tf,
205  const VectorX<T> & xe0,
206  const VectorX<T> & xef,
207  const VectorX<T> & q) -> VectorX<T> {
208  return ce.template operator()<T>(tf, rplus(xl_fun(0.), xe0), rplus(xl_fun(tf), xef), q);
209  };
210 
211  return FlatOCP<
212  decltype(theta_new),
213  decltype(f_new),
214  decltype(g_new),
215  decltype(cr_new),
216  decltype(ce_new)>{
217  .nx = ocp.nx,
218  .nu = ocp.nu,
219  .nq = ocp.nq,
220  .ncr = ocp.ncr,
221  .nce = ocp.nce,
222  .theta = std::move(theta_new),
223  .f = std::move(f_new),
224  .g = std::move(g_new),
225  .cr = std::move(cr_new),
226  .crl = ocp.crl,
227  .cru = ocp.cru,
228  .ce = std::move(ce_new),
229  .cel = ocp.cel,
230  .ceu = ocp.ceu,
231  };
232 }

298 auto ocp_nlp_structure(const FlatOCPType auto & ocp, const MeshType auto & mesh)
299 {
300  std::size_t N = mesh.N_colloc();
301 
302  // variable layout
303  std::array<std::size_t, 4> var_len{
304  1, // tf
305  ocp.nq, // integrals
306  ocp.nx * (N + 1), // states
307  ocp.nu * N, // inputs
308  };
309 
310  // constraint layout
311  std::array<std::size_t, 4> con_len{
312  ocp.nx * N, // derivatives
313  ocp.nq, // other integrals
314  ocp.ncr * N, // running constraints
315  ocp.nce, // end constraints
316  };
317 
318  std::array<std::size_t, 5> var_beg{0};
319  std::partial_sum(var_len.begin(), var_len.end(), var_beg.begin() + 1);
320 
321  std::array<std::size_t, 5> con_beg{0};
322  std::partial_sum(con_len.begin(), con_len.end(), con_beg.begin() + 1);
323 
324  return std::make_tuple(var_beg, var_len, con_beg, con_len);
325 }

338 NLP ocp_to_nlp(const FlatOCPType auto & ocp, const MeshType auto & mesh)
339 {
340  const auto [var_beg, var_len, con_beg, con_len] = detail::ocp_nlp_structure(ocp, mesh);
341 
342  // OBJECTIVE FUNCTION
343 
344  auto f = [var_beg = var_beg, var_len = var_len, ocp = ocp](const Eigen::VectorXd & x) -> double {
345  const auto [tfvar_B, qvar_B, xvar_B, uvar_B, n] = var_beg;
346  const auto [tfvar_L, qvar_L, xvar_L, uvar_L] = var_len;
347 
348  assert(std::size_t(x.size()) == n);
349 
350  const double t0 = 0;
351  const double tf = x(tfvar_B);
352  const Eigen::VectorXd Q = x.segment(qvar_B, qvar_L);
353  const Eigen::MatrixXd X = x.segment(xvar_B, xvar_L).reshaped(ocp.nx, xvar_L / ocp.nx);
354 
355  return colloc_eval_endpt<false>(1, ocp.nx, ocp.theta, t0, tf, X, Q);
356  };
357 
358  // OBJECTIVE JACOBIAN
359 
360  auto df_dx = [var_beg = var_beg, var_len = var_len, ocp = ocp](
361  const Eigen::VectorXd & x) -> Eigen::SparseMatrix<double> {
362  const auto [tfvar_B, qvar_B, xvar_B, uvar_B, n] = var_beg;
363  const auto [tfvar_L, qvar_L, xvar_L, uvar_L] = var_len;
364 
365  const double t0 = 0;
366  const double tf = x(tfvar_B);
367  const Eigen::VectorXd Q = x.segment(qvar_B, qvar_L);
368  const Eigen::MatrixXd X = x.segment(xvar_B, xvar_L).reshaped(ocp.nx, xvar_L / ocp.nx);
369 
370  const auto [fval, df_dt0, df_dtf, df_dvecX, df_dQ] =
371  colloc_eval_endpt<true>(1, ocp.nx, ocp.theta, t0, tf, X, Q);
372 
373  return sparse_block_matrix({
374  {df_dtf, df_dQ, df_dvecX, Eigen::SparseMatrix<double>(1, uvar_L)},
375  });
376  };
377 
378  // CONSTRAINT FUNCTION
379 
380  auto g = [var_beg = var_beg,
381  var_len = var_len,
382  con_beg = con_beg,
383  con_len = con_len,
384  mesh = mesh,
385  ocp = ocp](const Eigen::VectorXd & x) -> Eigen::VectorXd {
386  const auto [tfvar_B, qvar_B, xvar_B, uvar_B, n] = var_beg;
387  const auto [tfvar_L, qvar_L, xvar_L, uvar_L] = var_len;
388 
389  const auto [dcon_B, qcon_B, crcon_B, cecon_B, m] = con_beg;
390  const auto [dcon_L, qcon_L, crcon_L, cecon_L] = con_len;
391 
392  assert(std::size_t(x.size()) == n);
393 
394  const double t0 = 0;
395  const double tf = x(tfvar_B);
396  const Eigen::VectorXd Q = x.segment(qvar_B, qvar_L);
397  const Eigen::MatrixXd X = x.segment(xvar_B, xvar_L).reshaped(ocp.nx, xvar_L / ocp.nx);
398  const Eigen::MatrixXd U = x.segment(uvar_B, uvar_L).reshaped(ocp.nu, uvar_L / ocp.nu);
399 
400  Eigen::VectorXd ret(m);
401  // clang-format off
402  ret.segment(dcon_B, dcon_L) = colloc_dyn<false>(ocp.nx, ocp.f, mesh, t0, tf, X, U);
403  ret.segment(qcon_B, qcon_L) = colloc_int<false>(ocp.nq, ocp.g, mesh, t0, tf, Q, X, U).reshaped();
404  ret.segment(crcon_B, crcon_L) = colloc_eval<false>(ocp.ncr, ocp.cr, mesh, t0, tf, X, U).reshaped();
405  ret.segment(cecon_B, cecon_L) = colloc_eval_endpt<false>(ocp.nce, ocp.nx, ocp.ce, t0, tf, X, Q).reshaped();
406  // clang-format on
407  return ret;
408  };
409 
410  // CONSTRAINT JACOBIAN
411  auto dg_dx =
412  [var_beg = var_beg, var_len = var_len, mesh = mesh, ocp = ocp](const Eigen::VectorXd & x) {
413  const auto [tfvar_B, qvar_B, xvar_B, uvar_B, n] = var_beg;
414  const auto [tfvar_L, qvar_L, xvar_L, uvar_L] = var_len;
415 
416  assert(std::size_t(x.size()) == n);
417 
418  const double t0 = 0;
419  const double tf = x(tfvar_B);
420  const Eigen::VectorXd Qm = x.segment(qvar_B, qvar_L);
421  const Eigen::MatrixXd Xm = x.segment(xvar_B, xvar_L).reshaped(ocp.nx, xvar_L / ocp.nx);
422  const Eigen::MatrixXd Um = x.segment(uvar_B, uvar_L).reshaped(ocp.nu, uvar_L / ocp.nu);
423 
424  // clang-format off
425  const auto [Fval, dF_dt0, dF_dtf, dF_dX, dF_dU] = colloc_dyn<true>(ocp.nx, ocp.f, mesh, t0, tf, Xm, Um);
426  const auto [Gval, dG_dt0, dG_dtf, dG_dQ, dG_dX, dG_dU] = colloc_int<true>(ocp.nq, ocp.g, mesh, t0, tf, Qm, Xm, Um);
427  const auto [CRval, dCR_dt0, dCR_dtf, dCR_dX, dCR_dU] = colloc_eval<true>(ocp.ncr, ocp.cr, mesh, t0, tf, Xm, Um);
428  const auto [CEval, dCE_dt0, dCE_dtf, dCE_dX, dCE_dQ] = colloc_eval_endpt<true>(ocp.nce, ocp.nx, ocp.ce, t0, tf, Xm, Qm);
429  // clang-format on
430 
431  return sparse_block_matrix({
432  // clang-format off
433  { dF_dtf, {}, dF_dX, dF_dU},
434  { dG_dtf, dG_dQ, dG_dX, dG_dU},
435  {dCR_dtf, {}, dCR_dX, dCR_dU},
436  {dCE_dtf, dCE_dQ, dCE_dX, {}},
437  // clang-format on
438  });
439  };
440 
441  const auto [tfvar_B, qvar_B, xvar_B, uvar_B, n] = var_beg;
442  const auto [tfvar_L, qvar_L, xvar_L, uvar_L] = var_len;
443 
444  const auto [dcon_B, qcon_B, crcon_B, cecon_B, m] = con_beg;
445  const auto [dcon_L, qcon_L, crcon_L, cecon_L] = con_len;
446 
447  // VARIABLE BOUNDS
448 
449  Eigen::VectorXd xl = Eigen::VectorXd::Constant(n, -std::numeric_limits<double>::infinity());
450  Eigen::VectorXd xu = Eigen::VectorXd::Constant(n, std::numeric_limits<double>::infinity());
451 
452  xl.segment(tfvar_B, tfvar_L).setZero(); // tf lower bounded by zero
453 
454  // CONSTRAINT BOUNDS
455 
456  Eigen::VectorXd gl(m);
457  Eigen::VectorXd gu(m);
458 
459  // derivative constraints are equalities
460  gl.segment(dcon_B, dcon_L).setZero();
461  gu.segment(dcon_B, dcon_L).setZero();
462 
463  // integral constraints are equalities
464  gl.segment(qcon_B, qcon_L).setZero();
465  gu.segment(qcon_B, qcon_L).setZero();
466 
467  // running constraints
468  gl.segment(crcon_B, crcon_L) = ocp.crl.replicate(mesh.N_colloc(), 1);
469  gu.segment(crcon_B, crcon_L) = ocp.cru.replicate(mesh.N_colloc(), 1);
470 
471  // end constraints
472  gl.segment(cecon_B, cecon_L) = ocp.cel;
473  gu.segment(cecon_B, cecon_L) = ocp.ceu;
474 
475  return {
476  .n = n,
477  .m = m,
478  .f = std::move(f),
479  .xl = std::move(xl),
480  .xu = std::move(xu),
481  .g = std::move(g),
482  .gl = std::move(gl),
483  .gu = std::move(gu),
484  .df_dx = std::move(df_dx),
485  .dg_dx = std::move(dg_dx),
486  .d2f_dx2 = {},
487  .d2g_dx2 = {},
488  };
489 }

494 FlatOCPSolution nlpsol_to_ocpsol(
495  const FlatOCPType auto & ocp, const MeshType auto & mesh, const NLPSolution & nlp_sol)
496 {
497  const std::size_t N = mesh.N_colloc();
498  const auto [var_beg, var_len, con_beg, con_len] = detail::ocp_nlp_structure(ocp, mesh);
499 
500  const auto [tfvar_B, qvar_B, xvar_B, uvar_B, n] = var_beg;
501  const auto [tfvar_L, qvar_L, xvar_L, uvar_L] = var_len;
502 
503  const auto [dcon_B, qcon_B, crcon_B, cecon_B, m] = con_beg;
504  const auto [dcon_L, qcon_L, crcon_L, cecon_L] = con_len;
505 
506  const double t0 = 0;
507  const double tf = nlp_sol.x(tfvar_B);
508 
509  const Eigen::VectorXd Q = nlp_sol.x.segment(qvar_B, qvar_L);
510 
511  // state vector has a value at the endpoint
512 
513  Eigen::MatrixXd X(ocp.nx, N + 1);
514  X = nlp_sol.x.segment(xvar_B, xvar_L).reshaped(ocp.nx, xvar_L / ocp.nx);
515 
516  auto xfun = [t0 = t0, tf = tf, mesh = mesh, X = std::move(X)](double t) -> Eigen::VectorXd {
517  return mesh.template eval<Eigen::VectorXd>((t - t0) / (tf - t0), X.colwise(), 0, true);
518  };
519 
520  // for these we repeat last point since there are no values for endpoint
521 
522  Eigen::MatrixXd U(ocp.nu, N);
523  U = nlp_sol.x.segment(uvar_B, uvar_L).reshaped(ocp.nu, uvar_L / ocp.nu);
524 
525  auto ufun = [t0 = t0, tf = tf, mesh = mesh, U = std::move(U)](double t) -> Eigen::VectorXd {
526  return mesh.template eval<Eigen::VectorXd>((t - t0) / (tf - t0), U.colwise(), 0, false);
527  };
528 
529  Eigen::MatrixXd Ldyn(ocp.nx, N);
530  Ldyn = nlp_sol.lambda.segment(dcon_B, dcon_L).reshaped(ocp.nx, dcon_L / ocp.nx);
531 
532  auto ldfun =
533  [t0 = t0, tf = tf, mesh = mesh, Ldyn = std::move(Ldyn)](double t) -> Eigen::VectorXd {
534  return mesh.template eval<Eigen::VectorXd>((t - t0) / (tf - t0), Ldyn.colwise(), 0, false);
535  };
536 
537  Eigen::MatrixXd Lcr(ocp.ncr, N);
538  Lcr = nlp_sol.lambda.segment(crcon_B, crcon_L).reshaped(ocp.ncr, crcon_L / ocp.ncr);
539 
540  auto lcrfun = [t0 = t0, tf = tf, mesh = mesh, Lcr = std::move(Lcr)](double t) -> Eigen::VectorXd {
541  return mesh.template eval<Eigen::VectorXd>((t - t0) / (tf - t0), Lcr.colwise(), 0, false);
542  };
543 
544  return {
545  .t0 = t0,
546  .tf = tf,
547  .Q = std::move(Q),
548  .u = std::move(ufun),
549  .x = std::move(xfun),
550  .lambda_q = nlp_sol.lambda.segment(qcon_B, qcon_L),
551  .lambda_ce = nlp_sol.lambda.segment(cecon_B, cecon_L),
552  .lambda_dyn = std::move(ldfun),
553  .lambda_cr = std::move(lcrfun),
554  };
555 }

561  const FlatOCPType auto & ocp, const MeshType auto & mesh, const FlatOCPSolution & ocpsol)
562 {

564  const auto [var_beg, var_len, con_beg, con_len] = detail::ocp_nlp_structure(ocp, mesh);

566  const auto [tfvar_B, qvar_B, xvar_B, uvar_B, n] = var_beg;
