74template<LieGroup _X, Manifold _U, typename Theta, typename F, typename G, typename CR, typename CE>
75struct OCP
76{
78 using X = _X;
80 using U = _U;
81
83 static constexpr int Nx = Dof<X>;
85 static constexpr int Nu = Dof<U>;
87 static constexpr int Nq = std::invoke_result_t<G, double, X, U>::SizeAtCompileTime;
89 static constexpr int Ncr = std::invoke_result_t<CR, double, X, U>::SizeAtCompileTime;
91 static constexpr int Nce =
92 std::invoke_result_t<CE, double, X, X, Eigen::Matrix<double, Nq, 1>>::SizeAtCompileTime;
93
95 using Solution = OCPSolution<X, U, Nq, Ncr, Nce>;
96
97 static_assert(Nx > 0, "Static size required");
98 static_assert(Nu > 0, "Static size required");
99 static_assert(Nq > 0, "Static size required");
100 static_assert(Ncr > 0, "Static size required");
101 static_assert(Nce > 0, "Static size required");
102
104 Theta theta;
105
107 F f;
109 G g;
110
112 CR cr;
114 Eigen::Vector<double, Ncr> crl = Eigen::Vector<double, Ncr>::Zero();
116 Eigen::Vector<double, Ncr> cru = Eigen::Vector<double, Ncr>::Zero();
117
119 CE ce;
121 Eigen::Vector<double, Nce> cel = Eigen::Vector<double, Nce>::Zero();
123 Eigen::Vector<double, Nce> ceu = Eigen::Vector<double, Nce>::Zero();
124};

190template<diff::Type DT = diff::Type::Numerical>
191bool test_ocp_derivatives(OCPType auto & ocp, uint32_t num_trials = 1, double eps = 1e-4)
192{
193 using OCP = std::decay_t<decltype(ocp)>;
194
195 using X = typename OCP::X;
196 using U = typename OCP::U;
197 using Q = Eigen::Vector<double, OCP::Nq>;
198
199 if (!diff::detail::diffable_order1<decltype(ocp.theta), std::tuple<double, X, X, Q>>) {
200 std::cout << "no jacobian for theta\n";
201 }

203 std::cout << "no hessian for theta\n";
204 }

206 std::cout << "no jacobian for f\n";
207 }

209 std::cout << "no hessian for f\n";
210 }
211 if (!diff::detail::diffable_order1<decltype(ocp.g), std::tuple<double, X, U>>) {
212 std::cout << "no jacobian for g\n";
213 }
214 if (!diff::detail::diffable_order2<decltype(ocp.g), std::tuple<double, X, U>>) {
215 std::cout << "no hessian for g\n";
216 }

230 const auto cmp = [&eps](const auto & m1, const auto & m2) {
231 return (
232 // clang-format off
233 (m1.cols() == m2.cols())
234 && (m1.rows() == m2.rows())
235 && (
236 m1.isApprox(m2, 1e-4) ||
237 Eigen::MatrixXd(m1 - m2).cwiseAbs().maxCoeff() < eps
238 )
239 // clang-format on
240 );
241 };
242
243 bool success = true;
244
245 for (auto trial = 0u; trial < num_trials; ++trial) {
246 // endpt parameters
247 const double tf = 1 + static_cast<double>(std::rand()) / RAND_MAX;
248 const X x0 = Random<X>();
249 const X xf = Random<X>();
250 const Eigen::Vector<double, OCP::Nq> q = Eigen::Vector<double, OCP::Nq>::Random();
251 const double t = 1 + static_cast<double>(std::rand()) / RAND_MAX;
252 const X x = Random<X>();
253 const U u = Random<U>();
254
255 // theta
256 if constexpr (diff::detail::diffable_order1<decltype(ocp.theta), std::tuple<double, X, X, Q>>) {
257 const auto [f_def, df_def] = diff::dr<1, diff::Type::Analytic>(ocp.theta, wrt(tf, x0, xf, q));
258 const auto [f_num, df_num] = diff::dr<1, DT>(ocp.theta, wrt(tf, x0, xf, q));
259
260 if (!cmp(df_def, df_num)) {
261 std::cout << "Error in 1st derivative of theta: got\n"
262 << Eigen::MatrixXd(df_def) << "\nbut expected\n"
263 << Eigen::MatrixXd(df_num) << '\n';
264 success = false;
265 };
266 }
267 if constexpr (diff::detail::diffable_order2<decltype(ocp.theta), std::tuple<double, X, X, Q>>) {
268 const auto [f_def, df_def, d2f_def] =
269 diff::dr<2, diff::Type::Analytic>(ocp.theta, wrt(tf, x0, xf, q));
270 const auto [f_num, df_num, d2f_num] = diff::dr<2, DT>(ocp.theta, wrt(tf, x0, xf, q));
271
272 if (!cmp(d2f_def, d2f_num)) {
273 std::cout << "Error in 2nd derivative of theta: got\n"
274 << Eigen::MatrixXd(d2f_def) << "\nbut expected\n"
275 << Eigen::MatrixXd(d2f_num) << '\n';
276 success = false;
277 };
278 }
