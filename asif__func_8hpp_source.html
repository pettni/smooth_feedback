98template<LieGroup G, Manifold U>
99 requires(Dof<G> > 0 && Dof<U> > 0)
100void asif_to_qp_allocate(
101 std::size_t K, std::size_t nu_ineq, std::size_t nh, QuadraticProgram<-1, -1, double> & qp)
102{
103 static constexpr int nx = Dof<G>;
104 static constexpr int nu = Dof<U>;
105
106 static_assert(nx > 0, "State space dimension must be static");
107 static_assert(nu > 0, "Input space dimension must be static");
108
109 const int M = K * nh + nu_ineq + 1;
110 const int N = nu + 1;
111
112 qp.A.setZero(M, N);
113 qp.l.setZero(M);
114 qp.u.setZero(M);
115
116 qp.P.setZero(N, N);
117 qp.q.setZero(N);
118}

125template<
126 LieGroup G,
127 Manifold U,
128 typename Dyn,
129 typename SafeSet,
130 typename BackupU,
131 diff::Type DT = diff::Type::Default>
132 requires(Dof<G> > 0 && Dof<U> > 0)
133void asif_to_qp_fill(
134 const ASIFProblem<G, U> & pbm,
135 const ASIFtoQPParams & prm,
136 Dyn && f,
137 SafeSet && h,
138 BackupU && bu,
139 QuadraticProgram<-1, -1, double> & qp)
140{
141 using boost::numeric::odeint::euler, boost::numeric::odeint::vector_space_algebra;
142 using std::placeholders::_1;
143
144 static constexpr int nx = Dof<G>;
145 static constexpr int nu = Dof<U>;
146 static constexpr int nh = std::invoke_result_t<SafeSet, double, G>::SizeAtCompileTime;
147
148 euler<G, double, Tangent<G>, double, vector_space_algebra> state_stepper{};
149 euler<TangentMap<G>, double, TangentMap<G>, double, vector_space_algebra> sensi_stepper{};
150
151 const int nu_ineq = pbm.ulim.A.rows();
152
153 [[maybe_unused]] const int M = prm.K * nh + nu_ineq + 1;
154 [[maybe_unused]] const int N = nu + 1;
155
156 assert(qp.A.rows() == M);
157 assert(qp.A.cols() == N);
158 assert(qp.l.rows() == M);
159 assert(qp.u.rows() == M);
160
161 assert(qp.P.rows() == N);
162 assert(qp.P.cols() == N);
163 assert(qp.q.rows() == N);
164
165 // iteration variables
166 const double tau = pbm.T / static_cast<double>(prm.K);
167 const double dt = std::min<double>(prm.dt, tau);
168 double t = 0;
169 G x = pbm.x0;
170 TangentMap<G> dx_dx0 = TangentMap<G>::Identity();
171
172 // define ODEs for closed-loop dynamics and its sensitivity
173 const auto x_ode = [&f, &bu](const G & xx, Tangent<G> & dd, double tt) {
174 dd = f(tt, xx, bu(tt, xx));
175 };
176
177 const auto dx_dx0_ode = [&f, &bu, &x](const auto & S_v, auto & dS_dt_v, double tt) {
178 auto f_cl = [&]<typename T>(const CastT<T, G> & vx) { return f(T(tt), vx, bu(T(tt), vx)); };
179 const auto [fcl, dr_fcl_dx] = diff::dr<1, DT>(std::move(f_cl), wrt(x));
180 dS_dt_v = (-ad<G>(fcl) + dr_fcl_dx) * S_v;
181 };
182
183 // value of dynamics at call time
184 const auto [f0, d_f0_du] = diff::dr<1, DT>(
185 [&]<typename T>(const CastT<T, U> & vu) { return f(T(t), cast<T>(x), vu); }, wrt(pbm.u_des));
186
187 // loop over constraint number
188 for (auto k = 0u; k != prm.K; ++k) {
189 // differentiate barrier function w.r.t. x
190 const auto [hval, dh_dtx] = diff::dr<1, DT>(
191 [&h]<typename T>(const T & vt, const CastT<T, G> & vx) { return h(vt, vx); }, wrt(t, x));
192
193 const Eigen::Matrix<double, nh, 1> dh_dt = dh_dtx.template leftCols<1>();
194 const Eigen::Matrix<double, nh, nx> dh_dx = dh_dtx.template rightCols<nx>();
195
196 // insert barrier constraint
197 const Eigen::Matrix<double, nh, nx> dh_dx0 = dh_dx * dx_dx0;
198 qp.A.template block<nh, nu>(k * nh, 0) = dh_dx0 * d_f0_du;
199 qp.l.template segment<nh>(k * nh) = -dh_dt - prm.alpha * hval - dh_dx0 * f0;
200 qp.u.template segment<nh>(k * nh).setConstant(std::numeric_limits<double>::infinity());
201
202 // integrate system and sensitivity forward until next constraint
203 double dt_act = std::min(dt, tau * (k + 1) - t);
204 while (t < tau * (k + 1)) {
205 state_stepper.do_step(x_ode, x, t, dt_act);
206 sensi_stepper.do_step(dx_dx0_ode, dx_dx0, t, dt_act);
207 t += dt_act;
208 }
209 }
210
211 // relaxation of barrier constraints
212 qp.A.block(0, nu, prm.K * nh, 1).setConstant(1);
213
214 // input bounds
215 qp.A.block(prm.K * nh, 0, nu_ineq, nu) = pbm.ulim.A;
216 qp.l.segment(prm.K * nh, nu_ineq) = pbm.ulim.l - pbm.ulim.A * rminus(pbm.u_des, pbm.ulim.c);
217 qp.u.segment(prm.K * nh, nu_ineq) = pbm.ulim.u - pbm.ulim.A * rminus(pbm.u_des, pbm.ulim.c);
218
219 // upper and lower bounds on delta
220 qp.A(prm.K * nh + nu_ineq, nu) = 1;
221 qp.l(prm.K * nh + nu_ineq) = 0;
222 qp.u(prm.K * nh + nu_ineq) = std::numeric_limits<double>::infinity();
223
224 qp.P.template block<nu, nu>(0, 0) = pbm.W_u.asDiagonal();
225
226 qp.P(nu, nu) = prm.relax_cost;
227 qp.q(nu) = 0;
228}

273template<
274 LieGroup G,
275 Manifold U,
276 typename Dyn,
277 typename SS,
278 typename BackupU,
279 diff::Type DT = diff::Type::Default>
280QuadraticProgram<-1, -1, double> asif_to_qp(
281 const ASIFProblem<G, U> & pbm, const ASIFtoQPParams & prm, Dyn && f, SS && h, BackupU && bu)
282{
283 static constexpr int nh = std::invoke_result_t<SS, double, G>::SizeAtCompileTime;
284
285 static_assert(Dof<G> > 0, "State space dimension must be static");
286 static_assert(Dof<U> > 0, "Input space dimension must be static");
287 static_assert(nh > 0, "Safe set dimension must be static");
288
289 const int nu_ineq = pbm.ulim.A.rows();
290 QuadraticProgram<-1, -1, double> qp;
291 asif_to_qp_allocate<G, U>(prm.K, nu_ineq, nh, qp);
292 asif_to_qp_fill(pbm, prm, f, h, bu, qp);
293 return qp;
294}
