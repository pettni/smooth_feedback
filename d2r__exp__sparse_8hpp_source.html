53template<typename Scalar>
54void d2r_exp_sparse_se2(Eigen::SparseMatrix<Scalar> & out, const Eigen::Vector3<Scalar> & a)
55{
56 const auto & vx = a.x();
57 const auto & vy = a.y();
58 const auto & th = a.z();
59
60 const Scalar th2 = th * th;
61 const Scalar th3 = th2 * th;
62 const Scalar th4 = th2 * th2;
63
64 const auto [A, B, dA, dB] = [&]() -> std::array<Scalar, 4> {
65 if (th2 < Scalar(eps2)) {
66 return {
67 Scalar(1) / Scalar(2) - th2 / Scalar(24),
68 Scalar(1) / Scalar(6) - th2 / Scalar(120),
69 -th / Scalar(12),
70 -th / Scalar(60),
71 };
72 } else {
73 return {
74 (Scalar(1) - cos(th)) / th2,
75 (th - sin(th)) / (th2 * th),
76 sin(th) / th2 - Scalar(2) * (Scalar(1) - cos(th)) / th3,
77 (Scalar(1) - cos(th)) / th3 - Scalar(3) * (th - sin(th)) / th4,
78 };
79 }
80 }();
81
82 out.resize(3, 9);
83
84 if (out.isCompressed()) {
85 out.coeffs().setZero();
86 } else {
87 out.setZero();
88 out.reserve(Eigen::VectorXi{{1, 1, 3, 1, 1, 3, 0, 0, 0}});
89 }
90
91 // first row
92 out.coeffRef(2, 3 * 0 + 0) = B * th;
93 out.coeffRef(2, 3 * 0 + 1) = -A;
94 out.coeffRef(0, 3 * 0 + 2) = -Scalar(2) * B * th - dB * th2;
95 out.coeffRef(1, 3 * 0 + 2) = A + dA * th;
96 out.coeffRef(2, 3 * 0 + 2) = -dA * vy + B * vx + dB * th * vx;
97
98 // second row
99 out.coeffRef(2, 3 * 1 + 0) = A;
100 out.coeffRef(2, 3 * 1 + 1) = out.coeff(2, 0);
101 out.coeffRef(0, 3 * 1 + 2) = -out.coeff(1, 2);
102 out.coeffRef(1, 3 * 1 + 2) = out.coeff(0, 2);
103 out.coeffRef(2, 3 * 1 + 2) = dA * vx + B * vy + dB * th * vy;
104
105 out.makeCompressed();
106}

108template<typename Scalar>
109void d2r_expinv_sparse_se2(Eigen::SparseMatrix<Scalar> & out, const Eigen::Vector3<Scalar> & a)
110{
111 const auto & vx = a.x();
112 const auto & vy = a.y();
113 const auto & th = a.z();
114
115 const Scalar th2 = th * th;
116 const Scalar th3 = th2 * th;
117
118 const auto [A, dA] = [&]() -> std::array<Scalar, 2> {
119 if (th2 < Scalar(eps2)) {
120 return {
121 Scalar(1) / Scalar(12) + th2 / Scalar(720),
122 th / Scalar(360),
123 };
124 } else {
125 const Scalar q = Scalar(2) * th * sin(th);
126 const Scalar dq = Scalar(2) * (sin(th) + th * cos(th));
127 return {
128 (Scalar(1) / th2) - (Scalar(1) + cos(th)) / q,
129 (-Scalar(2) / th3) + sin(th) / q + (Scalar(1) + cos(th)) / (q * q) * dq,
130 };
131 }
132 }();
133
134 out.resize(3, 9);
135
136 if (out.isCompressed()) {
137 out.coeffs().setZero();
138 } else {
139 out.setZero();
140 out.reserve(Eigen::VectorXi{{1, 1, 3, 1, 1, 3, 0, 0, 0}});
141 }
142
143 // first row
144 out.coeffRef(2, 3 * 0 + 0) = A * th;
145 out.coeffRef(2, 3 * 0 + 1) = Scalar(0.5);
146 out.coeffRef(0, 3 * 0 + 2) = -dA * th2 - A * Scalar(2) * th;
147 out.coeffRef(1, 3 * 0 + 2) = -Scalar(0.5);
148 out.coeffRef(2, 3 * 0 + 2) = dA * th * vx + A * vx;
149
150 // second row
151 out.coeffRef(2, 3 * 1 + 0) = -Scalar(0.5);
152 out.coeffRef(2, 3 * 1 + 1) = out.coeff(2, 0);
153 out.coeffRef(0, 3 * 1 + 2) = Scalar(0.5);
154 out.coeffRef(1, 3 * 1 + 2) = out.coeff(0, 2);
155 out.coeffRef(2, 3 * 1 + 2) = dA * th * vy + A * vy;
156
157 out.makeCompressed();
158}

160template<typename G, bool Inv>
161 requires(std::is_base_of_v<BundleBase<G>, G>)
162void d2r_exp_sparse_bundle(Eigen::SparseMatrix<Scalar<G>> & out, const Tangent<G> & a)
163{
164 static constexpr auto BundleSize = G::BundleSize;
165 std::size_t cntr = 0;
166 utils::static_for<BundleSize>([&](auto I) {
167 using GI = typename G::template PartType<I>;
168 Eigen::SparseMatrix<Scalar<G>> tmp;
169 d2r_exp_impl<GI, Inv>(tmp, a.segment(cntr, Dof<GI>));
170 for (auto j = 0u; j < Dof<GI>; ++j) {
171 block_add(out, cntr, Dof<G> * (cntr + j) + cntr, tmp.middleCols(Dof<GI> * j, Dof<GI>));
172 }
173 cntr += Dof<GI>;
174 });
175 out.makeCompressed();
176}

178template<LieGroup G, bool Inv>
179void d2r_exp_impl(Eigen::SparseMatrix<Scalar<G>> & out, const Tangent<G> & a)
180{
181 out.resize(a.size(), a.size() * a.size());
182
183 if constexpr (smooth::traits::RnType<G> || smooth::traits::ScalarType<G>) {
184 out.setZero();
185 } else if constexpr (std::is_base_of_v<smooth::SE2Base<G>, G>) {
186 if constexpr (Inv == false) {
187 d2r_exp_sparse_se2<Scalar<G>>(out, a);
188 } else {
189 d2r_expinv_sparse_se2<Scalar<G>>(out, a);
190 }
191 } else if constexpr (std::is_base_of_v<smooth::BundleBase<G>, G>) {
192 d2r_exp_sparse_bundle<G, Inv>(out, a);
193 } else {
194 if (out.isCompressed()) {
195 out.coeffs().setZero();
196 } else {
197 out.setZero();
198 out.reserve(Eigen::VectorXi::Constant(a.size() * a.size(), a.size()));
199 }
200 if constexpr (Inv == false) {
201 block_add(out, 0, 0, smooth::d2r_exp<G>(a));
202 } else {
203 block_add(out, 0, 0, smooth::d2r_expinv<G>(a));
204 }
205 out.makeCompressed();
206 }
207}
