169  inline std::pair<U, QPSolutionStatus> operator()(const Time & t,
170  const G & g,
171  std::optional<std::reference_wrapper<std::vector<U>>> u_traj = std::nullopt,
172  std::optional<std::reference_wrapper<std::vector<G>>> x_traj = std::nullopt)
173  {
174  using std::chrono::duration, std::chrono::duration_cast, std::chrono::nanoseconds;
175 
176  static constexpr int Nx = Dof<G>;
177  static constexpr int Nu = Dof<U>;
178  const int NU = prm_.K * Nu;
179 
180  // update problem with functions defined in "MPC time"
181  ocp_.x0 = g;
182  ocp_.gdes = [this, &t](double t_loc) -> G {
183  return x_des_(t + duration_cast<nanoseconds>(duration<double>(t_loc))).first;
184  };
185  ocp_.udes = [this, &t](double t_loc) -> U {
186  return u_des_(t + duration_cast<nanoseconds>(duration<double>(t_loc)));
187  };
188 
189  // define linearization if not already updated around previous solution
190  if (!prm_.relinearize_around_solution || new_desired_) {
191  lin_.u = [this, &t](double t_loc) -> U {
192  return u_des_(t + duration_cast<nanoseconds>(duration<double>(t_loc)));
193  };
194  lin_.g = [this, &t](double t_loc) -> std::pair<G, Tangent<G>> {
195  return x_des_(t + duration_cast<nanoseconds>(duration<double>(t_loc)));
196  };
197  new_desired_ = false;
198  }
199 
200  const double dt = ocp_.T / static_cast<double>(prm_.K);
201 
202  // define dynamics in "MPC time"
203  const auto dyn = [this, &t]<typename S>(
204  double t_loc, const CastT<S, G> & vx, const CastT<S, U> & vu) {
205  return dyn_(t + duration_cast<nanoseconds>(duration<double>(t_loc)), vx, vu);
206  };
207 
208  ocp_to_qp_fill<G, U, decltype(dyn), DT>(ocp_, prm_.K, dyn, lin_, qp_);
209  auto sol = solve_qp(qp_, prm_.qp, warmstart_);
210 
211  for (auto i = 0u; i < prm_.iterative_relinearization; ++i) {
212  // check if solution touches linearization domain
213  bool touches = false;
214  for (auto k = 0u; !touches && k < prm_.K; ++k) {
215  // clang-format off
216  if (((1. - 1e-6) * lin_.g_domain - sol.primal.template segment<Nx>(NU + k * Nx).cwiseAbs()).minCoeff() < 0) { touches = true; }
217  // clang-format on
218  }
219  if (touches) {
220  // relinearize around solution and solve again
221  relinearize_around_sol(sol);
222  ocp_to_qp_fill<G, U, decltype(dyn), DT>(ocp_, prm_.K, dyn, lin_, qp_);
223  sol = solve_qp(qp_, prm_.qp, warmstart_);
224  } else {
225  // solution seems fine
226  break;
227  }
228  }
229 
230  // output solution trajectories
231  if (u_traj.has_value()) {
232  u_traj.value().get().resize(prm_.K);
233  for (auto i = 0u; i < prm_.K; ++i) {
234  u_traj.value().get()[i] = lin_.u(i * dt) + sol.primal.template segment<Nu>(i * Nu);
235  }
236  }
237  if (x_traj.has_value()) {
238  x_traj.value().get().resize(prm_.K + 1);
239  x_traj.value().get()[0] = ocp_.x0;
240  for (auto i = 1u; i < prm_.K + 1; ++i) {
241  x_traj.value().get()[i] =
242  lin_.g(i * dt).first + sol.primal.template segment<Nx>(NU + (i - 1) * Nx);
243  }
244  }
245 
246  // update linearization for next iteration
247  if (sol.code == QPSolutionStatus::Optimal) {
248  if (prm_.relinearize_around_solution) relinearize_around_sol(sol);
249  }
250 
251  // save solution for warmstart
252  if (prm_.warmstart) {
253  if (sol.code == QPSolutionStatus::Optimal || sol.code == QPSolutionStatus::MaxTime
254  || sol.code == QPSolutionStatus::MaxIterations) {
255  warmstart_ = sol;
256  }
257  }
258 
259  return {rplus(lin_.u(0), sol.primal.template head<Nu>()), sol.code};
260  }

357  inline void relinearize_around_sol(const QPSolution<-1, -1, double> & sol)
358  {
359  const double dt = ocp_.T / static_cast<double>(prm_.K);
360 
361  // clang-format off
362  auto g_spline = smooth::fit_cubic_bezier(
363  std::views::iota(0u, prm_.K + 1) | std::views::transform([&](auto k) -> double { return dt * k; }),
364  std::views::iota(0u, prm_.K + 1) | std::views::transform([&](auto k) -> G {
365  if (k == 0) {
366  return ocp_.x0;
367  } else {
368  return rplus(lin_.g(dt * k).first, sol.primal.template segment<Dof<G>>(prm_.K * Dof<U> + (k - 1) * Dof<G>));
369  }
370  })
371  );
372  // clang-format on
373 
374  lin_.g = [g_spline = std::move(g_spline)](double t) -> std::pair<G, Tangent<G>> {
375  Tangent<G> dg;
376  auto g = g_spline(t, dg);
377  return std::make_pair(g, dg);
378  };
379  }
