47template<Time T, LieGroup X>
48struct XDes
49{
50 T t0;
51 std::function<X(T)> xdes = [](T) -> X { return Default<X>(); };
52 std::function<Tangent<X>(T)> dxdes = [](T) -> Tangent<X> { return Tangent<X>::Zero(); };
53
54 X operator()(const double t_rel) const
55 {
56 const T t_abs = time_trait<T>::plus(t0, t_rel);
57 return xdes(t_abs);
58 };
59
60 Tangent<X> jacobian(const double t_rel) const
61 {
62 const T t_abs = time_trait<T>::plus(t0, t_rel);
63 return dxdes(t_abs);
64 }
65};

93template<LieGroup X>
94struct MPCObj
95{
96 static constexpr auto Nx = Dof<X>;
97
98 // public members
99 X xf_des = Default<X>();
100 Eigen::Matrix<double, Nx, Nx> Qtf = Eigen::Matrix<double, Nx, Nx>::Identity();
101
102 // private members
103 Eigen::SparseMatrix<double> hess{1 + 2 * Nx + 1, 1 + 2 * Nx + 1};
104
105 // functor members
106
107 // function f(t, x0, xf, q) = (1/2) xf' Q xf + q_0
108 double operator()(
109 const double,
110 const X &,
111 [[maybe_unused]] const X & xf,
112 [[maybe_unused]] const Eigen::Vector<double, 1> & q) const
113 {
114 assert(q(0) == 1.);
115 assert(xf_des.isApprox(xf, 1e-4));
116 if constexpr (false) {
117 const auto e = rminus(xf, xf_des);
118 return 0.5 * e.dot(Qtf * e) + q(0);
119 } else {
120 return 1.;
121 }
122 }
123
124 Eigen::RowVector<double, 1 + 2 * Nx + 1>
125 jacobian(const double, const X &, [[maybe_unused]] const X & xf, const Eigen::Vector<double, 1> &)
126 {
127 assert(xf_des.isApprox(xf, 1e-4));
128 return Eigen::RowVector<double, 1 + 2 * Nx + 1>::Unit(1 + 2 * Nx);
129 }
130
131 std::reference_wrapper<const Eigen::SparseMatrix<double>>
132 hessian(const double, const X &, [[maybe_unused]] const X & xf, const Eigen::Vector<double, 1> &)
133 {
134 assert(xf_des.isApprox(xf, 1e-4));
135
136 if (hess.isCompressed()) { set_zero(hess); }
137
138 block_add(hess, 1 + Nx, 1 + Nx, Qtf);
139
140 hess.makeCompressed();
141 return hess;
142 }
143};

186template<Time T, LieGroup X, Manifold U>
187struct MPCIntegrand
188{
189 static constexpr auto Nx = Dof<X>;
190 static constexpr auto Nu = Dof<U>;
191
192 // public members
193 std::shared_ptr<XDes<T, X>> xdes;
194 std::shared_ptr<UDes<T, U>> udes;
195
196 Eigen::Matrix<double, Nx, Nx> Q = Eigen::Matrix<double, Nx, Nx>::Identity();
197 Eigen::Matrix<double, Nu, Nu> R = Eigen::Matrix<double, Nu, Nu>::Identity();
198
199 // private members
200 Eigen::SparseMatrix<double> hess{1 + Nx + Nu, 1 + Nx + Nu};
201
202 // functor members
203
204 // function f(x, t, u) = (1/2) * (x' Q x + u' R u)
205 Eigen::Vector<double, 1> operator()(
206 [[maybe_unused]] const double t_rel,
207 [[maybe_unused]] const X & x,
208 [[maybe_unused]] const U & u) const
209 {
210 assert((*xdes)(t_rel).isApprox(x, 1e-4));
211 assert((*udes)(t_rel).isApprox(u, 1e-4));
212 if constexpr (false) {
213 const auto ex = rminus(x, (*xdes)(t_rel));
214 const auto eu = rminus(u, (*udes)(t_rel));
215 return Eigen::Vector<double, 1>{0.5 * ex.dot(Q * ex) + 0.5 * eu.dot(R * eu)};
216 } else {
217 return Eigen::Vector<double, 1>{0.};
218 }
219 }
220
221 Eigen::RowVector<double, 1 + Nx + Nu> jacobian(
222 [[maybe_unused]] const double t_rel, [[maybe_unused]] const X & x, [[maybe_unused]] const U & u)
223 {
224 assert((*xdes)(t_rel).isApprox(x, 1e-4));
225 assert((*udes)(t_rel).isApprox(u, 1e-4));
226 return Eigen::RowVector<double, 1 + Nx + Nu>::Zero();
227 }
228
229 std::reference_wrapper<const Eigen::SparseMatrix<double>> hessian(
230 [[maybe_unused]] const double t_rel, [[maybe_unused]] const X & x, [[maybe_unused]] const U & u)
231 {
232 assert((*xdes)(t_rel).isApprox(x, 1e-4));
233 assert((*udes)(t_rel).isApprox(u, 1e-4));
234
235 if (hess.isCompressed()) { set_zero(hess); }
236
237 block_add(hess, 1, 1, Q);
238 block_add(hess, 1 + Nx, 1 + Nx, R);
239
240 hess.makeCompressed();
241 return hess;
242 }
243};

481 inline std::pair<U, QPSolutionStatus> operator()(
482 const T & t,
483 const X & x,
484 std::optional<std::reference_wrapper<std::vector<U>>> u_traj = std::nullopt,
485 std::optional<std::reference_wrapper<std::vector<X>>> x_traj = std::nullopt)
486 {
487 static constexpr auto Nx = Dof<X>;
488 static constexpr auto Nu = Dof<U>;
489
490 const auto N = mesh_.N_colloc();
491 const auto xvar_L = Nx * (N + 1);
492 const auto xvar_B = 0u;
493 const auto uvar_B = xvar_L;
494
495 // update problem
496 xdes_->t0 = t;
497 udes_->t0 = t;
498 ocp_.ce.x0_fix = x;
499 ocp_.theta.xf_des = (*xdes_)(prm_.tf);
500
501 // transcribe to QP
502 ocp_to_qp_update<diff::Type::Analytic>(qp_, work_, ocp_, mesh_, prm_.tf, *xdes_, *udes_);
503 qp_.A.makeCompressed();
504 qp_.P.makeCompressed();
505
506 // solve QP
507 const auto & sol = qp_solver_.solve(qp_, warmstart_);
508
509 // output solution trajectories
510 if (u_traj.has_value()) {
511 u_traj.value().get().resize(N);
512 for (const auto & [i, tau] : zip(std::views::iota(0u, N), mesh_.all_nodes())) {
513 const double t_rel = prm_.tf * tau;
514 u_traj.value().get()[i] =
515 (*udes_)(t_rel) + sol.primal.template segment<Nu>(uvar_B + i * Nu);
516 }
517 }
518 if (x_traj.has_value()) {
519 x_traj.value().get().resize(N + 1);
520 for (const auto & [i, tau] : zip(std::views::iota(0u, N + 1), mesh_.all_nodes())) {
521 const double t_rel = prm_.tf * tau;
522 x_traj.value().get()[i] =
523 (*xdes_)(t_rel) + sol.primal.template segment<Nx>(xvar_B + i * Nx);
524 }
525 }
526
527 // save solution to warmstart next iteration
528 if (prm_.warmstart) {
529 // clang-format off
530 if (sol.code == QPSolutionStatus::Optimal || sol.code == QPSolutionStatus::MaxTime || sol.code == QPSolutionStatus::MaxIterations) {
531 warmstart_ = sol;
532 }
533 // clang-format on
534 }
535
536 return {rplus((*udes_)(0), sol.primal.template segment<Nu>(uvar_B)), sol.code};
537 }
