46template<Time T, LieGroup X>
47struct XDes
48{
49 T t0;
50 std::function<X(T)> xdes = [](T) -> X { return Default<X>(); };
51 std::function<Tangent<X>(T)> dxdes = [](T) -> Tangent<X> { return Tangent<X>::Zero(); };
52
53 X operator()(const double t_rel) const
54 {
55 const T t_abs = time_trait<T>::plus(t0, t_rel);
56 return xdes(t_abs);
57 };
58
59 Tangent<X> jacobian(const double t_rel) const
60 {
61 const T t_abs = time_trait<T>::plus(t0, t_rel);
62 return dxdes(t_abs);
63 }
64};

69template<Time T, Manifold U>
70struct UDes
71{
72 T t0;
73 std::function<U(T)> udes = [](T) -> U { return Default<U>(); };
74
75 U operator()(const double t_rel) const
76 {
77 const T t_abs = time_trait<T>::plus(t0, t_rel);
78 return udes(t_abs);
79 };

92template<LieGroup X>
93struct MPCObj
94{
95 static constexpr auto Nx = Dof<X>;
96
97 // public members
98 X xf_des = Default<X>();
99 Eigen::Matrix<double, Nx, Nx> Qtf = Eigen::Matrix<double, Nx, Nx>::Identity();
100
101 // private members
102 Eigen::SparseMatrix<double> hess{1 + 2 * Nx + 1, 1 + 2 * Nx + 1};
103
104 // functor members
105 double operator()(
106 const double,
107 const X &,
108 [[maybe_unused]] const X & xf,
109 [[maybe_unused]] const Eigen::Vector<double, 1> & q) const
110 {
111 assert(q(0) == 1.);
112 assert(xf_des.isApprox(xf, 1e-4));
113 return 0.;
114 }
115
116 Eigen::RowVector<double, 1 + 2 * Nx + 1>
117 jacobian(const double, const X &, [[maybe_unused]] const X & xf, const Eigen::Vector<double, 1> &)
118 {
119 assert(xf_des.isApprox(xf, 1e-4));
120 return Eigen::RowVector<double, 1 + 2 * Nx + 1>::Unit(1 + 2 * Nx);
121 }
122
123 std::reference_wrapper<const Eigen::SparseMatrix<double>>
124 hessian(const double, const X &, [[maybe_unused]] const X & xf, const Eigen::Vector<double, 1> &)
125 {
126 assert(xf_des.isApprox(xf, 1e-4));
127
128 set_zero(hess);
129 block_add(hess, 1 + Nx, 1 + Nx, Qtf);
130
131 hess.makeCompressed();
132 return hess;
133 }
134};

177template<Time T, LieGroup X, Manifold U>
178struct MPCIntegrand
179{
180 static constexpr auto Nx = Dof<X>;
181 static constexpr auto Nu = Dof<U>;
182
183 // public members
184 std::shared_ptr<XDes<T, X>> xdes;
185 std::shared_ptr<UDes<T, U>> udes;
186
187 Eigen::Matrix<double, Nx, Nx> Q = Eigen::Matrix<double, Nx, Nx>::Identity();
188 Eigen::Matrix<double, Nu, Nu> R = Eigen::Matrix<double, Nu, Nu>::Identity();
189
190 // private members
191 Eigen::SparseMatrix<double> hess{1 + Nx + Nu, 1 + Nx + Nu};
192
193 // functor members
194 Eigen::Vector<double, 1> operator()(
195 [[maybe_unused]] const double t_rel,
196 [[maybe_unused]] const X & x,
197 [[maybe_unused]] const U & u) const
198 {
199 assert((*xdes)(t_rel).isApprox(x, 1e-4));
200 assert((*udes)(t_rel).isApprox(u, 1e-4));
201 return Eigen::Vector<double, 1>{0.};
202 }
203
204 Eigen::RowVector<double, 1 + Nx + Nu> jacobian(
205 [[maybe_unused]] const double t_rel, [[maybe_unused]] const X & x, [[maybe_unused]] const U & u)
206 {
207 assert((*xdes)(t_rel).isApprox(x, 1e-4));
208 assert((*udes)(t_rel).isApprox(u, 1e-4));
209 return Eigen::RowVector<double, 1 + Nx + Nu>::Zero();
210 }
211
212 std::reference_wrapper<const Eigen::SparseMatrix<double>> hessian(
213 [[maybe_unused]] const double t_rel, [[maybe_unused]] const X & x, [[maybe_unused]] const U & u)
214 {
215 assert((*xdes)(t_rel).isApprox(x, 1e-4));
216 assert((*udes)(t_rel).isApprox(u, 1e-4));
217
218 set_zero(hess);
219 block_add(hess, 1, 1, Q);
220 block_add(hess, 1 + Nx, 1 + Nx, R);
221
222 hess.makeCompressed();
223 return hess;
224 }

461 inline std::pair<U, QPSolutionStatus> operator()(
462 const T & t,
463 const X & x,
464 std::optional<std::reference_wrapper<std::vector<U>>> u_traj = std::nullopt,
465 std::optional<std::reference_wrapper<std::vector<X>>> x_traj = std::nullopt)
466 {
467 static constexpr auto Nx = Dof<X>;
468 static constexpr auto Nu = Dof<U>;
469
470 const auto N = mesh_.N_colloc();
471 const auto xvar_L = Nx * (N + 1);
472 const auto xvar_B = 0u;
473 const auto uvar_B = xvar_L;
474
475 // update problem
476 xdes_->t0 = t;
477 udes_->t0 = t;
478 ocp_.ce.x0_fix = x;
479 ocp_.theta.xf_des = (*xdes_)(prm_.tf);
480
481 // transcribe to QP
482 ocp_to_qp_update<diff::Type::Analytic>(qp_, work_, ocp_, mesh_, prm_.tf, *xdes_, *udes_);
483 qp_.A.makeCompressed();
484 qp_.P.makeCompressed();
485
486 // solve QP
487 auto sol = solve_qp(qp_, prm_.qp, warmstart_);
488
489 // output solution trajectories
490 if (u_traj.has_value()) {
491 u_traj.value().get().resize(N);
492 for (const auto & [i, tau] : zip(std::views::iota(0u, N), mesh_.all_nodes())) {
493 const double t_rel = prm_.tf * tau;
494 u_traj.value().get()[i] =
495 (*udes_)(t_rel) + sol.primal.template segment<Nu>(uvar_B + i * Nu);
496 }
497 }
498 if (x_traj.has_value()) {
499 x_traj.value().get().resize(N + 1);
500 for (const auto & [i, tau] : zip(std::views::iota(0u, N + 1), mesh_.all_nodes())) {
501 const double t_rel = prm_.tf * tau;
502 x_traj.value().get()[i] =
503 (*xdes_)(t_rel) + sol.primal.template segment<Nx>(xvar_B + i * Nx);
504 }
505 }
506
507 // save solution for warmstart
508 if (prm_.warmstart) {
509 if (
510 // clang-format off
511 sol.code == QPSolutionStatus::Optimal || sol.code == QPSolutionStatus::MaxTime || sol.code == QPSolutionStatus::MaxIterations
512 // clang-format on
513 ) {
514 warmstart_ = sol;
515 }
516 }
517
518 return {rplus((*udes_)(0), sol.primal.template segment<Nu>(uvar_B)), sol.code};
519 }
