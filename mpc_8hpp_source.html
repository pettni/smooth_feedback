173  inline std::pair<U, QPSolutionStatus> operator()(
174  const T & t,
175  const G & g,
176  std::optional<std::reference_wrapper<std::vector<U>>> u_traj = std::nullopt,
177  std::optional<std::reference_wrapper<std::vector<G>>> x_traj = std::nullopt)
178  {
179  using std::chrono::duration, std::chrono::duration_cast, std::chrono::nanoseconds;
180 
181  static constexpr int Nx = Dof<G>;
182  static constexpr int Nu = Dof<U>;
183  const int NU = prm_.K * Nu;
184 
185  // update problem with functions defined in "MPC time"
186  ocp_.x0 = g;
187  ocp_.gdes = [this, &t](double t_loc) -> G {
188  return x_des_(t + duration_cast<nanoseconds>(duration<double>(t_loc))).first;
189  };
190  ocp_.udes = [this, &t](double t_loc) -> U {
191  return u_des_(t + duration_cast<nanoseconds>(duration<double>(t_loc)));
192  };
193 
194  // define linearization if not already updated around previous solution
195  if (!prm_.relinearize_around_solution || new_desired_) {
196  lin_.u = [this, &t](double t_loc) -> U { return u_des_(time_trait<T>::plus(t, t_loc)); };
197  lin_.g = [this, &t](double t_loc) -> std::pair<G, Tangent<G>> {
198  return x_des_(time_trait<T>::plus(t, t_loc));
199  };
200  new_desired_ = false;
201  }
202 
203  const double dt = ocp_.T / static_cast<double>(prm_.K);
204 
205  // define dynamics in "MPC time"
206  const auto dyn =
207  [this, &t]<typename S>(double t_loc, const CastT<S, G> & vx, const CastT<S, U> & vu) {
208  return dyn_(t + duration_cast<nanoseconds>(duration<double>(t_loc)), vx, vu);
209  };
210 
211  ocp_to_qp_fill<G, U, decltype(dyn), DiffT>(ocp_, prm_.K, dyn, lin_, qp_);
212  auto sol = solve_qp(qp_, prm_.qp, warmstart_);
213 
214  for (auto i = 0u; i < prm_.iterative_relinearization; ++i) {
215  // check if solution touches linearization domain
216  bool touches = false;
217  for (auto k = 0u; !touches && k < prm_.K; ++k) {
218  // clang-format off
219  if (((1. - 1e-6) * lin_.g_domain - sol.primal.template segment<Nx>(NU + k * Nx).cwiseAbs()).minCoeff() < 0) { touches = true; }
220  // clang-format on
221  }
222  if (touches) {
223  // relinearize around solution and solve again
224  relinearize_around_sol(sol);
225  ocp_to_qp_fill<G, U, decltype(dyn), DiffT>(ocp_, prm_.K, dyn, lin_, qp_);
226  sol = solve_qp(qp_, prm_.qp, warmstart_);
227  } else {
228  // solution seems fine
229  break;
230  }
231  }
232 
233  // output solution trajectories
234  if (u_traj.has_value()) {
235  u_traj.value().get().resize(prm_.K);
236  for (auto i = 0u; i < prm_.K; ++i) {
237  u_traj.value().get()[i] = lin_.u(i * dt) + sol.primal.template segment<Nu>(i * Nu);
238  }
239  }
240  if (x_traj.has_value()) {
241  x_traj.value().get().resize(prm_.K + 1);
242  x_traj.value().get()[0] = ocp_.x0;
243  for (auto i = 1u; i < prm_.K + 1; ++i) {
244  x_traj.value().get()[i] =
245  lin_.g(i * dt).first + sol.primal.template segment<Nx>(NU + (i - 1) * Nx);
246  }
247  }
248 
249  // update linearization for next iteration
250  if (sol.code == QPSolutionStatus::Optimal) {
251  if (prm_.relinearize_around_solution) relinearize_around_sol(sol);
252  }
253 
254  // save solution for warmstart
255  if (prm_.warmstart) {
256  // clang-format off
257  if (sol.code == QPSolutionStatus::Optimal || sol.code == QPSolutionStatus::MaxTime || sol.code == QPSolutionStatus::MaxIterations) {
258  warmstart_ = sol;
259  }
260  // clang-format on
261  }
262 
263  return {rplus(lin_.u(0), sol.primal.template head<Nu>()), sol.code};
264  }

358  inline void relinearize_around_sol(const QPSolution<-1, -1, double> & sol)
359  {
360  const double dt = ocp_.T / static_cast<double>(prm_.K);
361 
362  // clang-format off
363  auto g_spline = smooth::fit_spline_cubic(
364  std::views::iota(0u, prm_.K + 1) | std::views::transform([&](auto k) -> double { return dt * k; }),
365  std::views::iota(0u, prm_.K + 1) | std::views::transform([&](auto k) -> G {
366  if (k == 0) {
367  return ocp_.x0;
368  } else {
369  return rplus(lin_.g(dt * k).first, sol.primal.template segment<Dof<G>>(prm_.K * Dof<U> + (k - 1) * Dof<G>));
370  }
371  })
372  );
373  // clang-format on
374 
375  lin_.g = [g_spline = std::move(g_spline)](double t) -> std::pair<G, Tangent<G>> {
376  Tangent<G> dg;
377  auto g = g_spline(t, dg);
378  return std::make_pair(g, dg);
379  };
380  }
