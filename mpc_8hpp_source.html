143 template<std::size_t K,
144  LieGroup G,
145  Manifold U,
146  typename Dyn,
147  typename GLin,
148  typename DGLin,
149  typename ULin>
150 QuadraticProgramSparse<double> ocp_to_qp(
151  const OptimalControlProblem<G, U> & pbm, Dyn && f, GLin && glin, DGLin && dglin, ULin && ulin)
152 {
153  using std::placeholders::_1;
154 
155  // problem info
156  static constexpr int nx = G::SizeAtCompileTime;
157  static constexpr int nu = U::SizeAtCompileTime;
158 
159  static constexpr int nX = K * nx;
160  static constexpr int nU = K * nu;
161  static constexpr int nvar = nX + nU;
162 
163  static constexpr int n_eq = nX; // equality constraints from dynamics
164  static constexpr int n_u_iq = nU; // input bounds
165  static constexpr int n_x_iq = nX; // state bounds
166  static constexpr int ncon = n_eq + n_u_iq + n_x_iq;
167 
168  using AT = Eigen::Matrix<double, nx, nx>;
169  using BT = Eigen::Matrix<double, nx, nu>;
170  using ET = Eigen::Matrix<double, nx, 1>;
171 
172  const double dt = pbm.T / static_cast<double>(K);
173 
174  QuadraticProgramSparse ret;
175  ret.P.resize(nvar, nvar);
176  ret.q.setZero(nvar);
177 
178  ret.A.resize(ncon, nvar);
179  ret.l.setZero(ncon);
180  ret.u.setZero(ncon);
181 
182  // SET SPARSITY PATTERNS
183 
184  Eigen::Matrix<int, ncon, 1> A_sparsity; // row-wise
185  Eigen::Matrix<int, nU + nX, 1> P_sparsity; // column-wise (but symmetric so doesn't matter)
186 
187  // dynamics
188  A_sparsity.template head<nx>().setConstant(nu + 1); // B, I
189  A_sparsity.template segment<nx *(K - 1)>(nx).setConstant(nx + nu + 1); // B, A, I
190 
191  // input and state bounds
192  A_sparsity.template segment<n_u_iq>(n_eq).setConstant(1);
193  A_sparsity.template segment<n_x_iq>(n_eq + n_u_iq).setConstant(1);
194 
195  // cost matrix is block-diagonal with blocks of size nu x nu and nx x nx
196  P_sparsity.template head<nU>().setConstant(nu);
197  P_sparsity.template segment<nX>(nU).setConstant(nx);
198 
199  ret.A.reserve(A_sparsity);
200  ret.P.reserve(P_sparsity);
201 
202  // DYNAMICS CONSTRAINTS
203 
204  for (auto k = 0u; k != K; ++k) {
205  const double t = k * dt;
206 
207  // LINEARIZATION
208 
209  const auto xl = glin(t);
210  const auto dxl = dglin(t);
211  const auto ul = ulin(t);
212 
213  const auto [flin, df_xu] = diff::dr(f, wrt(xl, ul));
214 
215  // cltv system \dot x = At x(t) + Bt u(t) + Et
216  const AT At = (-0.5 * G::ad(flin) - 0.5 * G::ad(dxl) + df_xu.template leftCols<nx>());
217  const BT Bt = df_xu.template rightCols<nu>();
218  const ET Et = flin - dxl;
219 
220  // TIME DISCRETIZATION
221 
222  const AT At2 = At * At;
223  const AT At3 = At2 * At;
224  const double dt2 = dt * dt;
225  const double dt3 = dt2 * dt;
226 
227  // dltv system x^+ = Ak x + Bk u + Ek by truncated taylor expansion of the matrix exponential
228  const AT Ak = AT::Identity() + At * dt + At2 * dt2 / 2. + At3 * dt3 / 6.;
229  const BT Bk = Bt * dt + At * Bt * dt2 / 2. + At2 * Bt * dt3 / 6.;
230  const ET Ek = Et * dt + At * Et * dt2 / 2. + At2 * Et * dt3 / 6.;
231 
232  // DYNAMICS CONSTRANTS
233 
234  if (k == 0) {
235  // x(1) - B u(0) = A x0 + E
236 
237  // identity matrix on x(1)
238  for (auto i = 0u; i != nx; ++i) { ret.A.insert(nx * k + i, nU + nx * k + i) = 1; }
239 
240  // B matrix on u(0)
241  for (auto i = 0u; i != nx; ++i) {
242  for (auto j = 0u; j != nu; ++j) { ret.A.insert(nx * k + i, nu * k + j) = -Bk(i, j); }
243  }
244 
245  ret.u.template segment<nx>(nx * k) = Ak * (pbm.x0 - glin(t)) + Ek;
246  ret.l.template segment<nx>(nx * k) = ret.u.template segment<nx>(nx * k);
247 
248  } else {
249  // x(k) - A x(k-1) - B u(k-1) = E
250 
251  // identity matrix on x(k)
252  for (auto i = 0u; i != nx; ++i) { ret.A.insert(nx * k + i, nU + nx * k + i) = 1; }
253 
254  // A matrix on x(k-1)
255  for (auto i = 0u; i != nx; ++i) {
256  for (auto j = 0u; j != nx; ++j) {
257  ret.A.insert(nx * k + i, nU + nx * (k - 1) + j) = -Ak(i, j);
258  }
259  }
260 
261  // B matrix on u(k-1)
262  for (auto i = 0u; i != nx; ++i) {
263  for (auto j = 0u; j != nu; ++j) { ret.A.insert(nx * k + i, nu * k + j) = -Bk(i, j); }
264  }
265 
266  ret.u.template segment<nx>(nx * k) = Ek;
267  ret.l.template segment<nx>(nx * k) = Ek;
268  }
269  }
270 
271  // INPUT CONSTRAINTS
272 
273  for (auto i = 0u; i != n_u_iq; ++i) { ret.A.insert(n_eq + i, i) = 1; }
274 
275  if (pbm.umin) {
276  for (auto k = 0u; k < K; ++k) {
277  ret.l.template segment<nu>(n_eq + k * nu) = pbm.umin.value() - ulin(k * dt);
278  }
279  } else {
280  ret.l.template segment<nU>(n_eq).setConstant(-std::numeric_limits<double>::infinity());
281  }
282  if (pbm.umax) {
283  for (auto k = 0u; k < K; ++k) {
284  ret.u.template segment<nu>(n_eq + k * nu) = pbm.umax.value() - ulin(k * dt);
285  }
286  } else {
287  ret.u.template segment<nU>(n_eq).setConstant(std::numeric_limits<double>::infinity());
288  }
289 
290  // STATE CONSTRAINTS
291 
292  for (auto i = 0u; i != n_x_iq; ++i) { ret.A.insert(n_eq + n_u_iq + i, nU + i) = 1; }
293 
294  if (pbm.gmin) {
295  for (auto k = 1u; k < K; ++k) {
296  ret.l.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) = pbm.gmin.value() - glin(k * dt);
297  }
298  } else {
299  ret.l.template segment<nX>(n_eq + n_u_iq).setConstant(-std::numeric_limits<double>::infinity());
300  }
301  if (pbm.gmax) {
302  for (auto k = 1u; k < K; ++k) {
303  ret.u.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) = pbm.gmax.value() - glin(k * dt);
304  }
305  } else {
306  ret.u.template segment<nX>(n_eq + n_u_iq).setConstant(std::numeric_limits<double>::infinity());
307  }
308 
309  // INPUT COSTS
310 
311  for (auto k = 0u; k < K; ++k) {
312  for (auto i = 0u; i != nu; ++i) {
313  for (auto j = 0u; j != nu; ++j) { ret.P.insert(k * nu + i, k * nu + j) = pbm.R(i, j) * dt; }
314  }
315  ret.q.template segment<nu>(k * nu) = pbm.R * (ulin(k * dt) - pbm.udes(k * dt));
316  }
317 
318  // STATE COSTS
319 
320  // intermediate states x(1) ... x(K-1)
321  for (auto k = 1u; k < K; ++k) {
322  for (auto i = 0u; i != nx; ++i) {
323  for (auto j = 0u; j != nx; ++j) {
324  ret.P.insert(nU + (k - 1) * nx + i, nU + (k - 1) * nx + j) = pbm.Q(i, j) * dt;
325  }
326  }
327  ret.q.template segment<nx>(nU + (k - 1) * nx) = pbm.Q * (glin(k * dt) - pbm.gdes(k * dt)) * dt;
328  }
329 
330  // last state x(K) ~ x(T)
331  for (auto i = 0u; i != nx; ++i) {
332  for (auto j = 0u; j != nx; ++j) {
333  ret.P.insert(nU + (K - 1) * nx + i, nU + (K - 1) * nx + j) = pbm.QT(i, j);
334  }
335  }
336  ret.q.template segment<nx>(nU + (K - 1) * nx) = pbm.QT * (glin(pbm.T) - pbm.gdes(pbm.T));
337 
338  ret.A.makeCompressed();
339  ret.P.makeCompressed();
340 
341  return ret;
342 }

364  MPC(const Dyn & dyn, const SolverParams & qp_prm = SolverParams{}) : MPC(Dyn(dyn), qp_prm) {}
365  MPC() = default;

372  U operator()(const nanoseconds & t_abs, const G & g)
373  {
374  ocp_.x0 = g;
375  ocp_.gdes = [this, &t_abs](double t) {
376  return x_des_(t_abs + duration_cast<nanoseconds>(std::chrono::duration<double>(t)));
377  };
378  ocp_.udes = [this, &t_abs](double t) {
379  return u_des_(t_abs + duration_cast<nanoseconds>(std::chrono::duration<double>(t)));
380  };
381 
382  const auto glin = [](double) -> G { return G::Identity(); };
383  const auto dglin = [](double) -> typename G::Tangent { return G::Tangent::Zero(); };
384  const auto ulin = [](double) -> U {
385  if constexpr (LieGroup<U>) {
386  return U::Identity();
387  } else {
388  return U::Zero();
389  }
390  };
391 
392  const auto qp = smooth::feedback::ocp_to_qp<K>(ocp_, dyn_, glin, dglin, ulin);
393  const auto sol = smooth::feedback::solve_qp(qp, qp_prm_);
394 
395  return ulin(0) + sol.primal.template head<U::SizeAtCompileTime>();
396 
397  // TODO: update linearization point here
398  }
