223 template<std::size_t K,
224  LieGroup G,
225  Manifold U,
226  typename Dyn,
227  smooth::diff::Type DiffType = smooth::diff::Type::DEFAULT>
228 QuadraticProgramSparse<double> ocp_to_qp(const OptimalControlProblem<G, U> & pbm,
229  const Dyn & f,
230  const LinearizationInfo<G, U> & lin = LinearizationInfo<G, U>{})
231 {
232  using std::placeholders::_1;
233 
234  // problem info
235  static constexpr int nx = G::SizeAtCompileTime;
236  static constexpr int nu = U::SizeAtCompileTime;
237 
238  static constexpr int nX = K * nx;
239  static constexpr int nU = K * nu;
240  static constexpr int nvar = nX + nU;
241 
242  static constexpr int n_eq = nX; // equality constraints from dynamics
243 
244  uint32_t n_u_iq = pbm.ulim ? pbm.ulim.value().A.rows() * K : 0;
245  uint32_t n_g_iq = pbm.glim ? pbm.glim.value().A.rows() * K : 0;
246 
247  const uint32_t ncon = n_eq + n_u_iq + n_g_iq;
248 
249  QuadraticProgramSparse ret;
250  ret.P.resize(nvar, nvar);
251  ret.q.resize(nvar);
252 
253  ret.A.resize(ncon, nvar);
254  ret.l.resize(ncon);
255  ret.u.resize(ncon);
256 
257  // SET SPARSITY PATTERNS
258 
259  static constexpr auto Pp = mpc_nnz_pattern_P<K, nx, nu>();
260  ret.P.reserve(Eigen::Map<const Eigen::Matrix<int, nvar, 1>>(Pp.data()));
261 
262  Eigen::Matrix<int, -1, 1> Ap(ncon);
263  int Arow = 0;
264  Ap.segment(Arow, nx).setConstant(1 + nu);
265  Arow += nx;
266  Ap.segment(Arow, (K - 1) * nx).setConstant(1 + nu + nx);
267  Arow += (K - 1) * nx;
268  if (pbm.ulim) {
269  Ap.segment(Arow, n_u_iq).setConstant(nu);
270  Arow += n_u_iq;
271  }
272  if (pbm.glim) {
273  Ap.tail(n_g_iq).setConstant(nx);
274  Arow += n_g_iq;
275  }
276  ret.A.reserve(Ap);
277 
278  const double dt = pbm.T / static_cast<double>(K);
279 
280  // DYNAMICS CONSTRAINTS
281 
282  Arow = 0;
283 
284  for (auto k = 0u; k != K; ++k) {
285  using AT = Eigen::Matrix<double, nx, nx>;
286  using BT = Eigen::Matrix<double, nx, nu>;
287  using ET = Eigen::Matrix<double, nx, 1>;
288 
289  const double t = k * dt;
290 
291  // LINEARIZATION
292 
293  const auto xl = lin.g(t);
294  const auto dxl = lin.dg(t);
295  const auto ul = lin.u(t);
296 
297  const auto [flin, df_xu] = diff::dr<DiffType>(
298  [&f, &t](const auto & vx, const auto & vu) { return f(t, vx, vu); }, wrt(xl, ul));
299 
300  // cltv system \dot x = At x(t) + Bt u(t) + Et
301  const AT At = (-0.5 * G::ad(flin) - 0.5 * G::ad(dxl) + df_xu.template leftCols<nx>());
302  const BT Bt = df_xu.template rightCols<nu>();
303  const ET Et = flin - dxl;
304 
305  // TIME DISCRETIZATION
306 
307  const AT At2 = At * At;
308  const double dt2 = dt * dt;
309  const double dt3 = dt2 * dt;
310 
311  // dltv system x^+ = Ak x + Bk u + Ek by truncated taylor expansion of the matrix exponential
312  const AT Ak = AT::Identity() + At * dt + At2 * dt2 / 2. + At2 * At * dt3 / 6.;
313  const BT Bk = Bt * dt + At * Bt * dt2 / 2. + At2 * Bt * dt3 / 6.;
314  const ET Ek = Et * dt + At * Et * dt2 / 2. + At2 * Et * dt3 / 6.;
315 
316  // DYNAMICS CONSTRANTS
317 
318  if (k == 0) {
319  // x(1) - B u(0) = A x0 + E
320 
321  // identity matrix on x(1)
322  for (auto i = 0u; i != nx; ++i) { ret.A.insert(nx * k + i, nU + nx * k + i) = 1; }
323 
324  // B matrix on u(0)
325  for (auto i = 0u; i != nx; ++i) {
326  for (auto j = 0u; j != nu; ++j) { ret.A.insert(nx * k + i, nu * k + j) = -Bk(i, j); }
327  }
328 
329  ret.u.template segment<nx>(nx * k) = Ak * (pbm.x0 - lin.g(t)) + Ek;
330  ret.l.template segment<nx>(nx * k) = ret.u.template segment<nx>(nx * k);
331 
332  } else {
333  // x(k+1) - A x(k) - B u(k) = E
334 
335  // identity matrix on x(k+1)
336  for (auto i = 0u; i != nx; ++i) { ret.A.insert(nx * k + i, nU + nx * k + i) = 1; }
337 
338  // A matrix on x(k)
339  for (auto i = 0u; i != nx; ++i) {
340  for (auto j = 0u; j != nx; ++j) {
341  ret.A.insert(nx * k + i, nU + nx * (k - 1) + j) = -Ak(i, j);
342  }
343  }
344 
345  // B matrix on u(k)
346  for (auto i = 0u; i != nx; ++i) {
347  for (auto j = 0u; j != nu; ++j) { ret.A.insert(nx * k + i, nu * k + j) = -Bk(i, j); }
348  }
349 
350  ret.u.template segment<nx>(nx * k) = Ek;
351  ret.l.template segment<nx>(nx * k) = Ek;
352  }
353  }
354 
355  Arow += K * nx;
356 
357  // INPUT CONSTRAINTS
358 
359  if (pbm.ulim) {
360  for (auto k = 0u; k < K; ++k) {
361  for (auto i = 0u; i != pbm.ulim.value().A.rows(); ++i) {
362  for (auto j = 0u; j != nu; ++j) {
363  ret.A.insert(Arow + k * nu + i, k * nu + j) = pbm.ulim.value().A(i, j);
364  }
365  }
366  ret.l.template segment<nu>(Arow + k * nu) =
367  pbm.ulim.value().l - pbm.ulim.value().A * (lin.u(k * dt) - identity<U>());
368  ret.u.template segment<nu>(Arow + k * nu) =
369  pbm.ulim.value().u - pbm.ulim.value().A * (lin.u(k * dt) - identity<U>());
370  }
371  Arow += n_u_iq;
372  }
373 
374  // linearization bounds
375  /* for (auto k = 0u; k < K; ++k) {
376  ret.l.template segment<nu>(n_eq + k * nu) =
377  ret.l.template segment<nu>(n_eq + k * nu).cwiseMax(-lin.u_domain);
378  ret.u.template segment<nu>(n_eq + k * nu) =
379  ret.u.template segment<nu>(n_eq + k * nu).cwiseMin(lin.u_domain);
380  } */
381 
382  // STATE CONSTRAINTS
383 
384  if (pbm.glim) {
385  for (auto k = 1u; k != K + 1; ++k) {
386  for (auto i = 0u; i != pbm.glim.value().A.rows(); ++i) {
387  for (auto j = 0u; j != nx; ++j) {
388  ret.A.insert(Arow + (k - 1) * nx + i, nU + (k - 1) * nx + j) = pbm.glim.value().A(i, j);
389  }
390  }
391  ret.l.template segment<nx>(Arow + (k - 1) * nx) =
392  pbm.glim.value().l - pbm.glim.value().A * (lin.g(k * dt) - identity<G>());
393  ret.u.template segment<nx>(Arow + (k - 1) * nx) =
394  pbm.glim.value().u - pbm.glim.value().A * (lin.g(k * dt) - identity<G>());
395  }
396  Arow += n_g_iq;
397  }
398 
399  // linearization bounds
400  /* for (auto k = 1u; k < K + 1; ++k) {
401  ret.l.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) =
402  ret.l.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx).cwiseMax(-lin.g_domain);
403  ret.u.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) =
404  ret.u.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx).cwiseMin(lin.g_domain);
405  } */
406 
407  // INPUT COSTS
408 
409  for (auto k = 0u; k < K; ++k) {
410  for (auto i = 0u; i != nu; ++i) {
411  for (auto j = 0u; j != nu; ++j) { ret.P.insert(k * nu + i, k * nu + j) = pbm.R(i, j) * dt; }
412  }
413  ret.q.template segment<nu>(k * nu) = pbm.R * (lin.u(k * dt) - pbm.udes(k * dt)) * dt;
414  }
415 
416  // STATE COSTS
417 
418  // intermediate states x(1) ... x(K-1)
419  for (auto k = 1u; k < K; ++k) {
420  for (auto i = 0u; i != nx; ++i) {
421  for (auto j = 0u; j != nx; ++j) {
422  ret.P.insert(nU + (k - 1) * nx + i, nU + (k - 1) * nx + j) = pbm.Q(i, j) * dt;
423  }
424  }
425  ret.q.template segment<nx>(nU + (k - 1) * nx) = pbm.Q * (lin.g(k * dt) - pbm.gdes(k * dt)) * dt;
426  }
427 
428  // last state x(K) ~ x(T)
429  for (auto i = 0u; i != nx; ++i) {
430  for (auto j = 0u; j != nx; ++j) {
431  ret.P.insert(nU + (K - 1) * nx + i, nU + (K - 1) * nx + j) = pbm.QT(i, j);
432  }
433  }
434  ret.q.template segment<nx>(nU + (K - 1) * nx) = pbm.QT * (lin.g(pbm.T) - pbm.gdes(pbm.T));
435 
436  ret.A.makeCompressed();
437  ret.P.makeCompressed();
438 
439  return ret;
440 }

507 template<std::size_t K,
508  typename T,
509  LieGroup G,
510  Manifold U,
511  typename Dyn,
512  smooth::diff::Type DiffType = smooth::diff::Type::DEFAULT>
513 class MPC
514 {
515 public:
527  MPC(Dyn && f, MPCParams && prm) : prm_(std::move(prm)), dyn_(std::move(f)) {}
528 
530  MPC(const Dyn & f, const MPCParams & prm) : MPC(Dyn(f), MPCParams(prm)) {}
531 
533  MPC() = default;
535  MPC(const MPC &) = default;
537  MPC(MPC &&) = default;
539  MPC & operator=(const MPC &) = default;
541  MPC & operator=(MPC &&) = default;
543  ~MPC() = default;
544 
565  QPSolutionStatus operator()(U & u,
566  const T & t,
567  const G & g,
568  std::optional<std::reference_wrapper<std::vector<U>>> u_traj = std::nullopt,
569  std::optional<std::reference_wrapper<std::vector<G>>> x_traj = std::nullopt)
570  {
571  using std::chrono::nanoseconds;
572 
573  static constexpr int nx = G::SizeAtCompileTime;
574  static constexpr int nu = U::SizeAtCompileTime;
575  static constexpr int nU = K * nu;
576 
577  // update problem with funcitons defined in "MPC time"
578  ocp_.x0 = g;
579  ocp_.T = prm_.T;
580  ocp_.gdes = [this, &t](double t_loc) -> G {
581  return x_des_(t + duration_cast<nanoseconds>(std::chrono::duration<double>(t_loc)));
582  };
583  ocp_.udes = [this, &t](double t_loc) -> U {
584  return u_des_(t + duration_cast<nanoseconds>(std::chrono::duration<double>(t_loc)));
585  };
586 
587  // linearize around desired trajectory
588  if (!prm_.relinearize_input_around_sol || relinearize_around_desired_) { lin_.u = ocp_.udes; }
589  if (!prm_.relinearize_state_around_sol || relinearize_around_desired_) {
590  lin_.g = ocp_.gdes;
591  lin_.dg = [this](double t) -> typename G::Tangent {
592  // need numerical derivative here since gdes is not templated...
593  Eigen::Matrix<double, 1, 1> tvec(t);
594  auto [gv, dg] = diff::dr<diff::Type::NUMERICAL>(
595  [this](const auto & v) { return lin_.g(v.x()); }, wrt(tvec));
596  return dg;
597  };
598  }
599 
600  relinearize_around_desired_ = false;
601 
602  const double dt = ocp_.T / static_cast<double>(K);
603 
604  // define dynamics in "MPC time"
605  const auto dyn = [this, &t](double t_loc, const auto & vx, const auto & vu) {
606  return dyn_(t + duration_cast<nanoseconds>(std::chrono::duration<double>(t_loc)), vx, vu);
607  };
608 
609  const auto qp = ocp_to_qp<K, G, U, decltype(dyn), DiffType>(ocp_, dyn, lin_);
610  auto sol = solve_qp(qp, prm_.qp, warmstart_);
611 
612  for (auto i = 0u; i < prm_.iterative_relinearization; ++i) {
613  // check if solution touches linearization domain
614  bool touches = false;
615  for (int k = 0; !touches && k != K; ++k) {
616  // clang-format off
617  if (((1. - 1e-6) * lin_.u_domain - sol.primal.template segment<nu>( k * nu).cwiseAbs()).minCoeff() < 0) { touches = true; }
618  if (((1. - 1e-6) * lin_.g_domain - sol.primal.template segment<nx>(nU + k * nx).cwiseAbs()).minCoeff() < 0) { touches = true; }
619  // clang-format on
620  }
621  if (touches) {
622  // relinearize around solution and solve again
623  relinearize_state_around_sol(sol);
624  relinearize_input_around_sol(sol);
625 
626  const auto qp = ocp_to_qp<K, G, U, decltype(dyn), DiffType>(ocp_, dyn, lin_);
627  sol = solve_qp(qp, prm_.qp, warmstart_);
628  } else {
629  // solution seems fine
630  break;
631  }
632  }
633 
634  // output result
635  u = lin_.u(0) + sol.primal.template head<nu>();
636 
637  // output solution trajectories
638  if (u_traj.has_value()) {
639  u_traj.value().get().resize(K);
640  for (auto i = 0u; i < K; ++i) {
641  u_traj.value().get()[i] = lin_.u(i * dt) + sol.primal.template segment<nu>(i * nu);
642  }
643  }
644 
645  if (x_traj.has_value()) {
646  x_traj.value().get().resize(K + 1);
647  x_traj.value().get()[0] = ocp_.x0;
648  for (auto i = 1u; i < K + 1; ++i) {
649  x_traj.value().get()[i] =
650  lin_.g(i * dt) + sol.primal.template segment<nx>(nU + (i - 1) * nx);
651  }
652  }
653 
654  // update linearization for next iteration
655  if (sol.code == QPSolutionStatus::Optimal) {
656  if (prm_.relinearize_input_around_sol) relinearize_input_around_sol(sol);
657  if (prm_.relinearize_state_around_sol) relinearize_state_around_sol(sol);
658  }
659 
660  // save solution for warmstart
661  if (prm_.warmstart
662  && (sol.code == QPSolutionStatus::Optimal || sol.code == QPSolutionStatus::MaxTime
663  || sol.code == QPSolutionStatus::MaxIterations)) {
664  warmstart_ = sol; // store successful solution for later warmstart
665  }
666 
667  return sol.code;
668  }

758  void relinearize_state_around_sol(const QPSolution<-1, -1, double> & sol)
759  {
760  static constexpr int nx = G::SizeAtCompileTime;
761  static constexpr int nu = U::SizeAtCompileTime;
762  static constexpr int nU = K * nu;
763  const double dt = ocp_.T / static_cast<double>(K);
764 
765  // STATE LINEARIZATION
766  std::vector<double> tt(K + 1);
767  std::vector<G> gg(K + 1);
768  tt[0] = 0;
769  gg[0] = ocp_.x0;
770 
771  for (auto k = 1u; k != K + 1; ++k) {
772  tt[k] = dt * k;
773  gg[k] = lin_.g(k * dt) + sol.primal.template segment<nx>(nU + (k - 1) * nx);
774  }
775 
776  auto g_spline = smooth::fit_cubic_bezier(tt, gg);
777  lin_.g = [g_spline = g_spline](double t) -> G { return g_spline.eval(t); };
778  lin_.dg = [g_spline = std::move(g_spline)](double t) -> typename G::Tangent {
779  typename G::Tangent ret;
780  g_spline.eval(t, ret);
781  return ret;
782  };
783  }

788  void relinearize_input_around_sol(const QPSolution<-1, -1, double> & sol)
789  {
790  static constexpr int nu = U::SizeAtCompileTime;
791  const double dt = ocp_.T / static_cast<double>(K);
792 
793  if constexpr (LieGroup<U>) {
794  std::vector<double> tt(K);
795  std::vector<U> uu(K);
796 
797  for (auto k = 0u; k != K; ++k) {
798  tt[k] = k * dt;
799  uu[k] = lin_.u(k * dt) + sol.primal.template segment<nu>(k * nu);
800  }
801 
802  auto u_spline = smooth::fit_linear_bezier(tt, uu);
803  lin_.u = [u_spline = std::move(u_spline)](double t) -> U { return u_spline.eval(t); };
804  } else {
805  std::vector<double> tt(K);
806  std::vector<smooth::Tn<U::SizeAtCompileTime, double>> uu(K);
807 
808  for (auto k = 0u; k != K; ++k) {
809  tt[k] = k * dt;
