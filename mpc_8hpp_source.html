193 template<std::size_t K, LieGroup G, Manifold U, typename Dyn>
194 QuadraticProgramSparse<double> ocp_to_qp(const OptimalControlProblem<G, U> & pbm,
195  Dyn && f,
196  const LinearizationInfo<G, U> & lin = LinearizationInfo<G, U>{})
197 {
198  using std::placeholders::_1;
199 
200  // problem info
201  static constexpr int nx = G::SizeAtCompileTime;
202  static constexpr int nu = U::SizeAtCompileTime;
203 
204  static constexpr int nX = K * nx;
205  static constexpr int nU = K * nu;
206  static constexpr int nvar = nX + nU;
207 
208  static constexpr int n_eq = nX; // equality constraints from dynamics
209  static constexpr int n_u_iq = nU; // input bounds
210  static constexpr int n_x_iq = nX; // state bounds
211  static constexpr int ncon = n_eq + n_u_iq + n_x_iq;
212 
213  using AT = Eigen::Matrix<double, nx, nx>;
214  using BT = Eigen::Matrix<double, nx, nu>;
215  using ET = Eigen::Matrix<double, nx, 1>;
216 
217  const double dt = pbm.T / static_cast<double>(K);
218 
219  QuadraticProgramSparse ret;
220  ret.P.resize(nvar, nvar);
221  ret.q.resize(nvar);
222  ret.A.resize(ncon, nvar);
223  ret.l.resize(ncon);
224  ret.u.resize(ncon);
225 
226  // SET SPARSITY PATTERNS
227 
228  static constexpr auto Ap = mpc_nnz_pattern_A<K, nx, nu>();
229  static constexpr auto Pp = mpc_nnz_pattern_P<K, nx, nu>();
230  ret.A.reserve(Eigen::Map<const Eigen::Matrix<int, ncon, 1>>(Ap.data()));
231  ret.P.reserve(Eigen::Map<const Eigen::Matrix<int, nvar, 1>>(Pp.data()));
232 
233  // DYNAMICS CONSTRAINTS
234 
235  for (auto k = 0u; k != K; ++k) {
236  const double t = k * dt;
237 
238  // LINEARIZATION
239 
240  const auto xl = lin.g(t);
241  const auto dxl = lin.dg(t);
242  const auto ul = lin.u(t);
243 
244  const auto [flin, df_xu] = diff::dr(f, wrt(xl, ul));
245 
246  // cltv system \dot x = At x(t) + Bt u(t) + Et
247  const AT At = (-0.5 * G::ad(flin) - 0.5 * G::ad(dxl) + df_xu.template leftCols<nx>());
248  const BT Bt = df_xu.template rightCols<nu>();
249  const ET Et = flin - dxl;
250 
251  // TIME DISCRETIZATION
252 
253  const AT At2 = At * At;
254  const double dt2 = dt * dt;
255  const double dt3 = dt2 * dt;
256 
257  // dltv system x^+ = Ak x + Bk u + Ek by truncated taylor expansion of the matrix exponential
258  const AT Ak = AT::Identity() + At * dt + At2 * dt2 / 2. + At2 * At * dt3 / 6.;
259  const BT Bk = Bt * dt + At * Bt * dt2 / 2. + At2 * Bt * dt3 / 6.;
260  const ET Ek = Et * dt + At * Et * dt2 / 2. + At2 * Et * dt3 / 6.;
261 
262  // DYNAMICS CONSTRANTS
263 
264  if (k == 0) {
265  // x(1) - B u(0) = A x0 + E
266 
267  // identity matrix on x(1)
268  for (auto i = 0u; i != nx; ++i) { ret.A.insert(nx * k + i, nU + nx * k + i) = 1; }
269 
270  // B matrix on u(0)
271  for (auto i = 0u; i != nx; ++i) {
272  for (auto j = 0u; j != nu; ++j) { ret.A.insert(nx * k + i, nu * k + j) = -Bk(i, j); }
273  }
274 
275  ret.u.template segment<nx>(nx * k) = Ak * (pbm.x0 - lin.g(t)) + Ek;
276  ret.l.template segment<nx>(nx * k) = ret.u.template segment<nx>(nx * k);
277 
278  } else {
279  // x(k+1) - A x(k) - B u(k) = E
280 
281  // identity matrix on x(k+1)
282  for (auto i = 0u; i != nx; ++i) { ret.A.insert(nx * k + i, nU + nx * k + i) = 1; }
283 
284  // A matrix on x(k)
285  for (auto i = 0u; i != nx; ++i) {
286  for (auto j = 0u; j != nx; ++j) {
287  ret.A.insert(nx * k + i, nU + nx * (k - 1) + j) = -Ak(i, j);
288  }
289  }
290 
291  // B matrix on u(k)
292  for (auto i = 0u; i != nx; ++i) {
293  for (auto j = 0u; j != nu; ++j) { ret.A.insert(nx * k + i, nu * k + j) = -Bk(i, j); }
294  }
295 
296  ret.u.template segment<nx>(nx * k) = Ek;
297  ret.l.template segment<nx>(nx * k) = Ek;
298  }
299  }
300 
301  // INPUT CONSTRAINTS
302 
303  for (auto i = 0u; i != n_u_iq; ++i) { ret.A.insert(n_eq + i, i) = 1; }
304 
305  if (pbm.umin) {
306  for (auto k = 0u; k < K; ++k) {
307  ret.l.template segment<nu>(n_eq + k * nu) = pbm.umin.value() - lin.u(k * dt);
308  }
309  } else {
310  ret.l.template segment<nU>(n_eq).setConstant(-std::numeric_limits<double>::infinity());
311  }
312  if (pbm.umax) {
313  for (auto k = 0u; k < K; ++k) {
314  ret.u.template segment<nu>(n_eq + k * nu) = pbm.umax.value() - lin.u(k * dt);
315  }
316  } else {
317  ret.u.template segment<nU>(n_eq).setConstant(std::numeric_limits<double>::infinity());
318  }
319 
320  // STATE CONSTRAINTS
321 
322  for (auto i = 0u; i != n_x_iq; ++i) { ret.A.insert(n_eq + n_u_iq + i, nU + i) = 1; }
323 
324  if (pbm.gmin) {
325  for (auto k = 1u; k < K + 1; ++k) {
326  ret.l.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) = pbm.gmin.value() - lin.g(k * dt);
327  }
328  } else {
329  ret.l.template segment<nX>(n_eq + n_u_iq).setConstant(-std::numeric_limits<double>::infinity());
330  }
331  if (pbm.gmax) {
332  for (auto k = 1u; k < K + 1; ++k) {
333  ret.u.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) = pbm.gmax.value() - lin.g(k * dt);
334  }
335  } else {
336  ret.u.template segment<nX>(n_eq + n_u_iq).setConstant(std::numeric_limits<double>::infinity());
337  }
338 
339  // INPUT COSTS
340 
341  for (auto k = 0u; k < K; ++k) {
342  for (auto i = 0u; i != nu; ++i) {
343  for (auto j = 0u; j != nu; ++j) { ret.P.insert(k * nu + i, k * nu + j) = pbm.R(i, j) * dt; }
344  }
345  ret.q.template segment<nu>(k * nu) = pbm.R * (lin.u(k * dt) - pbm.udes(k * dt)) * dt;
346  }
347 
348  // STATE COSTS
349 
350  // intermediate states x(1) ... x(K-1)
351  for (auto k = 1u; k < K; ++k) {
352  for (auto i = 0u; i != nx; ++i) {
353  for (auto j = 0u; j != nx; ++j) {
354  ret.P.insert(nU + (k - 1) * nx + i, nU + (k - 1) * nx + j) = pbm.Q(i, j) * dt;
355  }
356  }
357  ret.q.template segment<nx>(nU + (k - 1) * nx) = pbm.Q * (lin.g(k * dt) - pbm.gdes(k * dt)) * dt;
358  }
359 
360  // last state x(K) ~ x(T)
361  for (auto i = 0u; i != nx; ++i) {
362  for (auto j = 0u; j != nx; ++j) {
363  ret.P.insert(nU + (K - 1) * nx + i, nU + (K - 1) * nx + j) = pbm.QT(i, j);
364  }
365  }
366  ret.q.template segment<nx>(nU + (K - 1) * nx) = pbm.QT * (lin.g(pbm.T) - pbm.gdes(pbm.T));
367 
368  ret.A.makeCompressed();
369  ret.P.makeCompressed();
370 
371  return ret;
372 }

386 template<std::size_t K, typename T, LieGroup G, Manifold U, typename Dyn>
387 class MPC
388 {
389 public:
397  MPC(Dyn && f, T t, const SolverParams & qp_prm = SolverParams{})
398  : dyn_(std::forward<Dyn>(f)), qp_prm_(qp_prm)
399  {
400  ocp_.T = std::chrono::duration_cast<std::chrono::duration<double>>(t).count();
401  }
403  MPC(const Dyn & f, T t, const SolverParams & qp_prm = SolverParams{}) : MPC(Dyn(f), t, qp_prm) {}
405  MPC() = default;
407  MPC(const MPC &) = default;
409  MPC(MPC &&) = default;
411  MPC & operator=(const MPC &) = default;
413  MPC & operator=(MPC &&) = default;
415  ~MPC() = default;
416 
425  ExitCode operator()(U & u,
426  const T & t,
427  const G & g,
428  std::optional<std::reference_wrapper<std::vector<G>>> x_traj = std::nullopt)
429  {
430  static constexpr int nx = G::SizeAtCompileTime;
431  static constexpr int nu = U::SizeAtCompileTime;
432  static constexpr int nU = K * nu;
433 
434  ocp_.x0 = g;
435  ocp_.gdes = [this, &t](double t_loc) {
436  return x_des_(t + duration_cast<T>(std::chrono::duration<double>(t_loc)));
437  };
438  ocp_.udes = [this, &t](double t_loc) {
439  return u_des_(t + duration_cast<T>(std::chrono::duration<double>(t_loc)));
440  };
441 
442  const auto qp = smooth::feedback::ocp_to_qp<K>(ocp_, dyn_, lin_);
443  const auto sol = smooth::feedback::solve_qp(qp, qp_prm_);
444 
445  u = lin_.u(0) + sol.primal.template head<nu>();
446 
447  if (x_traj.has_value()) {
448  const double dt = ocp_.T / static_cast<double>(K);
449  x_traj.value().get().resize(K + 1);
450  x_traj.value().get()[0] = ocp_.x0;
451  for (auto i = 1; i < K + 1; ++i) {
452  x_traj.value().get()[i] =
453  lin_.g(i * dt) + sol.primal.template segment<nx>(nU + (i - 1) * nx);
454  }
455  }
456  return sol.code;
457 
458  // TODO: update linearization point here
459  }
