46template<Time T, LieGroup X>
47struct XDes
48{
49 T t0;
50 std::function<X(T)> xdes = [](T) -> X { return Default<X>(); };
51 std::function<Tangent<X>(T)> dxdes = [](T) -> Tangent<X> { return Tangent<X>::Zero(); };
52
53 X operator()(const double t_rel) const
54 {
55 const T t_abs = time_trait<T>::plus(t0, t_rel);
56 return xdes(t_abs);
57 };
58
59 Tangent<X> jacobian(const double t_rel) const
60 {
61 const T t_abs = time_trait<T>::plus(t0, t_rel);
62 return dxdes(t_abs);
63 }
64};

69template<Time T, Manifold U>
70struct UDes
71{
72 T t0;
73 std::function<U(T)> udes = [](T) -> U { return Default<U>(); };
74
75 U operator()(const double t_rel) const
76 {
77 const T t_abs = time_trait<T>::plus(t0, t_rel);
78 return udes(t_abs);
79 };

92template<LieGroup X>
93struct MPCObj
94{
95 static constexpr auto Nx = Dof<X>;
96
97 // public members
98 X xf_des = Default<X>();
99 Eigen::Matrix<double, Nx, Nx> Qtf = Eigen::Matrix<double, Nx, Nx>::Identity();
100
101 // private members
102 Eigen::SparseMatrix<double> hess{1 + 2 * Nx + 1, 1 + 2 * Nx + 1};
103
104 // functor members
105
106 // function f(t, x0, xf, q) = (1/2) xf' Q xf + q_0
107 double operator()(
108 const double,
109 const X &,
110 [[maybe_unused]] const X & xf,
111 [[maybe_unused]] const Eigen::Vector<double, 1> & q) const
112 {
113 assert(q(0) == 1.);
114 assert(xf_des.isApprox(xf, 1e-4));
115 if constexpr (false) {
116 const auto e = rminus(xf, xf_des);
117 return 0.5 * e.dot(Qtf * e) + q(0);
118 } else {
119 return 1.;
120 }
121 }
122
123 Eigen::RowVector<double, 1 + 2 * Nx + 1>
124 jacobian(const double, const X &, [[maybe_unused]] const X & xf, const Eigen::Vector<double, 1> &)
125 {
126 assert(xf_des.isApprox(xf, 1e-4));
127 return Eigen::RowVector<double, 1 + 2 * Nx + 1>::Unit(1 + 2 * Nx);
128 }
129
130 std::reference_wrapper<const Eigen::SparseMatrix<double>>
131 hessian(const double, const X &, [[maybe_unused]] const X & xf, const Eigen::Vector<double, 1> &)
132 {
133 assert(xf_des.isApprox(xf, 1e-4));
134
135 set_zero(hess);
136
137 for (auto i = 0u; i < Nx; ++i) {
138 for (auto j = 0u; j < Nx; ++j) {
139 if (Qtf(i, j) != 0) { hess.coeffRef(1 + i, 1 + j) = Qtf(i, j); }
140 }
141 }
142
143 hess.makeCompressed();
144 return hess;
145 }
146};

151template<Time T, LieGroup X, Manifold U, typename F, diff::Type DT>
152struct MPCDyn
153{
154 static constexpr auto Nx = Dof<X>;
155 static constexpr auto Nu = Dof<U>;
156
157 // public members
158 F f;
159 T t0{};
160
161 // functor members
162 Tangent<X> operator()(const double t, const X & x, const U & u)
163 {
164 if constexpr (requires(F & fvar, T tvar) { fvar.set_time(tvar); }) {
165 f.set_time(time_trait<T>::plus(t0, t));
166 }
167
168 return f(x, u);
169 }
170
171 Eigen::Matrix<double, Nx, 1 + Nx + Nu> jacobian(const double t, const X & x, const U & u)
172 {
173 if constexpr (requires(F & fvar, T tvar) { fvar.set_time(tvar); }) {
174 f.set_time(time_trait<T>::plus(t0, t));
175 }
176
177 const auto & [fval, df] = diff::dr<1, DT>(f, smooth::wrt(x, u));
178
179 Eigen::Matrix<double, Nx, 1 + Nx + Nu> ret;
180 ret << Eigen::Vector<double, Nx>::Zero(), df;
181 return ret;
182 }

196template<Time T, LieGroup X, Manifold U>
197struct MPCIntegrand
198{
199 static constexpr auto Nx = Dof<X>;
200 static constexpr auto Nu = Dof<U>;
201
202 // public members
203 std::shared_ptr<XDes<T, X>> xdes;
204 std::shared_ptr<UDes<T, U>> udes;
205
206 Eigen::Matrix<double, Nx, Nx> Q = Eigen::Matrix<double, Nx, Nx>::Identity();
207 Eigen::Matrix<double, Nu, Nu> R = Eigen::Matrix<double, Nu, Nu>::Identity();
208
209 // private members
210 Eigen::SparseMatrix<double> hess{1 + Nx + Nu, 1 + Nx + Nu};
211
212 // functor members
213
214 // function f(x, t, u) = (1/2) * (x' Q x + u' R u)
215 Eigen::Vector<double, 1> operator()(
216 [[maybe_unused]] const double t_rel,
217 [[maybe_unused]] const X & x,
218 [[maybe_unused]] const U & u) const
219 {
220 assert((*xdes)(t_rel).isApprox(x, 1e-4));
221 assert((*udes)(t_rel).isApprox(u, 1e-4));
222 if constexpr (false) {
223 const auto ex = rminus(x, (*xdes)(t_rel));
224 const auto eu = rminus(u, (*udes)(t_rel));
225 return Eigen::Vector<double, 1>{0.5 * ex.dot(Q * ex) + 0.5 * eu.dot(R * eu)};
226 } else {
227 return Eigen::Vector<double, 1>{0.};
228 }
229 }
230
231 Eigen::RowVector<double, 1 + Nx + Nu> jacobian(
232 [[maybe_unused]] const double t_rel, [[maybe_unused]] const X & x, [[maybe_unused]] const U & u)
233 {
234 assert((*xdes)(t_rel).isApprox(x, 1e-4));
235 assert((*udes)(t_rel).isApprox(u, 1e-4));
236 return Eigen::RowVector<double, 1 + Nx + Nu>::Zero();
237 }
238
239 std::reference_wrapper<const Eigen::SparseMatrix<double>> hessian(
240 [[maybe_unused]] const double t_rel, [[maybe_unused]] const X & x, [[maybe_unused]] const U & u)
241 {
242 assert((*xdes)(t_rel).isApprox(x, 1e-4));
243 assert((*udes)(t_rel).isApprox(u, 1e-4));
244
245 set_zero(hess);
246
247 for (auto i = 0u; i < Nx; ++i) {
248 for (auto j = 0u; j < Nx; ++j) {
249 if (Q(i, j) != 0) { hess.coeffRef(1 + i, 1 + j) = Q(i, j); }
250 }
251 }
252 for (auto i = 0u; i < Nu; ++i) {
253 for (auto j = 0u; j < Nu; ++j) {
254 if (Q(i, j) != 0) { hess.coeffRef(1 + Nx + i, 1 + Nx + j) = R(i, j); }
255 }
256 }
257
258 hess.makeCompressed();
259 return hess;
260 }
261};

266template<Time T, LieGroup X, Manifold U, typename F, diff::Type DT>
267struct MPCCR
268{
269 static constexpr auto Nx = Dof<X>;
270 static constexpr auto Nu = Dof<U>;
271 static constexpr auto Ncr = std::invoke_result_t<F, X, U>::SizeAtCompileTime;
272
273 // public members
274 F f;
275 T t0{};
276
277 // private members
278 Eigen::SparseMatrix<double> jac{Ncr, 1 + Nx + Nu};
279
280 // functor members
281 Eigen::Vector<double, Ncr> operator()(const double t, const X & x, const U & u)
282 {
283 if constexpr (requires(F & fvar, T tvar) { fvar.set_time(tvar); }) {
284 f.set_time(time_trait<T>::plus(t0, t));
285 }
286
287 return f(x, u);
288 }
289
290 std::reference_wrapper<const Eigen::SparseMatrix<double>>
291 jacobian(const double t, const X & x, const U & u)
292 {
293 if constexpr (requires(F & fvar, T tvar) { fvar.set_time(tvar); }) {
294 f.set_time(time_trait<T>::plus(t0, t));
295 }
296
297 const auto & [fval, df] = diff::dr<1, DT>(f, smooth::wrt(x, u));
298 block_write(jac, 0, 1, df);
299
300 jac.makeCompressed();
301 return jac;
302 }

313template<LieGroup X>
314struct MPCCE
315{
316 static constexpr auto Nx = Dof<X>;
317
318 // public members
319 X x0_fix = Default<X>();
320
321 // private members
322 Eigen::SparseMatrix<double> dexpinv_tmp = d_exp_sparse_pattern<X>;
323 Eigen::SparseMatrix<double> jac{Nx, 1 + 2 * Nx + 1};
324
325 // functor members
326 Tangent<X>
327 operator()(const double, const X & x0, const X &, const Eigen::Vector<double, 1> &) const
328 {
329 return rminus(x0, x0_fix);
330 }
331
332 std::reference_wrapper<const Eigen::SparseMatrix<double>>
333 jacobian(const double, const X & x0, const X &, const Eigen::Vector<double, 1> &)
334 {
335 dr_expinv_sparse<X>(dexpinv_tmp, rminus(x0, x0_fix));
336
337 jac.middleCols(1, Nx) = dexpinv_tmp;
338 jac.makeCompressed();
339 return jac;
340 }

419class MPC
420{
421 static constexpr auto Ncr = std::invoke_result_t<CR, X, U>::SizeAtCompileTime;
422
423public:
444 inline MPC(
445 F && f,
446 CR && cr,
447 Eigen::Vector<double, Ncr> && crl,
448 Eigen::Vector<double, Ncr> && cru,
449 MPCParams && prm = {})
450 : xdes_{std::make_shared<detail::XDes<T, X>>()},
451 udes_{std::make_shared<detail::UDes<T, U>>()}, mesh_{(prm.K + Kmesh - 1) / Kmesh},
452 ocp_{
453 .theta = {},
454 .f = {.f = std::forward<F>(f)},
455 .g = {.xdes = xdes_, .udes = udes_},
456 .cr = {.f = std::forward<CR>(cr)},
457 .crl = std::move(crl),
458 .cru = std::move(cru),
459 .ce = {},
460 .cel = Eigen::Vector<double, Dof<X>>::Zero(),
461 .ceu = Eigen::Vector<double, Dof<X>>::Zero(),
462 },
463 prm_{std::move(prm)}, qp_solver_{prm_.qp}
464 {
465 detail::ocp_to_qp_allocate<DT>(qp_, work_, ocp_, mesh_);
466 ocp_to_qp_update<diff::Type::Analytic>(qp_, work_, ocp_, mesh_, prm_.tf, *xdes_, *udes_);
467 qp_solver_.analyze(qp_);
468 }

506 inline std::pair<U, QPSolutionStatus> operator()(
507 const T & t,
508 const X & x,
509 std::optional<std::reference_wrapper<std::vector<U>>> u_traj = std::nullopt,
510 std::optional<std::reference_wrapper<std::vector<X>>> x_traj = std::nullopt)
511 {
512 static constexpr auto Nx = Dof<X>;
513 static constexpr auto Nu = Dof<U>;
514
515 const auto N = mesh_.N_colloc();
516 const auto xvar_L = Nx * (N + 1);
517 const auto xvar_B = 0u;
518 const auto uvar_B = xvar_L;
519
520 // update problem
521 xdes_->t0 = t;
522 udes_->t0 = t;
523 ocp_.theta.xf_des = (*xdes_)(prm_.tf);
524 ocp_.f.t0 = t;
525 ocp_.cr.t0 = t;
526 ocp_.ce.x0_fix = x;
527
528 // transcribe to QP
529 ocp_to_qp_update_dyn<diff::Type::Analytic>(qp_, work_, ocp_, mesh_, prm_.tf, *xdes_, *udes_);
530 if constexpr (requires(CR & crvar, T tvar) { crvar.set_time(tvar); }) {
531 // only update if running constraints are time-dependent
532 ocp_to_qp_update_cr<diff::Type::Analytic>(qp_, work_, ocp_, mesh_, prm_.tf, *xdes_, *udes_);
533 }
534 ocp_to_qp_update_ce<diff::Type::Analytic>(qp_, work_, ocp_, mesh_, prm_.tf, *xdes_, *udes_);
535 qp_.A.makeCompressed();
536 qp_.P.makeCompressed();
537
538 // solve QP
539 const auto & sol = qp_solver_.solve(qp_, warmstart_);
540
541 // output solution trajectories
542 if (u_traj.has_value()) {
543 u_traj.value().get().resize(N);
544 for (const auto & [i, tau] : zip(std::views::iota(0u, N), mesh_.all_nodes())) {
545 const double t_rel = prm_.tf * tau;
546 u_traj.value().get()[i] =
547 (*udes_)(t_rel) + sol.primal.template segment<Nu>(uvar_B + i * Nu);
548 }
549 }
550 if (x_traj.has_value()) {
551 x_traj.value().get().resize(N + 1);
552 for (const auto & [i, tau] : zip(std::views::iota(0u, N + 1), mesh_.all_nodes())) {
553 const double t_rel = prm_.tf * tau;
554 x_traj.value().get()[i] =
555 (*xdes_)(t_rel) + sol.primal.template segment<Nx>(xvar_B + i * Nx);
556 }
557 }
558
559 // save solution to warmstart next iteration
560 if (prm_.warmstart) {
561 // clang-format off
562 if (sol.code == QPSolutionStatus::Optimal || sol.code == QPSolutionStatus::MaxTime || sol.code == QPSolutionStatus::MaxIterations) {
563 warmstart_ = sol;
564 }
565 // clang-format on
566 }
567
568 return {rplus((*udes_)(0), sol.primal.template segment<Nu>(uvar_B)), sol.code};
569 }
