222 template<std::size_t K, LieGroup G, Manifold U, typename Dyn>
223 QuadraticProgramSparse<double> ocp_to_qp(const OptimalControlProblem<G, U> & pbm,
224  Dyn && f,
225  const LinearizationInfo<G, U> & lin = LinearizationInfo<G, U>{})
226 {
227  using std::placeholders::_1;
228 
229  // problem info
230  static constexpr int nx = G::SizeAtCompileTime;
231  static constexpr int nu = U::SizeAtCompileTime;
232 
233  static constexpr int nX = K * nx;
234  static constexpr int nU = K * nu;
235  static constexpr int nvar = nX + nU;
236 
237  static constexpr int n_eq = nX; // equality constraints from dynamics
238  static constexpr int n_u_iq = nU; // input bounds
239  static constexpr int n_x_iq = nX; // state bounds
240  static constexpr int ncon = n_eq + n_u_iq + n_x_iq;
241 
242  using AT = Eigen::Matrix<double, nx, nx>;
243  using BT = Eigen::Matrix<double, nx, nu>;
244  using ET = Eigen::Matrix<double, nx, 1>;
245 
246  const double dt = pbm.T / static_cast<double>(K);
247 
248  QuadraticProgramSparse ret;
249  ret.P.resize(nvar, nvar);
250  ret.q.resize(nvar);
251  ret.A.resize(ncon, nvar);
252  ret.l.resize(ncon);
253  ret.u.resize(ncon);
254 
255  // SET SPARSITY PATTERNS
256 
257  static constexpr auto Ap = mpc_nnz_pattern_A<K, nx, nu>();
258  static constexpr auto Pp = mpc_nnz_pattern_P<K, nx, nu>();
259  ret.A.reserve(Eigen::Map<const Eigen::Matrix<int, ncon, 1>>(Ap.data()));
260  ret.P.reserve(Eigen::Map<const Eigen::Matrix<int, nvar, 1>>(Pp.data()));
261 
262  // DYNAMICS CONSTRAINTS
263 
264  for (auto k = 0u; k != K; ++k) {
265  const double t = k * dt;
266 
267  // LINEARIZATION
268 
269  const auto xl = lin.g(t);
270  const auto dxl = lin.dg(t);
271  const auto ul = lin.u(t);
272 
273  const auto [flin, df_xu] = diff::dr(f, wrt(xl, ul));
274 
275  // cltv system \dot x = At x(t) + Bt u(t) + Et
276  const AT At = (-0.5 * G::ad(flin) - 0.5 * G::ad(dxl) + df_xu.template leftCols<nx>());
277  const BT Bt = df_xu.template rightCols<nu>();
278  const ET Et = flin - dxl;
279 
280  // TIME DISCRETIZATION
281 
282  const AT At2 = At * At;
283  const double dt2 = dt * dt;
284  const double dt3 = dt2 * dt;
285 
286  // dltv system x^+ = Ak x + Bk u + Ek by truncated taylor expansion of the matrix exponential
287  const AT Ak = AT::Identity() + At * dt + At2 * dt2 / 2. + At2 * At * dt3 / 6.;
288  const BT Bk = Bt * dt + At * Bt * dt2 / 2. + At2 * Bt * dt3 / 6.;
289  const ET Ek = Et * dt + At * Et * dt2 / 2. + At2 * Et * dt3 / 6.;
290 
291  // DYNAMICS CONSTRANTS
292 
293  if (k == 0) {
294  // x(1) - B u(0) = A x0 + E
295 
296  // identity matrix on x(1)
297  for (auto i = 0u; i != nx; ++i) { ret.A.insert(nx * k + i, nU + nx * k + i) = 1; }
298 
299  // B matrix on u(0)
300  for (auto i = 0u; i != nx; ++i) {
301  for (auto j = 0u; j != nu; ++j) { ret.A.insert(nx * k + i, nu * k + j) = -Bk(i, j); }
302  }
303 
304  ret.u.template segment<nx>(nx * k) = Ak * (pbm.x0 - lin.g(t)) + Ek;
305  ret.l.template segment<nx>(nx * k) = ret.u.template segment<nx>(nx * k);
306 
307  } else {
308  // x(k+1) - A x(k) - B u(k) = E
309 
310  // identity matrix on x(k+1)
311  for (auto i = 0u; i != nx; ++i) { ret.A.insert(nx * k + i, nU + nx * k + i) = 1; }
312 
313  // A matrix on x(k)
314  for (auto i = 0u; i != nx; ++i) {
315  for (auto j = 0u; j != nx; ++j) {
316  ret.A.insert(nx * k + i, nU + nx * (k - 1) + j) = -Ak(i, j);
317  }
318  }
319 
320  // B matrix on u(k)
321  for (auto i = 0u; i != nx; ++i) {
322  for (auto j = 0u; j != nu; ++j) { ret.A.insert(nx * k + i, nu * k + j) = -Bk(i, j); }
323  }
324 
325  ret.u.template segment<nx>(nx * k) = Ek;
326  ret.l.template segment<nx>(nx * k) = Ek;
327  }
328  }
329 
330  // INPUT CONSTRAINTS
331 
332  for (auto i = 0u; i != n_u_iq; ++i) { ret.A.insert(n_eq + i, i) = 1; }
333 
334  if (pbm.umin) {
335  for (auto k = 0u; k < K; ++k) {
336  ret.l.template segment<nu>(n_eq + k * nu) = pbm.umin.value() - lin.u(k * dt);
337  }
338  } else {
339  ret.l.template segment<nU>(n_eq).setConstant(-std::numeric_limits<double>::infinity());
340  }
341  if (pbm.umax) {
342  for (auto k = 0u; k < K; ++k) {
343  ret.u.template segment<nu>(n_eq + k * nu) = pbm.umax.value() - lin.u(k * dt);
344  }
345  } else {
346  ret.u.template segment<nU>(n_eq).setConstant(std::numeric_limits<double>::infinity());
347  }
348 
349  // linearization bounds
350  for (auto k = 0u; k < K; ++k) {
351  ret.l.template segment<nu>(n_eq + k * nu) =
352  ret.l.template segment<nu>(n_eq + k * nu).cwiseMax(-lin.u_domain);
353  ret.u.template segment<nu>(n_eq + k * nu) =
354  ret.u.template segment<nu>(n_eq + k * nu).cwiseMin(lin.u_domain);
355  }
356 
357  // STATE CONSTRAINTS
358 
359  for (auto i = 0u; i != n_x_iq; ++i) { ret.A.insert(n_eq + n_u_iq + i, nU + i) = 1; }
360 
361  if (pbm.gmin) {
362  for (auto k = 1u; k < K + 1; ++k) {
363  ret.l.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) = pbm.gmin.value() - lin.g(k * dt);
364  }
365  } else {
366  ret.l.template segment<nX>(n_eq + n_u_iq).setConstant(-std::numeric_limits<double>::infinity());
367  }
368  if (pbm.gmax) {
369  for (auto k = 1u; k < K + 1; ++k) {
370  ret.u.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) = pbm.gmax.value() - lin.g(k * dt);
371  }
372  } else {
373  ret.u.template segment<nX>(n_eq + n_u_iq).setConstant(std::numeric_limits<double>::infinity());
374  }
375 
376  // linearization bounds
377  for (auto k = 1u; k < K + 1; ++k) {
378  ret.l.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) =
379  ret.l.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx).cwiseMax(-lin.g_domain);
380  ret.u.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx) =
381  ret.u.template segment<nx>(n_eq + n_u_iq + (k - 1) * nx).cwiseMin(lin.g_domain);
382  }
383 
384  // INPUT COSTS
385 
386  for (auto k = 0u; k < K; ++k) {
387  for (auto i = 0u; i != nu; ++i) {
388  for (auto j = 0u; j != nu; ++j) { ret.P.insert(k * nu + i, k * nu + j) = pbm.R(i, j) * dt; }
389  }
390  ret.q.template segment<nu>(k * nu) = pbm.R * (lin.u(k * dt) - pbm.udes(k * dt)) * dt;
391  }
392 
393  // STATE COSTS
394 
395  // intermediate states x(1) ... x(K-1)
396  for (auto k = 1u; k < K; ++k) {
397  for (auto i = 0u; i != nx; ++i) {
398  for (auto j = 0u; j != nx; ++j) {
399  ret.P.insert(nU + (k - 1) * nx + i, nU + (k - 1) * nx + j) = pbm.Q(i, j) * dt;
400  }
401  }
402  ret.q.template segment<nx>(nU + (k - 1) * nx) = pbm.Q * (lin.g(k * dt) - pbm.gdes(k * dt)) * dt;
403  }
404 
405  // last state x(K) ~ x(T)
406  for (auto i = 0u; i != nx; ++i) {
407  for (auto j = 0u; j != nx; ++j) {
408  ret.P.insert(nU + (K - 1) * nx + i, nU + (K - 1) * nx + j) = pbm.QT(i, j);
409  }
410  }
411  ret.q.template segment<nx>(nU + (K - 1) * nx) = pbm.QT * (lin.g(pbm.T) - pbm.gdes(pbm.T));
412 
413  ret.A.makeCompressed();
414  ret.P.makeCompressed();
415 
416  return ret;
417 }

480 template<std::size_t K, typename T, LieGroup G, Manifold U, typename Dyn>
481 class MPC
482 {
483 public:
491  MPC(Dyn && f, const MPCParams & prm) : prm_(prm), dyn_(std::forward<Dyn>(f)) {}
492 
494  MPC(const Dyn & f, const MPCParams & prm) : MPC(Dyn(f), prm) {}
496  MPC() = default;
498  MPC(const MPC &) = default;
500  MPC(MPC &&) = default;
502  MPC & operator=(const MPC &) = default;
504  MPC & operator=(MPC &&) = default;
506  ~MPC() = default;

528  ExitCode operator()(U & u,
529  const T & t,
530  const G & g,
531  std::optional<std::reference_wrapper<std::vector<U>>> u_traj = std::nullopt,
532  std::optional<std::reference_wrapper<std::vector<G>>> x_traj = std::nullopt)
533  {
534  using std::chrono::nanoseconds;
535 
536  static constexpr int nx = G::SizeAtCompileTime;
537  static constexpr int nu = U::SizeAtCompileTime;
538  static constexpr int nU = K * nu;
539 
540  // update problem
541  ocp_.x0 = g;
542  ocp_.T = prm_.T;
543  ocp_.gdes = [this, &t](double t_loc) -> G {
544  return x_des_(t + duration_cast<nanoseconds>(std::chrono::duration<double>(t_loc)));
545  };
546  ocp_.udes = [this, &t](double t_loc) -> U {
547  return u_des_(t + duration_cast<nanoseconds>(std::chrono::duration<double>(t_loc)));
548  };
549 
550  // linearize around desired trajectory
551  if (!prm_.relinearize_input_around_sol || relinearize_around_desired_) { lin_.u = ocp_.udes; }
552  if (!prm_.relinearize_state_around_sol || relinearize_around_desired_) {
553  lin_.g = ocp_.gdes;
554  lin_.dg = [this](double t) -> typename G::Tangent {
555  // need numerical derivative here since gdes is not templated...
556  Eigen::Matrix<double, 1, 1> tvec(t);
557  auto [gv, dg] = diff::dr<diff::Type::NUMERICAL>(
558  [this](const auto & v) { return lin_.g(v.x()); }, wrt(tvec));
559  return dg;
560  };
561  }
562 
563  relinearize_around_desired_ = false;
564 
565  const double dt = ocp_.T / static_cast<double>(K);
566 
567  const auto qp = smooth::feedback::ocp_to_qp<K>(ocp_, dyn_, lin_);
568 
569 #ifdef SMOOTH__FEEDBACK__COMPAT__OSQP_HPP_
570 #pragma message("Using OSQP qp solver in MPC")
571  auto sol = smooth::feedback::solve_qp_osqp(qp, prm_.qp, warmstart_);
572 #else
573  auto sol = smooth::feedback::solve_qp(qp, prm_.qp, warmstart_);
574 #endif
575 
576  for (auto i = 0u; i < prm_.iterative_relinearization; ++i) {
577  // check if solution touches linearization domain
578  bool touches = false;
579  for (int k = 0; !touches && k != K; ++k) {
580  // clang-format off
581  if (((1. - 1e-6) * lin_.u_domain - sol.primal.template segment<nu>( k * nu).cwiseAbs()).minCoeff() < 0) { touches = true; }
582  if (((1. - 1e-6) * lin_.g_domain - sol.primal.template segment<nx>(nU + k * nx).cwiseAbs()).minCoeff() < 0) { touches = true; }
583  // clang-format on
584  }
585  if (touches) {
586  // relinearize around solution and solve again
587  relinearize_state_around_sol(sol);
588  relinearize_input_around_sol(sol);
589 
590  const auto qp = smooth::feedback::ocp_to_qp<K>(ocp_, dyn_, lin_);
591 #ifdef SMOOTH__FEEDBACK__COMPAT__OSQP_HPP_
592  sol = smooth::feedback::solve_qp_osqp(qp, prm_.qp, warmstart_);
593 #else
594  sol = smooth::feedback::solve_qp(qp, prm_.qp, warmstart_);
595 #endif
596  } else {
597  // solution seems fine
598  break;
599  }
600  }
601 
602  // output result
603  u = lin_.u(0) + sol.primal.template head<nu>();
604 
605  // output solution trajectories
606  if (u_traj.has_value()) {
607  u_traj.value().get().resize(K);
608  for (auto i = 0u; i < K; ++i) {
609  u_traj.value().get()[i] = lin_.u(i * dt) + sol.primal.template segment<nu>(i * nu);
610  }
611  }
612 
613  if (x_traj.has_value()) {
614  x_traj.value().get().resize(K + 1);
615  x_traj.value().get()[0] = ocp_.x0;
616  for (auto i = 1u; i < K + 1; ++i) {
617  x_traj.value().get()[i] =
618  lin_.g(i * dt) + sol.primal.template segment<nx>(nU + (i - 1) * nx);
619  }
620  }
621 
622  // update linearization for next iteration
623  if (sol.code == ExitCode::Optimal) {
624  if (prm_.relinearize_input_around_sol) relinearize_input_around_sol(sol);
625  if (prm_.relinearize_state_around_sol) relinearize_state_around_sol(sol);
626  }
627 
628  // save solution for warmstart
629  if (prm_.warmstart
630  && (sol.code == ExitCode::Optimal || sol.code == ExitCode::MaxTime
631  || sol.code == ExitCode::MaxIterations)) {
632  warmstart_ = sol; // store successful solution for later warmstart
633  }
634 
635  return sol.code;
636  }

725  void relinearize_state_around_sol(const Solution<-1, -1, double> & sol)
726  {
727  static constexpr int nx = G::SizeAtCompileTime;
728  static constexpr int nu = U::SizeAtCompileTime;
729  static constexpr int nU = K * nu;
730  const double dt = ocp_.T / static_cast<double>(K);
731 
732  // STATE LINEARIZATION
733  std::vector<double> tt(K + 1);
734  std::vector<G> gg(K + 1);
735  tt[0] = 0;
736  gg[0] = ocp_.x0;
737 
738  for (auto k = 1u; k != K + 1; ++k) {
739  tt[k] = dt * k;
740  gg[k] = lin_.g(k * dt) + sol.primal.template segment<nx>(nU + (k - 1) * nx);
741  }
742 
743  auto g_spline = smooth::fit_cubic_bezier(tt, gg);
744  lin_.g = [g_spline = g_spline](double t) -> G { return g_spline.eval(t); };
745  lin_.dg = [g_spline = std::move(g_spline)](double t) -> typename G::Tangent {
746  typename G::Tangent ret;
747  g_spline.eval(t, ret);
748  return ret;
749  };
750  }

755  void relinearize_input_around_sol(const Solution<-1, -1, double> & sol)
756  {
757  static constexpr int nu = U::SizeAtCompileTime;
758  const double dt = ocp_.T / static_cast<double>(K);
759 
760  if constexpr (LieGroup<U>) {
761  std::vector<double> tt(K);
762  std::vector<U> uu(K);
763 
764  for (auto k = 0u; k != K; ++k) {
765  tt[k] = k * dt;
766  uu[k] = lin_.u(k * dt) + sol.primal.template segment<nu>(k * nu);
767  }
768 
769  auto u_spline = smooth::fit_linear_bezier(tt, uu);
770  lin_.u = [u_spline = std::move(u_spline)](double t) -> U { return u_spline.eval(t); };
771  } else {
772  std::vector<double> tt(K);
773  std::vector<smooth::Tn<U::SizeAtCompileTime, double>> uu(K);
774 
775  for (auto k = 0u; k != K; ++k) {
776  tt[k] = k * dt;
777  uu[k] = lin_.u(k * dt) + sol.primal.template segment<nu>(k * nu);
778  }
779 
780  auto u_spline = smooth::fit_linear_bezier(tt, uu);
781  lin_.u = [u_spline = std::move(u_spline)](double t) -> U { return u_spline.eval(t).rn(); };
782  }
783  }
