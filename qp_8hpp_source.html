241template<typename Pbm>
242auto scale_qp(const Pbm & pbm)
243{
244 using AmatT = decltype(Pbm::A);
245 using Scalar = typename AmatT::Scalar;
246 static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
247
248 Pbm ret = pbm;
249
250 static constexpr Eigen::Index M = AmatT::RowsAtCompileTime;
251 static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
252 static constexpr Eigen::Index K = (N == -1 || M == -1) ? Eigen::Index(-1) : N + M;
253
254 const Eigen::Index n = ret.A.cols(), m = ret.A.rows(), k = n + m;
255 const auto norm = [](auto && t) -> Scalar { return t.template lpNorm<Eigen::Infinity>(); };
256
257 Eigen::Matrix<Scalar, K, 1> scale = Eigen::Matrix<Scalar, K, 1>::Ones(k); // scaling
258 Eigen::Matrix<Scalar, K, 1> d_scale(k); // incremental scaling
259
260 // find "norm" of cost function
261 if constexpr (sparse) {
262 for (auto i = 0u; i != n; ++i) {
263 d_scale(i) = 0;
264 // traverse each col of P
265 for (Eigen::SparseMatrix<double>::InnerIterator it(ret.P, i); it; ++it) {
266 d_scale(i) = std::max(d_scale(i), std::abs(it.value()));
267 }
268 }
269 } else {
270 d_scale.template head<N>(n) = ret.P.colwise().template lpNorm<Eigen::Infinity>();
271 }
272
273 // if there are "zero cols"
274 for (auto i = 0u; i != n; ++i) {
275 if (d_scale(i) == 0) { d_scale(i) = 1; }
276 }
277
278 // scale cost function
279 Scalar c = Scalar(1) / std::max({1e-6, d_scale.template head<N>(n).mean(), norm(ret.q)});
280 ret.P *= c;
281 ret.q *= c;
282
283 int iter = 0;
284
285 // calculate norm for every column of [P A' ; A 0]
286 do {
287 if constexpr (sparse) {
288 // P is stored col wise
289 for (auto i = 0u; i != n; ++i) {
290 d_scale(i) = 0;
291 for (Eigen::SparseMatrix<double>::InnerIterator it(ret.P, i); it; ++it) {
292 // upper left block of H
293 d_scale(i) = std::max(d_scale(i), std::abs(it.value()));
294 }
295 }
296 // A is stored row wise
297 for (auto i = 0u; i != m; ++i) {
298 d_scale(n + i) = 0;
299 for (Eigen::SparseMatrix<double, Eigen::RowMajor>::InnerIterator it(ret.A, i); it; ++it) {
300 // bottom left block of H
301 d_scale(it.index()) = std::max(d_scale(it.index()), std::abs(it.value()));
302 // upper right block of H
303 d_scale(n + i) = std::max(d_scale(n + i), std::abs(it.value()));
304 }
305 }
306 } else {
307 d_scale.template head<N>(n) = ret.P.colwise().template lpNorm<Eigen::Infinity>().cwiseMax(
308 ret.A.colwise().template lpNorm<Eigen::Infinity>());
309 d_scale.template segment<M>(n, m) = ret.A.rowwise().template lpNorm<Eigen::Infinity>();
310 }
311
312 // if there are "zero cols" we don't scale
313 for (auto i = 0u; i != k; ++i) {
314 if (d_scale(i) == 0) { d_scale(i) = 1; }
315 }
316
317 d_scale = d_scale.cwiseMax(1e-8).cwiseInverse().cwiseSqrt();
318
319 // perform scaling
320 if constexpr (sparse) {
321 ret.P =
322 d_scale.template head<N>(n).asDiagonal() * ret.P * d_scale.template head<N>(n).asDiagonal();
323 ret.A = d_scale.template segment<M>(n, m).asDiagonal() * ret.A
324 * d_scale.template head<N>(n).asDiagonal();
325 } else {
326 ret.P.applyOnTheLeft(d_scale.template head<N>(n).asDiagonal());
327 ret.P.applyOnTheRight(d_scale.template head<N>(n).asDiagonal());
328 ret.A.applyOnTheLeft(d_scale.template segment<M>(n, m).asDiagonal());
329 ret.A.applyOnTheRight(d_scale.template head<N>(n).asDiagonal());
330 }
331 ret.q.applyOnTheLeft(d_scale.template head<N>(n).asDiagonal());
332 ret.l.applyOnTheLeft(d_scale.template segment<M>(n, m).asDiagonal());
333 ret.u.applyOnTheLeft(d_scale.template segment<M>(n, m).asDiagonal());
334
335 scale.applyOnTheLeft(d_scale.asDiagonal());
336 } while (iter++ < 10 && (d_scale.array() - 1).abs().maxCoeff() > 0.1);
337
338 return std::make_tuple(std::move(ret), std::move(scale), c);
339}

352template<typename Pbm>
353bool polish_qp(const Pbm & pbm, qp_solution_t<Pbm> & sol, const QPSolverParams & prm)
354{
355 using AmatT = decltype(Pbm::A);
356 using Scalar = typename AmatT::Scalar;
357 using VecX = Eigen::Matrix<Scalar, -1, 1>;
358 static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
359
360 static constexpr Scalar inf = std::numeric_limits<Scalar>::infinity();
361 static constexpr Scalar eps = std::numeric_limits<Scalar>::epsilon();
362
363 static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
364 const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows();
365
366 // FIND ACTIVE CONSTRAINT SETS
367
368 Eigen::Index nl = 0, nu = 0;
369 for (Eigen::Index idx = 0; idx < m; ++idx) {
370 if (sol.dual[idx] < -100 * eps && pbm.l[idx] != -inf) { nl++; }
371 if (sol.dual[idx] > 100 * eps && pbm.u[idx] != inf) { nu++; }
372 }
373
374 Eigen::Matrix<Eigen::Index, -1, 1> LU_idx(nl + nu);
375 for (Eigen::Index idx = 0, lcntr = 0, ucntr = 0; idx < m; ++idx) {
376 if (sol.dual[idx] < -100 * eps && pbm.l[idx] != -inf) { LU_idx(lcntr++) = idx; }
377 if (sol.dual[idx] > 100 * eps && pbm.u[idx] != inf) { LU_idx(nl + ucntr++) = idx; }
378 }
379
380 // FORM REDUCED SYSTEMS (27) AND (30)
381
382 // square symmetric system matrix
383 using HT = std::conditional_t<sparse, Eigen::SparseMatrix<Scalar>, Eigen::Matrix<Scalar, -1, -1>>;
384 HT H(n + nl + nu, n + nl + nu), Hp(n + nl + nu, n + nl + nu);
385
386 // fill up H
387 if constexpr (sparse) {
388 // preallocate nonzeros
389 Eigen::Matrix<int, -1, 1> nnz(n + nl + nu);
390 for (auto i = 0u; i != n; ++i) {
391 nnz(i) = pbm.P.outerIndexPtr()[i + 1] - pbm.P.outerIndexPtr()[i];
392 }
393 for (auto i = 0u; i != nl + nu; ++i) {
394 nnz(n + i) = pbm.A.outerIndexPtr()[LU_idx(i) + 1] - pbm.A.outerIndexPtr()[LU_idx(i)];
395 }
396 H.reserve(nnz);
397 Hp.reserve(nnz + Eigen::Matrix<int, -1, 1>::Ones(n + nl + nu));
398
399 using PIter = typename Eigen::SparseMatrix<Scalar, Eigen::ColMajor>::InnerIterator;
400 using AIter = typename Eigen::SparseMatrix<Scalar, Eigen::RowMajor>::InnerIterator;
401
402 // fill P in top left block
403 for (Eigen::Index p_col = 0u; p_col != n; ++p_col) {
404 for (PIter it(pbm.P, p_col); it && it.index() <= p_col; ++it) {
405 H.insert(it.index(), p_col) = it.value();
406 Hp.insert(it.index(), p_col) = it.value();
407 }
408 }
409
410 // fill selected rows of A in top right block
411 for (auto a_row = 0u; a_row != nl + nu; ++a_row) {
412 for (AIter it(pbm.A, LU_idx(a_row)); it; ++it) {
413 H.insert(it.index(), n + a_row) = it.value();
414 Hp.insert(it.index(), n + a_row) = it.value();
415 }
416 }
417 } else {
418 H.setZero();
419 H.topLeftCorner(n, n) = pbm.P;
420 for (auto i = 0u; i != nl + nu; ++i) {
421 H.col(n + i).template head<N>(n) = pbm.A.row(LU_idx(i));
422 }
423 Hp = H;
424 }
425
426 // add perturbing diagonal elements to Hp
427 if constexpr (sparse) {
428 for (auto i = 0u; i != n; ++i) { Hp.coeffRef(i, i) += prm.delta; }
429 for (auto i = 0u; i != nl + nu; ++i) { Hp.coeffRef(n + i, n + i) -= prm.delta; }
430 H.makeCompressed();
431 Hp.makeCompressed();
432 } else {
433 Hp.topLeftCorner(n, n) += VecX::Constant(n, prm.delta).asDiagonal();
434 Hp.bottomRightCorner(nl + nu, nl + nu) -= VecX::Constant(nl + nu, prm.delta).asDiagonal();
435 }
436
437 VecX h(n + nl + nu);
438 h.head(n) = -pbm.q;
439 for (auto i = 0u; i != nl; ++i) { h(n + i) = pbm.l(LU_idx(i)); }
440 for (auto i = 0u; i != nu; ++i) { h(n + nl + i) = pbm.u(LU_idx(nl + i)); }
441
442 // ITERATIVE REFINEMENT
443
444 // factorize Hp
445 std::conditional_t<
446 sparse,
447 Eigen::SimplicialLDLT<decltype(H), Eigen::Upper>,
448 Eigen::LDLT<Eigen::Ref<decltype(H)>, Eigen::Upper>>
449 ldlt(Hp);
450
451 if (ldlt.info()) { return false; }
452
453 VecX t_hat = VecX::Zero(n + nl + nu);
454 for (auto i = 0u; i != prm.polish_iter; ++i) {
455 t_hat += ldlt.solve(h - H.template selfadjointView<Eigen::Upper>() * t_hat);
456 }
457
458 // UPDATE SOLUTION
459
460 sol.primal = t_hat.template head<N>(n);
461 for (Eigen::Index i = 0; i < nl; ++i) { sol.dual(LU_idx(i)) = t_hat(n + i); }
462 for (Eigen::Index i = 0; i < nu; ++i) { sol.dual(LU_idx(nl + i)) = t_hat(n + nl + i); }
463 sol.objective = sol.primal.dot(0.5 * pbm.P * sol.primal + pbm.q);
464
465 return true;
466}

471template<typename Pbm, typename D1, typename D2, typename D3, typename D4, typename D5>
472std::optional<QPSolutionStatus> qp_check_stopping(
473 const Pbm & pbm,
474 const Eigen::MatrixBase<D1> & x,
475 const Eigen::MatrixBase<D2> & y,
476 const Eigen::MatrixBase<D3> & z,
477 const Eigen::MatrixBase<D4> & dx,
478 const Eigen::MatrixBase<D5> & dy,
479 const QPSolverParams & prm)
480{
481 using Scalar = typename decltype(Pbm::A)::Scalar;
482
483 const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows();
484 static constexpr Scalar inf = std::numeric_limits<Scalar>::infinity();
485 const auto norm = [](auto && t) -> Scalar { return t.template lpNorm<Eigen::Infinity>(); };
486
487 // working memory
488 Eigen::Matrix<Scalar, decltype(Pbm::A)::ColsAtCompileTime, 1> Px(n), Aty(n);
489 Eigen::Matrix<Scalar, decltype(Pbm::A)::RowsAtCompileTime, 1> Ax(m);
490
491 // OPTIMALITY
492
493 // check primal
494 Ax.noalias() = pbm.A * x;
495 if (norm(Ax - z) <= prm.eps_abs + prm.eps_rel * std::max<Scalar>(norm(Ax), norm(z))) {
496 // primal succeeded, check dual
497 Px.noalias() = pbm.P * x;
498 Aty.noalias() = pbm.A.transpose() * y;
499 const Scalar dual_scale = std::max<Scalar>({norm(Px), norm(pbm.q), norm(Aty)});
500 if (norm(Px + pbm.q + Aty) <= prm.eps_abs + prm.eps_rel * dual_scale) {
501 return QPSolutionStatus::Optimal;
502 }
503 }
504
505 // PRIMAL INFEASIBILITY
506
507 Aty.noalias() = pbm.A.transpose() * dy; // note new value A' * dy
508 const Scalar Edy_norm = norm(dy);
509
510 Scalar u_dyp_plus_l_dyn = Scalar(0);
511 for (auto i = 0u; i != m; ++i) {
512 if (pbm.u(i) != inf) {
513 u_dyp_plus_l_dyn += pbm.u(i) * std::max<Scalar>(Scalar(0), dy(i));
514 } else if (dy(i) > prm.eps_primal_inf * Edy_norm) {
515 // contributes +inf to sum --> no certificate
516 u_dyp_plus_l_dyn = inf;
517 break;
518 }
519 if (pbm.l(i) != -inf) {
520 u_dyp_plus_l_dyn += pbm.l(i) * std::min<Scalar>(Scalar(0), dy(i));
521 } else if (dy(i) < -prm.eps_primal_inf * Edy_norm) {
522 // contributes +inf to sum --> no certificate
523 u_dyp_plus_l_dyn = inf;
524 break;
525 }
526 }
527
528 if (std::max<Scalar>(norm(Aty), u_dyp_plus_l_dyn) < prm.eps_primal_inf * Edy_norm) {
529 return QPSolutionStatus::PrimalInfeasible;
530 }
531
532 // DUAL INFEASIBILITY
533
534 Ax.noalias() = pbm.A * dx; // note new value A * dx
535 const Scalar dx_norm = norm(dx);
536
537 bool dual_infeasible = (norm(pbm.P * dx) <= prm.eps_dual_inf * dx_norm)
538 && (pbm.q.dot(dx) <= prm.eps_dual_inf * dx_norm);
539 for (auto i = 0u; i != m && dual_infeasible; ++i) {
540 if (pbm.u(i) == inf) {
541 dual_infeasible &= (Ax(i) >= -prm.eps_dual_inf * dx_norm);
542 } else if (pbm.l(i) == -inf) {
543 dual_infeasible &= (Ax(i) <= prm.eps_dual_inf * dx_norm);
544 } else {
545 dual_infeasible &= std::abs(Ax(i)) < prm.eps_dual_inf * dx_norm;
546 }
547 }
548
549 if (dual_infeasible) { return QPSolutionStatus::DualInfeasible; }
550
551 return std::nullopt;
552}

576template<typename Pbm>
577detail::qp_solution_t<Pbm> solve_qp(
578 const Pbm & pbm,
579 const QPSolverParams & prm,
580 std::optional<std::reference_wrapper<const detail::qp_solution_t<Pbm>>> warmstart = {})
581{
582 using AmatT = decltype(Pbm::A);
583 using Scalar = typename AmatT::Scalar;
584 static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
585
586 // static sizes
587 static constexpr Eigen::Index M = AmatT::RowsAtCompileTime;
588 static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
589 static constexpr Eigen::Index K = (N == -1 || M == -1) ? Eigen::Index(-1) : N + M;
590
591 // typedefs
592 using Rn = Eigen::Matrix<Scalar, N, 1>;
593 using Rm = Eigen::Matrix<Scalar, M, 1>;
594 using Rk = Eigen::Matrix<Scalar, K, 1>;
595
596 // dynamic sizes
597 const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows(), k = n + m;
598
599 static constexpr Scalar inf = std::numeric_limits<Scalar>::infinity();
600
601 // cast parameters to scalar type
602 const Scalar rho_bar = static_cast<Scalar>(prm.rho);
603 const Scalar alpha = static_cast<Scalar>(prm.alpha);
604 const Scalar alpha_comp = Scalar(1) - alpha;
605 const Scalar sigma = static_cast<Scalar>(prm.sigma);
606
607 // return code: when set algorithm is finished
608 std::optional<QPSolutionStatus> ret_code = std::nullopt;
609
610 // allocate working arrays
611 Rn x_us(n), dx_us(n);
612 Rm z_next(m), y_us(m), z_us(m), dy_us(m), rho(m);
613 Rk p(k);
614
615 // scale problem
616 Scalar c = 1;
617 Eigen::Matrix<Scalar, K, 1> S = Eigen::Matrix<Scalar, K, 1>::Ones(k);
618 Pbm spbm = pbm;
619 if (prm.scaling) { std::tie(spbm, S, c) = detail::scale_qp(pbm); }
620
621 for (auto i = 0u; i != m; ++i) {
622 if (spbm.l(i) == inf || spbm.u(i) == -inf || spbm.u(i) - spbm.l(i) < Scalar(0.)) {
623 ret_code = QPSolutionStatus::PrimalInfeasible; // feasible set trivially empty
624 }
625
626 // set rho depending on constraint type
627 if (spbm.l(i) == -inf && spbm.u(i) == inf) {
628 rho(i) = Scalar(1e-6); // unbounded
629 } else if (abs(spbm.l(i) - spbm.u(i)) < 1e-5) {
630 rho(i) = Scalar(1e3) * rho_bar; // equality
631 } else {
632 rho(i) = rho_bar; // inequality
633 }
634 }
635
636 const auto t0 = std::chrono::high_resolution_clock::now();
637
638 // fill square symmetric system matrix H
639 std::conditional_t<sparse, Eigen::SparseMatrix<Scalar>, Eigen::Matrix<Scalar, K, K>> H(k, k);
640 if constexpr (sparse) {
641 // preallocate nonzeros in H
642 Eigen::Matrix<int, -1, 1> nnz(k);
643 for (auto i = 0u; i != n; ++i) {
644 nnz(i) = spbm.P.outerIndexPtr()[i + 1] - spbm.P.outerIndexPtr()[i] + 1;
645 }
646 for (auto i = 0u; i != m; ++i) {
647 nnz(n + i) = spbm.A.outerIndexPtr()[i + 1] - spbm.A.outerIndexPtr()[i] + 1;
648 }
649 H.reserve(nnz);
650
651 // fill nonzeros in H
652 using PIter = typename Eigen::SparseMatrix<Scalar, Eigen::ColMajor>::InnerIterator;
653 using AIter = typename Eigen::SparseMatrix<Scalar, Eigen::RowMajor>::InnerIterator;
654 for (Eigen::Index col = 0u; col != n; ++col) {
655 for (PIter it(spbm.P, col); it && it.index() <= col; ++it) {
656 H.insert(it.index(), col) = it.value();
657 }
658 H.coeffRef(col, col) += sigma;
659 }
660 for (auto row = 0u; row != m; ++row) {
661 for (AIter it(spbm.A, row); it; ++it) { H.insert(it.index(), n + row) = it.value(); }
662 H.insert(n + row, n + row) = Scalar(-1) / rho(row);
663 }
664 H.makeCompressed();
665 } else {
666 H.template topLeftCorner<N, N>(n, n) = spbm.P;
667 H.template topLeftCorner<N, N>(n, n) += Rn::Constant(n, sigma).asDiagonal();
668 H.template topRightCorner<N, M>(n, m) = spbm.A.transpose();
669 H.template bottomRightCorner<M, M>(m, m) = (-rho).cwiseInverse().asDiagonal();
670 }
671
672 const auto t_fill = std::chrono::high_resolution_clock::now();
673
674 if (prm.verbose) {
675 using std::cout, std::left, std::setw, std::right;
676 // clang-format off
677 cout << "========================= QP Solver =========================" << '\n';
678 cout << "Solving " << (sparse ? "sparse" : "dense") << " QP with n=" << n << ", m=" << m << '\n';
679 cout << setw(8) << right << "ITER"
680 << setw(14) << right << "OBJ"
681 << setw(14) << right << "PRI_RES"
682 << setw(14) << right << "DUA_RES"
683 << setw(10) << right << "TIME" << '\n';
684 // clang-format on
685 }
686
687 // factorize H
688 std::conditional_t<
689 sparse,
690 Eigen::SimplicialLDLT<decltype(H), Eigen::Upper>,
691 Eigen::LDLT<Eigen::Ref<decltype(H)>, Eigen::Upper>>
692 ldlt(H);
693
694 const auto t_factor = std::chrono::high_resolution_clock::now();
695
696 if (ldlt.info()) { ret_code = QPSolutionStatus::Unknown; }
697
698 // initialize solver variables
699 Rn x;
700 Rm z, y;
701 if (warmstart.has_value()) {
702 // warmstart variables must be scaled
703 x = warmstart.value().get().primal;
704 x.applyOnTheLeft(S.template head<N>(n).cwiseInverse().asDiagonal());
705 y = warmstart.value().get().dual;
706 y.applyOnTheLeft(S.template segment<M>(n, m).cwiseInverse().asDiagonal());
707 y *= c;
708 z.noalias() = spbm.A * x;
709 } else {
710 x.setZero(n);
711 y.setZero(m);
712 z.setZero(m);
713 }
714
715 // main optimization loop
716 auto iter = 0u;
717 for (; (!prm.max_iter || iter != prm.max_iter.value()) && !ret_code; ++iter) {
718 p.template head<N>(n) = sigma * x - spbm.q;
719 p.template segment<M>(n, m) = z - rho.cwiseInverse().cwiseProduct(y);
720 p = ldlt.solve(p);
721
722 if (iter % prm.stop_check_iter == 1) {
723 // termination checking requires difference, store old scaled values
724 dx_us = x, dy_us = y;
725 }
726
727 x = alpha * p.template head<N>(n) + alpha_comp * x;
728 z_next = (alpha * rho.cwiseInverse().cwiseProduct(p.template segment<M>(n, m))
729 + alpha_comp * rho.cwiseInverse().cwiseProduct(y) + z)
730 .cwiseMax(spbm.l)
731 .cwiseMin(spbm.u);
732 y = alpha_comp * y + alpha * p.template segment<M>(n, m) + rho.cwiseProduct(z)
733 - rho.cwiseProduct(z_next);
734 z = z_next;
735
736 if (iter % prm.stop_check_iter == 1) {
737 // check stopping criteria for unscaled problem and unscaled variables
738 x_us = S.template head<N>(n).cwiseProduct(x);
739 y_us = S.template segment<M>(n, m).cwiseProduct(y) / c;
740 z_us = S.template segment<M>(n, m).cwiseInverse().cwiseProduct(z);
741 dx_us = S.template head<N>(n).cwiseProduct(x - dx_us);
742 dy_us = S.template segment<M>(n, m).cwiseProduct(y - dy_us) / c;
743 ret_code = detail::qp_check_stopping(pbm, x_us, y_us, z_us, dx_us, dy_us, prm);
744
745 if (prm.verbose) {
746 using std::cout, std::setw, std::right, std::chrono::microseconds;
747 // clang-format off
748 cout << setw(7) << right << iter << ":"
749 << std::scientific
750 << setw(14) << right << (0.5 * pbm.P * x_us + pbm.q).dot(x_us)
751 << setw(14) << right << (pbm.A * x_us - z_us).template lpNorm<Eigen::Infinity>()
752 << setw(14) << right << (pbm.P * x_us + pbm.q + pbm.A.transpose() * y_us).template lpNorm<Eigen::Infinity>()
753 << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count()
754 << '\n';
755 // clang-format on
756 }
757
758 // check for timeout
759 if (!ret_code) {
760 if (prm.max_time && std::chrono::high_resolution_clock::now() > t0 + prm.max_time.value()) {
761 ret_code = QPSolutionStatus::MaxTime;
762 }
763 }
764 }
765 }
766
767 double obj = x.dot(0.5 * pbm.P * x + pbm.q);
768
769 detail::qp_solution_t<Pbm> sol{
770 .code = ret_code.value_or(QPSolutionStatus::MaxIterations),
771 .iter = iter - 1,
772 .primal = std::move(x),
773 .dual = std::move(y),
774 .objective = obj,
775 };
776
777 const auto t_iter = std::chrono::high_resolution_clock::now();
778
779 // polish solution if optimal
780 if (sol.code == QPSolutionStatus::Optimal && prm.polish) {
781 if (detail::polish_qp(spbm, sol, prm)) {
782 if (prm.verbose) {
783 using std::cout, std::setw, std::right, std::chrono::microseconds;
784 x_us = S.template head<N>(n).cwiseProduct(sol.primal); // NOTE: x std::moved to sol
785 y_us = S.template segment<M>(n, m).cwiseProduct(sol.dual) / c; // NOTE: y std::moved to sol
786 z_us = S.template segment<M>(n, m).cwiseInverse().cwiseProduct(z);
787 // clang-format off
788 cout << setw(8) << right << "polish:"
789 << std::scientific
790 << setw(14) << right << (0.5 * pbm.P * x_us + pbm.q).dot(x_us)
791 << setw(14) << right << (pbm.A * x_us - z_us).template lpNorm<Eigen::Infinity>()
792 << setw(14) << right << (pbm.P * x_us + pbm.q + pbm.A.transpose() * y_us).template lpNorm<Eigen::Infinity>()
793 << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count()
794 << '\n';
795 // clang-format on
796 }
797
798 } else {
799 if (prm.verbose) { std::cout << "Polish failed" << '\n'; }
800 sol.code = QPSolutionStatus::PolishFailed;
801 }
802 }
803
804 const auto t_polish = std::chrono::high_resolution_clock::now();
805
806 // unscale solution
807 sol.primal.applyOnTheLeft(S.template head<N>(n).asDiagonal());
808 sol.dual.applyOnTheLeft(S.template segment<M>(n, m).asDiagonal());
809 sol.dual /= c;
810 sol.objective = sol.primal.dot(0.5 * pbm.P * sol.primal + pbm.q);
811
812 if (prm.verbose) {
813 using std::cout, std::left, std::right, std::setw, std::chrono::microseconds;
814
815 // clang-format off
816 cout << "QP solver summary:" << '\n';
817 cout << "Result " << static_cast<int>(sol.code) << '\n';
818
819 cout << setw(25) << left << "Iterations" << setw(10) << right << iter - 1 << '\n';
820 cout << setw(26) << left << "Total time (µs)" << setw(10) << right << duration_cast<microseconds>(t_polish - t0).count() << '\n';
821 cout << setw(25) << left << " Matrix filling" << setw(10) << right << duration_cast<microseconds>(t_fill - t0).count() << '\n';
822 cout << setw(25) << left << " Factorization" << setw(10) << right << duration_cast<microseconds>(t_factor - t_fill).count() << '\n';
823 cout << setw(25) << left << " Iteration" << setw(10) << right << duration_cast<microseconds>(t_iter - t_factor).count() << '\n';
824 cout << setw(25) << left << " Polish" << setw(10) << right << duration_cast<microseconds>(t_polish - t_iter).count() << '\n';
825 cout << "=============================================================" << '\n';
826 // clang-format on
827 }
828
829 return sol;
830}
