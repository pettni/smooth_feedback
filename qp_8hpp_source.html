241 template<typename Pbm>
242 auto scale_qp(const Pbm & pbm)
243 {
244  using AmatT = decltype(Pbm::A);
245  using Scalar = typename AmatT::Scalar;
246  static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
247 
248  Pbm ret = pbm;
249 
250  static constexpr Eigen::Index M = AmatT::RowsAtCompileTime;
251  static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
252  static constexpr Eigen::Index K = (N == -1 || M == -1) ? Eigen::Index(-1) : N + M;
253 
254  const Eigen::Index n = ret.A.cols(), m = ret.A.rows(), k = n + m;
255  const auto norm = [](auto && t) -> Scalar { return t.template lpNorm<Eigen::Infinity>(); };
256 
257  Eigen::Matrix<Scalar, K, 1> scale = Eigen::Matrix<Scalar, K, 1>::Ones(k); // scaling
258  Eigen::Matrix<Scalar, K, 1> d_scale(k); // incremental scaling
259 
260  // find "norm" of cost function
261  if constexpr (sparse) {
262  for (auto i = 0u; i != n; ++i) {
263  d_scale(i) = 0;
264  // traverse each col of P
265  for (Eigen::SparseMatrix<double>::InnerIterator it(ret.P, i); it; ++it) {
266  d_scale(i) = std::max(d_scale(i), std::abs(it.value()));
267  }
268  }
269  } else {
270  d_scale.template head<N>(n) = ret.P.colwise().template lpNorm<Eigen::Infinity>();
271  }
272 
273  // if there are "zero cols"
274  for (auto i = 0u; i != n; ++i) {
275  if (d_scale(i) == 0) { d_scale(i) = 1; }
276  }
277 
278  // scale cost function
279  Scalar c = Scalar(1) / std::max({1e-6, d_scale.template head<N>(n).mean(), norm(ret.q)});
280  ret.P *= c;
281  ret.q *= c;
282 
283  int iter = 0;
284 
285  // calculate norm for every column of [P A' ; A 0]
286  do {
287  if constexpr (sparse) {
288  // P is stored col wise
289  for (auto i = 0u; i != n; ++i) {
290  d_scale(i) = 0;
291  for (Eigen::SparseMatrix<double>::InnerIterator it(ret.P, i); it; ++it) {
292  // upper left block of H
293  d_scale(i) = std::max(d_scale(i), std::abs(it.value()));
294  }
295  }
296  // A is stored row wise
297  for (auto i = 0u; i != m; ++i) {
298  d_scale(n + i) = 0;
299  for (Eigen::SparseMatrix<double, Eigen::RowMajor>::InnerIterator it(ret.A, i); it; ++it) {
300  // bottom left block of H
301  d_scale(it.index()) = std::max(d_scale(it.index()), std::abs(it.value()));
302  // upper right block of H
303  d_scale(n + i) = std::max(d_scale(n + i), std::abs(it.value()));
304  }
305  }
306  } else {
307  d_scale.template head<N>(n) = ret.P.colwise().template lpNorm<Eigen::Infinity>().cwiseMax(
308  ret.A.colwise().template lpNorm<Eigen::Infinity>());
309  d_scale.template segment<M>(n, m) = ret.A.rowwise().template lpNorm<Eigen::Infinity>();
310  }
311 
312  // if there are "zero cols" we don't scale
313  for (auto i = 0u; i != k; ++i) {
314  if (d_scale(i) == 0) { d_scale(i) = 1; }
315  }
316 
317  d_scale = d_scale.cwiseMax(1e-8).cwiseInverse().cwiseSqrt();
318 
319  // perform scaling
320  if constexpr (sparse) {
321  ret.P =
322  d_scale.template head<N>(n).asDiagonal() * ret.P * d_scale.template head<N>(n).asDiagonal();
323  ret.A = d_scale.template segment<M>(n, m).asDiagonal() * ret.A
324  * d_scale.template head<N>(n).asDiagonal();
325  } else {
326  ret.P.applyOnTheLeft(d_scale.template head<N>(n).asDiagonal());
327  ret.P.applyOnTheRight(d_scale.template head<N>(n).asDiagonal());
328  ret.A.applyOnTheLeft(d_scale.template segment<M>(n, m).asDiagonal());
329  ret.A.applyOnTheRight(d_scale.template head<N>(n).asDiagonal());
330  }
331  ret.q.applyOnTheLeft(d_scale.template head<N>(n).asDiagonal());
332  ret.l.applyOnTheLeft(d_scale.template segment<M>(n, m).asDiagonal());
333  ret.u.applyOnTheLeft(d_scale.template segment<M>(n, m).asDiagonal());
334 
335  scale.applyOnTheLeft(d_scale.asDiagonal());
336  } while (iter++ < 10 && (d_scale.array() - 1).abs().maxCoeff() > 0.1);
337 
338  return std::make_tuple(std::move(ret), std::move(scale), c);
339 }

352 template<typename Pbm>
353 bool polish_qp(const Pbm & pbm, qp_solution_t<Pbm> & sol, const QPSolverParams & prm)
354 {
355  using AmatT = decltype(Pbm::A);
356  using Scalar = typename AmatT::Scalar;
357  using VecX = Eigen::Matrix<Scalar, -1, 1>;
358  static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
359  using LDLT =
360  std::conditional_t<sparse, detail::LDLTSparse<Scalar>, detail::LDLTLapack<Scalar, -1>>;
361 
362  static constexpr Scalar inf = std::numeric_limits<Scalar>::infinity();
363  static constexpr Scalar eps = std::numeric_limits<Scalar>::epsilon();
364 
365  static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
366  const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows();
367 
368  // FIND ACTIVE CONSTRAINT SETS
369 
370  Eigen::Index nl = 0, nu = 0;
371  for (Eigen::Index idx = 0; idx < m; ++idx) {
372  if (sol.dual[idx] < -100 * eps && pbm.l[idx] != -inf) { nl++; }
373  if (sol.dual[idx] > 100 * eps && pbm.u[idx] != inf) { nu++; }
374  }
375 
376  Eigen::Matrix<Eigen::Index, -1, 1> LU_idx(nl + nu);
377  for (Eigen::Index idx = 0, lcntr = 0, ucntr = 0; idx < m; ++idx) {
378  if (sol.dual[idx] < -100 * eps && pbm.l[idx] != -inf) { LU_idx(lcntr++) = idx; }
379  if (sol.dual[idx] > 100 * eps && pbm.u[idx] != inf) { LU_idx(nl + ucntr++) = idx; }
380  }
381 
382  // FORM REDUCED SYSTEMS (27) AND (30)
383 
384  // square symmetric system matrix
385  using HT = std::conditional_t<sparse, Eigen::SparseMatrix<Scalar>, Eigen::Matrix<Scalar, -1, -1>>;
386  HT H(n + nl + nu, n + nl + nu), Hp(n + nl + nu, n + nl + nu);
387 
388  // fill up H
389  if constexpr (sparse) {
390  // preallocate nonzeros
391  Eigen::Matrix<int, -1, 1> nnz(n + nl + nu);
392  for (auto i = 0u; i != n; ++i) {
393  nnz(i) = pbm.P.outerIndexPtr()[i + 1] - pbm.P.outerIndexPtr()[i];
394  }
395  for (auto i = 0u; i != nl + nu; ++i) {
396  nnz(n + i) = pbm.A.outerIndexPtr()[LU_idx(i) + 1] - pbm.A.outerIndexPtr()[LU_idx(i)];
397  }
398  H.reserve(nnz);
399  Hp.reserve(nnz + Eigen::Matrix<int, -1, 1>::Ones(n + nl + nu));
400 
401  using PIter = typename Eigen::SparseMatrix<Scalar, Eigen::ColMajor>::InnerIterator;
402  using AIter = typename Eigen::SparseMatrix<Scalar, Eigen::RowMajor>::InnerIterator;
403 
404  // fill P in top left block
405  for (Eigen::Index p_col = 0u; p_col != n; ++p_col) {
406  for (PIter it(pbm.P, p_col); it && it.index() <= p_col; ++it) {
407  H.insert(it.index(), p_col) = it.value();
408  Hp.insert(it.index(), p_col) = it.value();
409  }
410  }
411 
412  // fill selected rows of A in top right block
413  for (auto a_row = 0u; a_row != nl + nu; ++a_row) {
414  for (AIter it(pbm.A, LU_idx(a_row)); it; ++it) {
415  H.insert(it.index(), n + a_row) = it.value();
416  Hp.insert(it.index(), n + a_row) = it.value();
417  }
418  }
419  } else {
420  H.setZero();
421  H.topLeftCorner(n, n) = pbm.P;
422  for (auto i = 0u; i != nl + nu; ++i) {
423  H.col(n + i).template head<N>(n) = pbm.A.row(LU_idx(i));
424  }
425  Hp = H;
426  }
427 
428  // add perturbing diagonal elements to Hp
429  if constexpr (sparse) {
430  for (auto i = 0u; i != n; ++i) { Hp.coeffRef(i, i) += prm.delta; }
431  for (auto i = 0u; i != nl + nu; ++i) { Hp.coeffRef(n + i, n + i) -= prm.delta; }
432  H.makeCompressed();
433  Hp.makeCompressed();
434  } else {
435  Hp.topLeftCorner(n, n) += VecX::Constant(n, prm.delta).asDiagonal();
436  Hp.bottomRightCorner(nl + nu, nl + nu) -= VecX::Constant(nl + nu, prm.delta).asDiagonal();
437  }
438 
439  VecX h(n + nl + nu);
440  h.head(n) = -pbm.q;
441  for (auto i = 0u; i != nl; ++i) { h(n + i) = pbm.l(LU_idx(i)); }
442  for (auto i = 0u; i != nu; ++i) { h(n + nl + i) = pbm.u(LU_idx(nl + i)); }
443 
444  // ITERATIVE REFINEMENT
445 
446  // factorize Hp
447  LDLT ldlt(std::move(Hp));
448  if (ldlt.info()) { return false; }
449 
450  VecX t_hat = VecX::Zero(n + nl + nu);
451  for (auto i = 0u; i != prm.polish_iter; ++i) {
452  t_hat += ldlt.solve(h - H.template selfadjointView<Eigen::Upper>() * t_hat);
453  }
454 
455  // UPDATE SOLUTION
456 
457  sol.primal = t_hat.template head<N>(n);
458  for (Eigen::Index i = 0; i < nl; ++i) { sol.dual(LU_idx(i)) = t_hat(n + i); }
459  for (Eigen::Index i = 0; i < nu; ++i) { sol.dual(LU_idx(nl + i)) = t_hat(n + nl + i); }
460  sol.objective = sol.primal.dot(0.5 * pbm.P * sol.primal + pbm.q);
461 
462  return true;
463 }

468 template<typename Pbm, typename D1, typename D2, typename D3, typename D4, typename D5>
469 std::optional<QPSolutionStatus> qp_check_stopping(const Pbm & pbm,
470  const Eigen::MatrixBase<D1> & x,
471  const Eigen::MatrixBase<D2> & y,
472  const Eigen::MatrixBase<D3> & z,
473  const Eigen::MatrixBase<D4> & dx,
474  const Eigen::MatrixBase<D5> & dy,
475  const QPSolverParams & prm)
476 {
477  using Scalar = typename decltype(Pbm::A)::Scalar;
478 
479  const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows();
480  static constexpr Scalar inf = std::numeric_limits<Scalar>::infinity();
481  const auto norm = [](auto && t) -> Scalar { return t.template lpNorm<Eigen::Infinity>(); };
482 
483  // working memory
484  Eigen::Matrix<Scalar, decltype(Pbm::A)::ColsAtCompileTime, 1> Px(n), Aty(n);
485  Eigen::Matrix<Scalar, decltype(Pbm::A)::RowsAtCompileTime, 1> Ax(m);
486 
487  // OPTIMALITY
488 
489  // check primal
490  Ax.noalias() = pbm.A * x;
491  if (norm(Ax - z) <= prm.eps_abs + prm.eps_rel * std::max<Scalar>(norm(Ax), norm(z))) {
492  // primal succeeded, check dual
493  Px.noalias() = pbm.P * x;
494  Aty.noalias() = pbm.A.transpose() * y;
495  const Scalar dual_scale = std::max<Scalar>({norm(Px), norm(pbm.q), norm(Aty)});
496  if (norm(Px + pbm.q + Aty) <= prm.eps_abs + prm.eps_rel * dual_scale) {
497  return QPSolutionStatus::Optimal;
498  }
499  }
500 
501  // PRIMAL INFEASIBILITY
502 
503  Aty.noalias() = pbm.A.transpose() * dy; // note new value A' * dy
504  const Scalar Edy_norm = norm(dy);
505 
506  Scalar u_dyp_plus_l_dyn = Scalar(0);
507  for (auto i = 0u; i != m; ++i) {
508  if (pbm.u(i) != inf) {
509  u_dyp_plus_l_dyn += pbm.u(i) * std::max<Scalar>(Scalar(0), dy(i));
510  } else if (dy(i) > prm.eps_primal_inf * Edy_norm) {
511  // contributes +inf to sum --> no certificate
512  u_dyp_plus_l_dyn = inf;
513  break;
514  }
515  if (pbm.l(i) != -inf) {
516  u_dyp_plus_l_dyn += pbm.l(i) * std::min<Scalar>(Scalar(0), dy(i));
517  } else if (dy(i) < -prm.eps_primal_inf * Edy_norm) {
518  // contributes +inf to sum --> no certificate
519  u_dyp_plus_l_dyn = inf;
520  break;
521  }
522  }
523 
524  if (std::max<Scalar>(norm(Aty), u_dyp_plus_l_dyn) < prm.eps_primal_inf * Edy_norm) {
525  return QPSolutionStatus::PrimalInfeasible;
526  }
527 
528  // DUAL INFEASIBILITY
529 
530  Ax.noalias() = pbm.A * dx; // note new value A * dx
531  const Scalar dx_norm = norm(dx);
532 
533  bool dual_infeasible = (norm(pbm.P * dx) <= prm.eps_dual_inf * dx_norm)
534  && (pbm.q.dot(dx) <= prm.eps_dual_inf * dx_norm);
535  for (auto i = 0u; i != m && dual_infeasible; ++i) {
536  if (pbm.u(i) == inf) {
537  dual_infeasible &= (Ax(i) >= -prm.eps_dual_inf * dx_norm);
538  } else if (pbm.l(i) == -inf) {
539  dual_infeasible &= (Ax(i) <= prm.eps_dual_inf * dx_norm);
540  } else {
541  dual_infeasible &= std::abs(Ax(i)) < prm.eps_dual_inf * dx_norm;
542  }
543  }
544 
545  if (dual_infeasible) { return QPSolutionStatus::DualInfeasible; }
546 
547  return std::nullopt;
548 }

572 template<typename Pbm>
573 detail::qp_solution_t<Pbm> solve_qp(const Pbm & pbm,
574  const QPSolverParams & prm,
575  std::optional<std::reference_wrapper<const detail::qp_solution_t<Pbm>>> warmstart = {})
576 {
577  using AmatT = decltype(Pbm::A);
578  using Scalar = typename AmatT::Scalar;
579  static constexpr bool sparse = std::is_base_of_v<Eigen::SparseMatrixBase<AmatT>, AmatT>;
580 
581  // static sizes
582  static constexpr Eigen::Index M = AmatT::RowsAtCompileTime;
583  static constexpr Eigen::Index N = AmatT::ColsAtCompileTime;
584  static constexpr Eigen::Index K = (N == -1 || M == -1) ? Eigen::Index(-1) : N + M;
585 
586  // factorization method
587  using LDLT =
588  std::conditional_t<sparse, detail::LDLTSparse<Scalar>, detail::LDLTLapack<Scalar, K>>;
589 
590  // typedefs
591  using Rn = Eigen::Matrix<Scalar, N, 1>;
592  using Rm = Eigen::Matrix<Scalar, M, 1>;
593  using Rk = Eigen::Matrix<Scalar, K, 1>;
594 
595  // dynamic sizes
596  const Eigen::Index n = pbm.A.cols(), m = pbm.A.rows(), k = n + m;
597 
598  static constexpr Scalar inf = std::numeric_limits<Scalar>::infinity();
599 
600  // cast parameters to scalar type
601  const Scalar rho_bar = static_cast<Scalar>(prm.rho);
602  const Scalar alpha = static_cast<Scalar>(prm.alpha);
603  const Scalar alpha_comp = Scalar(1) - alpha;
604  const Scalar sigma = static_cast<Scalar>(prm.sigma);
605 
606  // return code: when set algorithm is finished
607  std::optional<QPSolutionStatus> ret_code = std::nullopt;
608 
609  // allocate working arrays
610  Rn x_us(n), dx_us(n);
611  Rm z_next(m), y_us(m), z_us(m), dy_us(m), rho(m);
612  Rk p(k);
613 
614  // scale problem
615  Scalar c = 1;
616  Eigen::Matrix<Scalar, K, 1> S = Eigen::Matrix<Scalar, K, 1>::Ones(k);
617  Pbm spbm = pbm;
618  if (prm.scaling) { std::tie(spbm, S, c) = detail::scale_qp(pbm); }
619 
620  for (auto i = 0u; i != m; ++i) {
621  if (spbm.l(i) == inf || spbm.u(i) == -inf || spbm.u(i) - spbm.l(i) < Scalar(0.)) {
622  ret_code = QPSolutionStatus::PrimalInfeasible; // feasible set trivially empty
623  }
624 
625  // set rho depending on constraint type
626  if (spbm.l(i) == -inf && spbm.u(i) == inf) {
627  rho(i) = Scalar(1e-6); // unbounded
628  } else if (abs(spbm.l(i) - spbm.u(i)) < 1e-5) {
629  rho(i) = Scalar(1e3) * rho_bar; // equality
630  } else {
631  rho(i) = rho_bar; // inequality
632  }
633  }
634 
635  const auto t0 = std::chrono::high_resolution_clock::now();
636 
637  // fill square symmetric system matrix H
638  std::conditional_t<sparse, Eigen::SparseMatrix<Scalar>, Eigen::Matrix<Scalar, K, K>> H(k, k);
639  if constexpr (sparse) {
640  // preallocate nonzeros in H
641  Eigen::Matrix<int, -1, 1> nnz(k);
642  for (auto i = 0u; i != n; ++i) {
643  nnz(i) = spbm.P.outerIndexPtr()[i + 1] - spbm.P.outerIndexPtr()[i] + 1;
644  }
645  for (auto i = 0u; i != m; ++i) {
646  nnz(n + i) = spbm.A.outerIndexPtr()[i + 1] - spbm.A.outerIndexPtr()[i] + 1;
647  }
648  H.reserve(nnz);
649 
650  // fill nonzeros in H
651  using PIter = typename Eigen::SparseMatrix<Scalar, Eigen::ColMajor>::InnerIterator;
652  using AIter = typename Eigen::SparseMatrix<Scalar, Eigen::RowMajor>::InnerIterator;
653  for (Eigen::Index col = 0u; col != n; ++col) {
654  for (PIter it(spbm.P, col); it && it.index() <= col; ++it) {
655  H.insert(it.index(), col) = it.value();
656  }
657  H.coeffRef(col, col) += sigma;
658  }
659  for (auto row = 0u; row != m; ++row) {
660  for (AIter it(spbm.A, row); it; ++it) { H.insert(it.index(), n + row) = it.value(); }
661  H.insert(n + row, n + row) = Scalar(-1) / rho(row);
662  }
663  H.makeCompressed();
664  } else {
665  H.template topLeftCorner<N, N>(n, n) = spbm.P;
666  H.template topLeftCorner<N, N>(n, n) += Rn::Constant(n, sigma).asDiagonal();
667  H.template topRightCorner<N, M>(n, m) = spbm.A.transpose();
668  H.template bottomRightCorner<M, M>(m, m) = (-rho).cwiseInverse().asDiagonal();
669  }
670 
671  const auto t_fill = std::chrono::high_resolution_clock::now();
672 
673  if (prm.verbose) {
674  using std::cout, std::left, std::endl, std::setw, std::right;

701  y.applyOnTheLeft(S.template segment<M>(n, m).cwiseInverse().asDiagonal());

704  } else {

706  y.setZero(m);
707  z.setZero(m);
708  }

711  auto iter = 0u;
712  for (; (!prm.max_iter || iter != prm.max_iter.value()) && !ret_code; ++iter) {
713  p.template head<N>(n) = sigma * x - spbm.q;
714  p.template segment<M>(n, m) = z - rho.cwiseInverse().cwiseProduct(y);
715  ldlt.solve_inplace(p);
716 
717  if (iter % prm.stop_check_iter == 1) {
718  // termination checking requires difference, store old scaled values
719  dx_us = x, dy_us = y;
720  }
721 
722  x = alpha * p.template head<N>(n) + alpha_comp * x;
723  z_next = (alpha * rho.cwiseInverse().cwiseProduct(p.template segment<M>(n, m))
724  + alpha_comp * rho.cwiseInverse().cwiseProduct(y) + z)
725  .cwiseMax(spbm.l)
726  .cwiseMin(spbm.u);
727  y = alpha_comp * y + alpha * p.template segment<M>(n, m) + rho.cwiseProduct(z)
728  - rho.cwiseProduct(z_next);
729  z = z_next;
730 
731  if (iter % prm.stop_check_iter == 1) {
732  // check stopping criteria for unscaled problem and unscaled variables
733  x_us = S.template head<N>(n).cwiseProduct(x);
734  y_us = S.template segment<M>(n, m).cwiseProduct(y) / c;
735  z_us = S.template segment<M>(n, m).cwiseInverse().cwiseProduct(z);
736  dx_us = S.template head<N>(n).cwiseProduct(x - dx_us);
737  dy_us = S.template segment<M>(n, m).cwiseProduct(y - dy_us) / c;
738  ret_code = detail::qp_check_stopping(pbm, x_us, y_us, z_us, dx_us, dy_us, prm);
739 
740  if (prm.verbose) {

746  << setw(14) << right << (pbm.A * x_us - z_us).template lpNorm<Eigen::Infinity>()
747  << setw(14) << right << (pbm.P * x_us + pbm.q + pbm.A.transpose() * y_us).template lpNorm<Eigen::Infinity>()
748  << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count()
749  << std::endl;

751  }

754  if (!ret_code) {

756  ret_code = QPSolutionStatus::MaxTime;
757  }
758  }
759  }
760  }
761 
762  double obj = x.dot(0.5 * pbm.P * x + pbm.q);
763

765  .code = ret_code.value_or(QPSolutionStatus::MaxIterations),
766  .iter = iter,
767  .primal = std::move(x),
768  .dual = std::move(y),
769  .objective = obj,
770  };
771 
772  const auto t_iter = std::chrono::high_resolution_clock::now();
773 
774  // polish solution if optimal
775  if (sol.code == QPSolutionStatus::Optimal && prm.polish) {
776  if (detail::polish_qp(spbm, sol, prm)) {
777  if (prm.verbose) {
778  using std::cout, std::endl, std::setw, std::right, std::chrono::microseconds;
779  x_us = S.template head<N>(n).cwiseProduct(sol.primal); // NOTE: x std::moved to sol
780  y_us = S.template segment<M>(n, m).cwiseProduct(sol.dual) / c; // NOTE: y std::moved to sol
781  z_us = S.template segment<M>(n, m).cwiseInverse().cwiseProduct(z);
782  // clang-format off
783  cout << setw(8) << right << "polish:"
784  << std::scientific
785  << setw(14) << right << (0.5 * pbm.P * x_us + pbm.q).dot(x_us)
786  << setw(14) << right << (pbm.A * x_us - z_us).template lpNorm<Eigen::Infinity>()
787  << setw(14) << right << (pbm.P * x_us + pbm.q + pbm.A.transpose() * y_us).template lpNorm<Eigen::Infinity>()
788  << setw(10) << right << duration_cast<microseconds>(std::chrono::high_resolution_clock::now() - t0).count()
789  << std::endl;
790  // clang-format on
791  }
792 
793  } else {
794  if (prm.verbose) { std::cout << "Polish failed" << std::endl; }
795  sol.code = QPSolutionStatus::PolishFailed;
796  }
797  }
798 
799  const auto t_polish = std::chrono::high_resolution_clock::now();
800 
801  // unscale solution
802  sol.primal.applyOnTheLeft(S.template head<N>(n).asDiagonal());
803  sol.dual.applyOnTheLeft(S.template segment<M>(n, m).asDiagonal());
804  sol.dual /= c;
805  sol.objective = sol.primal.dot(0.5 * pbm.P * sol.primal + pbm.q);
806 
807  if (prm.verbose) {
808  using std::cout, std::left, std::right, std::setw, std::endl, std::chrono::microseconds;
809

815  cout << setw(25) << left << "Total time (microseconds)" << setw(10) << right << duration_cast<microseconds>(t_polish - t0).count() << endl;
816  cout << setw(25) << left << " Matrix filling" << setw(10) << right << duration_cast<microseconds>(t_fill - t0).count() << endl;
