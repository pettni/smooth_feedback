83template<LieGroup G>
84inline auto generators_sparse_reordered =
85 []() -> std::array<Eigen::SparseMatrix<double, Eigen::RowMajor>, Dof<G>> {
86 std::array<Eigen::SparseMatrix<double, Eigen::RowMajor>, Dof<G>> ret;
87 for (auto k = 0u; k < Dof<G>; ++k) {
88 ret[k].resize(Dof<G>, Dof<G>);
89 for (auto i = 0u; i < Dof<G>; ++i) { ret[k].row(i) = ::smooth::generators_sparse<G>[i].row(k); }
90 ret[k].makeCompressed();
91 }
92 return ret;
93}();

98template<LieGroup G>
99inline Eigen::SparseMatrix<double> d_ad = []() -> Eigen::SparseMatrix<double> {
100 Eigen::SparseMatrix<double> ret;
101 ret.resize(Dof<G>, Dof<G> * Dof<G>);
102 for (auto i = 0u; i < Dof<G>; ++i) {
103 for (auto j = 0u; j < Dof<G>; ++j) {
104 ret.col(j * Dof<G> + i) = smooth::generators_sparse<G>[i].row(j).transpose();
105 }
106 }
107 ret.makeCompressed();
108 return ret;
109}();

117template<LieGroup X, Manifold U, typename F, typename Xl, typename Ul>
118class FlatDyn
119{
120private:
121 using BundleT = smooth::Bundle<Eigen::Vector<double, 1>, X, U>;
122
123 static constexpr auto Nx = Dof<X>;
124 static constexpr auto Nu = Dof<U>;
125 static constexpr auto Nouts = Nx;
126 static constexpr auto Nvars = 1 + Nx + Nu;
127
128 static constexpr auto t_B = 0;
129 static constexpr auto x_B = t_B + 1;
130 static constexpr auto u_B = x_B + Nx;
131
132 using E = Tangent<X>;
133 using V = Tangent<U>;
134
135 F f;
136 Xl xl;
137 Ul ul;
138
139 Eigen::SparseMatrix<double> ad_e_ = smooth::ad_sparse_pattern<X>;
140 Eigen::SparseMatrix<double> ad_vi = smooth::ad_sparse_pattern<X>;
141
142 Eigen::SparseMatrix<double> Joplus_ = smooth::d_exp_sparse_pattern<BundleT>;
143 Eigen::SparseMatrix<double> Hoplus_ = smooth::d2_exp_sparse_pattern<BundleT>;
144
145 Eigen::SparseMatrix<double> dexpinv_e_ = smooth::d_exp_sparse_pattern<X>;
146 Eigen::SparseMatrix<double> d2expinv_e_ = smooth::d2_exp_sparse_pattern<X>;
147
148 Eigen::SparseMatrix<double> J_{Nouts, Nvars};
149 Eigen::SparseMatrix<double> H_{Nvars, Nouts * Nvars};
150
151 // Would ideally like to remove these temporaries...
152 Eigen::SparseMatrix<double> ji_{Nouts, Nvars}, ji_tmp_{Nouts, Nvars}, hi_{Nvars, Nouts *Nvars},
153 hi_tmp_{Nvars, Nouts *Nvars};
154
156 void update_joplus(const E & e, const V & v, const E & dxl, const V & dul)
157 {
158 dr_exp_sparse<BundleT>(Joplus_, (Tangent<BundleT>() << 1, e, v).finished());
159
160 block_write(Joplus_, x_B, t_B, Ad<X>(smooth::exp<X>(-e)) * dxl);
161 block_write(Joplus_, u_B, t_B, Ad<U>(smooth::exp<U>(-v)) * dul);
162
163 Joplus_.makeCompressed();
164 }
165
167 void update_hoplus(const E & e, const V & v, const E & dxl, const V & dul)
168 {
169 d2r_exp_sparse<BundleT>(Hoplus_, (Tangent<BundleT>() << 1, e, v).finished());
170
171 // d (Ad_X b) = -ad_(Ad_X b) * Ad_X
172 const TangentMap<X> Adexp_X = Ad<X>(smooth::exp<X>(-e));
173 const TangentMap<X> dAdexp_X = ad<X>(Adexp_X * dxl) * Adexp_X * dr_exp<X>(-e);
174 const TangentMap<U> Adexp_U = Ad<U>(smooth::exp<U>(-v));
175 const TangentMap<U> dAdexp_U = ad<U>(Adexp_U * dul) * Adexp_U * dr_exp<U>(-v);
176
177 for (auto nx = 0u; nx < Nx; ++nx) {
178 const auto b0 = Nvars * (x_B + nx);
179 block_write(Hoplus_, t_B, b0 + x_B, dAdexp_X.middleRows(nx, 1));
180 }
181 for (auto nu = 0u; nu < Nu; ++nu) {
182 const auto b0 = Nvars * (u_B + nu);
183 block_write(Hoplus_, t_B, b0 + u_B, dAdexp_U.middleRows(nu, 1));
184 }
185
186 Hoplus_.makeCompressed();
187 }
188
189public:
190 template<typename A1, typename A2, typename A3>
191 FlatDyn(A1 && a1, A2 && a2, A3 && a3)
192 : f(std::forward<A1>(a1)), xl(std::forward<A2>(a2)), ul(std::forward<A3>(a3))
193 {}
194
195 template<typename T>
196 CastT<T, E> operator()(const T & t, const CastT<T, E> & e, const CastT<T, V> & v) const
197 {
198 using XT = CastT<T, X>;
199
200 // can not double-differentiate, so we hide derivative of xl w.r.t. t
201 const double tdbl = static_cast<double>(t);
202 const auto [unused, dxlval] = diff::dr(xl, wrt(tdbl));
203
204 return dr_expinv<XT>(e) * (f(t, rplus(xl(t), e), rplus(ul(t), v)) - dxlval.template cast<T>())
205 + ad<XT>(e) * dxlval.template cast<T>();
206 }
207
208 // First derivative
209 std::reference_wrapper<const Eigen::SparseMatrix<double>>
210 jacobian(double t, const E & e, const V & v) requires(
211 diff::detail::diffable_order1<F, std::tuple<double, X, U>>)
212 {
213 const double tdbl = static_cast<double>(t);
214 const auto [xlval, dxlval] = diff::dr(xl, wrt(tdbl));
215 const auto [ulval, dulval] = diff::dr(ul, wrt(tdbl));
216 const auto x = rplus(xlval, e);
217 const auto u = rplus(ulval, v);
218
219 dr_expinv_sparse<X>(dexpinv_e_, e);
220 d2r_expinv_sparse<X>(d2expinv_e_, e);
221 update_joplus(e, v, dxlval, dulval);
222
223 // value and derivative of f
224 const auto fval = f(t, x, u);
225 const auto & Jf = f.jacobian(t, x, u);
226
227 // Want to differentiate drexpinv * (f o plus - dxl) + ad dxl
228
229 // Start with drexpinv * d (f \circ (+))
230 J_ = dexpinv_e_ * Jf * Joplus_;
231 // Add d ( drexpinv ) * (f \circ (+) - dxl)
232 for (auto i = 0u; i < d2expinv_e_.outerSize(); ++i) {
233 for (Eigen::InnerIterator it(d2expinv_e_, i); it; ++it) {
234 J_.coeffRef(it.col() / Nx, 1 + (it.col() % Nx)) +=
235 (fval(it.row()) - dxlval(it.row())) * it.value();
236 }
237 }
238 // Add d ( ad ) * dxl
239 for (auto i = 0u; i < d_ad<X>.outerSize(); ++i) {
240 for (Eigen::InnerIterator it(d_ad<X>, i); it; ++it) {
241 J_.coeffRef(it.col() / Nx, 1 + (it.col() % Nx)) += dxlval(it.row()) * it.value();
242 }
243 }
244
245 J_.makeCompressed();
246 return J_;
247 }
248
249 // Second derivative
250 // \sum Bn (-1)^n / n! d2r (ad_a^n f)_aa - \sum Bn / n! d2r(ad_a^n dxl)_aa
251 std::reference_wrapper<const Eigen::SparseMatrix<double>>
252 hessian(double t, const E & e, const V & v) requires(
253 diff::detail::diffable_order1<F, std::tuple<double, X, U>> &&
254 diff::detail::diffable_order2<F, std::tuple<double, X, U>>)
255 {
256 const double tdbl = static_cast<double>(t);
257 const auto [xlval, dxlval] = diff::dr(xl, wrt(tdbl));
258 const auto [ulval, dulval] = diff::dr(ul, wrt(tdbl));
259
260 const auto x = rplus(xlval, e);
261 const auto u = rplus(ul(t), v);
262 const auto & Jf = f.jacobian(t, x, u); // nx x (1 + nx + nu)
263 const auto & Hf = f.hessian(t, x, u); // (1 + nx + nu) x (nx * (1 + nx + nu))
264
265 ad_sparse<X>(ad_e_, e);
266 update_joplus(e, v, dxlval, dulval);
267 update_hoplus(e, v, dxlval, dulval);
268
269 double coef = 1; // hold (-1)^i / i!
270 Tangent<X> vi = f(t, x, u) - dxlval; // (ad_a)^i * (f - dxl)
271 ji_ = Jf * Joplus_; // dr (vi)_{t, e, v}
272 set_zero(hi_);
273 d2r_fog(hi_, Jf, Hf, Joplus_, Hoplus_); // d2r (vi)_{t, e, v}
274
275 set_zero(H_);
276 for (auto iter = 0u; iter < std::tuple_size_v<decltype(kBn)>; ++iter) {
277 if (kBn[iter] != 0) { block_add(H_, 0, 0, hi_, kBn[iter] * coef); }
278
279 // update hi_
280 hi_tmp_.setZero();
281 for (auto i = 0u; i < ad_e_.outerSize(); ++i) {
282 for (Eigen::InnerIterator it(ad_e_, i); it; ++it) {
283 const auto b0 = it.row() * Nvars;
284 block_add(hi_tmp_, 0, b0, hi_.middleCols(it.col() * Nvars, Nvars), it.value());
285 }
286 }
287 for (auto k = 0u; k < Nx; ++k) {
288 const auto b0 = k * Nvars;
289 block_add(hi_tmp_, 1, b0, generators_sparse_reordered<X>[k] * ji_);
290 block_add(hi_tmp_, 0, b0 + 1, ji_.transpose() * generators_sparse_reordered<X>[k], -1);
291 }
292 std::swap(hi_, hi_tmp_);
293
294 // update ji
295 ji_tmp_.setZero();
296 ji_tmp_ = ad_e_ * ji_;
297 ad_sparse<X>(ad_vi, vi);
298 block_add(ji_tmp_, 0, 1, ad_vi, -1);
299 std::swap(ji_, ji_tmp_);
300
301 // update vi
302 vi.applyOnTheLeft(ad_e_);
303
304 coef *= (-1.) / (iter + 1);
305 }
306
307 H_.makeCompressed();
308 return H_;
309 }
310};

433template<LieGroup X, Manifold U, std::size_t Nq, std::size_t Nouts, typename F, typename Xl>
434class FlatEndptFun
435{
436private:
437 using E = Tangent<X>;
438 using Q = Eigen::Vector<Scalar<X>, Nq>;
439 using BundleT = smooth::Bundle<Eigen::Vector<double, 1>, X, X, Q>;
440
441 F f;
442 Xl xl;
443
444 static constexpr auto Nx = Dof<X>;
445 static constexpr auto Nvars = 1 + 2 * Nx + Nq;
446
447 static constexpr auto tf_B = 0;
448 static constexpr auto x0_B = tf_B + 1;
449 static constexpr auto xf_B = x0_B + Nx;
450 static constexpr auto q_B = xf_B + Nx;
451
452 Eigen::SparseMatrix<double> Joplus_ = d_exp_sparse_pattern<BundleT>;
453 Eigen::SparseMatrix<double> Hoplus_ = d2_exp_sparse_pattern<BundleT>;
454
455 Eigen::SparseMatrix<double> J_{Nouts, Nvars};
456 Eigen::SparseMatrix<double> H_{Nvars, Nouts * Nvars};
457
459 void update_joplus(const E & e0, const E & ef, const E & dxlf)
460 {
461 dr_exp_sparse<BundleT>(Joplus_, (Tangent<BundleT>() << 1, e0, ef, Q::Ones()).finished());
462
463 block_write(Joplus_, xf_B, tf_B, Ad<X>(smooth::exp<X>(-ef)) * dxlf);
464
465 Joplus_.makeCompressed();
466 }
467
469 void update_hoplus(const E & e0, const E & ef, [[maybe_unused]] const E & dxlf)
470 {
471 d2r_exp_sparse<BundleT>(Hoplus_, (Tangent<BundleT>() << 1, e0, ef, Q::Ones()).finished());
472
473 // dr (Ad_X b)_X = -ad_{Ad_X b} Ad_X
474 const TangentMap<X> Adexp_f = Ad<X>(smooth::exp<X>(-ef));
475 const TangentMap<X> dAdexp_f = ad<X>(Adexp_f * dxlf) * Adexp_f * dr_exp<X>(-ef);
476
477 for (auto nx = 0u; nx < Nx; ++nx) {
478 const auto b0 = Nvars * (xf_B + nx);
479 block_write(Hoplus_, tf_B, b0 + xf_B, dAdexp_f.middleRows(nx, 1));
480 }
481
482 Hoplus_.makeCompressed();
483 }
484
485public:
486 template<typename A1, typename A2>
487 FlatEndptFun(A1 && a1, A2 && a2) : f(std::forward<A1>(a1)), xl(std::forward<A2>(a2))
488 {}
489
490 template<typename T>
491 auto operator()(
492 const T & tf, const CastT<T, E> & e0, const CastT<T, E> & ef, const CastT<T, Q> & q) const
493 {
494 return f.template operator()<T>(tf, rplus(xl(T(0.)), e0), rplus(xl(tf), ef), q);
495 }
496
497 std::reference_wrapper<const Eigen::SparseMatrix<double>>
498 jacobian(double tf, const E & e0, const E & ef, const Q & q) requires(
499 diff::detail::diffable_order1<F, std::tuple<double, X, X, Q>>)
500 {
501 const auto & [xlfval, dxlfval] = diff::dr(xl, wrt(tf));
502 const auto & Jf = f.jacobian(tf, rplus(xl(0.), e0), rplus(xlfval, ef), q);
503
504 update_joplus(e0, ef, dxlfval);
505
506 J_ = Jf * Joplus_;
507 J_.makeCompressed();
508 return J_;
509 }
510
511 std::reference_wrapper<const Eigen::SparseMatrix<double>>
512 hessian(double tf, const E & e0, const E & ef, const Q & q) requires(
513 diff::detail::diffable_order1<F, std::tuple<double, X, X, Q>> &&
514 diff::detail::diffable_order2<F, std::tuple<double, X, X, Q>>)
515 {
516 const auto & [xlfval, dxlfval] = diff::dr(xl, wrt(tf));
517 const auto x0 = rplus(xl(0.), e0);
518 const auto xf = rplus(xlfval, ef);
519 const auto & Jf = f.jacobian(tf, x0, xf, q); // Nouts x Nx
520 const auto & Hf = f.hessian(tf, x0, xf, q); // Nx x (Nouts * Nx)
521
522 update_joplus(e0, ef, dxlfval);
523 update_hoplus(e0, ef, dxlfval);
524
525 set_zero(H_);
526 d2r_fog(H_, Jf, Hf, Joplus_, Hoplus_);
527 H_.makeCompressed();
528 return H_;
529 }

549auto flatten_ocp(const OCPType auto & ocp, auto && xl, auto && ul)
550{
551 using ocp_t = std::decay_t<decltype(ocp)>;
552 using X = typename ocp_t::X;
553 using U = typename ocp_t::U;
554 using Xl = decltype(xl);
555 using Ul = decltype(ul);
556
557 static constexpr auto Nq = ocp_t::Nq;
558
559 return OCP<
560 Tangent<X>,
561 Tangent<U>,
562 detail::FlatEndptFun<X, U, Nq, 1, decltype(ocp.theta), Xl>,
563 detail::FlatDyn<X, U, decltype(ocp.f), Xl, Ul>,
564 detail::FlatInnerFun<X, U, Nq, decltype(ocp.g), Xl, Ul>,
565 detail::FlatInnerFun<X, U, ocp_t::Ncr, decltype(ocp.cr), Xl, Ul>,
566 detail::FlatEndptFun<X, U, Nq, ocp_t::Nce, decltype(ocp.ce), Xl>>{
567 .theta = detail::FlatEndptFun<X, U, Nq, 1, decltype(ocp.theta), Xl>{ocp.theta, xl},
568 .f = detail::FlatDyn<X, U, decltype(ocp.f), Xl, Ul>{ocp.f, xl, ul},
569 .g = detail::FlatInnerFun<X, U, Nq, decltype(ocp.g), Xl, Ul>{ocp.g, xl, ul},
570 .cr = detail::FlatInnerFun<X, U, ocp_t::Ncr, decltype(ocp.cr), Xl, Ul>{ocp.cr, xl, ul},
571 .crl = ocp.crl,
572 .cru = ocp.cru,
573 .ce = detail::FlatEndptFun<X, U, Nq, ocp_t::Nce, decltype(ocp.ce), Xl>{ocp.ce, xl},
574 .cel = ocp.cel,
575 .ceu = ocp.ceu,
576 };

585template<LieGroup X, Manifold U>
586auto unflatten_ocpsol(const auto & flatsol, auto && xl_fun, auto && ul_fun)
587{
588 using ocpsol_t = std::decay_t<decltype(flatsol)>;
589
590 auto u_unflat = [ul_fun = std::forward<decltype(ul_fun)>(ul_fun),
591 usol = flatsol.u](double t) -> U { return rplus(ul_fun(t), usol(t)); };
592
593 auto x_unflat = [xl_fun = std::forward<decltype(xl_fun)>(xl_fun),
