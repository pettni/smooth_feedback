106 template<std::size_t K, LieGroup G, Manifold U, typename Dyn, typename SafeSet, typename BackupU>
107 auto asif_to_qp(const G & x0,
108  const U & u_des,
109  Dyn && f,
110  SafeSet && h,
111  BackupU && bu,
112  const AsifParams & prm,
113  const U & u_lin = U::Zero())
114 {
115  using boost::numeric::odeint::euler, boost::numeric::odeint::vector_space_algebra;
116  using std::placeholders::_1;
117  using Tangent = typename G::Tangent;
118  using TangentMap = typename G::TangentMap;
119 
120  static constexpr int nx = G::SizeAtCompileTime;
121  static constexpr int nu = U::SizeAtCompileTime;
122  static constexpr int nh = std::invoke_result_t<SafeSet, G>::SizeAtCompileTime;
123 
124  euler<G, double, Tangent, double, vector_space_algebra> state_stepper{};
125  euler<TangentMap, double, TangentMap, double, vector_space_algebra> sensi_stepper{};
126 
127  static constexpr int M = K * nh + nu;
128  static constexpr int N = nu + 1;
129 
130  // iteration variables
131  double t = 0;
132  G x = x0;
133  TangentMap dx_dx0 = TangentMap::Identity();
134 
135  // define ODEs for closed-loop dynamics and its sensitivity
136  const auto x_ode = [&f, &bu](const auto & xx, auto & dd, double) { dd = f(xx, bu(xx)); };
137  const auto dx_dx0_ode = [&f, &bu, &x](const auto & S_v, auto & dS_dt_v, double) {
138  const auto [fcl, dr_fcl_dx] = diff::dr([&](const auto & xx) { return f(xx, bu(xx)); }, wrt(x));
139  dS_dt_v = (-G::ad(fcl) + dr_fcl_dx) * S_v;
140  };
141 
142  // value of dynamics at call time
143  const auto [f0, d_f0_du] = diff::dr(std::bind(f, x, _1), wrt(u_lin));
144 
145  QuadraticProgram<M, N> ret;
146 
147  // loop over constraint number
148  for (auto k = 0u; k != K; ++k) {
149  // differentiate barrier function w.r.t. x
150  const auto [hval, dh_dx] = diff::dr([&](const auto & xx) { return h(xx); }, wrt(x));
151 
152  // insert barrier constraint
153  const Eigen::Matrix<double, nh, nx> dh_dx0 = dh_dx * dx_dx0;
154  ret.A.template block<nh, nu>(k * nh, 0) = dh_dx0 * d_f0_du;
155  ret.l.template segment<nh>(k * nh) = -prm.alpha * hval - dh_dx0 * f0;
156  ret.u.template segment<nh>(k * nh).setZero();
157 
158  // integrate system and sensitivity forward until next constraint
159  while (t < prm.tau * (k + 1)) {
160  state_stepper.do_step(x_ode, x, t, prm.dt);
161  sensi_stepper.do_step(dx_dx0_ode, dx_dx0, t, prm.dt);
162  t += std::min(prm.dt, prm.tau);
163  }
164  }
165 
166  // set input constraints
167  ret.A.template block<K * nh, 1>(0, nu).setConstant(1); // coeffs for delta (upper part)
168  ret.A.template block<nu, 1>(K * nh, nu).setConstant(0); // coeffs for delta (lower part)
169  ret.A.template bottomRows<nu>().setIdentity();
170  ret.l.template tail<nu>().setConstant(-1);
171  ret.u.template tail<nu>().setConstant(1);
172 
173  ret.P.setIdentity();
174  ret.P(nu, nu) = prm.relax_cost;
175  ret.q.template head<nu>() = (u_des - u_lin);
176  ret.q(nu) = 0;
177 
178  return ret;
179 }
