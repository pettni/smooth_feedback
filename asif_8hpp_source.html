108 template<std::size_t K, LieGroup G, Manifold U, typename Dyn, typename SafeSet, typename BackupU>
109 auto asif_to_qp(const G & x0,
110  const U & u_des,
111  Dyn && f,
112  SafeSet && h,
113  BackupU && bu,
114  const AsifParams & prm,
115  const U & u_lin = identity<U>())
116 {
117  using boost::numeric::odeint::euler, boost::numeric::odeint::vector_space_algebra;
118  using std::placeholders::_1;
119  using Tangent = typename G::Tangent;
120  using TangentMap = typename G::TangentMap;
121 
122  static constexpr int nx = G::SizeAtCompileTime;
123  static constexpr int nu = U::SizeAtCompileTime;
124  static constexpr int nh = std::invoke_result_t<SafeSet, double, G>::SizeAtCompileTime;
125 
126  euler<G, double, Tangent, double, vector_space_algebra> state_stepper{};
127  euler<TangentMap, double, TangentMap, double, vector_space_algebra> sensi_stepper{};
128 
129  static constexpr int M = K * nh + nu + 1;
130  static constexpr int N = nu + 1;
131 
132  // iteration variables
133  double t = 0;
134  G x = x0;
135  TangentMap dx_dx0 = TangentMap::Identity();
136 
137  // define ODEs for closed-loop dynamics and its sensitivity
138  const auto x_ode = [&f, &bu](const G & xx, Tangent & dd, double tt) {
139  dd = f(tt, xx, bu(tt, xx));
140  };
141 
142  const auto dx_dx0_ode = [&f, &bu, &x](const auto & S_v, auto & dS_dt_v, double tt) {
143  auto f_cl = [&](const auto & vx) {
144  using T = typename std::decay_t<decltype(vx)>::Scalar;
145  return f(T(tt), vx, bu(T(tt), vx));
146  };
147  const auto [fcl, dr_fcl_dx] = diff::dr(std::move(f_cl), wrt(x));
148  dS_dt_v = (-G::ad(fcl) + dr_fcl_dx) * S_v;
149  };
150 
151  // value of dynamics at call time
152  const auto [f0, d_f0_du] = diff::dr(
153  [&](const auto & vu) {
154  using T = typename std::decay_t<decltype(vu)>::Scalar;
155  return f(T(t), x.template cast<T>(), vu);
156  },
157  wrt(u_lin));
158 
159  QuadraticProgram<-1, -1> ret;
160 
161  ret.A.resize(M, N);
162  ret.l.resize(M);
163  ret.u.resize(M);
164 
165  ret.P.resize(N, N);
166  ret.q.resize(N);
167 
168  // loop over constraint number
169  for (auto k = 0u; k != K; ++k) {
170  // differentiate barrier function w.r.t. x
171  Eigen::Matrix<double, 1, 1> tvec(t);
172  const auto [hval, dh_dtx] =
173  diff::dr([&h](const auto & vt, const auto & vx) { return h(vt(0), vx); }, wrt(tvec, x));
174 
175  const Eigen::Matrix<double, nh, 1> dh_dt = dh_dtx.template leftCols<1>();
176  const Eigen::Matrix<double, nh, nx> dh_dx = dh_dtx.template rightCols<nx>();
177 
178  // insert barrier constraint
179  const Eigen::Matrix<double, nh, nx> dh_dx0 = dh_dx * dx_dx0;
180  ret.A.template block<nh, nu>(k * nh, 0) = dh_dx0 * d_f0_du;
181  ret.l.template segment<nh>(k * nh) = -dh_dt - prm.alpha * hval - dh_dx0 * f0;
182  ret.u.template segment<nh>(k * nh).setConstant(std::numeric_limits<double>::infinity());
183 
184  // integrate system and sensitivity forward until next constraint
185  while (t < prm.tau * (k + 1)) {
186  state_stepper.do_step(x_ode, x, t, prm.dt);
187  sensi_stepper.do_step(dx_dx0_ode, dx_dx0, t, prm.dt);
188  t += std::min(prm.dt, prm.tau);
189  }
190  }
191 
192  ret.A.template block<K * nh, 1>(0, nu).setConstant(1);
193 
194  // identity matrix for inquality constraints
195  ret.A.template block<nu + 1, nu + 1>(K * nh, 0).setIdentity();
196 
197  // upper and lower bounds on u
198  ret.l.template segment<nu>(K * nh, 0).setConstant(-1);
199  ret.u.template segment<nu>(K * nh, 0).setConstant(1);
200 
201  // upper and lower bounds on delta
202  ret.l(K * nh + nu) = 0;
203  ret.u(K * nh + nu) = std::numeric_limits<double>::infinity();
204 
205  ret.P.setIdentity();
206  ret.P(nu, nu) = prm.relax_cost;
207  ret.q.template head<nu>() = (u_lin - u_des);
208  ret.q(nu) = 0;
209 
210  return ret;
211 }
