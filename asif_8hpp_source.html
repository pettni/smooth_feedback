106 std::pair<U, QPSolutionStatus> operator()(const G & g, const U & u_des, auto && h, auto && bu)
107 {
108 using std::chrono::duration, std::chrono::duration_cast, std::chrono::nanoseconds;
109
110 assert((prm_.nh == std::invoke_result_t<decltype(h), Scalar<G>, G>::RowsAtCompileTime));
111
112 ASIFProblem<G, U> pbm{
113 .T = prm_.T,
114 .x0 = g,
115 .u_des = u_des,
116 .W_u = prm_.u_weight,
117 .ulim = prm_.ulim,
118 };
119
120 asif_to_qp_update<G, U, DT>(
121 qp_, pbm, prm_.asif, f_, std::forward<decltype(h)>(h), std::forward<decltype(bu)>(bu));
122 auto sol = feedback::solve_qp(qp_, prm_.qp, warmstart_);
123
124 if (sol.code == QPSolutionStatus::Optimal) { warmstart_ = sol; }
125
126 return {rplus(u_des, sol.primal.template head<Dof<U>>()), sol.code};
127 }
128
129private:
130 Dyn f_;
131
132 QuadraticProgram<-1, -1, double> qp_;
133 ASIFilterParams<U> prm_;
134 std::optional<QPSolution<-1, -1, double>> warmstart_;
135};
