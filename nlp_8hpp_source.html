54template<typename T>
55concept NLP = requires(
56 std::decay_t<T> & nlp,
57 const Eigen::Ref<const Eigen::VectorXd> x,
58 const Eigen::Ref<const Eigen::VectorXd> lambda)
59{
60 // clang-format off
61 {nlp.n()} -> std::convertible_to<std::size_t>;
62 {nlp.m()} -> std::convertible_to<std::size_t>;
63
64 // variable bounds
65 {nlp.xl()} -> std::convertible_to<Eigen::VectorXd>;
66 {nlp.xu()} -> std::convertible_to<Eigen::VectorXd>;
67
68 // objective and its derivatives
69 {nlp.f(x)} -> std::convertible_to<double>;
70 {nlp.df_dx(x)} -> std::convertible_to<Eigen::SparseMatrix<double>>;
71
72 // constraint, its bounds, and its derivatives
73 {nlp.g(x)} -> std::convertible_to<Eigen::VectorXd>;
74 {nlp.gl()} -> std::convertible_to<Eigen::VectorXd>;
75 {nlp.gu()} -> std::convertible_to<Eigen::VectorXd>;
76 {nlp.dg_dx(x)} -> std::convertible_to<Eigen::SparseMatrix<double>>;
77 // clang-format on
78};
