103  template<typename F, typename QDer>
104  void predict(
105  F && f, const Eigen::MatrixBase<QDer> & Q, Scalar<G> tau, std::optional<Scalar<G>> dt = {})
106  {
107  const auto state_ode = [&f](const G & g, Tangent<G> & dg, Scalar<G> t) { dg = f(t, g); };
108 
109  const auto cov_ode = [this, &f, &Q](const CovT & cov, CovT & dcov, Scalar<G> t) {
110  const auto f_x = [&f, &t]<typename T>(
111  const CastT<T, G> & x) -> Tangent<CastT<T, G>> { return f(t, x); };
112  const auto [fv, dr] = diff::dr<DiffType>(f_x, wrt(g_hat_));
113  const CovT A = -ad<G>(fv) + dr;
114  dcov = (A * cov + cov * A.transpose() + Q).template selfadjointView<Eigen::Upper>();
115  };
116 
117  Scalar<G> t = 0;
118  const Scalar<G> dt_v = dt.value_or(2 * tau);
119  while (t + dt_v < tau) {
120  // step covariance first since it depends on g_hat_
121  cst_.do_step(cov_ode, P_, t, dt_v);
122  sst_.do_step(state_ode, g_hat_, t, dt_v);
123  t += dt_v;
124  }
125 
126  // last step up to time t
127  cst_.do_step(cov_ode, P_, t, tau - t);
128  sst_.do_step(state_ode, g_hat_, t, tau - t);
129  }

142  template<typename F, typename RDev, Manifold Y = std::invoke_result_t<F, G>>
143  void update(F && h, const Y & y, const Eigen::MatrixBase<RDev> & R)
144  {
145  const auto [hval, H] = diff::dr<DiffType>(h, wrt(g_hat_));
146 
147  using Result = std::decay_t<decltype(hval)>;
148 
149  static_assert(Manifold<Result>, "h(x) is not a Manifold");
150 
151  static constexpr Eigen::Index Ny = Dof<Result>;
152 
153  static_assert(Ny > 0, "h(x) must be statically sized");
154 
155  const Eigen::Matrix<Scalar<G>, Ny, Ny> S =
156  (H * P_.template selfadjointView<Eigen::Upper>() * H.transpose() + R)
157  .template triangularView<Eigen::Upper>();
158 
159  // solve for Kalman gain
160  const Eigen::Matrix<Scalar<G>, Dof<G>, Ny> K =
161  S.template selfadjointView<Eigen::Upper>().ldlt().solve(H * P_).transpose();
162 
163  // update estimate and covariance
164  g_hat_ += K * (y - hval);
165  P_ = ((CovT::Identity() - K * H) * P_).template selfadjointView<Eigen::Upper>();
166  }

169  // filter estimate and covariance
