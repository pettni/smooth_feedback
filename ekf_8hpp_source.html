103 template<typename F, typename QDer>
104 void predict(
105 F && f, const Eigen::MatrixBase<QDer> & Q, Scalar<G> tau, std::optional<Scalar<G>> dt = {})
106 {
107 const auto state_ode = [&f](const G & g, Tangent<G> & dg, Scalar<G> t) { dg = f(t, g); };
108
109 const auto cov_ode = [this, &f, &Q](const CovT & cov, CovT & dcov, Scalar<G> t) {
110 const auto f_x = [&f, &t]<typename _T>(const CastT<_T, G> & x) -> Tangent<CastT<_T, G>> {
111 return f(t, x);
112 };
113 const auto [fv, dr] = diff::dr<1, DiffType>(f_x, wrt(g_hat_));
114 const CovT A = -ad<G>(fv) + dr;
115 dcov = (A * cov + cov * A.transpose() + Q).template selfadjointView<Eigen::Upper>();
116 };
117
118 Scalar<G> t = 0;
119 const Scalar<G> dt_v = dt.value_or(2 * tau);
120 while (t + dt_v < tau) {
121 // step covariance first since it depends on g_hat_
122 cst_.do_step(cov_ode, P_, t, dt_v);
123 sst_.do_step(state_ode, g_hat_, t, dt_v);
124 t += dt_v;
125 }
126
127 // last step up to time t
128 cst_.do_step(cov_ode, P_, t, tau - t);
129 sst_.do_step(state_ode, g_hat_, t, tau - t);
130 }

143 template<typename F, typename RDev, Manifold Y = std::invoke_result_t<F, G>>
144 void update(F && h, const Y & y, const Eigen::MatrixBase<RDev> & R)
145 {
146 const auto [hval, H] = diff::dr<1, DiffType>(h, wrt(g_hat_));
147
148 using Result = std::decay_t<decltype(hval)>;
149
150 static_assert(Manifold<Result>, "h(x) is not a Manifold");
151
152 static constexpr Eigen::Index Ny = Dof<Result>;
153
154 static_assert(Ny > 0, "h(x) must be statically sized");
155
156 const Eigen::Matrix<Scalar<G>, Ny, Ny> S =
157 (H * P_.template selfadjointView<Eigen::Upper>() * H.transpose() + R)
158 .template triangularView<Eigen::Upper>();
159
160 // solve for Kalman gain
161 const Eigen::Matrix<Scalar<G>, Dof<G>, Ny> K =
162 S.template selfadjointView<Eigen::Upper>().ldlt().solve(H * P_).transpose();
163
164 // update estimate and covariance
165 g_hat_ += K * (y - hval);
166 P_ = ((CovT::Identity() - K * H) * P_).template selfadjointView<Eigen::Upper>();
167 }
168
169private:
170 // filter estimate and covariance
171 G g_hat_ = Default<G>();
172 CovT P_ = CovT::Identity();
173
174 // steppers for numerical ODE solutions
175 Stpr<G, Scalar<G>, Tangent<G>, Scalar<G>, boost::numeric::odeint::vector_space_algebra> sst_{};
176 Stpr<CovT, Scalar<G>, CovT, Scalar<G>, boost::numeric::odeint::vector_space_algebra> cst_{};
177};
