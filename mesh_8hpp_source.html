58template<std::size_t K, std::size_t I = 8>
59constexpr std::pair<std::array<double, K + 1>, std::array<double, K + 1>> lgr_plus_one()
60{
61 auto lgr_norm = ::smooth::lgr_nodes<K, I>();
62
63 std::array<double, K + 1> ns, ws;
64 for (auto i = 0u; i < K; ++i) {
65 ns[i] = lgr_norm.first[i];
66 ws[i] = lgr_norm.second[i];
67 }
68 ns[K] = 1;
69 ws[K] = 0;
70 return {ns, ws};
71}

149 inline void refine_ph(std::size_t i, std::size_t D)
150 {
151 assert(i < intervals_.size());
152 if (D > Kmax || intervals_[i].K > Kmax) {
153 // refine by splitting interval into n intervals, each with degree Kmin_
154 std::size_t n = std::max<std::size_t>(2u, (D + Kmin - 1) / Kmin);
155
156 const double tau0 = intervals_[i].tau0;
157 const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
158 const double taum = (tauf - tau0) / n;
159
160 while (n-- > 1) {
161 intervals_.insert(intervals_.begin() + i + 1, Interval{.K = Kmin, .tau0 = tau0 + n * taum});
162 }
163 } else if (D < intervals_[i].K) {
164 return;
165 } else if (D <= Kmax) {
166 // refine by increasing degree in interval
167 intervals_[i].K = D;
168 }
169 }

176 inline void refine_errors(std::ranges::sized_range auto && errs, double target_err)
177 {
178 const auto N = N_ivals();
179
180 assert(N == std::size_t(std::ranges::size(errs)));
181
182 for (const auto & [i, e] : zip(iota(0u, N) | reverse, errs | reverse)) {
183
184 const auto Ki = N_colloc_ival(i);
185
186 if (e > target_err) {
187 const auto Ktarget = Ki + std::lround(std::log(e / target_err) / std::log(Ki) + 1);
188 refine_ph(i, Ktarget);
189 }
190 }
191 }

210 inline auto interval_nodes(std::size_t i) const
211 {
212 const std::size_t k = intervals_[i].K;
213
214 assert(Kmin <= k && k <= Kmax + 1);
215
216 std::span<const double> sp;
217
218 utils::static_for<Kmax + 2 - Kmin>([&](auto i) {
219 static constexpr auto K = Kmin + i;
220 if (K == k) {
221 static constexpr auto nw_ext_s = detail::lgr_plus_one<K>();
222
223 sp = std::span<const double>(nw_ext_s.first.data(), k + 1);
224 }
225 });
226
227 const double tau0 = intervals_[i].tau0;
228 const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
229 const double al = (tauf - tau0) / 2;
230
231 return transform(std::move(sp), [tau0, al](double d) -> double { return tau0 + al * (d + 1); });
232 }

241 inline auto all_nodes() const
242 {

314 inline Eigen::MatrixXd interval_diffmat(std::size_t i) const
315 {
316 const std::size_t k = intervals_[i].K;
317
318 const double tau0 = intervals_[i].tau0;
319 const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
320
321 Eigen::MatrixXd ret(k + 1, k);
322
323 utils::static_for<Kmax + 2 - Kmin>([&](auto i) {
324 static constexpr auto K = Kmin + i;
325 if (K == k) {
326 static constexpr auto nw_ext_s = detail::lgr_plus_one<K>();
327 static constexpr auto B_ext_s = lagrange_basis<K>(nw_ext_s.first);
328 static constexpr auto D_ext_s =
329 polynomial_basis_derivatives<K, K + 1>(B_ext_s, nw_ext_s.first)
330 .template block<K + 1, K>(0, 0);
331 ret = MatMap(D_ext_s[0].data(), k + 1, k);
332 }
333 });
334
335 ret *= 2. / (tauf - tau0);
336 return ret;
337 }

400 inline std::size_t interval_find(double t) const
401 {
402 if (t < 0) { return 0; }
403 if (t > 1) { return intervals_.size() - 1; }
404 auto it = utils::binary_interval_search(
405 intervals_, t, [](const auto & ival, double _t) { return ival.tau0 <=> _t; });
406 if (it != intervals_.end()) { return std::distance(intervals_.begin(), it); }
407 return 0;
408 }

436 template<smooth::traits::RnType RetT>
437 RetT eval(double t, std::ranges::range auto && r, std::size_t p = 0, bool extend = true) const
438 {
439 const std::size_t ival = interval_find(t);
440 const std::size_t k = intervals_[ival].K;
441
442 const double tau0 = intervals_[ival].tau0;
443 const double tauf = ival + 1 < intervals_.size() ? intervals_[ival + 1].tau0 : 1.;
444
445 const double u = 2 * (t - tau0) / (tauf - tau0) - 1;
446
447 int64_t N_before = 0;
448 for (auto i = 0u; i < ival; ++i) { N_before += intervals_[i].K; }
449
450 // initialize output variable
451 RetT ret = RetT::Zero(dof(*std::ranges::begin(r)));
452
453 utils::static_for<Kmax + 2 - Kmin>([&](auto i) {
454 static constexpr auto K = Kmin + i;
455 if (K == k) {
456 if (extend || ival + 1 < intervals_.size()) {
457 static constexpr auto nw_ext_s = detail::lgr_plus_one<K>();
458 static constexpr auto B_ext_s = lagrange_basis<K>(nw_ext_s.first); // K+1 x K+1
459 const auto U = monomial_derivative<K>(u, p); // 1 x K+1
460 const auto W = U * B_ext_s; // 1 x K+1
461
462 for (const auto & [w, v] : zip(std::span(W[0].data(), k + 1), r | drop(N_before))) {
463 ret += w * v;
464 }
465 } else {
466 static constexpr auto nw_s = lgr_nodes<K>();
467 static constexpr auto B_s = lagrange_basis<K - 1>(nw_s.first); // K x K
468 const auto U = monomial_derivative<K - 1>(u, p); // 1 x K
469 const auto W = U * B_s; // 1 x K
470
471 for (const auto & [w, v] : zip(std::span(W[0].data(), k), r | drop(N_before))) {
472 ret += w * v;
473 }
474 }
475 }
476 });
477
478 return ret;
479 }
