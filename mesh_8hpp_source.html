58template<std::size_t K, std::size_t I = 8>
59constexpr std::pair<std::array<double, K + 1>, std::array<double, K + 1>> lgr_plus_one()
60{
61 auto lgr_norm = ::smooth::lgr_nodes<K, I>();
62
63 std::array<double, K + 1> ns, ws;
64 for (auto i = 0u; i < K; ++i) {
65 ns[i] = lgr_norm.first[i];
66 ws[i] = lgr_norm.second[i];
67 }
68 ns[K] = 1;
69 ws[K] = 0;
70 return {ns, ws};
71}

116 inline Mesh(const std::size_t n, const std::size_t k = Kmin)
117 {
118 assert(Kmin <= k && k <= Kmax + 1);
119
120 if (n < 2) {
121 intervals_.emplace_back(k, 0.);
122 } else {
123 const double dx = 1. / static_cast<double>(n);
124 intervals_.reserve(n);
125 for (std::size_t i = 0; i < n; ++i) {
126 intervals_.emplace_back(k, static_cast<double>(i) * dx);
127 }
128 }
129 }

172 inline void refine_ph(std::size_t i, std::size_t D)
173 {
174 assert(i < intervals_.size());
175 if (D > Kmax || intervals_[i].K > Kmax) {
176 // refine by splitting interval into n intervals, each with degree Kmin_
177 std::size_t n = std::max<std::size_t>(2u, (D + Kmin - 1) / Kmin);
178
179 const double tau0 = intervals_[i].tau0;
180 const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
181 const double taum = (tauf - tau0) / n;
182
183 while (n-- > 1) {
184 intervals_.insert(intervals_.begin() + i + 1, Interval{.K = Kmin, .tau0 = tau0 + n * taum});
185 }
186 } else if (D < intervals_[i].K) {
187 return;
188 } else if (D <= Kmax) {
189 // refine by increasing degree in interval
190 intervals_[i].K = D;
191 }
192 }

199 inline void refine_errors(std::ranges::sized_range auto && errs, double target_err)
200 {
201 const auto N = N_ivals();
202
203 assert(N == std::size_t(std::ranges::size(errs)));
204
205 for (const auto & [i, e] : zip(iota(0u, N) | reverse, errs | reverse)) {
206
207 const auto Ki = N_colloc_ival(i);
208
209 if (e > target_err) {
210 const auto Ktarget = Ki + std::lround(std::log(e / target_err) / std::log(Ki) + 1);
211 refine_ph(i, Ktarget);
212 }
213 }
214 }

233 inline auto interval_nodes(std::size_t i) const
234 {
235 const std::size_t k = intervals_[i].K;
236
237 assert(Kmin <= k && k <= Kmax + 1);
238
239 std::span<const double> sp;
240
241 utils::static_for<Kmax + 2 - Kmin>([&](auto ivar) {
242 static constexpr auto K = Kmin + ivar;
243 if (K == k) {
244 static constexpr auto nw_ext_s = detail::lgr_plus_one<K>();
245
246 sp = std::span<const double>(nw_ext_s.first.data(), k + 1);
247 }
248 });
249
250 const double tau0 = intervals_[i].tau0;
251 const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
252 const double al = (tauf - tau0) / 2;
253
254 return transform(std::move(sp), [tau0, al](double d) -> double { return tau0 + al * (d + 1); });
255 }

264 inline auto all_nodes() const
265 {
266 const auto n_ivals = N_ivals();
267 auto all_views = iota(0u, n_ivals) | transform([this, n_ivals = n_ivals](auto i) {
268 const auto n_ival = N_colloc_ival(i);
269 const int64_t n_take = i + 1 < n_ivals ? n_ival : n_ival + 1;
270 return interval_nodes(i) | take(n_take);

337 inline Eigen::MatrixXd interval_diffmat(std::size_t i) const
338 {
339 const std::size_t k = intervals_[i].K;
340
341 const double tau0 = intervals_[i].tau0;
342 const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
343
344 Eigen::MatrixXd ret(k + 1, k);
345
346 utils::static_for<Kmax + 2 - Kmin>([&](auto ivar) {
347 static constexpr auto K = Kmin + ivar;
348 if (K == k) {
349 static constexpr auto nw_ext_s = detail::lgr_plus_one<K>();
350 static constexpr auto B_ext_s = lagrange_basis<K>(nw_ext_s.first);
351 static constexpr auto D_ext_s =
352 polynomial_basis_derivatives<K, K + 1>(B_ext_s, nw_ext_s.first)
353 .template block<K + 1, K>(0, 0);
354 ret = MatMap(D_ext_s[0].data(), k + 1, k);
355 }
356 });
357
358 ret *= 2. / (tauf - tau0);
359 return ret;
360 }

423 inline std::size_t interval_find(double t) const
424 {
425 if (t < 0) { return 0; }
426 if (t > 1) { return intervals_.size() - 1; }
427 auto it = utils::binary_interval_search(
428 intervals_, t, [](const auto & ival, double _t) { return ival.tau0 <=> _t; });
429 if (it != intervals_.end()) { return std::distance(intervals_.begin(), it); }
430 return 0;
431 }

459 template<smooth::traits::RnType RetT>
460 RetT eval(double t, std::ranges::range auto && r, std::size_t p = 0, bool extend = true) const
461 {
462 const std::size_t ival = interval_find(t);
463 const std::size_t k = intervals_[ival].K;
464
465 const double tau0 = intervals_[ival].tau0;
466 const double tauf = ival + 1 < intervals_.size() ? intervals_[ival + 1].tau0 : 1.;
467
468 const double u = 2 * (t - tau0) / (tauf - tau0) - 1;
469
470 int64_t N_before = 0;
471 for (auto i = 0u; i < ival; ++i) { N_before += intervals_[i].K; }
472
473 // initialize output variable
474 RetT ret = RetT::Zero(dof(*std::ranges::begin(r)));
475
476 utils::static_for<Kmax + 2 - Kmin>([&](auto i) {
477 static constexpr auto K = Kmin + i;
478 if (K == k) {
479 if (extend || ival + 1 < intervals_.size()) {
480 static constexpr auto nw_ext_s = detail::lgr_plus_one<K>();
481 static constexpr auto B_ext_s = lagrange_basis<K>(nw_ext_s.first); // K+1 x K+1
482 const auto U = monomial_derivative<K>(u, p); // 1 x K+1
483 const auto W = U * B_ext_s; // 1 x K+1
484
485 for (const auto & [w, v] : zip(std::span(W[0].data(), k + 1), r | drop(N_before))) {
486 ret += w * v;
487 }
488 } else {
489 static constexpr auto nw_s = lgr_nodes<K>();
490 static constexpr auto B_s = lagrange_basis<K - 1>(nw_s.first); // K x K
491 const auto U = monomial_derivative<K - 1>(u, p); // 1 x K
492 const auto W = U * B_s; // 1 x K
493
494 for (const auto & [w, v] : zip(std::span(W[0].data(), k), r | drop(N_before))) {
495 ret += w * v;
496 }
497 }
498 }
499 });
500
501 return ret;
502 }
