51 template<std::size_t K, std::size_t I = 8>
52 constexpr std::pair<std::array<double, K + 1>, std::array<double, K + 1>> lgr_plus_one()
53 {
54  auto lgr_norm = ::smooth::lgr_nodes<K, I>();
55 
56  std::array<double, K + 1> ns, ws;
57  for (auto i = 0u; i < K; ++i) {
58  ns[i] = lgr_norm.first[i];
59  ws[i] = lgr_norm.second[i];
60  }
61  ns[K] = 1;
62  ws[K] = 0;
63  return {ns, ws};
64 }

134  inline void refine_ph(std::size_t i, std::size_t D)
135  {
136  if (D > Kmax || intervals_[i].K > Kmax) {
137  // refine by splitting interval into n intervals, each with degree Kmin_
138  std::size_t n = std::max<std::size_t>(2u, (D + Kmin - 1) / Kmin);
139 
140  const double tau0 = intervals_[i].tau0;
141  const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
142  const double taum = (tauf - tau0) / n;
143 
144  while (n-- > 1) {
145  intervals_.insert(intervals_.begin() + i + 1, Interval{.K = Kmin, .tau0 = tau0 + n * taum});
146  }
147  } else if (D < intervals_[i].K) {
148  return;
149  } else if (D <= Kmax) {
150  // refine by increasing degree in interval
151  intervals_[i].K = D;
152  }
153  }

170  inline std::pair<Eigen::VectorXd, Eigen::VectorXd> interval_nodes_and_weights(std::size_t i) const
171  {
172  const std::size_t k = intervals_[i].K;
173 
174  Eigen::VectorXd ns, ws;
175  utils::static_for<Kmax + 2 - Kmin>([&](auto i) {
176  static constexpr auto K = Kmin + i;
177  if (K == k) {
178  static constexpr auto nw_ext_s = detail::lgr_plus_one<K>();
179  ns = Eigen::Map<const Eigen::VectorXd>(nw_ext_s.first.data(), k + 1);
180  ws = Eigen::Map<const Eigen::VectorXd>(nw_ext_s.second.data(), k + 1);
181  }
182  });
183 
184  const double tau0 = intervals_[i].tau0;
185  const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
186  const double alpha = (tauf - tau0) / 2;
187 
188  return {
189  Eigen::VectorXd::Constant(ns.size(), tau0) + alpha * (ns + Eigen::VectorXd::Ones(ns.size())),
190  alpha * ws,
191  };
192  }

197  inline std::pair<Eigen::VectorXd, Eigen::VectorXd> all_nodes_and_weights() const
198  {
199  Eigen::VectorXd n(N_colloc() + 1), w(N_colloc() + 1);
200 
201  std::size_t cntr = 0;
202  for (auto i = 0u; i < intervals_.size(); ++i) {
203  auto [ni, wi] = interval_nodes_and_weights(i);
204 
205  const std::size_t Ni = ni.size();
206 
207  // exclude last point that belongs to next interval..
208  n.segment(cntr, Ni - 1) = ni.head(Ni - 1);
209  w.segment(cntr, Ni - 1) = wi.head(Ni - 1);
210 
211  cntr += Ni - 1;
212  }
213 
214  n.tail(1).setConstant(1);
215  w.tail(1).setConstant(0);
216 
217  return {n, w};
218  }

231  inline Eigen::MatrixXd interval_diffmat(std::size_t i) const
232  {
233  const std::size_t k = intervals_[i].K;
234 
235  const double tau0 = intervals_[i].tau0;
236  const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
237 
238  Eigen::MatrixXd ret;
239  utils::static_for<Kmax + 2 - Kmin>([&](auto i) {
240  static constexpr auto K = Kmin + i;
241  if (K == k) {
242  static constexpr auto nw_ext_s = detail::lgr_plus_one<K>();
243  static constexpr auto B_ext_s = lagrange_basis<K>(nw_ext_s.first);
244  static constexpr auto D_ext_s =
245  polynomial_basis_derivatives<K, K + 1>(B_ext_s, nw_ext_s.first)
246  .template block<K + 1, K>(0, 0);
247  ret = MatMap(D_ext_s[0].data(), k + 1, k);
248  ;
249  }
250  });
251 
252  return (2. / (tauf - tau0)) * ret;
253  }

280  inline std::size_t interval_find(double t) const
281  {
282  if (t < 0) { return 0; }

285  intervals_, t, [](const auto & ival, double _t) { return ival.tau0 <=> _t; });

300  template<typename RetT, std::ranges::sized_range R>
301  RetT eval(double t, const R & r, std::size_t p = 0, bool extend = true) const
302  {
303  [[maybe_unused]] const std::size_t N = N_colloc();
304 
305  if (extend) {
306  assert(std::ranges::size(r) == N + 1);
307  } else {
308  assert(std::ranges::size(r) == N);
309  }
310 
311  const std::size_t ival = interval_find(t);
312  const std::size_t k = intervals_[ival].K;
313 
314  const double tau0 = intervals_[ival].tau0;
315  const double tauf = ival + 1 < intervals_.size() ? intervals_[ival + 1].tau0 : 1.;
316 
317  const double u = 2 * (t - tau0) / (tauf - tau0) - 1;
318 
319  Eigen::RowVectorXd W;
320 
321  utils::static_for<Kmax + 2 - Kmin>([&](auto i) {
322  static constexpr auto K = Kmin + i;
323  if (K == k) {
324  if (extend || ival + 1 < intervals_.size()) {
325  static constexpr auto nw_ext_s = detail::lgr_plus_one<K>();
326  static constexpr auto B_ext_s = lagrange_basis<K>(nw_ext_s.first);
327  const auto U = monomial_derivative<K>(u, p);
328  W = MatMap(U[0].data(), 1, k + 1) * MatMap(B_ext_s[0].data(), k + 1, k + 1);
329  assert(std::size_t(W.size()) == k + 1);
330  } else {
331  static constexpr auto nw_s = lgr_nodes<K>();
332  static constexpr auto B_s = lagrange_basis<K - 1>(nw_s.first);
333  const auto U = monomial_derivative<K - 1>(u, p);
334  W = MatMap(U[0].data(), 1, k) * MatMap(B_s[0].data(), k, k);
335  assert(std::size_t(W.size()) == k);
336  }
337  }
338  });
339 
340  using namespace std::views;
341 
342  std::size_t N_before = 0;
343  for (auto i = 0u; i < ival; ++i) { N_before += intervals_[i].K; }
344  const auto r_ival = r | drop(int64_t(N_before));
345  RetT ret = W(0) * *std::ranges::begin(r_ival);
346 
347  for (auto i = 1u; const auto & v : r_ival | drop(1) | take(W.size() - 1)) { ret += W(i++) * v; }
348  return ret;
349  }

391 template<bool Deriv>
392 auto colloc_eval(
393  const std::size_t nf,
394  auto && f,
395  const MeshType auto & m,
396  const double t0,
397  const double tf,
398  const Eigen::MatrixXd & X,
399  const Eigen::MatrixXd & U)
400 {
401  assert(m.N_colloc() + 1 == static_cast<std::size_t>(X.cols())); // extra variable at the end
402  assert(m.N_colloc() == static_cast<std::size_t>(U.cols())); // one input per collocation point
403 
404  const std::size_t nx = X.rows();
405  const std::size_t nu = U.rows();
406 
407  // all nodes in mesh
408  const auto [tau_s, w_s] = m.all_nodes_and_weights();
409 
410  Eigen::MatrixXd Fval(nf, tau_s.size() - 1);
411 
412  Eigen::SparseMatrix<double> dvecF_dt0, dvecF_dtf, dvecF_dvecX, dvecF_dvecU;
413 
414  if constexpr (Deriv) {
415  dvecF_dt0.resize(Fval.size(), 1);
416  dvecF_dtf.resize(Fval.size(), 1);
417  dvecF_dvecX.resize(Fval.size(), X.size());
418  dvecF_dvecU.resize(Fval.size(), U.size());
419 
420  dvecF_dt0.reserve(Fval.size());
421  dvecF_dtf.reserve(Fval.size());
422 
423  Eigen::VectorXi FX_pattern = Eigen::VectorXi::Constant(X.size(), nf);
424  FX_pattern.tail(nx).setZero();
425  dvecF_dvecX.reserve(FX_pattern);
426  dvecF_dvecU.reserve(Eigen::VectorXi::Constant(U.size(), nf));
427  }
428 
429  for (auto i = 0u; i + 1 < tau_s.size(); ++i) {
430  const double T = t0 + (tf - t0) * tau_s(i);
431 
432  const Eigen::VectorXd x = X.col(i);
433  const Eigen::VectorXd u = U.col(i);
434 
435  if constexpr (Deriv) {
436  const auto [fval, dfval] = diff::dr(f, wrt(T, x, u));
437 
438  assert(fval.rows() == Eigen::Index(nf));
439  assert(dfval.rows() == Eigen::Index(nf));
440  assert(dfval.cols() == Eigen::Index(1 + nu + nx));
441 
442  Fval.col(i) = fval;
443 
444  for (auto row = 0u; row < nf; ++row) {
445  dvecF_dt0.insert(nf * i + row, 0) = dfval(row, 0) * (1. - tau_s(i));
446  dvecF_dtf.insert(nf * i + row, 0) = dfval(row, 0) * tau_s(i);
447  for (auto col = 0u; col < nx; ++col) {
448  dvecF_dvecX.insert(nf * i + row, i * nx + col) = dfval(row, 1 + col);
449  }
450  for (auto col = 0u; col < nu; ++col) {
451  dvecF_dvecU.insert(nf * i + row, i * nu + col) = dfval(row, 1 + nx + col);
452  }
453  }
454  } else {
455  Fval.col(i) = f(T, x, u);
456  }
457  }
458 
459  if constexpr (Deriv) {
460  dvecF_dt0.makeCompressed();
461  dvecF_dtf.makeCompressed();
462  dvecF_dvecX.makeCompressed();
463  dvecF_dvecU.makeCompressed();
464 
465  return std::make_tuple(
466  std::move(Fval),
467  std::move(dvecF_dt0),
468  std::move(dvecF_dtf),
469  std::move(dvecF_dvecX),
470  std::move(dvecF_dvecU));
471  } else {
472  return Fval;
473  }
474 }

496 template<bool Deriv>
497 auto colloc_eval_endpt(
498  const std::size_t nf,
499  const std::size_t nx,
500  auto && f,
501  [[maybe_unused]] const double t0,
502  const double tf,
503  const Eigen::MatrixXd & X,
504  const Eigen::VectorXd & Q)
505 {
506  assert(static_cast<std::size_t>(X.rows()) == nx);
507 
508  // NOTE: for now t0 = 0 and we don't want t0 in signatures
509  assert(t0 == 0);
510 
511  const Eigen::VectorXd x0 = X.leftCols(1);
512  const Eigen::VectorXd xf = X.rightCols(1);
513 
514  if constexpr (!Deriv) {
515  return f(tf, x0, xf, Q);
516  } else {
517  const auto [Fval, J] = diff::dr(f, wrt(tf, x0, xf, Q));
518 
519  assert(static_cast<std::size_t>(J.rows()) == nf);
520  assert(static_cast<std::size_t>(J.cols()) == 1 + 2 * nx + Q.size());
521 
522  Eigen::SparseMatrix<double> dF_dt0, dF_dtf, dF_dvecX, dF_dQ;
523 
524  dF_dt0.resize(nf, 1);
525  // dF_dt0.reserve(nf);
526  // for (auto i = 0u; i < nf; ++i) { dF_dt0.insert(i, 0) = J(i, 0); }
527 
528  dF_dtf.resize(nf, 1);
529  dF_dtf.reserve(nf);
530  for (auto i = 0u; i < nf; ++i) { dF_dtf.insert(i, 0) = J(i, 0); }
531 
532  dF_dvecX.resize(nf, X.size());
533  Eigen::VectorXi pattern = Eigen::VectorXi::Zero(X.size());
534  pattern.head(nx).setConstant(nf);
535  pattern.tail(nx).setConstant(nf);
536  dF_dvecX.reserve(pattern);
537 
538  for (auto row = 0u; row < nf; ++row) {
539  for (auto col = 0u; col < nx; ++col) {
540  dF_dvecX.insert(row, col) = J(row, 1 + col);
541  dF_dvecX.insert(row, X.size() - nx + col) = J(row, 1 + nx + col);
542  }
543  }
544 
545  dF_dQ.resize(nf, Q.size());
546  dF_dQ.reserve(Eigen::VectorXi::Constant(Q.size(), nf));
547 
548  for (auto row = 0u; row < nf; ++row) {
549  for (auto col = 0u; col < Q.size(); ++col) {
550  dF_dQ.insert(row, col) = J(row, 1 + 2 * nx + col);
551  }
552  }
553 
554  dF_dt0.makeCompressed();
555  dF_dtf.makeCompressed();
556  dF_dvecX.makeCompressed();
557  dF_dQ.makeCompressed();
558 
559  return std::make_tuple(Fval, dF_dt0, dF_dtf, dF_dvecX, dF_dQ);
560  }

579 template<bool Deriv>
580 auto colloc_dyn(
581  const std::size_t nx,
582  auto && f,
583  const MeshType auto & m,
584  const double t0,
585  const double tf,
586  const Eigen::MatrixXd & X,
587  const Eigen::MatrixXd & U)
588 {
589  assert(m.N_colloc() + 1 == static_cast<std::size_t>(X.cols())); // extra at the end
590  assert(m.N_colloc() == static_cast<std::size_t>(U.cols())); // one per collocation point
591  assert(nx == static_cast<std::size_t>(X.rows())); // one per collocation point
592 
593  Eigen::MatrixXd Fval;
594  Eigen::MatrixXd XD(nx, m.N_colloc());
595  Eigen::SparseMatrix<double> dvecF_dt0, dvecF_dtf, dvecF_dvecX, dvecF_dvecU, dvecXD_dvecX;
596 
597  if constexpr (!Deriv) {
598  Fval = colloc_eval<0>(nx, f, m, t0, tf, X, U);
599  } else {
600  std::tie(Fval, dvecF_dt0, dvecF_dtf, dvecF_dvecX, dvecF_dvecU) =
601  colloc_eval<1>(nx, f, m, t0, tf, X, U);
602 
603  dvecXD_dvecX.resize(XD.size(), X.size());
604 
605  // reserve sparsity pattern
606  Eigen::VectorXi pattern = Eigen::VectorXi::Zero(X.size());
607  for (auto M = 0u, i = 0u; i < m.N_ivals(); ++i) {
608  const std::size_t K = m.N_colloc_ival(i);
609  pattern.segment(M, (K + 1) * nx) += Eigen::VectorXi::Constant((K + 1) * nx, K);
610  M += K * nx;
611  }
612  dvecXD_dvecX.reserve(pattern);
613  }
614 
615  for (auto i = 0u, M = 0u; i < m.N_ivals(); M += m.N_colloc_ival(i), ++i) {
616  const std::size_t K = m.N_colloc_ival(i);
617  const Eigen::MatrixXd D = m.interval_diffmat(i);
618  XD.block(0, M, nx, K) = X.block(0, M, nx, K + 1) * D;
619 
620  if constexpr (Deriv) {
621  // vec(X * D) = kron(D', I) * vec(X), so derivative w.r.t vec(X) = kron(D', I)
622  for (auto i = 0u; i < K; ++i) {
623  for (auto j = 0u; j < K + 1; ++j) {
624  for (auto diag = 0u; diag < nx; ++diag) {
625  dvecXD_dvecX.coeffRef(M * nx + i * nx + diag, M * nx + j * nx + diag) += D(j, i);
626  }
627  }
628  }
629  }
630  }
631 
632  Eigen::VectorXd Fv = (XD - (tf - t0) * Fval).reshaped();
633 
634  // scale equalities by by quadrature weights
635  const auto N = m.N_colloc();
636  const auto [n, w] = m.all_nodes_and_weights();
637 
638  // vec(A * W) = kron(W', I) * vec(A), so we apply kron(W', I) on the left
639 
640  Eigen::SparseMatrix<double> W(N, N);
641  W.reserve(Eigen::VectorXi::Ones(N));
642  for (auto i = 0u; i < N; ++i) { W.insert(i, i) = w(i); }
643 
644  const Eigen::SparseMatrix<double> W_kron_I = kron_identity(W, nx);
645 
646  Fv.applyOnTheLeft(W_kron_I);
647 
648  if constexpr (!Deriv) {
649  return Fv;
650  } else {
651  dvecXD_dvecX.makeCompressed();
652 
653  Eigen::SparseMatrix<double> dF_dt0 = -(tf - t0) * dvecF_dt0;
654  dF_dt0 += Fval.reshaped().sparseView(); // OK since dvecF_dtf is dense
655  dF_dt0 = W_kron_I * dF_dt0;
656 
657  Eigen::SparseMatrix<double> dF_dtf = -(tf - t0) * dvecF_dtf;
658  dF_dtf -= Fval.reshaped().sparseView(); // OK since dvecF_dtf is dense
659  dF_dtf = W_kron_I * dF_dtf;
660 
661  Eigen::SparseMatrix<double> dF_dvecX = dvecXD_dvecX;
662  dF_dvecX -= (tf - t0) * dvecF_dvecX;
663  dF_dvecX = W_kron_I * dF_dvecX;
664 
665  Eigen::SparseMatrix<double> dF_dvecU = -(tf - t0) * W_kron_I * dvecF_dvecU;
666 
667  dF_dt0.makeCompressed();
668  dF_dtf.makeCompressed();
669  dF_dvecX.makeCompressed();
670  dF_dvecU.makeCompressed();
671 
672  return std::make_tuple(
673  std::move(Fv),
674  std::move(dF_dt0),
675  std::move(dF_dtf),
676  std::move(dF_dvecX),
677  std::move(dF_dvecU));
678  }
679 }

694 Eigen::VectorXd mesh_dyn_error(
695  const std::size_t nx,
696  auto && f,
697  const MeshType auto & m,
698  const double t0,
699  const double tf,
700  const std::function<Eigen::VectorXd(double)> xfun,
701  const std::function<Eigen::VectorXd(double)> ufun)
702 {
703  const auto N = m.N_ivals();
704 
705  // create a new mesh where each interval is extended
706  Mesh mext = m;
707  for (auto i = 0u; i < N; ++i) {
708  const std::size_t K = m.N_colloc_ival(i);
709  mext.set_N_colloc_ival(i, K + 1);
710  }
711 
712  Eigen::VectorXd ival_errs(N);
713 
714  // for each interval
715  for (auto i = 0u, M = 0u; i < N; M += m.N_colloc_ival(i), ++i) {
716  const std::size_t Kext = mext.N_colloc_ival(i);
717 
718  const auto [tau_s, weights] = mext.interval_nodes_and_weights(i);
719 
720  assert(std::size_t(tau_s.size()) == Kext + 1);
721 
722  // evaluate X and F at those points
723  Eigen::MatrixXd Fval(nx, Kext + 1);
724  Eigen::MatrixXd Xval(nx, Kext + 1);
725  for (auto j = 0u; j < Kext + 1; ++j) {
726  const double tj = t0 + (tf - t0) * tau_s(j);
727 
728  // evaluate x and u values at tj using current degree polynomials
729  const auto Xj = xfun(tj);
730  const auto Uj = ufun(tj);
731 
732  // evaluate right-hand side of dynamics at tj
733  Fval.col(j) = f(tj, Xj, Uj);
734 
735  // store x values for later comparison
736  Xval.col(j) = Xj;
737  }

739  // "integrate" system inside interval
740  const Eigen::MatrixXd Xval_est =
741  Xval.col(0).replicate(1, Kext) + (tf - t0) * Fval.leftCols(Kext) * mext.interval_intmat(i);
742 
743  // absolute error in interval
744  Eigen::VectorXd e_abs = (Xval_est - Xval.rightCols(Kext)).colwise().norm();
745  Eigen::VectorXd e_rel = e_abs / (1. + Xval.rightCols(Kext).colwise().norm().maxCoeff());
746 
747  // mex relative error on interval
748  ival_errs(i) = e_rel.maxCoeff();
749  }
750 
751  return ival_errs;

795 template<bool Deriv>
796 auto colloc_int(
797  const std::size_t nq,
798  auto && g,
799  const MeshType auto & m,
800  const double t0,
801  const double tf,
802  const Eigen::VectorXd & I,
803  const Eigen::MatrixXd & X,
804  const Eigen::MatrixXd & U)
805 {
806  assert(static_cast<std::size_t>(I.size()) == nq);
807 
808  const std::size_t N = m.N_colloc();
809 
810  const auto [n, w] = m.all_nodes_and_weights();
811 
812  if constexpr (Deriv == false) {
813  const auto Gv = colloc_eval<Deriv>(nq, g, m, t0, tf, X, U);
814  const Eigen::VectorXd Iest = Gv * w.head(N);
815  Eigen::VectorXd Rv = (tf - t0) * Iest - I;
816  return Rv;
817  } else {
818  const auto [Gv, dvecG_dt0, dvecG_dtf, dvecG_dvecX, dvecG_dvecU] =
819  colloc_eval<Deriv>(nq, g, m, t0, tf, X, U);
820  const Eigen::VectorXd Iest = Gv * w.head(N);
821 
822  Eigen::VectorXd Rv = (tf - t0) * Iest - I;
823 
824  const Eigen::SparseMatrix<double> w_kron_I =
825  (tf - t0) * kron_identity(w.head(N).transpose(), nq);
826 
827  Eigen::SparseMatrix<double> dR_dt0 = w_kron_I * dvecG_dt0;
828  for (auto i = 0u; i < Iest.size(); ++i) { dR_dt0.coeffRef(i, 0) -= Iest(i); }
829 
830  Eigen::SparseMatrix<double> dR_dtf = w_kron_I * dvecG_dtf;
831  for (auto i = 0u; i < Iest.size(); ++i) { dR_dtf.coeffRef(i, 0) += Iest(i); }
832 
833  Eigen::SparseMatrix<double> dR_dvecI = -sparse_identity(nq);
834 
835  Eigen::SparseMatrix<double> dR_dvecX = w_kron_I * dvecG_dvecX;
836 
837  Eigen::SparseMatrix<double> dR_dvecU = w_kron_I * dvecG_dvecU;
838 
839  dR_dt0.makeCompressed();
840  dR_dtf.makeCompressed();
841  dR_dvecI.makeCompressed();
842  dR_dvecX.makeCompressed();
843  dR_dvecU.makeCompressed();
844 
845  return std::make_tuple(
846  std::move(Rv),
847  std::move(dR_dt0),
848  std::move(dR_dtf),
849  std::move(dR_dvecI),
850  std::move(dR_dvecX),
851  std::move(dR_dvecU));
852  }
