64 template<std::size_t K, std::size_t I = 8>
65  requires(K <= 40)
66 constexpr std::pair<std::array<double, K + 1>, std::array<double, K + 1>> lgr_plus_one()
67 {
68  auto lgr_norm = ::smooth::lgr_nodes<K, I>();
69 
70  std::array<double, K + 1> ns, ws;
71  for (auto i = 0u; i < K; ++i) {
72  ns[i] = lgr_norm.first[i];
73  ws[i] = lgr_norm.second[i];
74  }
75  ns[K] = 1;
76  ws[K] = 0;
77  return {ns, ws};
78 }

123  inline void split(std::size_t i, std::size_t n = 2)
124  {
125  const double tau0 = intervals_[i].tau0;
126  const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
127  const double taum = tau0 + (tauf - tau0) / n;
128 
129  while (n-- > 1) {
130  intervals_.insert(intervals_.begin() + i + 1, Interval{.K = 5, .tau0 = n * taum});
131  }
132  }

146  inline std::pair<Eigen::VectorXd, Eigen::VectorXd> interval_nodes_and_weights(std::size_t i) const
147  {
148  const std::size_t K = intervals_[i].K;
149 
150  const double tau0 = intervals_[i].tau0;
151  const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
152 
153  Eigen::VectorXd n_n;
154  Eigen::VectorXd w_n;
155 
156  if (K == 5) {
157  constexpr auto Ks = 5u;
158  constexpr auto nw_s = detail::lgr_plus_one<Ks>();
159 
160  n_n = Eigen::Map<const Eigen::VectorXd>(nw_s.first.data(), Ks + 1);
161  w_n = Eigen::Map<const Eigen::VectorXd>(nw_s.second.data(), Ks + 1);
162  } else {
163  throw std::runtime_error("Size not available");
164  }
165 
166  return {
167  Eigen::VectorXd::Constant(n_n.size(), tau0)
168  + (tauf - tau0) * (n_n + Eigen::VectorXd::Constant(n_n.size(), 1)) / 2,
169  ((tauf - tau0) / 2.) * w_n,
170  };
171  }

176  inline std::pair<Eigen::VectorXd, Eigen::VectorXd> all_nodes_and_weights() const
177  {
178  Eigen::VectorXd n(N_colloc() + 1), w(N_colloc() + 1);
179 
180  std::size_t cntr = 0;
181  for (auto i = 0u; i < intervals_.size(); ++i) {
182  auto [ni, wi] = interval_nodes_and_weights(i);
183 
184  const std::size_t Ni = ni.size();
185 
186  // exclude last point that belongs to next interval..
187  n.segment(cntr, Ni - 1) = ni.head(Ni - 1);
188  w.segment(cntr, Ni - 1) = wi.head(Ni - 1);
189 
190  cntr += Ni - 1;
191  }
192 
193  n.tail(1).setConstant(1);
194  w.tail(1).setConstant(0);
195 
196  return {n, w};
197  }

211  inline Eigen::MatrixXd interval_diffmat(std::size_t i) const
212  {
213  const std::size_t K = intervals_[i].K;
214 
215  const double tau0 = intervals_[i].tau0;
216  const double tauf = i + 1 < intervals_.size() ? intervals_[i + 1].tau0 : 1.;
217 
218  if (K == 5) {
219  constexpr auto Ks = 5u;
220  constexpr auto nw_s = detail::lgr_plus_one<Ks>();
221  constexpr auto B = lagrange_basis<Ks>(nw_s.first);
222  constexpr auto D = polynomial_basis_derivatives<Ks, Ks + 1>(B, nw_s.first);
223 
224  Eigen::Map<const Eigen::Matrix<double, -1, -1, Eigen::RowMajor>> m(
225  D[0].data(), Ks + 1, Ks + 1);
226  return (2 / (tauf - tau0)) * m.block(0, 0, Ks + 1, Ks);
227  } else {
228  throw std::runtime_error("Wrong size");
229  }
230  }

265 template<typename Fun>
266 auto colloc_eval(std::size_t nf,
267  Fun && f,
268  const Mesh & m,
269  const double t0,
270  const double tf,
271  const Eigen::MatrixXd & X,
272  const Eigen::MatrixXd & U)
273 {
274  assert(m.N_colloc() + 1 == static_cast<std::size_t>(X.cols())); // extra variable at the end
275  assert(m.N_colloc() == static_cast<std::size_t>(U.cols())); // one input per collocation point
276 
277  const std::size_t nx = X.rows();
278  const std::size_t nu = U.rows();
279 
280  // all nodes in mesh
281  const auto [tau_s, w_s] = m.all_nodes_and_weights();
282 
283  Eigen::MatrixXd Fval(nf, tau_s.size() - 1);
284 
285  Eigen::SparseMatrix<double> dvecF_dt0(Fval.size(), 1);
286  Eigen::SparseMatrix<double> dvecF_dtf(Fval.size(), 1);
287  Eigen::SparseMatrix<double> dvecF_dvecX(Fval.size(), X.size());
288  Eigen::SparseMatrix<double> dvecF_dvecU(Fval.size(), U.size());
289 
290  Eigen::Matrix<int, -1, 1> FX_pattern = Eigen::Matrix<int, -1, 1>::Constant(X.size(), nf);
291  FX_pattern.tail(nx).setZero();
292  dvecF_dvecX.reserve(FX_pattern);
293  dvecF_dvecU.reserve(Eigen::Matrix<int, -1, 1>::Constant(U.size(), nf));
294 
295  Eigen::SparseMatrix<double> d2lF_dvecX2(X.size(), X.size());
296 
297  // TODO: calculate Hessian on demand
298 
299  for (auto i = 0u; i + 1 < tau_s.size(); ++i) {
300  autodiff::VectorXreal var_ad(1 + nx + nu);
301  var_ad << autodiff::real(t0 + (tf - t0) * tau_s(i)), X.col(i).template cast<autodiff::real>(),
302  U.col(i).template cast<autodiff::real>();
303 
304  autodiff::VectorXreal Fval_ad;
305  const Eigen::MatrixXd dF = autodiff::jacobian(
306  [&](const autodiff::VectorXreal & var) -> autodiff::VectorXreal {
307  return f.template operator()<autodiff::real>(
308  var(0), var.segment(1, nx), var.segment(1 + nx, nu));
309  },
310  autodiff::wrt(var_ad),
311  autodiff::at(var_ad),
312  Fval_ad);
313 
314  assert(Fval_ad.rows() == Eigen::Index(nf));
315  assert(dF.rows() == Eigen::Index(nf));
316  assert(dF.cols() == Eigen::Index(1 + nu + nx));
317 
318  Fval.col(i) = Fval_ad.template cast<double>();
319 
320  for (auto row = 0u; row < nf; ++row) {
321  dvecF_dt0.insert(nf * i + row, 0) = dF(row, 0) * (1. - tau_s(i));
322  dvecF_dtf.insert(nf * i + row, 0) = dF(row, 0) * tau_s(i);
323  for (auto col = 0u; col < nx; ++col) {
324  dvecF_dvecX.insert(nf * i + row, i * nx + col) = dF(row, 1 + col);
325  }
326  for (auto col = 0u; col < nu; ++col) {
327  dvecF_dvecU.insert(nf * i + row, i * nu + col) = dF(row, 1 + nx + col);
328  }
329  }
330  }
331 
332  dvecF_dt0.makeCompressed();
333  dvecF_dtf.makeCompressed();
334  dvecF_dvecX.makeCompressed();
335  dvecF_dvecU.makeCompressed();
336 
337  return std::make_tuple(std::move(Fval),
338  std::move(dvecF_dt0),
339  std::move(dvecF_dtf),
340  std::move(dvecF_dvecX),
341  std::move(dvecF_dvecU));
342 }

357 template<typename F>
358 auto dynamics_constraint(F && f,
359  const Mesh & m,
360  const double t0,
361  const double tf,
362  const Eigen::MatrixXd & X,
363  const Eigen::MatrixXd & U)
364 {
365  assert(m.N_colloc() + 1 == static_cast<std::size_t>(X.cols())); // extra at the end
366  assert(m.N_colloc() == static_cast<std::size_t>(U.cols())); // one per collocation point
367 
368  const std::size_t nx = X.rows();
369 
370  const auto [Fval, dvecF_dt0, dvecF_dtf, dFvec_dvecX, dvecF_dvecU] =
371  colloc_eval(nx, std::forward<F>(f), m, t0, tf, X, U);
372 
373  Eigen::MatrixXd XD(nx, m.N_colloc());
374  Eigen::SparseMatrix<double> dvecXD_dvecX(XD.size(), X.size());
375 
376  // reserve sparsity pattern
377  Eigen::Matrix<int, -1, 1> pattern = Eigen::Matrix<int, -1, 1>::Zero(X.size());
378  for (auto M = 0u, i = 0u; i < m.N_ivals(); ++i) {
379  const std::size_t K = m.N_colloc_ival(i);
380  pattern.segment(M, (K + 1) * nx) += Eigen::Matrix<int, -1, 1>::Constant((K + 1) * nx, K);
381  M += K * nx;
382  }
383  dvecXD_dvecX.reserve(pattern);
384 
385  for (auto i = 0u, M = 0u; i < m.N_ivals(); M += m.N_colloc_ival(i), ++i) {
386  const std::size_t K = m.N_colloc_ival(i);
387  const Eigen::MatrixXd D = m.interval_diffmat(i);
388  XD.block(0, M, nx, K) = X.block(0, M, nx, K + 1) * D;
389 
390  // vec(X * D) = kron(D', I) * vec(X), so derivative w.r.t vec(X) = kron(D', I)
391  for (auto i = 0u; i < K; ++i) {
392  for (auto j = 0u; j < K + 1; ++j) {
393  for (auto diag = 0u; diag < nx; ++diag) {
394  dvecXD_dvecX.coeffRef(M * nx + i * nx + diag, M * nx + j * nx + diag) += D(j, i);
395  }
396  }
397  }
398  }
399 
400  dvecXD_dvecX.makeCompressed();
401 
402  Eigen::VectorXd Fv = (XD - (tf - t0) * Fval).reshaped();
403 
404  Eigen::SparseMatrix<double> dF_dt0 = -(tf - t0) * dvecF_dt0;
405  dF_dt0 += Fval.reshaped().sparseView(); // OK since dvecF_dtf is dense
406 
407  Eigen::SparseMatrix<double> dF_dtf = -(tf - t0) * dvecF_dtf;
408  dF_dtf -= Fval.reshaped().sparseView(); // OK since dvecF_dtf is dense
409 
410  Eigen::SparseMatrix<double> dF_dvecX = dvecXD_dvecX;
411  dF_dvecX -= (tf - t0) * dFvec_dvecX;
412 
413  Eigen::SparseMatrix<double> dF_dvecU = -(tf - t0) * dvecF_dvecU;
414 
415  dF_dt0.makeCompressed();
416  dF_dtf.makeCompressed();
417  dF_dvecX.makeCompressed();
418  dF_dvecU.makeCompressed();
419 
420  return std::make_tuple(
421  std::move(Fv), std::move(dF_dt0), std::move(dF_dtf), std::move(dF_dvecX), std::move(dF_dvecU));
422 }

439 template<typename G>
440 auto integral_constraint(G && g,
441  const Mesh & m,
442  const double & t0,
443  const double & tf,
444  const Eigen::MatrixXd & I,
445  const Eigen::MatrixXd & X,
446  const Eigen::MatrixXd & U)
447 {
448  static constexpr std::size_t mi = 1;
449  const std::size_t N = m.N_colloc();
450 
451  const auto [Gv, dvecG_dt0, dvecG_dtf, dvecG_dvecX, dvecG_dvecU] =
452  colloc_eval(mi, std::forward<G>(g), m, t0, tf, X, U);
453 
454  const auto [n, w] = m.all_nodes_and_weights();
455  const Eigen::VectorXd Iest = Gv * w.head(N);
456 
457  const Eigen::SparseMatrix<double> w_kron_I = (tf - t0) * kron_identity(w.head(N).transpose(), mi);
458 
459  Eigen::VectorXd Rv = (tf - t0) * Iest - I;
460 
461  Eigen::SparseMatrix<double> dR_dt0 = w_kron_I * dvecG_dt0;
462  for (auto i = 0u; i < Iest.size(); ++i) { dR_dt0.coeffRef(i, 0) -= Iest(i); }
463 
464  Eigen::SparseMatrix<double> dR_dtf = w_kron_I * dvecG_dtf;
465  for (auto i = 0u; i < Iest.size(); ++i) { dR_dtf.coeffRef(i, 0) += Iest(i); }
466 
467  Eigen::SparseMatrix<double> dR_dvecI = -sparse_identity(mi);
468 
469  Eigen::SparseMatrix<double> dR_dvecX = w_kron_I * dvecG_dvecX;
470 
471  Eigen::SparseMatrix<double> dR_dvecU = w_kron_I * dvecG_dvecU;
472 
473  dR_dt0.makeCompressed();
474  dR_dtf.makeCompressed();
475  dR_dvecI.makeCompressed();
476  dR_dvecX.makeCompressed();
477  dR_dvecU.makeCompressed();
478 
479  return std::make_tuple(std::move(Rv),
480  std::move(dR_dt0),
481  std::move(dR_dtf),
482  std::move(dR_dvecI),
483  std::move(dR_dvecX),
484  std::move(dR_dvecU));
485 }
