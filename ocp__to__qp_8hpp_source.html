63template<diff::Type DT = diff::Type::Default>
64void ocp_to_qp_allocate(
65 QuadraticProgramSparse<double> & qp,
66 OcpToQpWorkmemory & work,
67 OCPType auto & ocp,
68 const MeshType auto & mesh)
69{
70 using ocp_t = typename std::decay_t<decltype(ocp)>;
71
72 using X = typename ocp_t::X;
73 using U = typename ocp_t::U;
74
75 static constexpr auto Nx = ocp_t::Nx;
76 static constexpr auto Nu = ocp_t::Nu;
77
81
82 // [x0 x1 ... xN u0 u1 ... uN-1]
83
84 const auto N = mesh.N_colloc();
85
86 const auto xvar_L = Nx * (N + 1);
87 const auto uvar_L = Nu * N;
88
89 const auto dcon_L = Nx * N;
90 const auto crcon_L = ocp_t::Ncr * N;
91 const auto cecon_L = ocp_t::Nce;
92
93 const auto dcon_B = 0u;
94 const auto crcon_B = dcon_L;
95 const auto cecon_B = crcon_B + crcon_L;
96
97 const auto Nvar = xvar_L + uvar_L;
98 const auto Ncon = dcon_L + crcon_L + cecon_L;
99
100 // resize qp
101 qp.P.resize(Nvar, Nvar);
102 qp.q.resize(Nvar);
103 qp.A.resize(Ncon, Nvar);
104 qp.l.resize(Ncon);
105 qp.u.resize(Ncon);
106
107 // sparsity pattern of A (row-major)
108 Eigen::VectorXi A_pattern = Eigen::VectorXi::Zero(Ncon);
109 for (auto ival = 0u, I0 = 0u; ival < mesh.N_ivals(); I0 += mesh.N_colloc_ival(ival), ++ival) {
110 const auto Ki = mesh.N_colloc_ival(ival); // number of nodes in interval
111 A_pattern.segment(dcon_B + I0 * Nx, Ki * Nx) +=
112 Eigen::VectorXi::Constant(Ki * Nx, Nx + Ki + Nu);
113 }
114 A_pattern.segment(crcon_B, crcon_L).setConstant(Nx + Nu);
115 A_pattern.segment(cecon_B, cecon_L).setConstant(2 * Nx);
116 qp.A.reserve(A_pattern);
117
118 // sparsity pattern of P
119 Eigen::VectorXi P_pattern = Eigen::VectorXi::Zero(Nvar);
120 for (auto i = 0u; i < N; ++i) {
121 for (auto j = 0u; j < Nx; ++j) { P_pattern(Nx * i + j) = j + 1; }
122 }
123 for (auto j = 0u; j < Nx; ++j) { P_pattern(Nx * N + j) = Nx + j + 1; }
124 for (auto i = 0u; i < N; ++i) {
125 for (auto j = 0u; j < Nu; ++j) { P_pattern(Nx * (N + 1) + Nu * i + j) = Nx + (j + 1); }
126 }
127 qp.P.reserve(P_pattern);
128
129 // compute work stuff once to allocate pattern
130 const double tf = 1.;
131 auto xslin = mesh.all_nodes() | transform([&](double) { return Identity<X>(); });
132 auto uslin = mesh.all_nodes() | transform([&](double) { return Identity<U>(); });
133
134 work.int_out.lambda.setConstant(1, 1);
135 mesh_eval<1, DT>(work.cr_out, mesh, ocp.cr, 0, tf, xslin, uslin); // allocates work.cr_out
136 mesh_integrate<2, DT>(work.int_out, mesh, ocp.g, 0, tf, xslin, uslin); // allocates work.int_out
137}

157template<diff::Type DT = diff::Type::Default>
158void ocp_to_qp_update(
159 QuadraticProgramSparse<double> & qp,
160 OcpToQpWorkmemory & work,
161 OCPType auto & ocp,
162 const MeshType auto & mesh,
163 double tf,
164 auto && xl_fun,
165 auto && ul_fun)
166{
167 using utils::zip;
168 using namespace std::views;
169
170 using ocp_t = typename std::decay_t<decltype(ocp)>;
171
172 using X = typename ocp_t::X;
173
174 static constexpr auto Nx = ocp_t::Nx;
175 static constexpr auto Nu = ocp_t::Nu;
176 static constexpr auto Nq = ocp_t::Nq;
177
178 const double t0 = 0.;
179
180 static_assert(ocp_t::Nq == 1, "exactly one integral supported in ocp_to_qp");
181
185
186 // [x0 x1 ... xN u0 u1 ... uN-1]
187
188 const auto N = mesh.N_colloc();
189
190 const auto xvar_L = Nx * (N + 1);
191 const auto uvar_L = Nu * N;
192
193 const auto xvar_B = 0u;
194 const auto uvar_B = xvar_L;
195
196 const auto dcon_L = Nx * N;
197 const auto crcon_L = ocp_t::Ncr * N;
198 const auto cecon_L = ocp_t::Nce;
199
200 const auto dcon_B = 0u;
201 const auto crcon_B = dcon_L;
202 const auto cecon_B = crcon_B + crcon_L;
203
207
208 set_zero(qp.A);
209 set_zero(qp.P);
210
211 qp.l.setConstant(-std::numeric_limits<double>::infinity());
212 qp.u.setConstant(std::numeric_limits<double>::infinity());
213 qp.q.setZero();
214
218
219 const X xl0 = xl_fun(0.);
220 const X xlf = xl_fun(tf);
221
222 auto xslin = mesh.all_nodes() | transform([&](double t) { return xl_fun(t0 + (tf - t0) * t); });
223 auto uslin = mesh.all_nodes() | transform([&](double t) { return ul_fun(t0 + (tf - t0) * t); });
224
225 const Eigen::Vector<double, 1> ql{1.};
226
230
231 const auto & [th, dth, d2th] = diff::dr<2, DT>(ocp.theta, wrt(tf, xl0, xlf, ql));
232
233 const Eigen::Vector<double, Nx> qo_x0 = dth.middleCols(1, Nx).transpose();
234 const Eigen::Vector<double, Nx> qo_xf = dth.middleCols(1 + Nx, Nx).transpose();
235 const Eigen::Vector<double, Nq> qo_q = dth.middleCols(1 + 2 * Nx, Nq).transpose();
236
237 const Eigen::Matrix<double, Nx, Nx> Qo_x0 = d2th.block(1, 1, Nx, Nx) / 2;
238 const Eigen::Matrix<double, Nx, Nx> Qo_x0f = d2th.block(1, 1 + Nx, Nx, Nx) / 2;
239 const Eigen::Matrix<double, Nx, Nx> Qo_xf = d2th.block(1 + Nx, 1 + Nx, Nx, Nx) / 2;
240
241 block_add(qp.P, 0, 0, Qo_x0, 1., true); // d2q / dx0x0
242 block_add(qp.P, 0, Nx * N, Qo_x0f, 1., true); // d2q / dx0xf
243 block_add(qp.P, Nx * N, Nx * N, Qo_xf, 1., true); // d2q / dxfxf
244
245 qp.q.segment(0, Nx) += qo_x0; // dq / dx0
246 qp.q.segment(Nx * N, Nx) += qo_xf; // dq / dxf
247
251
252 mesh_integrate<2, DT>(work.int_out, mesh, ocp.g, 0, tf, xslin, uslin);
253
254 // clang-format off
255 block_add(qp.P, 0, 0, work.int_out.d2F.block(2, 2, xvar_L + uvar_L, xvar_L + uvar_L), qo_q.x(), true);
256
257 qp.q.segment(xvar_B, xvar_L) = qo_q.x() * work.int_out.dF.middleCols(2, xvar_L).transpose();
258 qp.q.segment(uvar_B, uvar_L) = qo_q.x() * work.int_out.dF.middleCols(2 + xvar_L, uvar_L).transpose();
259 // clang-format on
260
264
265 for (auto ival = 0u, M = 0u; ival < mesh.N_ivals(); M += mesh.N_colloc_ival(ival), ++ival) {
266 const auto Ki = mesh.N_colloc_ival(ival); // number of nodes in interval
267
268 const auto [alpha, Dus] = mesh.interval_diffmat_unscaled(ival);
269
270 // in each interval the collocation constraint is
271 // [A0 x0 ... Ak-1 xk-1 0] + [B0 u0 ... Bk-1 uk-1] + [E0 ... Ek-1] = alpha * X Dus
272
273 for (const auto & [i, tau_i] : zip(iota(0u, Ki), mesh.interval_nodes(ival))) {
274 const auto t_i = tf * tau_i; // unscaled time
275 const auto & [xl_i, dxl_i] = diff::dr<1, DT>(xl_fun, wrt(t_i)); // x-lin
276 const auto ul_i = ul_fun(t_i); // u-lin
277
278 // LINEARIZE DYNAMICS
279 const auto & [f_i, df_i] = diff::dr<1, DT>(ocp.f, wrt(t_i, xl_i, ul_i));
280
281 const Eigen::Matrix<double, Nx, Nx> A =
282 tf * (-0.5 * ad<X>(f_i) - 0.5 * ad<X>(dxl_i) + df_i.template middleCols<Nx>(1));
283 const Eigen::Matrix<double, Nx, Nu> B = tf * df_i.template middleCols<Nu>(1 + Nx);
284 const Eigen::Vector<double, Nx> E = tf * (f_i - dxl_i);
285
286 // insert new constraint A xi + B ui + E = [x0 ... XNi] di
287
288 block_add(qp.A, dcon_B + (M + i) * Nx, xvar_B + (M + i) * Nx, A);
289 block_add(qp.A, dcon_B + (M + i) * Nx, uvar_B + (M + i) * Nu, B);
290
291 for (auto j = 0u; j < Ki + 1; ++j) {
292 for (auto diag = 0u; diag < Nx; ++diag) {
293 qp.A.coeffRef(dcon_B + (M + i) * Nx + diag, (M + j) * Nx + diag) -= alpha * Dus(j, i);
294 }
295 }
296
297 qp.l.segment(dcon_B + (M + i) * Nx, Nx) = -E;
298 qp.u.segment(dcon_B + (M + i) * Nx, Nx) = -E;
299 }
300 }
301
305
306 mesh_eval<1, DT>(work.cr_out, mesh, ocp.cr, 0, tf, xslin, uslin);
307
308 block_add(qp.A, crcon_B, 0, work.cr_out.dF.middleCols(2, xvar_L + uvar_L));
309 qp.l.segment(crcon_B, crcon_L) = ocp.crl.replicate(N, 1) - work.cr_out.F;
310 qp.u.segment(crcon_B, crcon_L) = ocp.cru.replicate(N, 1) - work.cr_out.F;
311
315
316 const auto & [ceval, dceval] = diff::dr<1, DT>(ocp.ce, wrt(tf, xl0, xlf, ql));
317
318 block_add(qp.A, cecon_B, xvar_B, dceval.middleCols(1, Nx)); // dce / dx0
319 block_add(qp.A, cecon_B, xvar_B + xvar_L - Nx, dceval.middleCols(1 + Nx, Nx)); // dce / dxf
320
321 qp.l.segment(cecon_B, cecon_L) = ocp.cel - ceval;
322 qp.u.segment(cecon_B, cecon_L) = ocp.ceu - ceval;
323}

343template<diff::Type DT = diff::Type::Default>
344QuadraticProgramSparse<double> ocp_to_qp(
345 const OCPType auto & ocp, const MeshType auto & mesh, double tf, auto && xl_fun, auto && ul_fun)
346{
347 QuadraticProgramSparse<double> qp;
348 detail::OcpToQpWorkmemory work;
349
350 detail::ocp_to_qp_allocate<DT>(qp, work, ocp, mesh);
351 detail::ocp_to_qp_update<DT>(qp, work, ocp, mesh, tf, xl_fun, ul_fun);
352
353 qp.A.makeCompressed();
354 qp.P.makeCompressed();
355
356 return qp;
357}

374auto qpsol_to_ocpsol(
375 const OCPType auto & ocp,
376 const MeshType auto & mesh,
377 const QPSolution<-1, -1, double> & qpsol,
378 double tf,
379 auto && xl_fun,
380 auto && ul_fun)
381{
382 using ocp_t = std::decay_t<decltype(ocp)>;
383
384 using X = typename ocp_t::X;
385 using U = typename ocp_t::U;
386
387 static constexpr int Nx = ocp_t::Nx;
388 static constexpr int Nu = ocp_t::Nu;
389 static constexpr int Nq = ocp_t::Nq;
390 static constexpr int Ncr = ocp_t::Ncr;
391 static constexpr int Nce = ocp_t::Nce;
392
393 const auto N = mesh.N_colloc();
394
395 const auto xvar_L = Nx * (N + 1);
396 const auto uvar_L = Nu * N;
397
398 const auto xvar_B = 0u;
399 const auto uvar_B = xvar_L;
400 Eigen::MatrixXd Xmat = qpsol.primal.segment(xvar_B, xvar_L).reshaped(Nx, N + 1);
401 Eigen::MatrixXd Umat = qpsol.primal.segment(uvar_B, uvar_L).reshaped(Nu, N);
402
403 auto xfun = [t0 = 0.,
404 tf = tf,
405 mesh = mesh,
406 Xmat = std::move(Xmat),
407 xl_fun = std::forward<decltype(xl_fun)>(xl_fun)](double t) -> X {
408 const auto tngnt =
409 mesh.template eval<Eigen::Vector<double, Nx>>((t - t0) / (tf - t0), Xmat.colwise(), 0, true);
410 return rplus(xl_fun(t), tngnt);
411 };
412
413 auto ufun = [t0 = 0.,
414 tf = tf,
415 mesh = mesh,
416 Umat = std::move(Umat),
417 ul_fun = std::forward<decltype(ul_fun)>(ul_fun)](double t) -> U {
418 const auto tngnt =
419 mesh.template eval<Eigen::Vector<double, Nu>>((t - t0) / (tf - t0), Umat.colwise(), 0, false);
420 return rplus(ul_fun(t), tngnt);
421 };
422
423 return OCPSolution<X, U, Nq, Nce, Ncr>{
424 .t0 = 0.,
425 .tf = tf,
426 .u = std::move(ufun),
427 .x = std::move(xfun),
428 };
429}
