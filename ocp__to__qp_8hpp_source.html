57template<diff::Type DT = diff::Type::Default>
58QuadraticProgramSparse<double> ocp_to_qp(
59 const OCPType auto & ocp, const MeshType auto & mesh, double tf, auto && xl_fun, auto && ul_fun)
60{
61 using utils::zip;
62 using namespace std::views;
63
64 using ocp_t = typename std::decay_t<decltype(ocp)>;
65
66 using X = typename ocp_t::X;
67
68 static constexpr auto Nx = ocp_t::Nx;
69 static constexpr auto Nu = ocp_t::Nu;
70 static constexpr auto Nq = ocp_t::Nq;
71
72 const double t0 = 0.;
73
74 const X xl0 = xl_fun(0.);
75 const X xlf = xl_fun(tf);
76 const Eigen::Matrix<double, 1, 1> ql(0);
77
78 static_assert(ocp_t::Nq == 1, "exactly one integral supported in ocp_to_qp");
79
83
84 // [x0 x1 ... xN u0 u1 ... uN-1]
85
86 const auto N = mesh.N_colloc();
87
88 const auto xvar_L = Nx * (N + 1);
89 const auto uvar_L = Nu * N;
90
91 const auto xvar_B = 0u;
92 const auto uvar_B = xvar_L;
93
94 const auto dcon_L = Nx * N;
95 const auto crcon_L = ocp_t::Ncr * N;
96 const auto cecon_L = ocp_t::Nce;
97
98 const auto dcon_B = 0u;
99 const auto crcon_B = dcon_L;
100 const auto cecon_B = crcon_B + crcon_L;
101
102 const auto Nvar = xvar_L + uvar_L;
103 const auto Ncon = dcon_L + crcon_L + cecon_L;
104
106
107 MeshValue<1> cr_out;
108 MeshValue<2> int_out;
109 int_out.lambda.setConstant(1, 1);
110
111 // output of this function
112 QuadraticProgramSparse<double> ret;
113 ret.P.resize(Nvar, Nvar);
114 ret.q.resize(Nvar);
115 ret.A.resize(Ncon, Nvar);
116 ret.l.resize(Ncon);
117 ret.u.resize(Ncon);
118
119 // sparsity pattern of A
120 Eigen::VectorXi A_pattern = Eigen::VectorXi::Zero(Ncon);
121 for (auto ival = 0u, I0 = 0u; ival < mesh.N_ivals(); I0 += mesh.N_colloc_ival(ival), ++ival) {
122 const auto Ki = mesh.N_colloc_ival(ival); // number of nodes in interval
123 A_pattern.segment(dcon_B + I0 * Nx, Ki * Nx) +=
124 Eigen::VectorXi::Constant(Ki * Nx, Nx + Ki + Nu);
125 }
126 A_pattern.segment(crcon_B, crcon_L).setConstant(Nx + Nu);
127 A_pattern.segment(cecon_B, cecon_L).setConstant(2 * Nx);
128 ret.A.reserve(A_pattern);
129
133
134 const auto & [th, dth, d2th] = diff::dr<2, DT>(ocp.theta, wrt(tf, xl0, xlf, ql));
135
136 const Eigen::Vector<double, Nx> qo_x0 = dth.middleCols(1, Nx).transpose();
137 const Eigen::Vector<double, Nx> qo_xf = dth.middleCols(1 + Nx, Nx).transpose();
138 const Eigen::Vector<double, Nq> qo_q = dth.middleCols(1 + 2 * Nx, Nq).transpose();
139
140 const Eigen::Matrix<double, Nx, Nx> Qo_x0 = d2th.block(1, 1, Nx, Nx) / 2;
141 const Eigen::Matrix<double, Nx, Nx> Qo_x0f = d2th.block(1, 1 + Nx, Nx, Nx) / 2;
142 const Eigen::Matrix<double, Nx, Nx> Qo_xf = d2th.block(1 + Nx, 1 + Nx, Nx, Nx) / 2;
143
147
148 for (auto ival = 0u, M = 0u; ival < mesh.N_ivals(); M += mesh.N_colloc_ival(ival), ++ival) {
149 const auto Ki = mesh.N_colloc_ival(ival); // number of nodes in interval
150
151 const auto [alpha, Dus] = mesh.interval_diffmat_unscaled(ival);
152
153 // in each interval the collocation constraint is
154 //
155 // [A0 x0 ... Ak-1 xk-1 0] + [B0 u0 ... Bk-1 uk-1] + [E0 ... Ek-1] = alpha * X Dus
156
157 for (const auto & [i, tau_i] : zip(iota(0u, Ki), mesh.interval_nodes(ival))) {
158 const auto t_i = tf * tau_i; // unscaled time
159 const auto & [xl_i, dxl_i] = diff::dr<1>(xl_fun, wrt(t_i)); // linearization trajectory
160 const auto ul_i = ul_fun(t_i); // linearization input
161
162 // LINEARIZE DYNAMICS
163 const auto & [f_i, df_i] = diff::dr<1, DT>(ocp.f, wrt(t_i, xl_i, ul_i));
164
165 const Eigen::Matrix<double, Nx, Nx> A =
166 tf * (-0.5 * ad<X>(f_i) - 0.5 * ad<X>(dxl_i) + df_i.template middleCols<Nx>(1));
167 const Eigen::Matrix<double, Nx, Nu> B = tf * df_i.template middleCols<Nu>(1 + Nx);
168 const Eigen::Vector<double, Nx> E = tf * (f_i - dxl_i);
169
170 // insert new constraint A xi + B ui + E = [x0 ... XNi] di
171
172 for (auto j = 0u; j < Ki + 1; ++j) {
173 for (auto diag = 0u; diag < Nx; ++diag) {
174 ret.A.coeffRef(dcon_B + (M + i) * Nx + diag, (M + j) * Nx + diag) -= alpha * Dus(j, i);
175 }
176 }
177
178 for (auto row = 0u; row < Nx; ++row) {
179 for (auto col = 0u; col < Nx; ++col) {
180 ret.A.coeffRef(dcon_B + (M + i) * Nx + row, (M + i) * Nx + col) += A(row, col);
181 }
182 for (auto col = 0u; col < Nu; ++col) {
183 ret.A.coeffRef(dcon_B + (M + i) * Nx + row, uvar_B + (M + i) * Nu + col) = B(row, col);
184 }
185 }
186
187 ret.l.segment(dcon_B + (M + i) * Nx, Nx) = -E;
188 ret.u.segment(dcon_B + (M + i) * Nx, Nx) = -E;
189 }
190 }
191
192 auto xslin = mesh.all_nodes() | transform([&](double t) { return xl_fun(t0 + (tf - t0) * t); });
193 auto uslin = mesh.all_nodes() | transform([&](double t) { return ul_fun(t0 + (tf - t0) * t); });
194
195 const Eigen::Vector<double, 1> qlin{1.};
196
200
201 mesh_eval<1, DT>(cr_out, mesh, ocp.cr, 0, tf, xslin, uslin);
202
203 block_add(ret.A, crcon_B, 0, cr_out.dF.middleCols(2, xvar_L + uvar_L));
204 ret.l.segment(crcon_B, crcon_L) = ocp.crl.replicate(N, 1) - cr_out.F;
205 ret.u.segment(crcon_B, crcon_L) = ocp.cru.replicate(N, 1) - cr_out.F;
206
210
211 const auto & [ceval, dceval] = diff::dr<1, DT>(ocp.ce, wrt(tf, xl0, xlf, qlin));
212
213 // integral constraints not supported
214 // assert(dceval.middleCols(1 + 2 * Nx, Nq).cwiseAbs().maxCoeff() < 1e-9);
215
216 block_add(ret.A, cecon_B, xvar_B, dceval.middleCols(1, Nx)); // dce / dx0
217 block_add(ret.A, cecon_B, xvar_B + xvar_L - Nx, dceval.middleCols(1 + Nx, Nx)); // dce / dxf
218
219 ret.l.segment(cecon_B, cecon_L) = ocp.cel - ceval;
220 ret.u.segment(cecon_B, cecon_L) = ocp.ceu - ceval;
221
225
226 mesh_integrate<2, DT>(int_out, mesh, ocp.g, 0, tf, xslin, uslin);
227
228 ret.P = qo_q.x() * int_out.d2F.block(2, 2, xvar_L + uvar_L, xvar_L + uvar_L);
229 ret.q.segment(xvar_B, xvar_L) = qo_q.x() * int_out.dF.middleCols(2, xvar_L).transpose();
230 ret.q.segment(uvar_B, uvar_L) = qo_q.x() * int_out.dF.middleCols(2 + xvar_L, uvar_L).transpose();
231
235
237 for (auto row = 0u; row < Nx; ++row) {
238 for (auto col = 0u; col < Nx; ++col) {
239 ret.P.coeffRef(row, col) += Qo_x0(row, col);
240 ret.P.coeffRef(row, Nx * N + col) += Qo_x0f(row, col); // upper-triangular
241 ret.P.coeffRef(Nx * N + row, Nx * N + col) += Qo_xf(row, col);
242 }
243 }
244
245 ret.q.segment(0, Nx) += qo_x0;
246 ret.q.segment(Nx * N, Nx) += qo_xf;
247
250
251 ret.A.makeCompressed();
252 ret.P.makeCompressed();
253
254 return ret;
255}

272auto qpsol_to_ocpsol(
273 const OCPType auto & ocp,
274 const MeshType auto & mesh,
275 const QPSolution<-1, -1, double> & qpsol,
276 double tf,
277 auto && xl_fun,
278 auto && ul_fun)
279{
280 using ocp_t = std::decay_t<decltype(ocp)>;
281
282 using X = typename ocp_t::X;
283 using U = typename ocp_t::U;
284
285 static constexpr int Nx = ocp_t::Nx;
286 static constexpr int Nu = ocp_t::Nu;
287 static constexpr int Nq = ocp_t::Nq;
288 static constexpr int Ncr = ocp_t::Ncr;
289 static constexpr int Nce = ocp_t::Nce;
290
291 const auto N = mesh.N_colloc();
292
293 const auto xvar_L = Nx * (N + 1);
294 const auto uvar_L = Nu * N;
295
296 const auto xvar_B = 0u;
297 const auto uvar_B = xvar_L;
298 Eigen::MatrixXd Xmat = qpsol.primal.segment(xvar_B, xvar_L).reshaped(Nx, N + 1);
299 Eigen::MatrixXd Umat = qpsol.primal.segment(uvar_B, uvar_L).reshaped(Nu, N);
300
301 auto xfun = [t0 = 0.,
302 tf = tf,
303 mesh = mesh,
304 Xmat = std::move(Xmat),
305 xl_fun = std::forward<decltype(xl_fun)>(xl_fun)](double t) -> X {
306 const auto tngnt =
307 mesh.template eval<Eigen::Vector<double, Nx>>((t - t0) / (tf - t0), Xmat.colwise(), 0, true);
308 return rplus(xl_fun(t), tngnt);
309 };
310
311 auto ufun = [t0 = 0.,
312 tf = tf,
313 mesh = mesh,
314 Umat = std::move(Umat),
315 ul_fun = std::forward<decltype(ul_fun)>(ul_fun)](double t) -> U {
316 const auto tngnt =
317 mesh.template eval<Eigen::Vector<double, Nu>>((t - t0) / (tf - t0), Umat.colwise(), 0, false);
318 return rplus(ul_fun(t), tngnt);
319 };
320
321 return OCPSolution<X, U, Nq, Nce, Ncr>{
322 .t0 = 0.,
323 .tf = tf,
324 .u = std::move(ufun),
325 .x = std::move(xfun),
326 };
327}
