53template<typename Problem>
54QPSolution<-1, -1, double> solve_qp_osqp(
55 const Problem & pbm,
56 const QPSolverParams & prm,
57 std::optional<std::reference_wrapper<const QPSolution<-1, -1, double>>> warmstart = {})
58{
59 // Covert to sparse matrices with OSQP indexing
60 Eigen::SparseMatrix<double, Eigen::ColMajor, c_int> P;
61 Eigen::SparseMatrix<double, Eigen::ColMajor, c_int> A;
62 if constexpr (std::is_base_of_v<Eigen::SparseMatrixBase<decltype(pbm.A)>, decltype(pbm.A)>) {
63 A = pbm.A;
64 P = pbm.P.template triangularView<Eigen::Upper>();
65 } else {
66 A = pbm.A.sparseView();
67 Eigen::MatrixXd Pup = pbm.P.template triangularView<Eigen::Upper>();
68 P = Pup.sparseView();
69 A.prune(1e-6);
70 P.prune(1e-6);
71 }
72 A.makeCompressed();
73 P.makeCompressed();
74
75 OSQPSettings * settings = (OSQPSettings *)c_malloc(sizeof(OSQPSettings));
76 osqp_set_default_settings(settings);
77
78 settings->verbose = prm.verbose;
79 settings->sigma = prm.sigma;
80 settings->alpha = prm.alpha;
81 settings->rho = prm.rho;
82 settings->eps_abs = prm.eps_abs;
83 settings->eps_rel = prm.eps_rel;
84 settings->eps_prim_inf = prm.eps_primal_inf;
85 settings->eps_dual_inf = prm.eps_dual_inf;
86 settings->scaling = prm.scaling;
87 settings->check_termination = prm.stop_check_iter;
88 settings->polish = prm.polish;
89 settings->polish_refine_iter = prm.polish_iter;
90 settings->delta = prm.delta;
91
92 settings->adaptive_rho = false;
93 settings->linsys_solver = QDLDL_SOLVER;
94 settings->scaled_termination = false;
95
96 if (prm.max_iter) {
97 settings->max_iter = prm.max_iter.value();
98 } else {
99 settings->max_iter = std::numeric_limits<c_int>::max();
100 }
101 if (prm.max_time) {
102 settings->time_limit =
103 duration_cast<std::chrono::duration<double>>(prm.max_time.value()).count();
104 } else {
105 settings->time_limit = 0;
106 }
107
108 OSQPData * data = (OSQPData *)c_malloc(sizeof(OSQPData));
109 data->n = A.cols();
110 data->m = A.rows();
111 data->A = csc_matrix(
112 A.rows(), A.cols(), A.nonZeros(), A.valuePtr(), A.innerIndexPtr(), A.outerIndexPtr());
113 data->q = const_cast<double *>(pbm.q.data());
114 data->P = csc_matrix(
115 P.rows(), P.cols(), P.nonZeros(), P.valuePtr(), P.innerIndexPtr(), P.outerIndexPtr());
116 data->l = const_cast<double *>(pbm.l.data());
117 data->u = const_cast<double *>(pbm.u.data());
118
119 OSQPWorkspace * work;
120
121 QPSolution<-1, -1, double> ret;
122 ret.code = QPSolutionStatus::Unknown;
123
124 c_int error = osqp_setup(&work, data, settings);
125
126 if (warmstart) {
127 osqp_warm_start(
128 work, warmstart.value().get().primal.data(), warmstart.value().get().dual.data());
129 settings->warm_start = 1;
130 } else {
131 settings->warm_start = 0;
132 }
133
134 if (!error) { error &= osqp_solve(work); }
135
136 if (!error) {
137 switch (work->info->status_val) {
138 case OSQP_SOLVED: {
139 ret.code = QPSolutionStatus::Optimal;
140 break;
141 }
142 case OSQP_PRIMAL_INFEASIBLE: {
143 ret.code = QPSolutionStatus::PrimalInfeasible;
144 break;
145 }
146 case OSQP_DUAL_INFEASIBLE: {
147 ret.code = QPSolutionStatus::DualInfeasible;
148 break;
149 }
150 case OSQP_MAX_ITER_REACHED: {
151 ret.code = QPSolutionStatus::MaxIterations;
152 break;
153 }
154 case OSQP_TIME_LIMIT_REACHED: {
155 ret.code = QPSolutionStatus::MaxTime;
156 break;
157 }
158 default: {
159 break;
160 }
161 }
162
163 ret.iter = work->info->iter;
164 ret.primal = Eigen::Map<const Eigen::Matrix<double, -1, 1>>(work->solution->x, data->n);
165 ret.dual = Eigen::Map<const Eigen::Matrix<double, -1, 1>>(work->solution->y, data->m);
166 ret.objective = work->info->obj_val;
167 }
168
169 osqp_cleanup(work);
170
171 c_free(data->A);
172 c_free(data->P);
173 c_free(data);
174 c_free(settings);
175
176 return ret;
177}
