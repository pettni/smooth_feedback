48auto ocp_nlp_structure(const FlatOCPType auto & ocp, const MeshType auto & mesh)
49{
50 std::size_t N = mesh.N_colloc();
51
52 // variable layout
53 std::array<std::size_t, 4> var_len{
54 1, // tf
55 ocp.Nq, // integrals
56 ocp.Nx * (N + 1), // states
57 ocp.Nu * N, // inputs
58 };
59
60 // constraint layout
61 std::array<std::size_t, 4> con_len{
62 ocp.Nx * N, // derivatives
63 ocp.Nq, // other integrals
64 ocp.Ncr * N, // running constraints
65 ocp.Nce, // end constraints
66 };
67
68 std::array<std::size_t, 5> var_beg{0};
69 std::partial_sum(var_len.begin(), var_len.end(), var_beg.begin() + 1);
70
71 std::array<std::size_t, 5> con_beg{0};
72 std::partial_sum(con_len.begin(), con_len.end(), con_beg.begin() + 1);
73
74 return std::make_tuple(var_beg, var_len, con_beg, con_len);
75}

82template<FlatOCPType Ocp, MeshType Mesh, diff::Type DT = diff::Type::Default>
83class OCPNLP
84{
85private:
86 static constexpr auto Nx = std::decay_t<Ocp>::Nx;
87 static constexpr auto Nu = std::decay_t<Ocp>::Nu;
88 static constexpr auto Nq = std::decay_t<Ocp>::Nq;
89
90 Ocp ocp_; // optimal control problem
91 Mesh mesh_; // discretization mesh
92 std::size_t N_; // number of collocation points in mesh
93
94 std::size_t tfvar_B, qvar_B, xvar_B, uvar_B, n_; // variable start indices
95 std::size_t tfvar_L, qvar_L, xvar_L, uvar_L; // variable lengths
96
97 std::size_t dcon_B, qcon_B, crcon_B, cecon_B, m_; // constraint start indices
98 std::size_t dcon_L, qcon_L, crcon_L, cecon_L; // constraint lengths
99
100 // scaling
101 double w_scaling_{1};
102
103 // variable and constraint bounds
104 Eigen::VectorXd xl_, xu_, gl_, gu_;
105
106 // allocated return arguments
107 Eigen::VectorXd g_;
108 Eigen::SparseMatrix<double> df_dx_, dg_dx_, d2f_dx2_, d2g_dx2_;
109
110 // allocated computation
111 MeshValue<0> dyn_out0_, int_out0_, cr_out0_;
112 MeshValue<1> dyn_out1_, int_out1_, cr_out1_;
113 MeshValue<2> dyn_out2_, int_out2_, cr_out2_;
114
115public:
117 template<typename OcpArg, typename MeshArg>
118 OCPNLP(OcpArg && ocp, MeshArg && mesh)
119 : ocp_(std::forward<OcpArg>(ocp)), mesh_(std::forward<MeshArg>(mesh)), N_(mesh_.N_colloc())
120 {
121 const auto [var_beg, var_len, con_beg, con_len] = detail::ocp_nlp_structure(ocp_, mesh_);
122
123 tfvar_B = var_beg[0];
124 qvar_B = var_beg[1];
125 xvar_B = var_beg[2];
126 uvar_B = var_beg[3];
127 n_ = var_beg[4];
128
129 tfvar_L = var_len[0];
130 qvar_L = var_len[1];
131 xvar_L = var_len[2];
132 uvar_L = var_len[3];
133
134 dcon_B = con_beg[0];
135 qcon_B = con_beg[1];
136 crcon_B = con_beg[2];
137 cecon_B = con_beg[3];
138 m_ = con_beg[4];
139
140 dcon_L = con_len[0];
141 qcon_L = con_len[1];
142 crcon_L = con_len[2];
143 cecon_L = con_len[3];
144
145 // Mesh weight
146 double max_weight = 1e-6;
147 for (auto w : mesh_.all_weights()) { max_weight = std::max(max_weight, w); }
148 w_scaling_ = 1. / max_weight;
149
150 // VARIABLE BOUNDS
151
152 xl_.setConstant(n_, -std::numeric_limits<double>::infinity());
153 xl_.segment(tfvar_B, tfvar_L).setZero(); // tf lower bounded by zero
154 xu_.setConstant(n_, std::numeric_limits<double>::infinity());
155
156 // CONSTRAINT BOUNDS
157
158 gl_.resize(m_);
159 gu_.resize(m_);
160
161 // derivative constraints are equalities
162 gl_.segment(dcon_B, dcon_L).setZero();
163 gu_.segment(dcon_B, dcon_L).setZero();
164
165 // integral constraints are equalities
166 gl_.segment(qcon_B, qcon_L).setZero();
167 gu_.segment(qcon_B, qcon_L).setZero();
168
169 // running constraints (scaled by quadrature weights)
170 gl_.segment(crcon_B, crcon_L) = ocp.crl.replicate(N_, 1);
171 gu_.segment(crcon_B, crcon_L) = ocp.cru.replicate(N_, 1);
172 for (const auto & [i, w] : utils::zip(std::views::iota(0u, N_), mesh_.all_weights())) {
173 gl_.segment(crcon_B + i * ocp_.Ncr, ocp_.Ncr) *= w_scaling_ * w;
174 gu_.segment(crcon_B + i * ocp_.Ncr, ocp_.Ncr) *= w_scaling_ * w;
175 }
176
177 // end constraints
178 gl_.segment(cecon_B, cecon_L) = ocp.cel;
179 gu_.segment(cecon_B, cecon_L) = ocp.ceu;
180
181 // allocate output args
182 g_.setZero(m_);
183
184 df_dx_.resize(1, n_);
185 df_dx_.reserve(Eigen::VectorXi::Constant(n_, 1));
186
187 d2f_dx2_.resize(n_, n_);
188 dg_dx_.resize(m_, n_);
189 d2g_dx2_.resize(n_, n_);
190
192 }

195 std::size_t m() const { return m_; }

197 const Eigen::VectorXd & xu() const { return xu_; }

199 double f(const Eigen::Ref<const Eigen::VectorXd> x) const
200 {
201 assert(static_cast<std::size_t>(x.size()) == n_);
202
203 const auto x0var_B = xvar_B;
204 const auto xfvar_B = xvar_B + xvar_L - Nx;
205
206 const double tf = x(tfvar_B);
207 const Eigen::Vector<double, Nx> x0 = x.segment(x0var_B, Nx);
208 const Eigen::Vector<double, Nx> xf = x.segment(xfvar_B, Nx);
209 const Eigen::Vector<double, Nq> q = x.segment(qvar_B, qvar_L);
210
211 return ocp_.theta(tf, x0, xf, q);
212 }

214 const Eigen::SparseMatrix<double> & df_dx(const Eigen::Ref<const Eigen::VectorXd> x)
215 {
216 assert(static_cast<std::size_t>(x.size()) == n_);
217
218 const auto x0var_B = xvar_B;
219 const auto xfvar_B = xvar_B + xvar_L - Nx;
220
221 const double tf = x(tfvar_B);
222 const Eigen::Vector<double, Nx> x0 = x.segment(x0var_B, Nx);
223 const Eigen::Vector<double, Nx> xf = x.segment(xfvar_B, Nx);
224 const Eigen::Vector<double, Nq> q = x.segment(qvar_B, qvar_L);
225
226 const auto & [fval, dfval] = diff::dr<1, DT>(ocp_.theta, wrt(tf, x0, xf, q));
227
228 set_zero(df_dx_);
229
230 block_add(df_dx_, 0, tfvar_B, dfval.middleCols(0, 1)); // df / dtf
231 block_add(df_dx_, 0, x0var_B, dfval.middleCols(1, Nx)); // df / dx0
232 block_add(df_dx_, 0, xfvar_B, dfval.middleCols(1 + Nx, Nx)); // df / dxf
233 block_add(df_dx_, 0, qvar_B, dfval.middleCols(1 + 2 * Nx, Nq)); // df / dq
234
235 df_dx_.makeCompressed();
236 return df_dx_;
237 }

239 const Eigen::SparseMatrix<double> & d2f_dx2(Eigen::Ref<const Eigen::VectorXd> x)
240 {
241 assert(static_cast<std::size_t>(x.size()) == n_);
242
243 const auto x0var_B = xvar_B;
244 const auto xfvar_B = xvar_B + xvar_L - Nx;
245
246 const double tf = x(tfvar_B);
247 const Eigen::Vector<double, Nx> x0 = x.segment(x0var_B, Nx);
248 const Eigen::Vector<double, Nx> xf = x.segment(xfvar_B, Nx);
249 const Eigen::Vector<double, Nq> q = x.segment(qvar_B, qvar_L);
250
251 const auto & [fval, dfval, d2fval] = diff::dr<2, DT>(ocp_.theta, wrt(tf, x0, xf, q));
252
253 set_zero(d2f_dx2_);
254
255 // clang-format off
256 block_add(d2f_dx2_, tfvar_B, tfvar_B, d2fval.block( 0, 0, 1, 1), 1, true); // tftf
257 block_add(d2f_dx2_, tfvar_B, x0var_B, d2fval.block( 0, 1, 1, Nx), 1, true); // tfx0
258 block_add(d2f_dx2_, tfvar_B, xfvar_B, d2fval.block( 0, 1 + Nx, 1, Nx), 1, true); // tfxf
259 block_add(d2f_dx2_, tfvar_B, qvar_B, d2fval.block( 0, 1 + 2 * Nx, 1, Nq), 1, true); // tfq
260
261 block_add(d2f_dx2_, x0var_B, x0var_B, d2fval.block( 1, 1, Nx, Nx), 1, true); // x0x0
262 block_add(d2f_dx2_, x0var_B, xfvar_B, d2fval.block( 1, 1 + Nx, Nx, Nx), 1, true); // x0xf
263 block_add(d2f_dx2_, x0var_B, qvar_B, d2fval.block( 1, 1 + 2 * Nx, Nx, Nq), 1, true); // x0q
264
265 block_add(d2f_dx2_, xfvar_B, xfvar_B, d2fval.block( 1 + Nx, 1 + Nx, Nx, Nx), 1, true); // xfxf
266 block_add(d2f_dx2_, xfvar_B, qvar_B, d2fval.block( 1 + Nx, 1 + 2 * Nx, Nx, Nq), 1, true); // xfq
267
268 block_add(d2f_dx2_, qvar_B, qvar_B, d2fval.block(1 + 2 * Nx, 1 + 2 * Nx, Nq, Nq), 1, true); // qq
269 // clang-format on
270
271 d2f_dx2_.makeCompressed();
272 return d2f_dx2_;
273 }

275 const Eigen::VectorXd & g(const Eigen::Ref<const Eigen::VectorXd> x)
276 {
277 assert(static_cast<std::size_t>(x.size()) == n_);
278
279 const auto x0var_B = xvar_B;
280 const auto xfvar_B = xvar_B + xvar_L - Nx;
281
282 const double t0 = 0;
283 const double tf = x(tfvar_B);
284 const Eigen::Vector<double, Nx> x0 = x.segment(x0var_B, Nx);
285 const Eigen::Vector<double, Nx> xf = x.segment(xfvar_B, Nx);
286 const Eigen::Vector<double, Nq> q = x.segment(qvar_B, qvar_L);
287
288 const Eigen::Map<const Eigen::Matrix<double, Nx, -1>> X(x.data() + xvar_B, Nx, N_ + 1);
289 const Eigen::Map<const Eigen::Matrix<double, Nu, -1>> U(x.data() + uvar_B, Nu, N_);
290
291 mesh_dyn<0>(dyn_out0_, mesh_, ocp_.f, t0, tf, X.colwise(), U.colwise());
292 mesh_integrate<0>(int_out0_, mesh_, ocp_.g, t0, tf, X.colwise(), U.colwise());
293 mesh_eval<0>(cr_out0_, mesh_, ocp_.cr, t0, tf, X.colwise(), U.colwise(), true);
294
295 g_.segment(dcon_B, dcon_L) = w_scaling_ * dyn_out0_.F;
296 g_.segment(qcon_B, qcon_L) = w_scaling_ * (int_out0_.F - q);
297 g_.segment(crcon_B, crcon_L) = w_scaling_ * cr_out0_.F;
298 g_.segment(cecon_B, cecon_L) = ocp_.ce(tf, x0, xf, q);
299
300 return g_;
301 }
302 const Eigen::VectorXd & gl() const { return gl_; }
303 const Eigen::VectorXd & gu() const { return gu_; }
304 const Eigen::SparseMatrix<double> & dg_dx(const Eigen::Ref<const Eigen::VectorXd> x)
305 {
306 assert(static_cast<std::size_t>(x.size()) == n_);
307
308 const auto x0var_B = xvar_B;
309 const auto xfvar_B = xvar_B + xvar_L - Nx;
310
311 const double t0 = 0;
312 const double tf = x(tfvar_B);
313 const Eigen::Vector<double, Nx> x0 = x.segment(x0var_B, Nx);
314 const Eigen::Vector<double, Nx> xf = x.segment(xfvar_B, Nx);
315 const Eigen::Vector<double, Nq> q = x.segment(qvar_B, qvar_L);
316
317 const Eigen::Map<const Eigen::Matrix<double, Nx, -1>> X(x.data() + xvar_B, Nx, N_ + 1);
318 const Eigen::Map<const Eigen::Matrix<double, Nu, -1>> U(x.data() + uvar_B, Nu, N_);
319
320 mesh_dyn<1, DT>(dyn_out1_, mesh_, ocp_.f, t0, tf, X.colwise(), U.colwise());
321 mesh_integrate<1, DT>(int_out1_, mesh_, ocp_.g, t0, tf, X.colwise(), U.colwise());
322 mesh_eval<1, DT>(cr_out1_, mesh_, ocp_.cr, t0, tf, X.colwise(), U.colwise(), true);
323 const auto & [ceval, dceval] = diff::dr<1, DT>(ocp_.ce, wrt(tf, x0, xf, q));
324
325 dyn_out1_.dF.makeCompressed();
326 int_out1_.dF.makeCompressed();
327 cr_out1_.dF.makeCompressed();
328
329 set_zero(dg_dx_);
330
331 // dynamics constraint
332 block_add(dg_dx_, dcon_B, tfvar_B, dyn_out1_.dF.middleCols(1, 1), w_scaling_);
333 block_add(dg_dx_, dcon_B, xvar_B, dyn_out1_.dF.middleCols(2, xvar_L), w_scaling_);
334 block_add(dg_dx_, dcon_B, uvar_B, dyn_out1_.dF.middleCols(2 + xvar_L, uvar_L), w_scaling_);
335
336 // integral constraint
337 block_add(dg_dx_, qcon_B, tfvar_B, int_out1_.dF.middleCols(1, 1), w_scaling_);
338 block_add(dg_dx_, qcon_B, xvar_B, int_out1_.dF.middleCols(2, xvar_L), w_scaling_);
339 block_add(dg_dx_, qcon_B, uvar_B, int_out1_.dF.middleCols(2 + xvar_L, uvar_L), w_scaling_);
340 block_add_identity(dg_dx_, qcon_B, qvar_B, qvar_L, -w_scaling_);
341
342 // running constraint
343 block_add(dg_dx_, crcon_B, tfvar_B, cr_out1_.dF.middleCols(1, 1), w_scaling_);
344 block_add(dg_dx_, crcon_B, xvar_B, cr_out1_.dF.middleCols(2, xvar_L), w_scaling_);
345 block_add(dg_dx_, crcon_B, uvar_B, cr_out1_.dF.middleCols(2 + xvar_L, uvar_L), w_scaling_);
346
347 // end constraint
348 block_add(dg_dx_, cecon_B, tfvar_B, dceval.middleCols(0, 1));
349 block_add(dg_dx_, cecon_B, xvar_B, dceval.middleCols(1, Nx));
350 block_add(dg_dx_, cecon_B, xvar_B + xvar_L - Nx, dceval.middleCols(1 + Nx, Nx));
351 block_add(dg_dx_, cecon_B, qvar_B, dceval.middleCols(1 + 2 * Nx, Nq));
352
353 dg_dx_.makeCompressed();
354 return dg_dx_;
355 }

357 const Eigen::SparseMatrix<double> &
358 d2g_dx2(const Eigen::Ref<const Eigen::VectorXd> x, const Eigen::Ref<const Eigen::VectorXd> lambda)
359 {
360 assert(static_cast<std::size_t>(x.size()) == n_);
361 assert(static_cast<std::size_t>(lambda.size()) == m_);
362
363 const auto x0var_B = xvar_B;
364 const auto xfvar_B = xvar_B + xvar_L - Nx;
365
366 const double t0 = 0;
367 const double tf = x(tfvar_B);
368 const Eigen::Vector<double, Nx> x0 = x.segment(x0var_B, Nx);
369 const Eigen::Vector<double, Nx> xf = x.segment(xfvar_B, Nx);
370 const Eigen::Vector<double, Nq> q = x.segment(qvar_B, qvar_L);
371
372 const Eigen::Map<const Eigen::Matrix<double, Nx, -1>> X(x.data() + xvar_B, Nx, N_ + 1);
373 const Eigen::Map<const Eigen::Matrix<double, Nu, -1>> U(x.data() + uvar_B, Nu, N_);
374
375 dyn_out2_.lambda = lambda.segment(dcon_B, dcon_L);
376 int_out2_.lambda = lambda.segment(qcon_B, qcon_L);
377 cr_out2_.lambda = lambda.segment(crcon_B, crcon_L);
378 mesh_dyn<2, DT>(dyn_out2_, mesh_, ocp_.f, t0, tf, X.colwise(), U.colwise());
379 mesh_integrate<2, DT>(int_out2_, mesh_, ocp_.g, t0, tf, X.colwise(), U.colwise());
380 mesh_eval<2, DT>(cr_out2_, mesh_, ocp_.cr, t0, tf, X.colwise(), U.colwise(), true);
381 const auto & [ceval, dceval, d2ceval] = diff::dr<2, DT>(ocp_.ce, wrt(tf, x0, xf, q));
382
383 dyn_out2_.dF.makeCompressed();
384 int_out2_.dF.makeCompressed();
385 cr_out2_.dF.makeCompressed();
386
387 dyn_out2_.d2F.makeCompressed();
388 int_out2_.d2F.makeCompressed();
389 cr_out2_.d2F.makeCompressed();
390
391 set_zero(d2g_dx2_);
392
393 // clang-format off
394 block_add(d2g_dx2_, tfvar_B, tfvar_B, dyn_out2_.d2F.block(1, 1, 1, 1), w_scaling_, true); // tftf
395 block_add(d2g_dx2_, tfvar_B, tfvar_B, int_out2_.d2F.block(1, 1, 1, 1), w_scaling_, true); // tftf
396 block_add(d2g_dx2_, tfvar_B, tfvar_B, cr_out2_.d2F.block(1, 1, 1, 1), w_scaling_, true); // tftf
397
398 block_add(d2g_dx2_, tfvar_B, xvar_B, dyn_out2_.d2F.block(1, 2, 1, xvar_L), w_scaling_, true); // tfx
399 block_add(d2g_dx2_, tfvar_B, xvar_B, int_out2_.d2F.block(1, 2, 1, xvar_L), w_scaling_, true); // tfx
400 block_add(d2g_dx2_, tfvar_B, xvar_B, cr_out2_.d2F.block(1, 2, 1, xvar_L), w_scaling_, true); // tfx
401
402 block_add(d2g_dx2_, tfvar_B, uvar_B, dyn_out2_.d2F.block(1, 2 + xvar_L, 1, uvar_L), w_scaling_, true); // tfu
403 block_add(d2g_dx2_, tfvar_B, uvar_B, int_out2_.d2F.block(1, 2 + xvar_L, 1, uvar_L), w_scaling_, true); // tfu
404 block_add(d2g_dx2_, tfvar_B, uvar_B, cr_out2_.d2F.block(1, 2 + xvar_L, 1, uvar_L), w_scaling_, true); // tfu
405
406 block_add(d2g_dx2_, xvar_B, xvar_B, dyn_out2_.d2F.block(2, 2, xvar_L, xvar_L), w_scaling_, true); // xx
407 block_add(d2g_dx2_, xvar_B, xvar_B, int_out2_.d2F.block(2, 2, xvar_L, xvar_L), w_scaling_, true); // xx
408 block_add(d2g_dx2_, xvar_B, xvar_B, cr_out2_.d2F.block(2, 2, xvar_L, xvar_L), w_scaling_, true); // xx
409
410 block_add(d2g_dx2_, xvar_B, uvar_B, dyn_out2_.d2F.block(2, 2 + xvar_L, xvar_L, uvar_L), w_scaling_, true); // xu
411 block_add(d2g_dx2_, xvar_B, uvar_B, int_out2_.d2F.block(2, 2 + xvar_L, xvar_L, uvar_L), w_scaling_, true); // xu
412 block_add(d2g_dx2_, xvar_B, uvar_B, cr_out2_.d2F.block(2, 2 + xvar_L, xvar_L, uvar_L), w_scaling_, true); // xu
413
414 block_add(d2g_dx2_, uvar_B, uvar_B, dyn_out2_.d2F.block(2 + xvar_L, 2 + xvar_L, uvar_L, uvar_L), w_scaling_, true); // uu
415 block_add(d2g_dx2_, uvar_B, uvar_B, int_out2_.d2F.block(2 + xvar_L, 2 + xvar_L, uvar_L, uvar_L), w_scaling_, true); // uu
416 block_add(d2g_dx2_, uvar_B, uvar_B, cr_out2_.d2F.block(2 + xvar_L, 2 + xvar_L, uvar_L, uvar_L), w_scaling_, true); // uu
417 // clang-format on
418
419 for (auto j = 0u; j < ocp_.Nce; ++j) {
420 const auto b0 = (1 + 2 * Nx + ocp_.Nq) * j;
421 // clang-format off
422 block_add(d2g_dx2_, tfvar_B, tfvar_B, d2ceval.block( 0, b0 + 0, 1, 1), lambda(cecon_B + j), true); // tftf
423 block_add(d2g_dx2_, tfvar_B, x0var_B, d2ceval.block( 0, b0 + 1, 1, Nx), lambda(cecon_B + j), true); // tfx0
424 block_add(d2g_dx2_, tfvar_B, xfvar_B, d2ceval.block( 0, b0 + 1 + Nx, 1, Nx), lambda(cecon_B + j), true); // tfxf
425 block_add(d2g_dx2_, tfvar_B, qvar_B, d2ceval.block( 0, b0 + 1 + 2 * Nx, 1, Nq), lambda(cecon_B + j), true); // tfq
426
427 block_add(d2g_dx2_, x0var_B, x0var_B, d2ceval.block( 1, b0 + 1, Nx, Nx), lambda(cecon_B + j), true); // x0x0
428 block_add(d2g_dx2_, x0var_B, xfvar_B, d2ceval.block( 1, b0 + 1 + Nx, Nx, Nx), lambda(cecon_B + j), true); // x0xf
429 block_add(d2g_dx2_, x0var_B, qvar_B, d2ceval.block( 1, b0 + 1 + 2 * Nx, Nx, Nq), lambda(cecon_B + j), true); // x0q
430
431 block_add(d2g_dx2_, xfvar_B, xfvar_B, d2ceval.block( 1 + Nx, b0 + 1 + Nx, Nx, Nx), lambda(cecon_B + j), true); // xfxf

466auto nlpsol_to_ocpsol(
467 const FlatOCPType auto & ocp, const MeshType auto & mesh, const NLPSolution & nlp_sol)
468{
469 using ocp_t = std::decay_t<decltype(ocp)>;
470
471 static constexpr auto Nx = ocp_t::Nx;
472 static constexpr auto Nu = ocp_t::Nu;
473 static constexpr auto Nq = ocp_t::Nq;
474 static constexpr auto Ncr = ocp_t::Ncr;
475
476 const std::size_t N = mesh.N_colloc();
477 const auto [var_beg, var_len, con_beg, con_len] = detail::ocp_nlp_structure(ocp, mesh);
478
479 const auto [tfvar_B, qvar_B, xvar_B, uvar_B, n] = var_beg;
480 const auto [tfvar_L, qvar_L, xvar_L, uvar_L] = var_len;
481
482 const auto [dcon_B, qcon_B, crcon_B, cecon_B, m] = con_beg;
483 const auto [dcon_L, qcon_L, crcon_L, cecon_L] = con_len;
484
485 const double t0 = 0;
486 const double tf = nlp_sol.x(tfvar_B);
487
488 const Eigen::Vector<double, Nq> Q = nlp_sol.x.segment(qvar_B, qvar_L);
489
490 // state vector has a value at the endpoint
491
492 Eigen::MatrixXd X(ocp.Nx, N + 1);
493 X = nlp_sol.x.segment(xvar_B, xvar_L).reshaped(ocp.Nx, xvar_L / ocp.Nx);
494
495 auto xfun =
496 [t0 = t0, tf = tf, mesh = mesh, X = std::move(X)](double t) -> Eigen::Vector<double, Nx> {
497 return mesh.template eval<Eigen::Vector<double, Nx>>(
498 (t - t0) / (tf - t0), X.colwise(), 0, true);
499 };
500
501 // for these we repeat last point since there are no values for endpoint
502
503 Eigen::MatrixXd U(ocp.Nu, N);
504 U = nlp_sol.x.segment(uvar_B, uvar_L).reshaped(ocp.Nu, uvar_L / ocp.Nu);
505
506 auto ufun =
507 [t0 = t0, tf = tf, mesh = mesh, U = std::move(U)](double t) -> Eigen::Vector<double, Nu> {
508 return mesh.template eval<Eigen::Vector<double, Nu>>(
509 (t - t0) / (tf - t0), U.colwise(), 0, false);
510 };
511
512 Eigen::MatrixXd Ldyn(ocp.Nx, N);
513 Ldyn = nlp_sol.lambda.segment(dcon_B, dcon_L).reshaped(ocp.Nx, dcon_L / ocp.Nx);
514
515 auto ldfun =
516 [t0 = t0, tf = tf, mesh = mesh, Ldyn = std::move(Ldyn)](double t) -> Eigen::Vector<double, Nx> {
517 return mesh.template eval<Eigen::Vector<double, Nx>>(
518 (t - t0) / (tf - t0), Ldyn.colwise(), 0, false);
519 };
520
521 Eigen::MatrixXd Lcr(ocp.Ncr, N);
522 Lcr = nlp_sol.lambda.segment(crcon_B, crcon_L).reshaped(ocp.Ncr, crcon_L / ocp.Ncr);
523
524 auto lcrfun =
525 [t0 = t0, tf = tf, mesh = mesh, Lcr = std::move(Lcr)](double t) -> Eigen::Vector<double, Ncr> {
526 return mesh.template eval<Eigen::Vector<double, Ncr>>(
527 (t - t0) / (tf - t0), Lcr.colwise(), 0, false);
528 };
529
530 return OCPSolution<typename ocp_t::X, typename ocp_t::U, ocp_t::Nq, ocp_t::Ncr, ocp_t::Nce>{
531 .t0 = t0,
532 .tf = tf,
533 .Q = std::move(Q),
534 .u = std::move(ufun),
535 .x = std::move(xfun),
536 .lambda_q = nlp_sol.lambda.segment(qcon_B, qcon_L),
537 .lambda_ce = nlp_sol.lambda.segment(cecon_B, cecon_L),
538 .lambda_dyn = std::move(ldfun),
539 .lambda_cr = std::move(lcrfun),
540 };
541}

548NLPSolution
549ocpsol_to_nlpsol(const FlatOCPType auto & ocp, const MeshType auto & mesh, const auto & ocpsol)
550{
551 const auto N = mesh.N_colloc();
552
553 const auto [var_beg, var_len, con_beg, con_len] = detail::ocp_nlp_structure(ocp, mesh);
554
555 const auto [tfvar_B, qvar_B, xvar_B, uvar_B, n] = var_beg;
556 const auto [tfvar_L, qvar_L, xvar_L, uvar_L] = var_len;
557
558 const auto [dcon_B, qcon_B, crcon_B, cecon_B, m] = con_beg;
559 const auto [dcon_L, qcon_L, crcon_L, cecon_L] = con_len;
560
561 const double t0 = 0;
562 const double tf = ocpsol.tf;
563
564 Eigen::VectorXd x(n), lambda(m);
565
566 x(tfvar_B) = ocpsol.tf;
567 x.segment(qvar_B, qvar_L) = ocpsol.Q;
568
569 lambda.segment(qcon_B, qcon_L) = ocpsol.lambda_q;
570 lambda.segment(cecon_B, cecon_L) = ocpsol.lambda_ce;
571
572 for (const auto & [i, tau] : utils::zip(std::views::iota(0u), mesh.all_nodes())) {
573 x.segment(xvar_B + i * ocp.Nx, ocp.Nx) = ocpsol.x(t0 + tau * (tf - t0));
574 if (i < N) {
575 x.segment(uvar_B + i * ocp.Nu, ocp.Nu) = ocpsol.u(t0 + tau * (tf - t0));
576 lambda.segment(dcon_B + i * ocp.Nx, ocp.Nx) = ocpsol.lambda_dyn(t0 + tau * (tf - t0));
577 lambda.segment(crcon_B + i * ocp.Ncr, ocp.Ncr) = ocpsol.lambda_cr(t0 + tau * (tf - t0));
578 }
579 }
580
581 return {
582 .status = NLPSolution::Status::Unknown,
583 .x = std::move(x),
584 .zl = Eigen::VectorXd::Zero(n),
585 .zu = Eigen::VectorXd::Zero(n),
586 .lambda = std::move(lambda),
587 };
588}
