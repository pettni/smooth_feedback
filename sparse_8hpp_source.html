57template<typename Source, int Options>
58 requires(std::is_base_of_v<Eigen::EigenBase<Source>, Source>)
59void block_add(
60 Eigen::SparseMatrix<double, Options> & dest,
61 Eigen::Index row0,
62 Eigen::Index col0,
63 const Source & source,
64 double scale = 1,
65 double upper_only = false)
66{
67 for (auto c = 0; c < source.outerSize(); ++c) {
68 for (Eigen::InnerIterator it(source, c); it; ++it) {
69 if (!upper_only || row0 + it.row() <= col0 + it.col()) {
70 dest.coeffRef(row0 + it.row(), col0 + it.col()) += scale * it.value();
71 }
72 }
73 }
74}

107template<typename Scalar, int Options>
108void set_zero(Eigen::SparseMatrix<Scalar, Options> & mat)
109{
110 if (mat.isCompressed()) {
111 mat.coeffs().setZero();
112 } else {
113 for (auto i = 0; i < mat.outerSize(); ++i) {
114 for (Eigen::InnerIterator it(mat, i); it; ++it) { mat.coeffRef(it.row(), it.col()) = 0; }
115 }
116 }
117}

128 uint64_t ret = 0;

147template<typename Mat>
148 requires(std::is_base_of_v<Eigen::EigenBase<Mat>, Mat>)
149Eigen::MatrixX<typename Mat::Scalar> mark_explicit_zeros(const Mat & mat)
150{
151 Eigen::MatrixX<typename Mat::Scalar> ret;
152 ret.setConstant(mat.rows(), mat.cols(), 0);
153 for (auto c = 0; c < mat.outerSize(); ++c) {
154 for (Eigen::InnerIterator it(mat, c); it; ++it) {
155 if (it.value() == 0) {
156 ret(it.row(), it.col()) = 9;
157 } else {
158 ret(it.row(), it.col()) = 1;
159 }
160 }
161 }
162 return ret;
163}

178template<typename S1, typename S2, typename S3, typename S4>
179 requires(
180 std::is_base_of_v<Eigen::EigenBase<S1>, S1> && std::is_base_of_v<Eigen::EigenBase<S2>, S2> &&
181 std::is_base_of_v<Eigen::EigenBase<S3>, S3> && std::is_base_of_v<Eigen::EigenBase<S4>, S4>)
182inline void d2r_fog(
183 Eigen::SparseMatrix<double> & out,
184 const S1 & Jf,
185 const S2 & Hf,
186 const S3 & Jg,
187 const S4 & Hg,
188 Eigen::Index r0 = 0,
189 Eigen::Index c0 = 0)
190{
191 const auto No = Jf.rows();
192 const auto Ny = Jf.cols();
193
194 [[maybe_unused]] const auto Ni = Jg.rows();
195 const auto Nx = Jg.cols();
196
197 // check some dimensions
198 assert(Ny == Ni);
199 assert(Hf.rows() == Ny);
200 assert(Hf.cols() == No * Ny);
201 assert(Hg.rows() == Nx);
202 assert(Hg.cols() == Ni * Nx);
203
204 for (auto no = 0u; no < No; ++no) {
205 // TODO sparse-sparse-sparse product is expensive and allocates temporary
206 block_add(out, r0, c0 + no * Nx, Jg.transpose() * Hf.middleCols(no * Ny, Ny) * Jg);
207 }
208
209 for (auto i = 0u; i < Jf.outerSize(); ++i) {
210 for (Eigen::InnerIterator it(Jf, i); it; ++it) {
211 block_add(out, r0, c0 + it.row() * Nx, Hg.middleCols(it.col() * Nx, Nx), it.value());
212 }
213 }
214}
