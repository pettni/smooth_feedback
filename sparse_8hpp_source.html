57template<typename Source, int Options>
58 requires(std::is_base_of_v<Eigen::EigenBase<Source>, Source>)
59inline void block_add(
60 Eigen::SparseMatrix<double, Options> & dest,
61 Eigen::Index row0,
62 Eigen::Index col0,
63 const Source & source,
64 double scale = 1,
65 double upper_only = false)
66{
67 for (auto c = 0; c < source.outerSize(); ++c) {
68 for (Eigen::InnerIterator it(source, c); it; ++it) {
69 if (!upper_only || row0 + it.row() <= col0 + it.col()) {
70 dest.coeffRef(row0 + it.row(), col0 + it.col()) += scale * it.value();
71 }
72 }
73 }
74}

170template<typename SparseMat>
171 requires(std::is_base_of_v<
172 Eigen::SparseCompressedBase<std::decay_t<SparseMat>>,
173 std::decay_t<SparseMat>>)
174inline void set_zero(SparseMat && mat)
175{
176 if (mat.isCompressed()) {
177 mat.coeffs().setZero();
178 } else {
179 for (auto i = 0; i < mat.outerSize(); ++i) {
180 for (typename std::decay_t<decltype(mat)>::InnerIterator it(mat, i); it; ++it) {
181 it.valueRef() = 0;
182 }
183 }
184 }
185}

215template<typename Mat>
216 requires(std::is_base_of_v<Eigen::EigenBase<Mat>, Mat>)
217Eigen::MatrixX<typename Mat::Scalar> mark_explicit_zeros(const Mat & mat)
218{
219 Eigen::MatrixX<typename Mat::Scalar> ret;
220 ret.setConstant(mat.rows(), mat.cols(), 0);
221 for (auto c = 0; c < mat.outerSize(); ++c) {
222 for (Eigen::InnerIterator it(mat, c); it; ++it) {
223 if (it.value() == 0) {
224 ret(it.row(), it.col()) = 9;
225 } else {
226 ret(it.row(), it.col()) = 1;
227 }
228 }
229 }
230 return ret;

246template<typename S1, typename S2, typename S3, typename S4>
247 requires(
248 std::is_base_of_v<Eigen::EigenBase<S1>, S1> && std::is_base_of_v<Eigen::EigenBase<S2>, S2> &&
249 std::is_base_of_v<Eigen::EigenBase<S3>, S3> && std::is_base_of_v<Eigen::EigenBase<S4>, S4>)
250inline void d2r_fog(
251 Eigen::SparseMatrix<double> & out,
252 const S1 & Jf,
253 const S2 & Hf,
254 const S3 & Jg,
255 const S4 & Hg,
256 Eigen::Index r0 = 0,
257 Eigen::Index c0 = 0)
258{
259 const auto No = Jf.rows();
260 const auto Ny = Jf.cols();
261
262 [[maybe_unused]] const auto Ni = Jg.rows();
263 const auto Nx = Jg.cols();
264
265 // check some dimensions
266 assert(Ny == Ni);
267 assert(Hf.rows() == Ny);
268 assert(Hf.cols() == No * Ny);
269 assert(Hg.rows() == Nx);
270 assert(Hg.cols() == Ni * Nx);
271
272 for (auto no = 0u; no < No; ++no) {
273 // TODO sparse-sparse-sparse product is expensive and allocates temporary
274 block_add(out, r0, c0 + no * Nx, Jg.transpose() * Hf.middleCols(no * Ny, Ny) * Jg);
275 }
276
277 for (auto i = 0u; i < Jf.outerSize(); ++i) {
278 for (Eigen::InnerIterator it(Jf, i); it; ++it) {
279 block_add(out, r0, c0 + it.row() * Nx, Hg.middleCols(it.col() * Nx, Nx), it.value());
280 }
281 }
282}
