23 inline Eigen::SparseMatrix<double> sparse_block_matrix(
24  const std::initializer_list<std::initializer_list<std::optional<Eigen::SparseMatrix<double>>>> &
25  l)
26 {
27  const auto n_rows = l.size();
28  const auto n_cols = std::begin(l)->size();
29 
30  Eigen::VectorXi dims_rows = Eigen::VectorXi::Constant(n_rows, -1);
31  Eigen::VectorXi dims_cols = Eigen::VectorXi::Constant(n_cols, -1);
32 
33  // figure block row and col dimensions
34  for (auto krow = 0u; const auto & row : l) {
35  for (auto kcol = 0u; const auto & item : row) {
36  if (item.has_value()) {
37  if (dims_cols(kcol) == -1) {
38  dims_cols(kcol) = item->cols();
39  } else {
40  assert(dims_cols(kcol) == item->cols());
41  }
42  if (dims_rows(krow) == -1) {
43  dims_rows(krow) = item->rows();
44  } else {
45  assert(dims_rows(krow) == item->rows());
46  }
47  }
48  ++kcol;
49  }
50  ++krow;
51  }
52 
53  // check that all dimensions are defined by input args
54  assert(dims_rows.minCoeff() > -1);
55  assert(dims_cols.minCoeff() > -1);
56 
57  // figure starting indices
58  const auto n_row = std::accumulate(std::cbegin(dims_rows), std::cend(dims_rows), 0u);
59  const auto n_col = std::accumulate(std::cbegin(dims_cols), std::cend(dims_cols), 0u);
60 
61  Eigen::SparseMatrix<double> ret(n_row, n_col);
62 
63  // allocate pattern
64  Eigen::Matrix<decltype(ret)::StorageIndex, -1, 1> pattern(n_col);
65  pattern.setZero();
66  for (const auto & row : l) {
67  for (auto kcol = 0u, col0 = 0u; const auto &item : row) {
68  if (item.has_value()) {
69  for (auto col = 0; col < dims_cols(kcol); ++col) {
70  pattern(col0 + col) += item->outerIndexPtr()[col + 1] - item->outerIndexPtr()[col];
71  }
72  }
73  col0 += dims_cols(kcol++);
74  }
75  }
76 
77  ret.reserve(pattern);
78 
79  // insert values
80  for (auto krow = 0u, row0 = 0u; const auto &row : l) {
81  for (auto kcol = 0u, col0 = 0u; const auto &item : row) {
82  if (item.has_value()) {
83  for (auto col = 0; col < dims_cols(kcol); ++col) {
84  for (typename std::decay_t<decltype(*item)>::InnerIterator it(*item, col); it; ++it) {
85  ret.insert(row0 + it.index(), col0 + col) = it.value();
86  }
87  }
88  }
89  col0 += dims_cols(kcol++);
90  }
91  row0 += dims_rows(krow++);
92  }
93 
94  ret.makeCompressed();
95 
96  return ret;
97 }

104 inline Eigen::SparseMatrix<double> sparse_identity(std::size_t n)
105 {
106  Eigen::SparseMatrix<double> ret(n, n);

121 inline auto kron_identity(const Eigen::SparseCompressedBase<Derived> & X, std::size_t n)
122 {
123  Eigen::
124  SparseMatrix<typename Derived::Scalar, Derived::IsRowMajor ? Eigen::RowMajor : Eigen::ColMajor>
125  ret(X.rows() * n, X.cols() * n);
126 
127  Eigen::Matrix<int, -1, 1> pattern(X.outerSize() * n);
128 
129  for (auto i0 = 0u, i = 0u; i < X.outerSize(); ++i) {
130  auto nnz_i = X.outerIndexPtr()[i + 1] - X.outerIndexPtr()[i];
131  pattern.segment(i0, n).setConstant(nnz_i);
132  i0 += n;
133  }
134 
135  ret.reserve(pattern);
136 
137  for (auto i0 = 0u; i0 < X.outerSize(); ++i0) {
138  for (typename std::decay_t<decltype(X)>::InnerIterator it(X, i0); it; ++it) {
139  for (auto diag = 0u; diag < n; ++diag) {
140  ret.insert(n * it.row() + diag, n * it.col() + diag) = it.value();
141  }
142  }
143  }
144 
145  ret.makeCompressed();
146 
147  return ret;
148 }

162 
163  ret.reserve(Eigen::Matrix<int, -1, 1>::Constant(ret.cols(), n * X.rows()));
164 
165  for (auto row = 0u; row < X.rows(); ++row) {
166  for (auto col = 0u; col < X.cols(); ++col) {
167  for (auto diag = 0u; diag < n; ++diag) {
168  ret.insert(n * row + diag, n * col + diag) = X(row, col);
