97  inline Tangent<G> operator()(const T & t, const G & x, const Tangent<G> & v)
98  {
99  const auto [g_des, v_des, a_des] = x_des_(t);
100  const Tangent<G> g_err = g_des - x;
101 
102  if (t_last && t > t_last.value()) {
103  // update integral state
104  i_err_ += time_trait<T>::minus(t, t_last.value()) * g_err;
105  i_err_ = i_err_.cwiseMax(-prm_.windup_limit).cwiseMin(prm_.windup_limit);
106  }
107  t_last = t;
108 
109  return a_des + kp_.cwiseProduct(g_err) + kd_.cwiseProduct(v_des - v) + ki_.cwiseProduct(i_err_);
110  }
