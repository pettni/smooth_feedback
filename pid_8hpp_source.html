121  typename G::Tangent operator()(const T & t, const G & g, const typename G::Tangent & dg_dt)
122  {
123  const auto [x_des, v_des, a_des] = x_des_(t);
124 
125  typename G::Tangent g_err = x_des - g;
126  typename G::Tangent v_err = v_des - dg_dt;
127 
128  if (t_last && t > t_last.value()) {
129  // update integral state
130  double dt =
131  std::chrono::duration_cast<std::chrono::duration<double>>(t - t_last.value()).count();
132  i_err_ += dt * g_err;
133  i_err_ = i_err_.cwiseMax(-prm_.windup_limit).cwiseMin(prm_.windup_limit);
134  }
135  t_last = t;
136 
137  return a_des + kp_.cwiseProduct(g_err) + kd_.cwiseProduct(v_err) + ki_.cwiseProduct(i_err_);
138  }
