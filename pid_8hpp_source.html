98 inline Tangent<G> operator()(const T & t, const G & x, const Tangent<G> & v)
99 {
100 const auto [g_des, v_des, a_des] = x_des_(t);
101 const Tangent<G> g_err = g_des - x;
102
103 if (t_last && t > t_last.value()) {
104 // update integral state
105 i_err_ += time_trait<T>::minus(t, t_last.value()) * g_err;
106 i_err_ = i_err_.cwiseMax(-prm_.windup_limit).cwiseMin(prm_.windup_limit);
107 }
108 t_last = t;
109
110 return a_des + kp_.cwiseProduct(g_err) + kd_.cwiseProduct(v_des - v) + ki_.cwiseProduct(i_err_);
111 }
