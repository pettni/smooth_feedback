175  typename G::Tangent operator()(const T & t, const G & g, const typename G::Tangent & v)
176  {
177  const auto [g_des, v_des, a_des] = x_des_(t);
178 
179  const typename G::Tangent g_err = g_des - g;
180 
181  if (t_last && t > t_last.value()) {
182  // update integral state
183  double dt =
184  std::chrono::duration_cast<std::chrono::duration<double>>(t - t_last.value()).count();
185  i_err_ += dt * g_err;
186  i_err_ = i_err_.cwiseMax(-prm_.windup_limit).cwiseMin(prm_.windup_limit);
187  }
188  t_last = t;
189 
190  return a_des + kp_.cwiseProduct(g_err) + kd_.cwiseProduct(v_des - v) + ki_.cwiseProduct(i_err_);
191  }
