103template<uint8_t Deriv>
104void setZero(MeshValue<Deriv> & mv)
105{
106 mv.F.setZero();
107
108 if constexpr (Deriv >= 1) {
109 if (mv.dF.isCompressed()) {
110 mv.dF.coeffs().setZero();
111 } else {
112 for (auto i = 0; i < mv.dF.outerSize(); ++i) {
113 for (Eigen::InnerIterator it(mv.dF, i); it; ++it) {
114 mv.dF.coeffRef(it.row(), it.col()) = 0;
115 }
116 }
117 }
118 }
119 if constexpr (Deriv >= 2) {
120 if (mv.d2F.isCompressed()) {
121 mv.d2F.coeffs().setZero();
122 } else {
123 for (auto i = 0; i < mv.d2F.outerSize(); ++i) {
124 for (Eigen::InnerIterator it(mv.d2F, i); it; ++it) {
125 mv.d2F.coeffRef(it.row(), it.col()) = 0;
126 }
127 }
128 }
129 }

159template<uint8_t Deriv, diff::Type DT = diff::Type::Default>
160 requires(Deriv <= 2)
161void mesh_eval(
162 MeshValue<Deriv> & out,
163 const MeshType auto & m,
164 auto & f,
165 const double t0,
166 const double tf,
167 std::ranges::range auto && xs,
168 std::ranges::range auto && us,
169 bool scale = false)
170{
171 using utils::zip, std::views::iota;
172 using X = PlainObject<std::decay_t<std::ranges::range_value_t<decltype(xs)>>>;
173 using U = PlainObject<std::decay_t<std::ranges::range_value_t<decltype(us)>>>;
174
175 static constexpr auto nx = Dof<X>;
176 static constexpr auto nu = Dof<U>;
177 static constexpr auto nf = Dof<std::invoke_result_t<decltype(f), double, X, U>>;
178
179 static_assert(nx > -1, "State dimension must be static");
180 static_assert(nu > -1, "Input dimension must be static");
181 static_assert(nf > -1, "Output size must be static");
182
183 const auto N = m.N_colloc();
184
185 const Eigen::Index numOuts = nf * N;
186 const Eigen::Index numVars = 2 + nx * (N + 1) + nu * N;
187
188 // ALLOCATION
189
190 if (!out.allocated) {
191 out.F.resize(numOuts);
192
193 if constexpr (Deriv >= 1) {
194 Eigen::VectorXi pattern = Eigen::VectorXi::Zero(numVars);
195 pattern.segment(0, 1).setConstant(numOuts); // t0 is dense
196 pattern.segment(1, 1).setConstant(numOuts); // tf is dense
197 pattern.segment(2, nx * N).setConstant(nf); // block diagonal, last x not used
198 pattern.segment(2 + nx * (N + 1), nu * N).setConstant(nf); // block diagonal
199
200 out.dF.resize(numOuts, numVars);
201 out.dF.reserve(pattern);
202 }
203
204 if constexpr (Deriv >= 2u) {
205 Eigen::VectorXi pattern = Eigen::VectorXi::Zero(numVars);
206 pattern.segment(0, 1).setConstant(1); // t0
207 pattern.segment(1, 1).setConstant(2); // t0, tf
208 for (auto i = 0u; i < N; ++i) {
209 for (auto j = 0u; j < nx; ++j) {
210 pattern(2 + nx * i + j) = 2 + (j + 1); // t0, tf, x upper diag
211 }
212 }
213 for (auto i = 0u; i < N; ++i) {
214 for (auto j = 0u; j < nu; ++j) {
215 pattern(2 + nx * (N + 1) + nu * i + j) = 2 + nx + (j + 1); // t0, tf, x, u upper diag
216 }
217 }
218 out.d2F.resize(numVars, numVars);
219 out.d2F.reserve(pattern);
220 }
221
222 out.allocated = true;
223 }
224
225 setZero(out);
226
227 for (const auto & [i, tau, w_quad, x, u] :
228 zip(iota(0u, N), m.all_nodes(), m.all_weights(), xs, us)) {
229 const double ti = t0 + (tf - t0) * tau;
230 const X xi = x;
231 const U ui = u;
232
233 const double mtau = 1. - tau;
234 const double w = scale ? w_quad : 1.;
235
236 const auto fvals = diff::dr<Deriv, DT>(f, wrt(ti, xi, ui));
237
238 out.F.segment(i * nf, nf) = w * std::get<0>(fvals);
239
240 if constexpr (Deriv >= 1u) {
241 const auto & df = std::get<1>(fvals);
242 const auto row0 = nf * i;
243
244 block_add(out.dF, row0, 0, df.middleCols(0, 1), w * mtau);
245 block_add(out.dF, row0, 1, df.middleCols(0, 1), w * tau);
246 block_add(out.dF, row0, 2 + i * nx, df.middleCols(1, nx), w);
247 block_add(out.dF, row0, 2 + nx * (N + 1) + nu * i, df.middleCols(1 + nx, nu), w);
248
249 if constexpr (Deriv >= 2u) {
250 assert(out.lambda.size() == numOuts);
251
252 const auto & d2f = std::get<2>(fvals);
253
254 for (auto j = 0u; j < nf; ++j) {
255 const double wl = w * out.lambda(row0 + j);
256
257 // destination block locations
258 const auto t0_d = 0;
259 const auto tf_d = 1;
260 const auto x_d = 2 + i * nx;
261 const auto u_d = 2 + (N + 1) * nx + i * nu;
262
263 // source block locations
264 const auto b_s = (1 + nx + nu) * j;
265 const auto t_s = 0;
266 const auto x_s = 1;
267 const auto u_s = 1 + nx;
268
269 // clang-format off
270 // t0 row
271 block_add(out.d2F, t0_d, t0_d, d2f.block(t_s, b_s + t_s, 1, 1 ), wl * mtau * mtau, true);
272 block_add(out.d2F, t0_d, tf_d, d2f.block(t_s, b_s + t_s, 1, 1 ), wl * mtau * tau, true);
273 block_add(out.d2F, t0_d, x_d, d2f.block(t_s, b_s + x_s, 1, nx), wl * mtau, true);
274 block_add(out.d2F, t0_d, u_d, d2f.block(t_s, b_s + u_s, 1, nu), wl * mtau, true);
275
276 // tf row
277 block_add(out.d2F, tf_d, tf_d, d2f.block(t_s, b_s + t_s, 1, 1 ), wl * tau * tau, true);
278 block_add(out.d2F, tf_d, x_d, d2f.block(t_s, b_s + x_s, 1, nx), wl * tau, true);
279 block_add(out.d2F, tf_d, u_d, d2f.block(t_s, b_s + u_s, 1, nu), wl * tau, true);
280
281 // x row
282 block_add(out.d2F, x_d, x_d, d2f.block(x_s, b_s + x_s, nx, nx), wl, true);
283 block_add(out.d2F, x_d, u_d, d2f.block(x_s, b_s + u_s, nx, nu), wl, true);
284
285 // u row
286 block_add(out.d2F, u_d, u_d, d2f.block(u_s, b_s + u_s, nu, nu), wl, true);
287 // clang-format on
288 }
289 }
290 }
291 }

319template<uint8_t Deriv, diff::Type DT = diff::Type::Default>
320 requires(Deriv <= 2)
321void mesh_integrate(
322 MeshValue<Deriv> & out,
323 const MeshType auto & m,
324 auto & f,
325 const double t0,
326 const double tf,
327 std::ranges::range auto && xs,
328 std::ranges::range auto && us)
329{
330 using utils::zip, std::views::iota;
331 using X = PlainObject<std::decay_t<std::ranges::range_value_t<decltype(xs)>>>;
332 using U = PlainObject<std::decay_t<std::ranges::range_value_t<decltype(us)>>>;
333
334 static constexpr auto nx = Dof<X>;
335 static constexpr auto nu = Dof<U>;
336 static constexpr auto nf = Dof<std::invoke_result_t<decltype(f), double, X, U>>;
337
338 static_assert(nx > -1, "State dimension must be static");
339 static_assert(nu > -1, "Input dimension must be static");
340 static_assert(nf > -1, "Output size must be static");
341
342 const auto N = m.N_colloc();
343
344 const Eigen::Index numOuts = nf;
345 const Eigen::Index numVars = 2 + nx * (N + 1) + nu * N;
346
347 // ALLOCATION
348
349 if (!out.allocated) {
350 out.F.resize(numOuts);
351
352 if constexpr (Deriv >= 1) {
353 Eigen::VectorXi pattern = Eigen::VectorXi::Constant(numVars, numOuts); // dense
354 pattern.segment(2 + nx * N, nx).setZero();
355
356 out.dF.resize(numOuts, numVars);
357 out.dF.reserve(pattern);
358 }
359
360 if constexpr (Deriv >= 2) {
361 Eigen::VectorXi pattern = Eigen::VectorXi::Zero(numVars); // dense
362 pattern(0) = 1; // t0 dense
363 pattern(1) = 2; // tf dense
364 for (auto i = 0u; i < N; ++i) {
365 for (auto j = 0u; j < nx; ++j) {
366 pattern(2 + nx * i + j) = 2 + (j + 1); // t0, tf, x upper diag
367 }
368 }
369 for (auto i = 0u; i < N; ++i) {
370 for (auto j = 0u; j < nu; ++j) {
371 pattern(2 + nx * (N + 1) + nu * i + j) = 2 + nx + (j + 1); // t0, tf, x, u upper diag
372 }
373 }
374
375 out.d2F.resize(numVars, numVars);
376 out.d2F.reserve(pattern.replicate(numOuts, 1).reshaped());
377 }
378
379 out.allocated = true;
380 }
381
382 setZero(out);
383
384 for (const auto & [i, tau, w, x, u] : zip(iota(0u, N), m.all_nodes(), m.all_weights(), xs, us)) {
385 const double ti = t0 + (tf - t0) * tau;
386 const X xi = x;
387 const U ui = u;
388
389 const double mtau = 1. - tau;
390
391 const auto fvals = diff::dr<Deriv, DT>(f, wrt(ti, xi, ui));
392
393 const auto & fval = std::get<0>(fvals);
394 out.F.noalias() += w * (tf - t0) * fval;
395
396 if constexpr (Deriv >= 1u) {
397 const auto & df = std::get<1>(fvals);
398 // t0
399 block_add(out.dF, 0, 0, df.middleCols(0, 1), w * (tf - t0) * mtau);
400 block_add(out.dF, 0, 0, fval, -w);
401 // tf
402 block_add(out.dF, 0, 1, df.middleCols(0, 1), w * (tf - t0) * tau);
403 block_add(out.dF, 0, 1, fval, w);
404 // x
405 block_add(out.dF, 0, 2 + i * nx, df.middleCols(1, nx), w * (tf - t0));
406 // u
407 block_add(out.dF, 0, 2 + nx * (N + 1) + nu * i, df.middleCols(1 + nx, nu), w * (tf - t0));
408
409 if constexpr (Deriv >= 2u) {
410 assert(out.lambda.size() == numOuts);
411
412 const auto & d2f = std::get<2>(fvals);
413
414 for (auto j = 0u; j < nf; ++j) {
415 const double wl = w * out.lambda(j);
416
417 // source block locations
418 const auto b_s = (1 + nx + nu) * j; // horizontal block
419 const auto t_s = 0; // t
420 const auto x_s = 1; // x
421 const auto u_s = 1 + nx; // u
422
423 // destination block locations
424 const auto t0_d = 0; // t0
425 const auto tf_d = 1; // tf
426 const auto x_d = 2 + nx * i; // x[i]
427 const auto u_d = 2 + nx * (N + 1) + nu * i; // u[i]
428
429 // clang-format off
430 // t0t0
431 block_add(out.d2F, t0_d, t0_d, d2f.block(t_s, b_s + t_s, 1, 1), wl * (tf - t0) * mtau * mtau, true);
432 block_add(out.d2F, t0_d, t0_d, df.block(j, t_s, 1, 1), -wl * 2 * mtau, true);
433 // t0tf
434 block_add(out.d2F, t0_d, tf_d, d2f.block(t_s, b_s + t_s, 1, 1), wl * (tf - t0) * mtau * tau, true);
435 block_add(out.d2F, t0_d, tf_d, df.block(j, t_s, 1, 1), wl * (1 - 2 * tau), true);
436 // t0x
437 block_add(out.d2F, t0_d, x_d, d2f.block(t_s, b_s + x_s, 1, nx), wl * (tf - t0) * mtau, true);
438 block_add(out.d2F, t0_d, x_d, df.block(j, x_s, 1, nx), -wl, true);
439 // t0u
440 block_add(out.d2F, t0_d, u_d, d2f.block(t_s, b_s + u_s, 1, nu), wl * (tf - t0) * mtau, true);
441 block_add(out.d2F, t0_d, u_d, df.block(j, u_s, 1, nu), -wl, true);
442
443 // tftf
444 block_add(out.d2F, tf_d, tf_d, d2f.block(t_s, b_s + t_s, 1, 1), wl * (tf - t0) * tau * tau, true);
445 block_add(out.d2F, tf_d, tf_d, df.block(j, t_s, 1, 1), wl * 2 * tau, true);
446 // tfx
447 block_add(out.d2F, tf_d, x_d, d2f.block(t_s, b_s + x_s, 1, nx), wl * (tf - t0) * tau, true);
448 block_add(out.d2F, tf_d, x_d, df.block(j, x_s, 1, nx), wl, true);
449 // tfu
450 block_add(out.d2F, tf_d, u_d, d2f.block(t_s, b_s + u_s, 1, nu), wl * (tf - t0) * tau, true);
451 block_add(out.d2F, tf_d, u_d, df.block(j, u_s, 1, nu), wl, true);
452
453 // xx
454 block_add(out.d2F, x_d, x_d, d2f.block(x_s, b_s + x_s, nx, nx), wl * (tf - t0), true);
455 // xu
456 block_add(out.d2F, x_d, u_d, d2f.block(x_s, b_s + u_s, nx, nu), wl * (tf - t0), true);
457
458 // uu
459 block_add(out.d2F, u_d, u_d, d2f.block(u_s, b_s + u_s, nu, nu), wl * (tf - t0), true);
460 // clang-format on
461 }
462 }
463 }
464 }

496template<uint8_t Deriv, diff::Type DT = diff::Type::Default>
497 requires(Deriv <= 2)
498void mesh_dyn(
499 MeshValue<Deriv> & out,
500 const MeshType auto & m,
501 auto & f,
502 const double t0,
503 const double tf,
504 std::ranges::range auto && xs,
505 std::ranges::range auto && us)
506{
507 using utils::zip, std::views::iota;
508 using X = PlainObject<std::decay_t<std::ranges::range_value_t<decltype(xs)>>>;
509 using U = PlainObject<std::decay_t<std::ranges::range_value_t<decltype(us)>>>;
510
511 static constexpr auto nx = Dof<X>;
512 static constexpr auto nu = Dof<U>;
513 static constexpr auto nf = Dof<std::invoke_result_t<decltype(f), double, X, U>>;
514
515 static_assert(nx > -1, "State dimension must be static");
516 static_assert(nu > -1, "Input dimension must be static");
517 static_assert(nf > -1, "Output size must be static");
518 static_assert(nx == nf, "Output dimension must be same as state dimension");
519
520 const auto N = m.N_colloc();
521 const Eigen::Index numOuts = nx * N;
522 const Eigen::Index numVars = 2 + nx * (N + 1) + nu * N;
523
524 // ALLOCATION
525
526 if (!out.allocated) {
527 out.F.resize(numOuts);
528
529 if constexpr (Deriv >= 1) {
530 Eigen::VectorXi pattern = Eigen::VectorXi::Zero(numVars);
531 pattern(0) = numOuts; // t0 dense
532 pattern(1) = numOuts; // tf dense
533
534 // x has blocks depending on mesh size
535 auto idx0 = 2u;
536 for (auto ival = 0u; ival < m.N_ivals(); ++ival) {
537 const std::size_t K = m.N_colloc_ival(ival);
538 pattern.segment(idx0, K * nx) += Eigen::VectorXi::Constant(K * nx, nx + K - 1);
539 pattern.segment(idx0 + K * nx, nx) += Eigen::VectorXi::Constant(nx, K);
540 idx0 += K * nx;
541 }
542
543 // u is block diagonal with small blocks
544 pattern.segment(2 + nx * (N + 1), nu * N).setConstant(nx);
545
546 out.dF.resize(numOuts, numVars);
547 out.dF.reserve(pattern);
548 }
549
550 if constexpr (Deriv >= 2) {
551 Eigen::VectorXi pattern = Eigen::VectorXi::Zero(numVars);
552 pattern(0) = 1;
553 pattern(1) = 2;
554 for (auto i = 0u; i < N; ++i) {
555 for (auto j = 0u; j < nx; ++j) {
556 pattern(2 + nx * i + j) = 2 + (j + 1); // t0, tf, x upper diag
557 }
558 }
559 for (auto i = 0u; i < N; ++i) {
560 for (auto j = 0u; j < nu; ++j) {
561 pattern(2 + nx * (N + 1) + nu * i + j) = 2 + nx + (j + 1); // t0, tf, x, u upper diag
562 }
563 }
564
565 out.d2F.resize(numVars, numVars);
566 out.d2F.reserve(pattern);
567 }
568
569 out.allocated = true;
570 }
571
572 setZero(out);
573
574 // We build the constraint through two loops over xi:
575 // - the first loop adds wk * f(tk, xk, uk)
576 // - the second loop adds -wk * [x0 ... Xni] * dk
577
578 // ADD FIRST PART
579
580 for (const auto & [i, tau, w, x, u] : zip(iota(0u, N), m.all_nodes(), m.all_weights(), xs, us)) {
581 const double ti = t0 + (tf - t0) * tau;
582 const X xi = x;
583 const U ui = u;
584
585 const auto row0 = nx * i;
586 const double mtau = 1. - tau;
587
588 const auto fvals = diff::dr<Deriv, DT>(f, wrt(ti, xi, ui));
589 const auto fval = std::get<0>(fvals);
590
591 out.F.segment(row0, nx) += w * (tf - t0) * fval;
592
593 if constexpr (Deriv >= 1) {
594 const auto & df = std::get<1>(fvals);
595
596 // clang-format off
597 // dF/dt0 = -f + (tf - t0) * df/dti * (1-tau)
598 block_add(out.dF, row0, 0, fval, -w);
599 block_add(out.dF, row0, 0, df.col(0), w * (tf - t0) * mtau);
600 // dF/dtf = f + (tf - t0) * df/dti * tau
601 block_add(out.dF, row0, 1, fval, w);
602 block_add(out.dF, row0, 1, df.col(0), w * (tf - t0) * tau);
603 // dF/dx
604 block_add(out.dF, row0, 2 + nx * i, df.middleCols(1, nx), w * (tf - t0));
605 // dF/du
606 block_add(out.dF, row0, 2 + nx * (N + 1) + nu * i, df.middleCols(1 + nx, nu), w * (tf - t0));
607 // clang-format on
608
609 if constexpr (Deriv >= 2) {
610 assert(out.lambda.size() == numOuts);
611
612 const auto & d2f = std::get<2>(fvals);
613
614 for (auto j = 0u; j < nx; ++j) {
615 const double wl = w * out.lambda(row0 + j);
616
617 // destination block locations
618 const auto t0_d = 0;
619 const auto tf_d = 1;
620 const auto x_d = 2 + i * nx;
621 const auto u_d = 2 + (N + 1) * nx + i * nu;
622
623 // source block locations
624 const auto b_s = (1 + nx + nu) * j;
625 const auto t_s = 0;
626 const auto x_s = 1;
627 const auto u_s = 1 + nx;
628
629 // clang-format off
630 // t0t0
631 block_add(out.d2F, t0_d, t0_d, d2f.block(t_s, b_s + t_s, 1, 1 ), wl * (tf - t0) * mtau * mtau, true);
632 block_add(out.d2F, t0_d, t0_d, df.block(j, t_s, 1, 1 ), -wl * 2 * mtau, true);
633 // t0tf
634 block_add(out.d2F, t0_d, tf_d, d2f.block(t_s, b_s + t_s, 1, 1 ), wl * (tf - t0) * mtau * tau, true);
635 block_add(out.d2F, t0_d, tf_d, df.block(j, t_s, 1, 1 ), wl * (1. - 2 * tau), true);
636 // t0x
637 block_add(out.d2F, t0_d, x_d, d2f.block(t_s, b_s + x_s, 1, nx), wl * (tf - t0) * mtau, true);
638 block_add(out.d2F, t0_d, x_d, df.block(j, x_s, 1, nx), -wl, true);
639 // t0u
640 block_add(out.d2F, t0_d, u_d, d2f.block(t_s, b_s + u_s, 1, nu), wl * (tf - t0) * mtau, true);
641 block_add(out.d2F, t0_d, u_d, df.block(j, u_s, 1, nu), -wl, true);
642
643 // tftf
644 block_add(out.d2F, tf_d, tf_d, d2f.block(t_s, b_s + t_s, 1, 1), wl * (tf - t0) * tau * tau, true);
645 block_add(out.d2F, tf_d, tf_d, df.block(j, t_s, 1, 1), wl * 2 * tau, true);
646 // tfx
647 block_add(out.d2F, tf_d, x_d, d2f.block(t_s, b_s + x_s, 1, nx), wl * (tf - t0) * tau, true);
